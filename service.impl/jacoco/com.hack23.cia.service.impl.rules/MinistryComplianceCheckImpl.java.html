package com.hack23.cia.service.impl.rules;

import com.hack23.cia.model.internal.application.data.ministry.impl.ViewRiksdagenMinistry;
import com.hack23.cia.model.internal.application.data.rules.impl.ResourceType;

/**
 * The Class MinistryComplianceCheckImpl.
 */
public final class MinistryComplianceCheckImpl extends AbstractComplianceCheckImpl {

	/** The Constant serialVersionUID. */
	private static final long serialVersionUID = 1L;

	/** The ministry. */
	private final ViewRiksdagenMinistry ministry;

	/** The name. */
	private final String name;

	/**
	 * Instantiates a new ministry compliance check impl.
	 *
	 * @param ministry the ministry
	 */
	public MinistryComplianceCheckImpl(final ViewRiksdagenMinistry ministry) {
		super(ResourceType.MINISTRY);
		this.ministry = ministry;
		this.name = ministry.getNameId();
	}

	/**
	 * Gets the ministry.
	 *
	 * @return the ministry
	 */
	public ViewRiksdagenMinistry getMinistry() {
		return ministry;
	}

	@Override
	public String getName() {
		return name;
	}

	@Override
	public String getId() {
		return ministry.getNameId();
	}

}
