19  package com.hack23.cia.service.impl.rules;
20  
21  import java.util.List;
22  
23  import org.slf4j.Logger;
24  import org.slf4j.LoggerFactory;
25  import org.springframework.beans.factory.annotation.Autowired;
26  import org.springframework.beans.factory.annotation.Qualifier;
27  import org.springframework.stereotype.Service;
28  import org.springframework.transaction.annotation.Propagation;
29  import org.springframework.transaction.annotation.Transactional;
30  
31  import com.hack23.cia.model.internal.application.data.rules.impl.RuleViolation;
32  import com.hack23.cia.service.api.action.kpi.ComplianceCheck;
33  import com.hack23.cia.service.data.api.RuleViolationDAO;
34  
35  /**
36   * The Class ComplianceCheckService.
37   */
38  @Service
39  @Transactional(propagation = Propagation.REQUIRED, timeout = 300)
40  public final class RulesManagerImpl implements RulesManager {
41  
42  	/** The Constant LOGGER. */
43  	private static final Logger LOGGER = LoggerFactory.getLogger(RulesManagerImpl.class);
44  
45  	@Autowired
46  	private RulesEngine rulesEngine;
47  
48  	@Autowired
49  	@Qualifier("RuleViolationDAO")
50  	private RuleViolationDAO ruleViolationDAO;
51  
52  	/**
53  	 * Instantiates a new compliance check service.
54  	 */
55  	public RulesManagerImpl() {
56  	}
57  
58  	@Override
59  	public void processService() {
60  		LOGGER.info("{}", RulesManagerImpl.class.getClass().getSimpleName());
61  
62  		final List<ComplianceCheck> complianceList = rulesEngine.checkRulesCompliance();
63  
64  		for (final RuleViolation ruleViolation : ruleViolationDAO.getAll()) {
65  			ruleViolationDAO.delete(ruleViolation);
66  		}
67  
68  		for (final ComplianceCheck complianceCheck : complianceList) {
69  			ruleViolationDAO.persist(complianceCheck.getRuleViolations());
70  		}
71  	}
72  }
