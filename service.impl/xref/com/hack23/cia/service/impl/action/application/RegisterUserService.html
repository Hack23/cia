63  /**
64   * The Class RegisterUserService.
65   */
66  @Service
67  @Transactional(propagation = Propagation.REQUIRED)
68  public final class RegisterUserService extends AbstractBusinessServiceImpl<RegisterUserRequest, RegisterUserResponse> {
69  
70  	/** The Constant LOGGER. */
71  	private static final Logger LOGGER = LoggerFactory.getLogger(RegisterUserService.class);
72  
73  	/** The application configuration service. */
74  	@Autowired
75  	private ApplicationConfigurationService applicationConfigurationService;
76  
77  	/** The password encoder. */
78  	private final PasswordEncoder passwordEncoder = new BCryptPasswordEncoder();
79  
80  	/** The password validator. */
81  	private final PasswordValidator passwordValidator = new PasswordValidator(new LengthRule(8, 64),
82  			new CharacterRule(EnglishCharacterData.UpperCase, 1), new CharacterRule(EnglishCharacterData.LowerCase, 1),
83  			new CharacterRule(EnglishCharacterData.Digit, 1), new CharacterRule(EnglishCharacterData.Special, 1),
84  			new WhitespaceRule());
85  
86  	/**
87  	 * Instantiates a new register user service.
88  	 */
89  	public RegisterUserService() {
90  		super(RegisterUserRequest.class);
91  	}
92  
93  	@Override
94  	@Secured({ "ROLE_ANONYMOUS" })
95  	public RegisterUserResponse processService(final RegisterUserRequest serviceRequest) {
96  		final RegisterUserResponse inputValidation = inputValidation(serviceRequest);
97  		if (inputValidation != null) {
98  			return inputValidation;
99  		}
100 
101 		final CreateApplicationEventRequest eventRequest = createApplicationEventForService(serviceRequest);
102 
103 		RegisterUserResponse response;
104 
105 		final UserAccount userNameExist = getUserDAO().findFirstByProperty(UserAccount_.username,
106 				serviceRequest.getUsername());
107 		final UserAccount userEmailExist = getUserDAO().findFirstByProperty(UserAccount_.email, serviceRequest.getEmail());
108 
109 		final RuleResult passwordRuleResults = passwordValidator
110 				.validate(new PasswordData(serviceRequest.getUserpassword()));
111 
112 		if (userEmailExist == null && userNameExist == null && passwordRuleResults.isValid()) {
113 			final UserAccount userAccount = createUserAccount(serviceRequest);
114 			eventRequest.setUserId(userAccount.getUserId());
115 			response = new RegisterUserResponse(ServiceResult.SUCCESS);
116 		} else {
117 			response = new RegisterUserResponse(ServiceResult.FAILURE);
118 			if (passwordRuleResults.isValid()) {
119 				response.setErrorMessage(RegisterUserResponse.ErrorMessage.USER_ALREADY_EXIST.toString());
120 				eventRequest.setErrorMessage(RegisterUserResponse.ErrorMessage.USER_ALREADY_EXIST.toString());
121 			} else {
122 				final String errorMessage = passwordValidator.getMessages(passwordRuleResults).toString();
123 				response.setErrorMessage(errorMessage);
124 				eventRequest.setErrorMessage(errorMessage);
125 			}
126 		}
127 
128 		eventRequest.setApplicationMessage(response.getResult().toString());
129 		createApplicationEventService.processService(eventRequest);
130 		LOGGER.info("Event: {}", eventRequest);
131 		return response;
132 	}
133 
134 	/**
135 	 * Creates the user account.
136 	 *
137 	 * @param serviceRequest the service request
138 	 * @return the user account
139 	 */
140 	private UserAccount createUserAccount(final RegisterUserRequest serviceRequest) {
141 
142 		final ApplicationConfiguration registeredUsersGetAdminConfig = applicationConfigurationService
143 				.checkValueOrLoadDefault("Registered User All get Role Admin", "Registered User All get Role Admin",
144 						ConfigurationGroup.AUTHORIZATION, RegisterUserService.class.getSimpleName(),
145 						"Register User Service", "Responsible for create of useraccounts", "registered.users.get.admin",
146 						"true");
147 
148 		final UserAccount userAccount = new UserAccount();
149 		userAccount.setCountry(serviceRequest.getCountry());
150 		userAccount.setEmail(serviceRequest.getEmail());
151 		userAccount.setUsername(serviceRequest.getUsername());
152 		userAccount.setUserId(UUID.randomUUID().toString());
153 		userAccount.setUserpassword(
154 				passwordEncoder.encode(userAccount.getUserId() + ".uuid" + serviceRequest.getUserpassword()));
155 		userAccount.setNumberOfVisits(1);
156 
157 		if ( serviceRequest.getUserType() == null) {
158 			userAccount.setUserType(UserType.PRIVATE);
159 		} else {
160 			userAccount.setUserType(serviceRequest.getUserType());
161 		}
162 
163 		userAccount.setUserEmailStatus(UserEmailStatus.UNKNOWN);
164 		userAccount.setUserLockStatus(UserLockStatus.UNLOCKED);
165 		userAccount.setCreatedDate(new Date());
166 		getUserDAO().persist(userAccount);
167 
168 		final Collection<SimpleGrantedAuthority> authorities = new ArrayList<>();
169 		if ("true".equals(registeredUsersGetAdminConfig.getPropertyValue())) {
170 			userAccount.setUserRole(UserRole.ADMIN);
171 			authorities.add(new SimpleGrantedAuthority("ROLE_ADMIN"));
172 		} else {
173 			userAccount.setUserRole(UserRole.USER);
174 			authorities.add(new SimpleGrantedAuthority("ROLE_USER"));
175 		}
176 
177 		SecurityContextHolder.getContext().setAuthentication(
178 				new UsernamePasswordAuthenticationToken(userAccount.getUserId(), "n/a", authorities));
179 		return userAccount;
180 	}
181 
182 	@Override
183 	protected CreateApplicationEventRequest createApplicationEventForService(
184 			final RegisterUserRequest serviceRequest) {
185 		final CreateApplicationEventRequest eventRequest = createBaseApplicationEventRequest();
186 		eventRequest.setEventGroup(ApplicationEventGroup.USER);
187 		eventRequest.setApplicationOperation(ApplicationOperationType.CREATE);
188 		eventRequest.setActionName(RegisterUserRequest.class.getSimpleName());
189 		eventRequest.setSessionId(serviceRequest.getSessionId());
190 		eventRequest.setElementId(serviceRequest.getEmail());
191 		return eventRequest;
192 	}
193 
194 	@Override
195 	protected RegisterUserResponse createErrorResponse() {
196 		return new RegisterUserResponse(ServiceResult.FAILURE);
197 	}
