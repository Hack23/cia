47  /**
48   * The Class SetGoogleAuthenticatorCredentialService.
49   */
50  @Service
51  @Transactional(propagation = Propagation.REQUIRED,timeout=600)
52  public final class SetGoogleAuthenticatorCredentialService extends
53  		AbstractBusinessServiceImpl<SetGoogleAuthenticatorCredentialRequest, SetGoogleAuthenticatorCredentialResponse> {
54  
55  	/** The Constant LOGGER. */
56  	private static final Logger LOGGER = LoggerFactory
57  			.getLogger(SetGoogleAuthenticatorCredentialService.class);
58  
59  	@Autowired
60  	private AgencyDAO agencyDAO;
61  
62  	/** The encrypted value DAO. */
63  	@Autowired
64  	private EncryptedValueDAO encryptedValueDAO;
65  
66  	@Autowired
67  	private VaultManager vaultManager;
68  
69  	/** The password encoder. */
70  	private final PasswordEncoder passwordEncoder = new BCryptPasswordEncoder();
71  
72  	/**
73  	 * Instantiates a new sets the google authenticator credential service.
74  	 */
75  	public SetGoogleAuthenticatorCredentialService() {
76  		super(SetGoogleAuthenticatorCredentialRequest.class);
77  	}
78  
79  
80  	@Override
81  	@Secured({ "ROLE_USER", "ROLE_ADMIN"})
82  	public SetGoogleAuthenticatorCredentialResponse processService(
83  			final SetGoogleAuthenticatorCredentialRequest serviceRequest) {
84  
85  		final SetGoogleAuthenticatorCredentialResponse inputValidation = inputValidation(serviceRequest);
86  		if (inputValidation != null) {
87  			return inputValidation;
88  		}
89  
90  		LOGGER.info("{}:{}",serviceRequest.getClass().getSimpleName(),serviceRequest.getSessionId());
91  		final CreateApplicationEventRequest eventRequest = createApplicationEventForService(serviceRequest);
92  
93  		final UserAccount userAccount = getUserAccountFromSecurityContext();
94  
95  		SetGoogleAuthenticatorCredentialResponse response = new SetGoogleAuthenticatorCredentialResponse(ServiceResult.SUCCESS);
96  		if (userAccount != null) {
97  
98  			final GoogleAuthenticator gAuth = new GoogleAuthenticator();
99  			final GoogleAuthenticatorKey gKey = gAuth.createCredentials();
100 
101 			if (passwordEncoder.matches(
102 					userAccount.getUserId() + ".uuid" + serviceRequest.getUserpassword(), userAccount.getUserpassword())) {
103 
104 				final EncryptedValue encryptedValue = new EncryptedValue();
105 				encryptedValue.setId(userAccount.getHjid());
106 				encryptedValue.setUserId(userAccount.getUserId());
107 				encryptedValue.setVaultName(GoogleAuthenticatorKey.class.getSimpleName());
108 				encryptedValue.setStorage(vaultManager.encryptValue(serviceRequest.getUserpassword(), userAccount.getUserId(), gKey.getKey()));
109 				encryptedValueDAO.persist(encryptedValue);
110 
111 				final String otpAuthTotpURL = GoogleAuthenticatorQRGenerator.getOtpAuthTotpURL(agencyDAO.getAll().get(0).getAgencyName(), userAccount.getEmail(), gKey);
112 
113 				response.setOtpAuthTotpURL(otpAuthTotpURL);
114 				response.setGoogleAuthKey(gKey.getKey());
115 				response.setGoogleAuthVerificationCode(gKey.getVerificationCode());
116 				response.setGoogleAuthScratchCodes(gKey.getScratchCodes());
117 			} else {
118 				response = new SetGoogleAuthenticatorCredentialResponse(ServiceResult.FAILURE);
119 			}
120 		}
121 
122 		eventRequest.setApplicationMessage(response.getResult().toString());
123 		createApplicationEventService.processService(eventRequest);
124 
125 		return response;
126 	}
127 
128 
129 	@Override
130 	protected CreateApplicationEventRequest createApplicationEventForService(
131 			final SetGoogleAuthenticatorCredentialRequest serviceRequest) {
132 		final CreateApplicationEventRequest eventRequest = createBaseApplicationEventRequest();
133 		eventRequest.setEventGroup(ApplicationEventGroup.USER);
134 		eventRequest.setApplicationOperation(ApplicationOperationType.CREATE);
135 		eventRequest.setActionName(SetGoogleAuthenticatorCredentialRequest.class.getSimpleName());
136 		eventRequest.setSessionId(serviceRequest.getSessionId());
137 		return eventRequest;
138 	}
139 
140 
141 	@Override
142 	protected SetGoogleAuthenticatorCredentialResponse createErrorResponse() {
143 		return new SetGoogleAuthenticatorCredentialResponse(ServiceResult.FAILURE);
144 	}
