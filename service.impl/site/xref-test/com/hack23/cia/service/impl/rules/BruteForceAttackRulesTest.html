44  /**
45   * Unit test for BruteForceAttack Drools rules.
46   *
47   * Validates brute force detection rules fire correctly at
48   * MINOR (5+), MAJOR (10+), and CRITICAL (20+) thresholds.
49   */
50  public class BruteForceAttackRulesTest {
51  
52  	/** The kie session. */
53  	private KieSession kieSession;
54  
55  	/** The kie container. */
56  	private KieContainer kieContainer;
57  
58  	/**
59  	 * Sets up.
60  	 */
61  	@Before
62  	public void setUp() {
63  		final KieServices kieServices = KieServices.Factory.get();
64  		final KieFileSystem kfs = kieServices.newKieFileSystem();
65  		kfs.write(ResourceFactory.newClassPathResource("META-INF/kmodule.xml"));
66  		kfs.write(ResourceFactory.newClassPathResource(
67  				"com/hack23/cia/service/impl/rules/application/BruteForceAttack.drl"));
68  		final KieBuilder kieBuilder = kieServices.newKieBuilder(kfs).buildAll();
69  		final Results results = kieBuilder.getResults();
70  		assertFalse("DRL compilation should have no errors: " + results.getMessages(),
71  				results.hasMessages(Message.Level.ERROR));
72  		kieContainer = kieServices.newKieContainer(kieServices.getRepository().getDefaultReleaseId());
73  		kieSession = kieContainer.newKieSession();
74  		assertNotNull("KieSession should be available", kieSession);
75  	}
76  
77  	/**
78  	 * Tears down.
79  	 */
80  	@After
81  	public void tearDown() {
82  		if (kieSession != null) {
83  			kieSession.dispose();
84  		}
85  	}

87  	/**
88  	 * Test no violation below threshold.
89  	 * Sessions with fewer than 5 failed attempts should not trigger any rule.
90  	 */
91  	@Test
92  	public void testNoViolationBelowThreshold() {
93  		final ApplicationComplianceCheckImpl check = new ApplicationComplianceCheckImpl(
94  				"session-1", "192.168.1.1", 4);
95  
96  		kieSession.insert(check);
97  		kieSession.fireAllRules();
98  
99  		final List<RuleViolation> violations = check.getRuleViolations();
100 		assertTrue("Should have no violations for 4 failed attempts", violations.isEmpty());
101 	}
102 
103 	/**
104 	 * Test minor threshold fires at 5 failed attempts.
105 	 */
106 	@Test
107 	public void testMinorThresholdFires() {
108 		final ApplicationComplianceCheckImpl check = new ApplicationComplianceCheckImpl(
109 				"session-2", "192.168.1.2", 5);
110 
111 		kieSession.insert(check);
112 		kieSession.fireAllRules();
113 
114 		final List<RuleViolation> violations = check.getRuleViolations();
115 		assertFalse("Should have violations for 5 failed attempts", violations.isEmpty());
116 
117 		final RuleViolation violation = violations.get(0);
118 		assertEquals("Status should be MINOR", Status.MINOR, violation.getStatus());
119 		assertEquals("Rule name should be BruteForceAttack", "BruteForceAttack", violation.getRuleName());
120 		assertEquals("Resource type should be APPLICATION", ResourceType.APPLICATION, violation.getResourceType());
121 	}

179 	/**
180 	 * Test only critical rule fires for 25 attempts.
181 	 * With non-overlapping conditions (5-9, 10-19, 20+), exactly one
182 	 * rule matches for 25 attempts, producing a single CRITICAL violation.
183 	 */
184 	@Test
185 	public void testOnlyCriticalRuleFires() {
186 		final ApplicationComplianceCheckImpl check = new ApplicationComplianceCheckImpl(
187 				"session-6", "192.168.0.100", 25);
188 
189 		kieSession.insert(check);
190 		final int firedRules = kieSession.fireAllRules();
191 
192 		assertEquals("Should have exactly 1 fired rule (only CRITICAL matches)", 1, firedRules);
193 
194 		final List<RuleViolation> violations = check.getRuleViolations();
195 		assertFalse("Should have violations", violations.isEmpty());
196 		assertEquals("Should have exactly 1 violation (only CRITICAL matches)", 1, violations.size());
197 
198 		final RuleViolation violation = violations.get(0);
199 		assertEquals("Should be CRITICAL for 25 attempts", Status.CRITICAL, violation.getStatus());
200 	}
