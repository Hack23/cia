59  	 * Test end-to-end workflow: Load data → Execute rules → Verify persistence.
60  	 * 
61  	 * @throws Exception the exception
62  	 */
63  	@Test
64  	public void testEndToEndRuleExecutionAndPersistence() throws Exception {
65  		setAuthenticatedAnonymousUser();
66  
67  		// Note: processService() clears all existing violations before persisting new ones.
68  		// Clear any existing violations
69  		ruleViolationDAO.getAll().forEach(ruleViolationDAO::delete);
70  
71  		// Execute the full compliance check workflow
72  		rulesManager.processService();
73  
74  		// Verify violations were persisted to database
75  		final List<RuleViolation> persistedViolations = ruleViolationDAO.getAll();
76  		assertNotNull("Persisted violations should not be null", persistedViolations);
77  		
78  		// Note: If database has no data, no violations will be created (expected behavior)
79  		// Only verify audit fields if violations exist
80  		if (!persistedViolations.isEmpty()) {
81  			// Verify audit trail fields are populated
82  			for (final RuleViolation violation : persistedViolations) {
83  				assertNotNull("Violation ID should be set", violation.getId());
84  				assertNotNull("Detected date should be set", violation.getDetectedDate());
85  				assertNotNull("Reference ID should be set", violation.getReferenceId());
86  				assertNotNull("Name should be set", violation.getName());
87  				assertNotNull("Resource type should be set", violation.getResourceType());
88  				assertNotNull("Rule name should be set", violation.getRuleName());
89  				assertNotNull("Status should be set", violation.getStatus());
90  			}
91  		}
92  	}

168 	public void testViolationSeverityLevels() throws Exception {
169 		setAuthenticatedAnonymousUser();
170 
171 		// Clear existing violations
172 		ruleViolationDAO.getAll().forEach(ruleViolationDAO::delete);
173 
174 		// Execute compliance checks
175 		rulesManager.processService();
176 
177 		// Collect violations by severity
178 		final List<RuleViolation> allViolations = ruleViolationDAO.getAll();
179 		
180 		// Only verify severity levels if violations exist
181 		if (!allViolations.isEmpty()) {
182 			boolean foundMinor = false;
183 			boolean foundMajor = false;
184 			boolean foundCritical = false;
185 
186 			for (final RuleViolation violation : allViolations) {
187 				if (Status.MINOR.equals(violation.getStatus())) {
188 					foundMinor = true;
189 				} else if (Status.MAJOR.equals(violation.getStatus())) {
190 					foundMajor = true;
191 				} else if (Status.CRITICAL.equals(violation.getStatus())) {
192 					foundCritical = true;
193 				}
194 			}
195 
196 			// Verify at least one severity level was found
197 			assertTrue("Should have at least one violation with a valid severity level", 
198 					foundMinor || foundMajor || foundCritical);
199 		}
200 	}

245 	/**
246 	 * Test that re-running the service clears old violations and creates new ones.
247 	 * Validates transaction management: old violations deleted, new ones persisted.
248 	 * 
249 	 * @throws Exception the exception
250 	 */
251 	@Test
252 	public void testViolationRefreshOnRerun() throws Exception {
253 		setAuthenticatedAnonymousUser();
254 
255 		// First run
256 		rulesManager.processService();
257 		final List<RuleViolation> firstRunViolations = ruleViolationDAO.getAll();
258 		final int firstRunCount = firstRunViolations.size();
259 		
260 		// Second run - should clear and recreate
261 		rulesManager.processService();
262 		final List<RuleViolation> secondRunViolations = ruleViolationDAO.getAll();
263 		final int secondRunCount = secondRunViolations.size();
264 
265 		// Verify that both runs produce the same number of violations
266 		assertEquals("Second run should produce same violation count as first run", firstRunCount, secondRunCount);
267 
268 		// Verify all violations have fresh timestamps if violations exist
269 		if (!secondRunViolations.isEmpty()) {
270 			for (final RuleViolation violation : secondRunViolations) {
271 				assertNotNull("Violation should have detected date", violation.getDetectedDate());
272 			}
273 		}
274 	}

277 	 * Test resource type coverage across all domains.
278 	 * 
279 	 * @throws Exception the exception
280 	 */
281 	@Test
282 	public void testResourceTypeCoverage() throws Exception {
283 		setAuthenticatedAnonymousUser();
284 
285 		// Clear existing violations
286 		ruleViolationDAO.getAll().forEach(ruleViolationDAO::delete);
287 
288 		// Execute compliance checks
289 		rulesManager.processService();
290 
291 		// Count violations by resource type
292 		final List<RuleViolation> allViolations = ruleViolationDAO.getAll();
293 		int politicianCount = 0;
294 		int partyCount = 0;
295 		int committeeCount = 0;
296 		int ministryCount = 0;
297 		int applicationCount = 0;
298 
299 		for (final RuleViolation violation : allViolations) {
300 			switch (violation.getResourceType()) {
301 				case POLITICIAN:
302 					politicianCount++;
303 					break;
304 				case PARTY:
305 					partyCount++;
306 					break;
307 				case COMMITTEE:
308 					committeeCount++;
309 					break;
310 				case MINISTRY:
311 					ministryCount++;
312 					break;
313 				case APPLICATION:
314 					applicationCount++;
315 					break;
316 				default:
317 					break;
318 			}
319 		}
320 
321 		// Note: All resource type counts are always non-negative since they are only incremented.
322 		// Empty results (zero counts) are acceptable depending on database content.
323 		
324 		// Note: Actual violation counts depend on database content
325 		// Politician violations are most common if data exists
326 	}
