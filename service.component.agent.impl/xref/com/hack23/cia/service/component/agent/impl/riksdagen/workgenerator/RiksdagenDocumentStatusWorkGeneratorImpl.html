19  package com.hack23.cia.service.component.agent.impl.riksdagen.workgenerator;
20  
21  import java.text.ParseException;
22  import java.text.SimpleDateFormat;
23  import java.util.ArrayList;
24  import java.util.List;
25  import java.util.Locale;
26  import java.util.Map;
27  import java.util.Set;
28  
29  import javax.jms.Destination;
30  
31  import org.slf4j.Logger;
32  import org.slf4j.LoggerFactory;
33  import org.springframework.beans.factory.annotation.Autowired;
34  import org.springframework.beans.factory.annotation.Qualifier;
35  import org.springframework.stereotype.Service;
36  
37  import com.hack23.cia.model.external.riksdagen.dokumentlista.impl.DocumentType;
38  import com.hack23.cia.model.internal.application.data.impl.RiksdagenDataSources;
39  
40  /**
41   * The Class RiksdagenDocumentStatusWorkGeneratorImpl.
42   */
43  @Service("RiksdagenDocumentStatusWorkGeneratorImpl")
44  final class RiksdagenDocumentStatusWorkGeneratorImpl extends AbstractRiksdagenDataSourcesWorkGenerator {
45  
46  	/** The Constant LOGGER. */
47  	private static final Logger LOGGER = LoggerFactory.getLogger(RiksdagenDocumentStatusWorkGeneratorImpl.class);
48  
49  	/** The Constant RIKSDAGEN_JAVA_SIMPLE_DATE_FORMAT. */
50  	private static final String RIKSDAGEN_JAVA_SIMPLE_DATE_FORMAT = "yyyy-MM-dd HH:mm:ss";
51  
52  	/** The document status container workdestination. */
53  	@Autowired
54  	@Qualifier("com.hack23.cia.model.external.riksdagen.dokumentstatus.impl.DocumentStatusContainer")
55  	private Destination documentStatusContainerWorkdestination;
56  
57  
58  	/**
59  	 * Instantiates a new riksdagen document status work generator impl.
60  	 */
61  	public RiksdagenDocumentStatusWorkGeneratorImpl() {
62  		super(RiksdagenDataSources.DOCUMENT_STATUS);
63  	}
64  
65  	@Override
66  	public void generateWorkOrders() {
67  		try {
68  			final Map<String, String> documentStatusContainerMap = getImportService()
69  					.getDocumentStatusContainerMap();
70  
71  
72  			final List<DocumentType> selectedDocumentTypes = new ArrayList<>();
73  			selectedDocumentTypes.add(DocumentType.BET);
74  			selectedDocumentTypes.add(DocumentType.PROP);
75  			selectedDocumentTypes.add(DocumentType.MOT);
76  
77  
78  
79  			final Map<String, String> documentElementMap = getImportService()
80  					.getDocumentElementMap(
81  							new SimpleDateFormat(
82  									RIKSDAGEN_JAVA_SIMPLE_DATE_FORMAT,Locale.ENGLISH).parse("1999-12-30 00:00:00"),
83  							selectedDocumentTypes, false);
84  
85  			final Set<String> avaibleDocumentStatus = documentElementMap
86  					.keySet();
87  
88  			for (final String id : avaibleDocumentStatus) {
89  				if (!documentStatusContainerMap.containsKey(id)) {
90  					getJmsSender().send(documentStatusContainerWorkdestination,
91  							id);
92  				}
93  			}
94  
95  		} catch (final ParseException e) {
96  			LOGGER.warn("Loading document status ", e);
97  		}
98  	}
99  
100 }
