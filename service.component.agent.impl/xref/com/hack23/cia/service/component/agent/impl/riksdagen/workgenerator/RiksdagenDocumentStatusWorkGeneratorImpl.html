22  import java.text.SimpleDateFormat;
23  import java.util.ArrayList;
24  import java.util.List;
25  import java.util.Locale;
26  import java.util.Map;
27  import java.util.Set;
28  
29  import javax.jms.Destination;
30  
31  import org.slf4j.Logger;
32  import org.slf4j.LoggerFactory;
33  import org.springframework.beans.factory.annotation.Autowired;
34  import org.springframework.beans.factory.annotation.Qualifier;
35  import org.springframework.stereotype.Service;
36  
37  import com.hack23.cia.model.external.riksdagen.dokumentlista.impl.DocumentType;
38  import com.hack23.cia.model.external.riksdagen.dokumentstatus.impl.DocumentStatusContainer;
39  import com.hack23.cia.model.internal.application.data.impl.RiksdagenDataSources;
40  
41  /**
42   * The Class RiksdagenDocumentStatusWorkGeneratorImpl.
43   */
44  @Service("RiksdagenDocumentStatusWorkGeneratorImpl")
45  final class RiksdagenDocumentStatusWorkGeneratorImpl extends AbstractRiksdagenDataSourcesWorkGenerator {
46  
47  	/** The Constant LOGGER. */
48  	private static final Logger LOGGER = LoggerFactory.getLogger(RiksdagenDocumentStatusWorkGeneratorImpl.class);
49  
50  	/** The Constant RIKSDAGEN_JAVA_SIMPLE_DATE_FORMAT. */
51  	private static final String RIKSDAGEN_JAVA_SIMPLE_DATE_FORMAT = "yyyy-MM-dd HH:mm:ss";
52  
53  	/** The document status container workdestination. */
54  	@Autowired
55  	@Qualifier("com.hack23.cia.model.external.riksdagen.dokumentstatus.impl.DocumentStatusContainer")
56  	private Destination documentStatusContainerWorkdestination;
57  
58  
59  	/**
60  	 * Instantiates a new riksdagen document status work generator impl.
61  	 */
62  	public RiksdagenDocumentStatusWorkGeneratorImpl() {
63  		super(RiksdagenDataSources.DOCUMENT_STATUS);
64  	}
65  
66  	@Override
67  	public void generateWorkOrders() {
68  		try {
69  			final Map<String, String> documentStatusContainerMap = getImportService()
70  					.getDocumentStatusContainerMap();
71  
72  
73  			final List<DocumentType> selectedDocumentTypes = new ArrayList<>();
74  			selectedDocumentTypes.add(DocumentType.BET);
75  			selectedDocumentTypes.add(DocumentType.PROP);
76  			selectedDocumentTypes.add(DocumentType.MOT);
77  
78  
79  
80  			final Map<String, String> documentElementMap = getImportService()
81  					.getDocumentElementMap(
82  							new SimpleDateFormat(
83  									RIKSDAGEN_JAVA_SIMPLE_DATE_FORMAT,Locale.ENGLISH).parse("1999-12-30 00:00:00"),
84  							selectedDocumentTypes, false);
85  
86  			final Set<String> avaibleDocumentStatus = documentElementMap
87  					.keySet();
88  
89  			for (final String id : avaibleDocumentStatus) {
90  				if (!documentStatusContainerMap.containsKey(id)) {
91  					getJmsSender().send(documentStatusContainerWorkdestination,
92  							id);
93  				}
94  			}
95  			
96  			for (final DocumentStatusContainer container : getImportService().getNoneCompletedDocumentStatusCommitteeReports()) {
97  				 if ("planerat".equals(container.getDocument().getStatus())) {
98  					 getJmsSender().send(documentStatusContainerWorkdestination,
99  							 container.getDocument().getId());
100 				 }
101 			}		
102 
103 		} catch (final ParseException e) {
104 			LOGGER.warn("Loading document status ", e);
105 		}
106 	}
107 
108 }
