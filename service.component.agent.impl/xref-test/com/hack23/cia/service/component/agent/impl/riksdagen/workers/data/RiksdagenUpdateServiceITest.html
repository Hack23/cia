73  	 * Update person data already exist success test.
74  	 *
75  	 * @throws Exception the exception
76  	 */
77  	@Test
78  	public void updatePersonDataAlreadyExistSuccessTest() throws Exception {
79  		final PersonDataDAO personDataDAOMock = mock(PersonDataDAO.class);
80  		ReflectionTestUtils.setField(riksdagenUpdateService, "personDataDAO", personDataDAOMock);
81  
82  		String personId = "0542160909628";
83  		
84  		PersonData existPersonData = new PersonData();
85  		List<AssignmentData> assignmentList = new ArrayList<>();
86  		assignmentList.add(new AssignmentData().withOrgCode("OrgCode").withRoleCode("RoleCode").withFromDate(new Date())
87  				.withToDate(new Date()).withIntressentId(personId));
88  		existPersonData.withPersonAssignmentData(new PersonAssignmentData().withAssignmentList(assignmentList));
89  		existPersonData.setId(personId);
90  		
91  
92  		PersonData personData = new PersonData();
93  		List<AssignmentData> assignmentListNew = new ArrayList<>();
94  		assignmentListNew.add(new AssignmentData().withOrgCode("OrgCode").withRoleCode("RoleCode").withFromDate(new Date())
95  				.withToDate(new Date()).withIntressentId(personId));
96  		assignmentListNew.add(new AssignmentData().withOrgCode("OrgCodeNew").withRoleCode("RoleCodeNew").withFromDate(new Date())
97  				.withToDate(new Date()).withIntressentId(personId));
98  		
99  		personData.withPersonAssignmentData(new PersonAssignmentData().withAssignmentList(assignmentListNew));
100 		personData.setId(personId);
101 
102 		when(personDataDAOMock.load(personId)).thenReturn(existPersonData);
103 
104 		riksdagenUpdateService.update(personData);
105 		verify(personDataDAOMock, times(1)).persist(existPersonData);
106 	}
