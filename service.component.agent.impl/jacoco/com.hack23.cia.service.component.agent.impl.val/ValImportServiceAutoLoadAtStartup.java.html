package com.hack23.cia.service.component.agent.impl.val;

import javax.annotation.PostConstruct;

import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Component;

/**
 * The Class ValImportServiceAutoLoadAtStartup.
 */
@Component
public final class ValImportServiceAutoLoadAtStartup {

	/** The val import service. */
	@Autowired
	private ValImportService valImportService;


	/**
	 * Instantiates a new val import service auto load at startup.
	 */
	public ValImportServiceAutoLoadAtStartup() {
		super();
	}


	/**
	 * Auto load.
	 */
	@PostConstruct
	public void autoLoad() {
		valImportService.loadPoliticalParties();
	}

}
