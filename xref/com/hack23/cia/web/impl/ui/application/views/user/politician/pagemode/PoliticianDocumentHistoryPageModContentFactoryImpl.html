39  /**
40   * The Class OverviewPageModContentFactoryImpl.
41   */
42  @Component
43  public final class PoliticianDocumentHistoryPageModContentFactoryImpl
44  		extends AbstractPoliticianPageModContentFactoryImpl {
45  
46  	/** The Constant COLUMN_ORDER. */
47  	private static final String[] COLUMN_ORDER = { "rm", "madePublicDate", "documentType", "subType", "title",
48  			"subTitle", "referenceName", "partyShortCode", "personReferenceId", "roleDescription", "org", "id", "docId",
49  			"tempLabel", "label", "numberValue", "orderNumber", "status" };
50  
51  	/** The Constant DOCUMENTS. */
52  	private static final String DOCUMENTS = "Documents";
53  
54  	/** The Constant HIDE_COLUMNS. */
55  	private static final String[] HIDE_COLUMNS = { "id", "partyShortCode", "personReferenceId", "numberValue",
56  			"orderNumber", "tempLabel", "referenceName", "docId", "label", "roleDescription" };
57  
58  	/** The Constant LISTENER. */
59  	private static final PageItemPropertyClickListener LISTENER = new PageItemPropertyClickListener(
60  			UserViews.DOCUMENT_VIEW_NAME, "docId", true);
61  
62  	/**
63  	 * Instantiates a new politician document history page mod content factory impl.
64  	 */
65  	public PoliticianDocumentHistoryPageModContentFactoryImpl() {
66  		super();
67  	}
68  
69  	@Secured({ "ROLE_ANONYMOUS", "ROLE_USER", "ROLE_ADMIN" })
70  	@Override
71  	public Layout createContent(final String parameters, final MenuBar menuBar, final Panel panel) {
72  		final VerticalLayout panelContent = createPanelContent();
73  
74  		final String pageId = getPageId(parameters);
75  
76  		final ViewRiksdagenPolitician viewRiksdagenPolitician = getItem(parameters);
77  
78  		getPoliticianMenuItemFactory().createPoliticianMenuBar(menuBar, pageId);
79  
80  		CardInfoRowUtil.createPageHeader(panel, panelContent,
81  				PoliticianPageTitleConstants.DOCUMENT_HISTORY_TITLE + viewRiksdagenPolitician.getFirstName() + ' '
82  						+ viewRiksdagenPolitician.getLastName() + '(' + viewRiksdagenPolitician.getParty() + ')',
83  				PoliticianPageTitleConstants.HISTORY_SUBTITLE, PoliticianPageTitleConstants.DOC_HISTORY_DESC);
84  
85  		final DataContainer<ViewRiksdagenPoliticianDocument, String> politicianDocumentDataContainer = getApplicationManager()
86  				.getDataContainer(ViewRiksdagenPoliticianDocument.class);
87  
88  		getGridFactory().createBasicBeanItemGrid(panelContent, ViewRiksdagenPoliticianDocument.class,
89  				politicianDocumentDataContainer.findOrderedListByProperty(
90  						ViewRiksdagenPoliticianDocument_.personReferenceId, viewRiksdagenPolitician.getPersonId(),
91  						ViewRiksdagenPoliticianDocument_.madePublicDate),
92  				DOCUMENTS, COLUMN_ORDER, HIDE_COLUMNS, LISTENER, null, null);
93  
94  		getPageActionEventHelper().createPageEvent(ViewAction.VISIT_POLITICIAN_VIEW, ApplicationEventGroup.USER,
95  				UserViews.POLITICIAN_VIEW_NAME, parameters, pageId);
96  		return panelContent;
97  
98  	}
99  
100 	@Override
101 	public boolean matches(final String page, final String parameters) {
102 		return PageCommandPoliticianConstants.COMMAND_POLITICIAN_DOCUMENT_HISTORY.matches(page, parameters);
103 	}
104 }
