52  /**
53   * The Class AdminUserAccountPageModContentFactoryImpl.
54   */
55  @Component
56  public final class AdminUserAccountPageModContentFactoryImpl extends AbstractAdminSystemPageModContentFactoryImpl {
57  
58  
59      /** The Constant BUTTON_ID_PATTERN. */
60      private static final String BUTTON_ID_PATTERN = "{0}.{1}";
61  
62      /** The Constant BUTTON_PATTERN. */
63      private static final String BUTTON_PATTERN = "Perform {0}";
64  
65      /** The Constant COLUMN_ORDER. */
66      private static final String[] COLUMN_ORDER = { "hjid", "modelObjectId", "modelObjectVersion", "createdDate",
67              "userId", "username", "userType", "userRole", "userpassword", "email", "country", "numberOfVisits" };
68  
69      /** The Constant HIDE_COLUMNS. */
70      private static final String[] HIDE_COLUMNS = { "hjid", "modelObjectId", "modelObjectVersion", "userId",
71              "userpassword", "address" };
72  
73      /** The Constant LISTENER. */
74      private static final PageItemPropertyClickListener LISTENER = new PageItemPropertyClickListener(
75              AdminViews.ADMIN_USERACCOUNT_VIEW_NAME, "hjid");
76  
77      /** The Constant NAME. */
78      public static final String NAME = AdminViews.ADMIN_USERACCOUNT_VIEW_NAME;
79  
80      /** The Constant USER_ACCOUNT. */
81      private static final String USER_ACCOUNT = "UserAccount";
82  
83      /**
84       * Instantiates a new admin user account page mod content factory impl.
85       */
86      public AdminUserAccountPageModContentFactoryImpl() {
87          super(NAME);
88      }
89  
90      @Secured({ "ROLE_ADMIN" })
91      @Override
92      public Layout createContent(final String parameters, final MenuBar menuBar, final Panel panel) {
93          final VerticalLayout content = createPanelContent();
94  
95          final String pageId = getPageId(parameters);
96          final int pageNr = getPageNr(parameters);
97  
98          getMenuItemFactory().createMainPageMenuBar(menuBar);
99  
100         CardInfoRowUtil.createPageHeader(panel, content, AdminViewConstants.ADMIN_USER_ACCOUNT_MANAGEMENT, AdminViewConstants.USER_ACCOUNT_OVERVIEW,
101                 AdminViewConstants.USER_ACCOUNT_MANAGEMENT_DESCRIPTION);
102 
103         final DataContainer<UserAccount, Long> dataContainer = getApplicationManager()
104                 .getDataContainer(UserAccount.class);
105 
106         final List<UserAccount> pageOrderBy = dataContainer.getPageOrderBy(pageNr, DEFAULT_RESULTS_PER_PAGE,
107                 UserAccount_.createdDate);
108 
109         getPagingUtil().createPagingControls(content, NAME, pageId, dataContainer.getSize(), pageNr,
110                 DEFAULT_RESULTS_PER_PAGE);
111 
112         getGridFactory().createBasicBeanItemGrid(content, UserAccount.class, pageOrderBy, USER_ACCOUNT,
113                 COLUMN_ORDER, HIDE_COLUMNS, LISTENER, null, null);
114 
115         if (pageId != null && !pageId.isEmpty()) {
116             final UserAccount userAccount = dataContainer.load(Long.valueOf(pageId));
117             if (userAccount != null) {
118                 // Instead of form fields, use a card layout similar to other views.
119                 final Panel cardPanel = new Panel();
120                 cardPanel.addStyleName("politician-overview-card"); // Reuse existing card style
121                 cardPanel.setWidth("100%");
122                 cardPanel.setHeightUndefined();
123                 Responsive.makeResponsive(cardPanel);
124 
125                 final VerticalLayout cardContent = new VerticalLayout();
126                 cardContent.setMargin(true);
127                 cardContent.setSpacing(true);
128                 cardContent.setWidth("100%");
129                 cardPanel.setContent(cardContent);
130 
131                 content.addComponent(cardPanel);
132 
133                 CardInfoRowUtil.createCardHeader(cardContent,"User Account Details");
134 
135                 // Attributes layout
136                 final VerticalLayout attributesLayout = new VerticalLayout();
137                 attributesLayout.setSpacing(true);
138                 attributesLayout.setWidth("100%");
139                 cardContent.addComponent(attributesLayout);
140 
141                 // Display fields using a card layout approach with null-check
142                 CardInfoRowUtil.addInfoRowIfNotNull(attributesLayout, "Username:", userAccount.getUsername(), VaadinIcons.USER);
143                 CardInfoRowUtil.addInfoRowIfNotNull(attributesLayout, "Created Date:", String.valueOf(userAccount.getCreatedDate()),
144                         VaadinIcons.CALENDAR);
145                 CardInfoRowUtil.addInfoRowIfNotNull(attributesLayout, "Email:", userAccount.getEmail(), VaadinIcons.ENVELOPE);
146                 CardInfoRowUtil.addInfoRowIfNotNull(attributesLayout, "Country:", userAccount.getCountry(), VaadinIcons.GLOBE);
147                 CardInfoRowUtil.addInfoRowIfNotNull(attributesLayout, "Number Of Visits:", String.valueOf(userAccount.getNumberOfVisits()),
148                         VaadinIcons.EYE);
149 
150                 final VerticalLayout overviewLayout = new VerticalLayout();
151                 overviewLayout.setSizeFull();
152                 overviewLayout.addStyleName("v-layout-content-overview-panel-level2");
153                 content.addComponent(overviewLayout);
154                 content.setExpandRatio(overviewLayout, ContentRatio.LARGE);
155 
156                 final ResponsiveRow grid = RowUtil.createGridLayout(overviewLayout);
157 
158                 for (final AccountOperation accountOperation : ManageUserAccountRequest.AccountOperation.values()) {
159                     final ManageUserAccountRequest request = new ManageUserAccountRequest();
160                     request.setSessionId(RequestContextHolder.currentRequestAttributes().getSessionId());
161                     request.setAccountOperation(accountOperation);
162                     request.setUserAcountId(userAccount.getUserId());
163                     final Button accountOperationButton = new Button(
164                             MessageFormat.format(BUTTON_PATTERN, accountOperation), VaadinIcons.BULLSEYE);
165                     accountOperationButton.addClickListener(new ManageUserAccountClickListener(request));
166                     accountOperationButton.setId(
167                             MessageFormat.format(BUTTON_ID_PATTERN, ViewAction.START_AGENT_BUTTON, accountOperation));
168                     RowUtil.createRowItem(grid, accountOperationButton, "Will perform useraccount action");
169                 }
170             }
171         }
172 
173         getPageActionEventHelper().createPageEvent(ViewAction.VISIT_ADMIN_USERACCOUNT_VIEW, ApplicationEventGroup.ADMIN,
174                 NAME, null, pageId);
175 
176         return content;
177     }
178 
179     @Override
180     public boolean matches(final String page, final String parameters) {
181     	return PageCommandAdminConstants.COMMAND_USERACCOUNT.matches(page, parameters);
182     }
