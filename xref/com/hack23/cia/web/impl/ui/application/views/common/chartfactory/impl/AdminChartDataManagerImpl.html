75  	/**
76  	 * Adds the view application action event page element daily summary values.
77  	 *
78  	 * @param label the label
79  	 * @param series the series
80  	 * @param list the list
81  	 * @param dataSeries the data series
82  	 * @param simpleDateFormat the simple date format
83  	 * @param t the t
84  	 */
85  	private static void addViewApplicationActionEventPageElementDailySummaryValues(final String label,
86  			final Series series, final List<ViewApplicationActionEventPageElementDailySummary> list,
87  			final DataSeries dataSeries, final SimpleDateFormat simpleDateFormat,
88  			final ToLongFunction<ViewApplicationActionEventPageElementDailySummary> t) {
89  		series.addSeries(new XYseries().setLabel(label));
90  		dataSeries.newSeries();
91  
92  		for (final ViewApplicationActionEventPageElementDailySummary item : list) {
93  			dataSeries.add(simpleDateFormat.format(item.getEmbeddedId().getCreatedDate()), t.applyAsLong(item));
94  		}
95  	}

97  	@Override
98  	public void createApplicationActionEventPageDailySummaryChart(final AbstractOrderedLayout content) {
99  
100 		final Map<String, List<ViewApplicationActionEventPageDailySummary>> map = getApplicationActionEventPageDailySummaryMap();
101 
102 		final DataSeries dataSeries = new DataSeries();
103 
104 		final SimpleDateFormat simpleDateFormat = new SimpleDateFormat(DD_MMM_YYYY, Locale.ENGLISH);
105 
106 		final Series series = new Series();
107 
108 		for (final Entry<String, List<ViewApplicationActionEventPageDailySummary>> entry : map.entrySet()) {
109 
110 			if (entry.getKey() != null) {
111 				series.addSeries(new XYseries().setLabel(entry.getKey()));
112 
113 				dataSeries.newSeries();
114 				final List<ViewApplicationActionEventPageDailySummary> list = entry.getValue();
115 				for (final ViewApplicationActionEventPageDailySummary dataSummaryDaily : list) {
116 					dataSeries.add(
117 							simpleDateFormat.format(
118 									dataSummaryDaily.getEmbeddedId().getCreatedDate()),
119 							dataSummaryDaily.getHits());
120 				}
121 			}
122 
123 		}
124 
125 		addChart(content, "Application Action Events daily Summary",
126 				new DCharts().setDataSeries(dataSeries)
127 						.setOptions(getChartOptions().createOptionsXYDateFloatLogYAxisLegendOutside(series)).show(),
128 				true);
129 	}

185 	/**
186 	 * Gets the application action event page daily summary map.
187 	 *
188 	 * @return the application action event page daily summary map
189 	 */
190 	private Map<String, List<ViewApplicationActionEventPageDailySummary>> getApplicationActionEventPageDailySummaryMap() {
191 		final DataContainer<ViewApplicationActionEventPageDailySummary, ApplicationActionEventPagePeriodSummaryEmbeddedId> documentTypeSummaryDailyDataContainer = getApplicationManager()
192 				.getDataContainer(ViewApplicationActionEventPageDailySummary.class);
193 
194 		return documentTypeSummaryDailyDataContainer.getAll().parallelStream().filter(Objects::nonNull)
195 				.collect(Collectors.groupingBy(t -> t.getEmbeddedId().getPage()));
196 	}
197 
198 	/**
199 	 * Gets the application action event page element daily summary list.
200 	 *
201 	 * @param page the page
202 	 * @param elementId the element id
203 	 * @return the application action event page element daily summary list
204 	 */
205 	private List<ViewApplicationActionEventPageElementDailySummary> getApplicationActionEventPageElementDailySummaryList(
206 			final String page, final String elementId) {
207 		final DataContainer<ViewApplicationActionEventPageElementDailySummary, ApplicationActionEventPageElementPeriodSummaryEmbeddedId> documentTypeSummaryDailyDataContainer = getApplicationManager()
208 				.getDataContainer(ViewApplicationActionEventPageElementDailySummary.class);
209 
210 		final List<ViewApplicationActionEventPageElementDailySummary> findOrderedListByEmbeddedProperty = documentTypeSummaryDailyDataContainer
211 				.findOrderedListByEmbeddedProperty(ViewApplicationActionEventPageElementDailySummary.class,
212 						ViewApplicationActionEventPageElementDailySummary_.embeddedId,
213 						ApplicationActionEventPageElementPeriodSummaryEmbeddedId.class,
214 						ApplicationActionEventPageElementPeriodSummaryEmbeddedId_.elementId, elementId,
215 						ApplicationActionEventPageElementPeriodSummaryEmbeddedId_.createdDate);
216 
217 		return findOrderedListByEmbeddedProperty.parallelStream()
218 				.filter(t -> t != null && t.getEmbeddedId().getPage().equals(page)).toList();
219 	}

243 	@Override
244 	public void createApplicationSessionPageDailySummaryChart(final VerticalLayout content) {
245 		final SimpleDateFormat simpleDateFormat = new SimpleDateFormat(DD_MMM_YYYY, Locale.ENGLISH);
246 
247 		final DataContainer<ApplicationSession, Serializable> dataContainer = getApplicationManager().getDataContainer(ApplicationSession.class);
248 		final Map <String,Long> sessionByDayMap = dataContainer.getAll().stream()
249 	                .collect(Collectors.groupingBy(p -> simpleDateFormat.format(p.getCreatedDate()), Collectors.counting()));
250 
251 		final DataSeries dataSeries = new DataSeries();
252 		final Series series = new Series();
253 		series.addSeries(new XYseries().setLabel("Daily Active Users"));
254 		dataSeries.newSeries();
255 		for (final Entry<String, Long> entry : sessionByDayMap.entrySet()) {
256 			dataSeries.add(
257 					entry.getKey(),
258 						entry.getValue());
259 		}
260 
261 		addChart(content, "Application Active Daily Users",
262 				new DCharts().setDataSeries(dataSeries)
263 						.setOptions(getChartOptions().createOptionsXYDateFloatLogYAxisLegendOutside(series)).show(),
264 				true);
265 	}
