49  @Component
50  public final class DocumentAttachementsPageModContentFactoryImpl extends AbstractDocumentPageModContentFactoryImpl {
51  
52  	/** The Constant COLUMN_ORDER. */
53  	private static final String[] COLUMN_ORDER = { "fileName", "fileSize", "fileType", "fileUrl" };
54  
55  	/** The Constant DOCUMENT_ATTACHMENTS. */
56  	private static final String DOCUMENT_ATTACHMENTS = "Document Attachments";
57  
58  	/** The Constant HIDE_COLUMNS. */
59  	private static final String[] HIDE_COLUMNS = { "hjid" };
60  
61  	/** The Constant PDF. */
62  	private static final String PDF = "pdf";
63  
64  	/**
65  	 * Instantiates a new document attachements page mod content factory impl.
66  	 */
67  	public DocumentAttachementsPageModContentFactoryImpl() {
68  		super();
69  	}
70  
71  	/**
72  	 * Display document attachements.
73  	 *
74  	 * @param panelContent the panel content
75  	 * @param documentAttachmentList the document attachment list
76  	 */
77  	private static void displayDocumentAttachements(final VerticalLayout panelContent,
78  			final List<DocumentAttachment> documentAttachmentList) {
79  		for (final DocumentAttachment documentAttachment : documentAttachmentList) {
80  
81  			if (PDF.equalsIgnoreCase(documentAttachment.getFileType())) {
82  				final WTPdfViewer wtPdfViewer = new WTPdfViewer();
83  				wtPdfViewer.setSizeFull();
84  
85  				wtPdfViewer.setResource(new StreamResource(new StreamSourceImplementation(documentAttachment.getFileUrl()), documentAttachment.getFileName()));
86  
87  				panelContent.addComponent(wtPdfViewer);
88  				panelContent.setExpandRatio(wtPdfViewer, ContentRatio.LARGE);
89  			} else {
90  				final VerticalLayout verticalLayout = new VerticalLayout();
91  				panelContent.addComponent(verticalLayout);
92  				panelContent.setExpandRatio(verticalLayout, ContentRatio.SMALL);
93  				final ExternalAttachmentDownloadLink link = new ExternalAttachmentDownloadLink(
94  						documentAttachment.getFileName(), documentAttachment.getFileType(),
95  						documentAttachment.getFileUrl());
96  				verticalLayout.addComponent(link);
97  			}
98  		}
99  	}
100 
101 	@Secured({ "ROLE_ANONYMOUS", "ROLE_USER", "ROLE_ADMIN" })
102 	@Override
103 	public Layout createContent(final String parameters, final MenuBar menuBar, final Panel panel) {
104 		final VerticalLayout panelContent = createPanelContent();
105 
106 		final String pageId = getPageId(parameters);
107 
108 		getDocumentMenuItemFactory().createDocumentMenuBar(menuBar, pageId);
109 
110 		final DataContainer<DocumentStatusContainer, String> documentStatusContainerDataContainer = getApplicationManager()
111 				.getDataContainer(DocumentStatusContainer.class);
112 
113 		final DocumentStatusContainer documentStatusContainer = documentStatusContainerDataContainer
114 				.findByQueryProperty(DocumentStatusContainer.class, DocumentStatusContainer_.document,
115 						DocumentData.class, DocumentData_.id, pageId);
116 
117 		createPageHeader(panel, panelContent, "Document Attachments", "Attachments Overview", "Explore and manage attachments associated with the document.");
118 
119 		if (documentStatusContainer != null && documentStatusContainer.getDocumentAttachmentContainer() != null
120 				&& documentStatusContainer.getDocumentAttachmentContainer().getDocumentAttachmentList() != null) {
121 
122 			getGridFactory().createBasicBeanItemGrid(panelContent, DocumentAttachment.class,
123 					documentStatusContainer.getDocumentAttachmentContainer().getDocumentAttachmentList(),
124 					DOCUMENT_ATTACHMENTS, COLUMN_ORDER, HIDE_COLUMNS, null, null, null);
125 
126 			displayDocumentAttachements(panelContent,
127 					documentStatusContainer.getDocumentAttachmentContainer().getDocumentAttachmentList());
128 		}
129 
130 		panel.setContent(panelContent);
131 		getPageActionEventHelper().createPageEvent(ViewAction.VISIT_DOCUMENT_VIEW, ApplicationEventGroup.USER, NAME,
132 				parameters, pageId);
133 		return panelContent;
134 
135 	}
136 
137 	@Override
138 	public boolean matches(final String page, final String parameters) {
139 		return NAME.equals(page) && StringUtils.contains(parameters, DocumentPageMode.DOCUMENTATTACHMENTS.toString());
140 	}
