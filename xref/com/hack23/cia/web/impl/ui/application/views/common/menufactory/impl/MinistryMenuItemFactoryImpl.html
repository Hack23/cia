57  	@Override
58  	public void createMinistryMenuBar(final MenuBar menuBar, final String pageId) {
59  		initApplicationMenuBar(menuBar);
60  
61  		applicationMenuItemFactory.addRankingMenu(menuBar);
62  
63  		ministryRankingMenuItemFactory
64  				.createMinistryRankingTopics(menuBar.addItem(MINISTRY_RANKING, VaadinIcons.GROUP, null));
65  
66  		final MenuItem ministryItem = menuBar.addItem("Ministry " + pageId, VaadinIcons.GROUP, null);
67  
68  		ministryItem.addItem(OVERVIEW_TEXT, VaadinIcons.GROUP,
69  				PageCommandMinistryConstants.COMMAND_MINISTRY_OVERVIEW.createItemPageCommand(pageId));
70  		final MenuItem rolesItem = ministryItem.addItem(ROLES_TEXT, VaadinIcons.GROUP, null);
71  
72  		rolesItem.addItem(CURRENT_MEMBERS_TEXT, VaadinIcons.GROUP,
73  				PageCommandMinistryConstants.COMMAND_MINISTRY_CURRENT_MEMBERS.createItemPageCommand(pageId));
74  
75  		rolesItem.addItem(MEMBER_HISTORY_TEXT, VaadinIcons.GROUP,
76  				PageCommandMinistryConstants.COMMAND_MINISTRY_MEMBER_HISTORY.createItemPageCommand(pageId));
77  
78  		rolesItem.addItem(ROLE_GHANT_TEXT, VaadinIcons.GROUP,
79  				PageCommandMinistryConstants.COMMAND_MINISTRY_ROLE_GHANT.createItemPageCommand(pageId));
80  
81  		rolesItem.addItem(GOVERNMENT_BODIES_HEADCOUNT, VaadinIcons.GROUP,
82  				PageCommandMinistryConstants.COMMAND_MINISTRY_GOVERNMENT_BODIES_HEADCOUNT.createItemPageCommand(pageId));
83  
84  		rolesItem.addItem(GOVERNMENT_BODIES_INCOME, VaadinIcons.GROUP,
85  				PageCommandMinistryConstants.COMMAND_MINISTRY_GOVERNMENT_BODIES_INCOME.createItemPageCommand(pageId));
86  
87  		rolesItem.addItem(GOVERNMENT_BODIES_EXPENDITURE, VaadinIcons.GROUP,
88  				PageCommandMinistryConstants.COMMAND_MINISTRY_GOVERNMENT_BODIES_EXPENDITURE.createItemPageCommand(pageId));
89  
90  		final MenuItem documentItem = ministryItem.addItem(DOCUMENTS_TEXT, VaadinIcons.GROUP, null);
91  
92  		documentItem.addItem(DOCUMENT_ACTIVITY_TEXT, VaadinIcons.GROUP,
93  				PageCommandMinistryConstants.COMMAND_MINISTRY_DOCUMENT_ACTIVITY.createItemPageCommand(pageId));
94  
95  		documentItem.addItem(DOCUMENT_HISTORY_TEXT, VaadinIcons.GROUP,
96  				PageCommandMinistryConstants.COMMAND_MINISTRY_DOCUMENT_HISTORY.createItemPageCommand(pageId));
97  
98  		ministryItem.addItem(RANKING_PAGE_VISIT_TEXT, VaadinIcons.GROUP,
99  				MINISTRY_RANKING_COMMAND_PAGEVISIT_HISTORY);
100 
101 	}
102
