50  /**
51   * The Class ParliamentDecisionFlowPageModContentFactoryImpl.
52   */
53  @Component
54  public final class ParliamentDecisionFlowPageModContentFactoryImpl extends AbstractParliamentPageModContentFactoryImpl {
55  
56  
57  	/** The Constant PARLIAMENT_DECISION_FLOW. */
58  	private static final String PARLIAMENT_DECISION_FLOW = "Parliament decision flow";
59  	
60  	@Autowired
61  	private DecisionFlowChartManager decisionFlowChartManager;
62  	
63  	/**
64  	 * Instantiates a new parliament decision flow page mod content factory impl.
65  	 */
66  	public ParliamentDecisionFlowPageModContentFactoryImpl() {
67  		super();
68  	}
69  
70  	@Override
71  	public boolean matches(final String page, final String parameters) {
72  		return NAME.equals(page) && StringUtils.contains(parameters, PageMode.CHARTS.toString())
73  				&& parameters.contains(ChartIndicators.DECISION_FLOW_CHART.toString());
74  	}
75  
76  	@Secured({ "ROLE_ANONYMOUS", "ROLE_USER", "ROLE_ADMIN" })
77  	@Override
78  	public Layout createContent(final String parameters, final MenuBar menuBar, final Panel panel) {
79  		final VerticalLayout panelContent = createPanelContent();
80  		getParliamentMenuItemFactory().createParliamentTopicMenu(menuBar);
81  
82  		String selectedYear = "2018/19";
83  		if (parameters != null && parameters.contains("[") && parameters.contains("]")) {
84  			selectedYear = parameters.substring(parameters.indexOf('[') + 1, parameters.lastIndexOf(']'));
85  		} 
86  		
87  		final DataContainer<ViewRiksdagenCommittee, String> dataContainer = getApplicationManager()
88  				.getDataContainer(ViewRiksdagenCommittee.class);
89  		final List<ViewRiksdagenCommittee> allCommittess = dataContainer.getAll();
90  
91  		final Map<String, List<ViewRiksdagenCommittee>> committeeMap = allCommittess.stream().collect(Collectors.groupingBy(c -> c.getEmbeddedId().getOrgCode().toUpperCase(Locale.ENGLISH)));
92  		
93  		final ComboBox<String> comboBox = new ComboBox<>("Select year", Collections.unmodifiableList(Arrays.asList("2018/19","2017/18","2016/17","2015/16","2014/15","2013/14","2012/13","2011/12","2010/11")));
94  		panelContent.addComponent(comboBox);
95  		panelContent.setExpandRatio(comboBox, ContentRatio.SMALL);
96  		comboBox.setSelectedItem(selectedYear);
97  		comboBox.addValueChangeListener(new DecisionFlowValueChangeListener(NAME,""));
98  		
99  		final SankeyChart chart = decisionFlowChartManager.createAllDecisionFlow(committeeMap,comboBox.getSelectedItem().orElse(selectedYear));
100 		panelContent.addComponent(chart);
101 		panelContent.setExpandRatio(chart, ContentRatio.LARGE);
102 
103 		final TextArea textarea = decisionFlowChartManager.createCommitteeeDecisionSummary(committeeMap,comboBox.getSelectedItem().orElse(selectedYear));
104 		textarea.setSizeFull();
105 		panelContent.addComponent(textarea);
106 		panelContent.setExpandRatio(textarea, ContentRatio.SMALL_GRID);
107 
108 
109 		getPageActionEventHelper().createPageEvent(ViewAction.VISIT_PARLIAMENT_RANKING_VIEW, ApplicationEventGroup.USER, NAME,
110 				parameters, selectedYear);
111 		panel.setCaption(new StringBuilder().append(NAME).append("::").append(PARLIAMENT_DECISION_FLOW).toString());
112 
113 		return panelContent;
114 
115 	}
