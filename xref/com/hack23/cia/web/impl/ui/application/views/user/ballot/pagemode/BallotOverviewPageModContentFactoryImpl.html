96  	/**
97  	 * Creates the content.
98  	 *
99  	 * @param parameters the parameters
100 	 * @param menuBar the menu bar
101 	 * @param panel the panel
102 	 * @return the layout
103 	 */
104 	@Secured({ "ROLE_ANONYMOUS", "ROLE_USER", "ROLE_ADMIN" })
105 	@Override
106 	public Layout createContent(final String parameters, final MenuBar menuBar, final Panel panel) {
107 		final VerticalLayout panelContent = createPanelContent();
108 		panel.setContent(panelContent);
109 
110 		final String pageId = getPageId(parameters);
111 		final List<ViewRiksdagenVoteDataBallotSummary> ballots = getItem(parameters);
112 
113 		if (!ballots.isEmpty()) {
114 			getBallotMenuItemFactory().createBallotMenuBar(menuBar, pageId);
115 
116 			final DataContainer<ViewRiksdagenVoteDataBallotPartySummary, RiksdagenVoteDataBallotPartyEmbeddedId> dataPartyContainer = getApplicationManager()
117 					.getDataContainer(ViewRiksdagenVoteDataBallotPartySummary.class);
118 
119 			final DataContainer<ViewRiksdagenCommitteeBallotDecisionSummary, ViewRiksdagenCommitteeBallotDecisionEmbeddedId> dataDecisionContainer = getApplicationManager()
120 					.getDataContainer(ViewRiksdagenCommitteeBallotDecisionSummary.class);
121 
122 			final List<ViewRiksdagenVoteDataBallotPartySummary> partyBallotList = dataPartyContainer
123 					.findListByEmbeddedProperty(ViewRiksdagenVoteDataBallotPartySummary.class,
124 							ViewRiksdagenVoteDataBallotPartySummary_.embeddedId,
125 							RiksdagenVoteDataBallotPartyEmbeddedId.class,
126 							RiksdagenVoteDataBallotPartyEmbeddedId_.ballotId, pageId);
127 
128 			final List<ViewRiksdagenCommitteeBallotDecisionSummary> decisionSummaries = dataDecisionContainer
129 					.getAllBy(ViewRiksdagenCommitteeBallotDecisionSummary_.ballotId, pageId);
130 
131 			final boolean useDecisionSummaries = !decisionSummaries.isEmpty();
132 
133 			String mainTitle;
134 			String subTitle;
135 			if (useDecisionSummaries) {
136 				mainTitle = "Ballot Overview " + decisionSummaries.get(0).getTitle() + " - "
137 						+ decisionSummaries.get(0).getSubTitle();
138 			} else {
139 				final ViewRiksdagenVoteDataBallotSummary firstBallot = ballots.get(FIRST_OBJECT);
140 				mainTitle = "Ballot Overview " + firstBallot.getEmbeddedId().getConcern();
141 			}
142 			subTitle = "Ballot Details";
143 
144 			createPageHeader(panel, panelContent, mainTitle, subTitle,
145 					"Explore and analyze ballot results and voting statistics.");
146 
147 			// Card panel
148 			final Panel cardPanel = new Panel();
149 			cardPanel.addStyleName("politician-overview-card");
150 			cardPanel.setWidth("100%");
151 			cardPanel.setHeightUndefined();
152 			Responsive.makeResponsive(cardPanel);
153 
154 			final VerticalLayout cardContent = new VerticalLayout();
155 			cardContent.setMargin(true);
156 			cardContent.setSpacing(true);
157 			cardContent.setWidth("100%");
158 			cardPanel.setContent(cardContent);
159 
160 			panelContent.addComponent(cardPanel);
161 			panelContent.setExpandRatio(cardPanel, ContentRatio.GRID);
162 
163 			// Header layout for the card
164 			final HorizontalLayout headerLayout = new HorizontalLayout();
165 			headerLayout.setSpacing(true);
166 			headerLayout.setWidth("100%");
167 			headerLayout.addStyleName("card-header-section");
168 
169 			final Label titleLabel = new Label("Ballot Information", ContentMode.HTML);
170 			titleLabel.addStyleName("card-title");
171 			titleLabel.setWidthUndefined();
172 			headerLayout.addComponent(titleLabel);
173 			cardContent.addComponent(headerLayout);
174 
175 			// Divider line
176 			final Label divider = new Label("<hr/>", ContentMode.HTML);
177 			divider.addStyleName("card-divider");
178 			divider.setWidth("100%");
179 			cardContent.addComponent(divider);
180 
181 			// Two-column layout
182 			final HorizontalLayout attributesLayout = new HorizontalLayout();
183 			attributesLayout.setSpacing(true);
184 			attributesLayout.setWidth("100%");
185 			cardContent.addComponent(attributesLayout);
186 
187 			// Left column: Ballot Profile (Textual and identifying attributes)
188 			final VerticalLayout profileDetailsLayout = new VerticalLayout();
189 			profileDetailsLayout.setSpacing(true);
190 			profileDetailsLayout.addStyleName("card-details-column");
191 			profileDetailsLayout.setWidthUndefined();
192 
193 			final Label profileDetailsHeader = new Label("Ballot Profile");
194 			profileDetailsHeader.addStyleName("card-section-title");
195 			profileDetailsLayout.addComponent(profileDetailsHeader);
196 
197 			// Right column: Voting Statistics (Numeric and outcome attributes)
198 			final VerticalLayout serviceStatsLayout = new VerticalLayout();
199 			serviceStatsLayout.setSpacing(true);
200 			serviceStatsLayout.addStyleName("card-details-column");
201 			serviceStatsLayout.setWidthUndefined();
202 
203 			final Label serviceStatsHeader = new Label("Voting Statistics");
204 			serviceStatsHeader.addStyleName("card-section-title");
205 			serviceStatsLayout.addComponent(serviceStatsHeader);
206 
207 			attributesLayout.addComponents(profileDetailsLayout, serviceStatsLayout);
208 
209 			// Populate fields depending on data
210 			if (useDecisionSummaries) {
211 				final ViewRiksdagenCommitteeBallotDecisionSummary ds = decisionSummaries.get(FIRST_OBJECT);
212 				// Show essential profile fields
213 				profileDetailsLayout.addComponent(createInfoRow("Vote Date:", String.valueOf(ds.getVoteDate()), VaadinIcons.CALENDAR, "Date of the vote"));
214 				profileDetailsLayout.addComponent(createInfoRow("Title:", ds.getTitle(), VaadinIcons.FILE_TEXT_O, "Title of the ballot"));
215 				profileDetailsLayout.addComponent(createInfoRow("SubTitle:", ds.getSubTitle(), VaadinIcons.FILE_TEXT, "Subtitle of the ballot"));
216 				profileDetailsLayout.addComponent(createInfoRow("Decision Type:", ds.getDecisionType(), VaadinIcons.QUESTION_CIRCLE, "Type of decision"));
217 
218 				// Stats fields
219 				serviceStatsLayout.addComponent(createInfoRow("Concern:", ds.getEmbeddedId().getConcern(), VaadinIcons.CLIPBOARD, "Concern or topic of the ballot"));
220 				serviceStatsLayout.addComponent(createInfoRow("Ballot Type:", ds.getBallotType(), VaadinIcons.BULLETS, "Type of ballot"));
221 				serviceStatsLayout.addComponent(createInfoRow("Winner:", ds.getWinner(), VaadinIcons.TROPHY, "Winner or outcome"));
222 				serviceStatsLayout.addComponent(createInfoRow("Approved:", String.valueOf(ds.isApproved()), VaadinIcons.CHECK, "Whether the proposal was approved"));
223 
224 				// Optional: If still too crowded, remove againstProposal fields
225 				// serviceStatsLayout.addComponent(createInfoRow("Against Proposal Parties:", ds.getAgainstProposalParties(), VaadinIcons.WARNING, "Parties against the proposal"));
226 				// serviceStatsLayout.addComponent(createInfoRow("Against Proposal Number:", String.valueOf(ds.getAgainstProposalNumber()), VaadinIcons.WARNING, "Number of parties against the proposal"));
227 			} else {
228 				// Non-decision scenario
229 				final ViewRiksdagenVoteDataBallotSummary bs = ballots.get(FIRST_OBJECT);
230 				// Profile fields
231 				profileDetailsLayout.addComponent(createInfoRow("Ballot ID:", bs.getEmbeddedId().getBallotId(), VaadinIcons.CLIPBOARD_USER, "Ballot identification"));
232 				profileDetailsLayout.addComponent(createInfoRow("RM:", bs.getRm(), VaadinIcons.CALENDAR, "Session (RM) identifier"));
233 				profileDetailsLayout.addComponent(createInfoRow("Vote Date:", String.valueOf(bs.getVoteDate()), VaadinIcons.CALENDAR, "Date of the vote"));
234 				profileDetailsLayout.addComponent(createInfoRow("Issue:", bs.getEmbeddedId().getIssue(), VaadinIcons.CLIPBOARD_TEXT, "The specific issue voted upon"));
235 				profileDetailsLayout.addComponent(createInfoRow("Concern:", bs.getEmbeddedId().getConcern(), VaadinIcons.CLIPBOARD, "Concern or topic of the ballot"));
236 				profileDetailsLayout.addComponent(createInfoRow("Ballot Type:", bs.getBallotType(), VaadinIcons.BULLETS, "Type of ballot"));
237 				profileDetailsLayout.addComponent(createInfoRow("Label:", bs.getLabel(), VaadinIcons.FILE_TEXT, "Short label or description"));
238 
239 				// Stats fields
240 				serviceStatsLayout.addComponent(createInfoRow("Total Votes:", String.valueOf(bs.getTotalVotes()), VaadinIcons.GROUP, "Total number of votes cast"));
241 				serviceStatsLayout.addComponent(createInfoRow("Yes Votes:", String.valueOf(bs.getYesVotes()), VaadinIcons.THUMBS_UP, "Number of 'yes' votes"));
242 				serviceStatsLayout.addComponent(createInfoRow("No Votes:", String.valueOf(bs.getNoVotes()), VaadinIcons.THUMBS_DOWN, "Number of 'no' votes"));
243 				serviceStatsLayout.addComponent(createInfoRow("Abstain Votes:", String.valueOf(bs.getAbstainVotes()), VaadinIcons.SPLIT, "Number of abstentions"));
244 				serviceStatsLayout.addComponent(createInfoRow("Absent Votes:", String.valueOf(bs.getAbsentVotes()), VaadinIcons.EXIT_O, "Number of absent voters"));
245 				serviceStatsLayout.addComponent(createInfoRow("Approved:", String.valueOf(bs.isApproved()), VaadinIcons.CHECK, "Whether the proposal was approved"));
246 			}
247 
248 			// Spacer before Party Ballot Summary
249 			final Label tableDivider = new Label("<hr/>", ContentMode.HTML);
250 			tableDivider.addStyleName("card-divider");
251 			tableDivider.setWidth("100%");
252 			panelContent.addComponent(tableDivider);
253 
254 			// Party Ballot Summary table
255 			getGridFactory().createBasicBeanItemNestedPropertiesGrid(panelContent,
256 					ViewRiksdagenVoteDataBallotPartySummary.class, partyBallotList, PARTY_BALLOT_SUMMARY,
257 					NESTED_PROPERTIES, COLUMN_ORDER, HIDE_COLUMNS, LISTENER, EMBEDDED_ID_PARTY, null);
258 
259 			// Overview layout after table
260 			final VerticalLayout overviewLayout = new VerticalLayout();
261 			overviewLayout.setSizeFull();
262 			panelContent.addComponent(overviewLayout);
263 			panelContent.setExpandRatio(overviewLayout, ContentRatio.LARGE_FORM);
264 			getBallotMenuItemFactory().createOverviewPage(overviewLayout, pageId);
265 
266 			getPageActionEventHelper().createPageEvent(ViewAction.VISIT_BALLOT_VIEW, ApplicationEventGroup.USER, NAME,
267 					parameters, pageId);
268 		}
269 
270 		return panelContent;
271 	}

273 	/**
274 	 * Creates the info row.
275 	 *
276 	 * @param caption the caption
277 	 * @param value the value
278 	 * @param icon the icon
279 	 * @param tooltip the tooltip
280 	 * @return the horizontal layout
281 	 */
282 	private HorizontalLayout createInfoRow(final String caption, final String value, VaadinIcons icon,
283 			final String tooltip) {
284 		final HorizontalLayout layout = new HorizontalLayout();
285 		layout.setSpacing(true);
286 		layout.addStyleName("metric-label");
287 		layout.setWidthUndefined();
288 
289 		if (icon != null) {
290 			final Label iconLabel = new Label(icon.getHtml(), ContentMode.HTML);
291 			iconLabel.addStyleName("card-info-icon");
292 			if (tooltip != null && !tooltip.isEmpty()) {
293 				iconLabel.setDescription(tooltip);
294 			}
295 			layout.addComponent(iconLabel);
296 		}
297 
298 		final Label captionLabel = new Label(caption);
299 		captionLabel.addStyleName("card-info-caption");
300 		if (tooltip != null && !tooltip.isEmpty()) {
301 			captionLabel.setDescription(tooltip);
302 		}
303 
304 		final Label valueLabel = new Label(value != null ? value : "");
305 		valueLabel.addStyleName("card-info-value");
306 
307 		layout.addComponents(captionLabel, valueLabel);
308 		return layout;
309 	}

311 	/**
312 	 * Matches.
313 	 *
314 	 * @param page the page
315 	 * @param parameters the parameters
316 	 * @return true, if successful
317 	 */
318 	@Override
319 	public boolean matches(final String page, final String parameters) {
320 		final String pageId = getPageId(parameters);
321 		return NAME.equals(page)
322 				&& (StringUtils.isEmpty(parameters) || parameters.equals(pageId) || parameters.contains(PageMode.OVERVIEW.toString()));
323 	}
