185 	/**
186 	 * Creates the basic layout with panel and footer.
187 	 *
188 	 * @param panelName
189 	 *            the panel name
190 	 */
191 	protected final void createBasicLayoutWithPanelAndFooter(final String panelName) {
192 		final VerticalLayout layout = createFullSizeVerticalLayout();
193 		final VerticalLayout pageModeContent = createFullSizeVerticalLayout(false,false);
194 		layout.addComponent(pageModeContent);
195 
196 		final HorizontalLayout topHeader = new HorizontalLayout();
197 
198 		addLogoToHeader(topHeader);
199 		createTopTitleHeader(topHeader);
200 
201 		topHeaderRightPanel.removeAllComponents();
202 		topHeader.addComponent(topHeaderRightPanel);
203 		topHeader.setComponentAlignment(topHeaderRightPanel, Alignment.MIDDLE_RIGHT);
204 		topHeader.setExpandRatio(topHeaderRightPanel, ContentRatio.LARGE);
205 
206 		createTopHeaderActionsForUserContext();
207 
208 		topHeaderRightPanel.setWidth("100%");
209 		topHeaderRightPanel.setHeight("50px");
210 
211 		topHeader.setWidth("100%");
212 		topHeader.setHeight("50px");
213 
214 		pageModeContent.addComponent(topHeader);
215 		pageModeContent.setComponentAlignment(topHeader, Alignment.TOP_CENTER);
216 
217 		pageModeContent.addComponent(getBarmenu());
218 		pageModeContent.setComponentAlignment(getBarmenu(), Alignment.TOP_CENTER);
219 
220 		panel = new Panel(panelName);
221 		panel.addStyleName("v-panel-page-panel");
222 
223 		panel.setSizeFull();
224 		pageModeContent.addComponent(panel);
225 		pageModeContent.setExpandRatio(panel, ContentRatio.FULL_SIZE);
226 
227 		pageModeContent.addComponent(pageLinkFactory.createMainViewPageLink());
228 		setContent(layout);
229 
230 		setWidth(100, Unit.PERCENTAGE);
231 		setHeight(100, Unit.PERCENTAGE);
232 		setSizeFull();
233 
234 	}

236 	/**
237 	 * Creates the top header actions for user context.
238 	 */
239 	private void createTopHeaderActionsForUserContext() {
240 		if (UserContextUtil.allowRoleInSecurityContext(ROLE_ADMIN) || UserContextUtil.allowRoleInSecurityContext(ROLE_USER)) {
241 			final Link userHomePageLink = pageLinkFactory.createUserHomeViewPageLink();
242 			topHeaderRightPanel.addComponent(userHomePageLink);
243 			topHeaderRightPanel.setComponentAlignment(userHomePageLink, Alignment.MIDDLE_RIGHT);
244 
245 			final Button logoutButton = new Button(LOGOUT,VaadinIcons.SIGN_OUT);
246 
247 			final LogoutRequest logoutRequest = new LogoutRequest();
248 			logoutRequest.setSessionId(RequestContextHolder.currentRequestAttributes().getSessionId());
249 			logoutButton.addClickListener(new LogoutClickListener(logoutRequest));
250 
251 			topHeaderRightPanel.addComponent(logoutButton);
252 			topHeaderRightPanel.setComponentAlignment(logoutButton, Alignment.MIDDLE_RIGHT);
253 
254 		} else {
255 			final Link createRegisterPageLink = pageLinkFactory.createRegisterPageLink();
256 			topHeaderRightPanel.addComponent(createRegisterPageLink);
257 			topHeaderRightPanel.setComponentAlignment(createRegisterPageLink, Alignment.MIDDLE_RIGHT);
258 
259 			final Link createLoginPageLink = pageLinkFactory.createLoginPageLink();
260 			topHeaderRightPanel.addComponent(createLoginPageLink);
261 			topHeaderRightPanel.setComponentAlignment(createLoginPageLink, Alignment.MIDDLE_RIGHT);
262 		}
263 	}

265 	@Override
266 	public final void enter(final ViewChangeEvent event) {
267 		try {
268 
269 			final String parameters = Jsoup.clean(event.getParameters(), Safelist.basic());
270 
271 			for (final PageModeContentFactory pageModeContentFactory : pageModeContentFactoryMap.values()) {
272 				if (pageModeContentFactory.matches(pageName, parameters) && pageModeContentFactory.validReference(parameters)) {
273 					getPanel().setContent(pageModeContentFactory.createContent(parameters, getBarmenu(), getPanel()));
274 					return;
275 				}
276 			}
277 
278 			LOGGER.warn("Invalid reference, content not found:{}/{}",pageName, parameters);
279 			final VerticalLayout panelContent = createFullSizeVerticalLayout();
280 
281 			menuItemFactory.createMainPageMenuBar(getBarmenu());
282 
283 			LabelFactory.createHeader2Label(panelContent,"Invalid reference, content not found:" +pageName+ "/"+ parameters);
284 
285 			getPanel().setContent(panelContent);
286 			getPanel().setCaption("Invalid Reference");
287 		} catch (final AccessDeniedException e ) {
288 			LOGGER.warn("Authorization Failure:: {} : {}  exception : {}",e.getMessage(),pageName,e.getClass().getName());
289 			final VerticalLayout panelContent = createFullSizeVerticalLayout();
290 			LabelFactory.createHeader2Label(panelContent,"Access denied:" +pageName);
291 			getPanel().setContent(panelContent);
292 			getPanel().setCaption("Access denied");
293 		}
294 	}
