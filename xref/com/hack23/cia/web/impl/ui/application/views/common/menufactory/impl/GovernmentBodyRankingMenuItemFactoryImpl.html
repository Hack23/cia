126     /**
127      * Creates the government body ranking menu bar and integrates it into the main
128      * application menu. Adds government body ranking topics as sub-items.
129      *
130      * @param menuBar the main application {@link MenuBar} to which the government
131      *                body ranking menu will be added.
132      */
133     @Override
134     public void createGovernmentBodyRankingMenuBar(final MenuBar menuBar) {
135         initApplicationMenuBar(menuBar);
136         applicationMenuItemFactory.addRankingMenu(menuBar);
137         createGovernmentBodyRankingTopics(menuBar.addItem(GOVERNMENT_BODY_RANKING, VaadinIcons.BUILDING_O, null));
138     }
139 
140     /**
141      * Adds menu items for various government body ranking topics:
142      * <ul>
143      *   <li>Overview: General dashboard icon</li>
144      *   <li>Government Bodies: Building icon for institutions</li>
145      *   <li>Headcount: Multiple users for staffing</li>
146      *   <li>Income: Money deposit icon for revenues</li>
147      *   <li>Expenditure: Money withdraw icon for spending</li>
148      *   <li>Page Visit History: History icon for past engagement</li>
149      * </ul>
150      *
151      * @param menuItem the {@link MenuBar.MenuItem} representing "GovernmentBody Ranking"
152      *                 under which these topics will be added.
153      */
154     @Override
155     public void createGovernmentBodyRankingTopics(final MenuBar.MenuItem menuItem) {
156         menuItem.addItem(OVERVIEW_TEXT, VaadinIcons.DASHBOARD, COMMAN_OVERVIEW);
157         menuItem.addItem(GOVERNMENT_BODIES, VaadinIcons.BUILDING_O, COMMAND_DATAGRID);
158         menuItem.addItem(GOVERNMENT_BODY_HEADCOUNT_PER_MINISTRY, VaadinIcons.USER_CHECK, COMMAND_GOVERNMENT_BODIES_HEADCOUNT);
159         menuItem.addItem(GOVERNMENT_BODY_INCOME_PER_MINISTRY, VaadinIcons.MONEY, COMMAND_GOVERNMENT_BODIES_INCOME);
160         menuItem.addItem(GOVERNMENT_BODY_EXPENDITURE_PER_MINISTRY, VaadinIcons.MONEY_WITHDRAW, COMMAND_GOVERNMENT_BODIES_EXPENDITURE);
161 
162         menuItem.addItem(HEADCOUNT, VaadinIcons.USERS, COMMAND_HEADCOUNT);
163         menuItem.addItem(INCOME, VaadinIcons.MONEY_DEPOSIT, COMMAND_INCOME);
164         menuItem.addItem(EXPENDITURE, VaadinIcons.MONEY_WITHDRAW, COMMAND_EXPENDITURE);
165         menuItem.addItem(PAGE_VISIT_HISTORY_TEXT, VaadinIcons.CHART, COMMAND_PAGEVISITHISTORY);
166     }

168     /**
169      * Creates the overview page with quick-access links to government body ranking topics.
170      * Each button uses an icon and a brief description to guide the user:
171      * <ul>
172      *   <li>Government Bodies: Building icon, structural roles</li>
173      *   <li>Headcount: Users icon, staffing influence</li>
174      *   <li>Income: Money deposit icon, revenue streams</li>
175      *   <li>Expenditure: Money withdraw icon, spending focus</li>
176      *   <li>Page Visit History: History icon, past engagement</li>
177      * </ul>
178      *
179      * @param panelContent the {@link VerticalLayout} container that holds the overview
180      *                     content. New UI components will be added here.
181      */
182     @Override
183     public void createOverviewPage(final VerticalLayout panelContent) {
184         final ResponsiveRow grid = RowUtil.createGridLayout(panelContent);
185 
186         createButtonLink(grid, GOVERNMENT_BODIES, VaadinIcons.BUILDING_O, COMMAND_DATAGRID, CURRENT_GOVERNMENT_BODIES_DESCRIPTION);
187 		createButtonLink(grid, GOVERNMENT_BODY_HEADCOUNT_PER_MINISTRY, VaadinIcons.USER_CHECK, COMMAND_GOVERNMENT_BODIES_HEADCOUNT, GOVERNMENT_BODY_HEADCOUNT_PER_MINISTRY_DESCRIPTION);
188 		createButtonLink(grid, GOVERNMENT_BODY_INCOME_PER_MINISTRY, VaadinIcons.MONEY, COMMAND_GOVERNMENT_BODIES_INCOME, GOVERNMENT_BODY_INCOME_PER_MINISTRY_DESCRIPTION);
189 		createButtonLink(grid, GOVERNMENT_BODY_EXPENDITURE_PER_MINISTRY, VaadinIcons.MONEY_WITHDRAW, COMMAND_GOVERNMENT_BODIES_EXPENDITURE, GOVERNMENT_BODY_EXPENDITURE_PER_MINISTRY_DESCRIPTION);
190 
191         createButtonLink(grid, HEADCOUNT, VaadinIcons.USERS, COMMAND_HEADCOUNT, HEADCOUNT_DESCRIPTION);
192         createButtonLink(grid, INCOME, VaadinIcons.MONEY_DEPOSIT, COMMAND_INCOME, INCOME_DESCRIPTION);
193         createButtonLink(grid, EXPENDITURE, VaadinIcons.MONEY_WITHDRAW, COMMAND_EXPENDITURE, EXPENDITURE_DESCRIPTION);
194         createButtonLink(grid, PAGE_VISIT_HISTORY_TEXT, VaadinIcons.CHART, COMMAND_PAGEVISITHISTORY, PAGE_VISIT_HISTORY_DESCRIPTION);
195     }
