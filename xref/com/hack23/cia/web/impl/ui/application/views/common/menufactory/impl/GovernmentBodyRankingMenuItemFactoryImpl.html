55  			GovernmentBodyPageMode.HEADCOUNT.toString());

57  	private static final PageModeMenuCommand COMMAND_INCOME = new PageModeMenuCommand(UserViews.GOVERNMENT_BODY_RANKING_VIEW_NAME,
58  			GovernmentBodyPageMode.INCOME.toString());

60  	/** The Constant COMMAND_PAGEVISITHISTORY. */

63  
64  	private static final String EXPENDITURE = "Expenditure";

67  	private static final String GOVERNMENT_BODIES = "Government bodies";

69  	/** The Constant GOVERNMENT_BODY_RANKING. */

75  	private static final String INCOME = "Income";

77  	/** The Constant OVERVIEW_TEXT. */

94  	@Override
95  	public void createGovernmentBodyRankingMenuBar(final MenuBar menuBar) {
96  		initApplicationMenuBar(menuBar);
97  
98  		applicationMenuItemFactory.addRankingMenu(menuBar);
99  
100 		createGovernmentBodyRankingTopics(menuBar.addItem(GOVERNMENT_BODY_RANKING, null, null));
101 
102 	}
103 
104 	@Override
105 	public void createGovernmentBodyRankingTopics(final MenuItem ministryMenuItem) {
106 
107 		ministryMenuItem.addItem(OVERVIEW_TEXT, VaadinIcons.GROUP, COMMAN_OVERVIEW);
108 
109 		ministryMenuItem.addItem(GOVERNMENT_BODIES, VaadinIcons.GROUP, COMMAND_DATAGRID);
110 
111 		ministryMenuItem.addItem(HEADCOUNT, VaadinIcons.GROUP, COMMAND_HEADCOUNT);
112 		ministryMenuItem.addItem(INCOME, VaadinIcons.GROUP, COMMAND_INCOME);
113 		ministryMenuItem.addItem(EXPENDITURE, VaadinIcons.GROUP, COMMAND_EXPENDITURE);
114 
115 		ministryMenuItem.addItem(PAGE_VISIT_HISTORY_TEXT, VaadinIcons.GROUP, COMMAND_PAGEVISITHISTORY);
116 
117 	}
118 
119 	@Override
120 	public void createOverviewPage(final VerticalLayout panelContent) {
121 		final ResponsiveRow grid = RowUtil.createGridLayout(panelContent);
122 
123 		createButtonLink(grid, GOVERNMENT_BODIES, VaadinIcons.GROUP, COMMAND_DATAGRID, "Current government bodies");
124 
125 		createButtonLink(grid, HEADCOUNT, VaadinIcons.GROUP, COMMAND_HEADCOUNT, "Desc");
126 		createButtonLink(grid, INCOME, VaadinIcons.GROUP, COMMAND_INCOME, "Desc");
127 		createButtonLink(grid, EXPENDITURE, VaadinIcons.GROUP, COMMAND_EXPENDITURE , "Desc");
128 
129 		createButtonLink(grid, PAGE_VISIT_HISTORY_TEXT, VaadinIcons.GROUP, COMMAND_PAGEVISITHISTORY, "View history of page visit for this page.");
130 
131 	}
