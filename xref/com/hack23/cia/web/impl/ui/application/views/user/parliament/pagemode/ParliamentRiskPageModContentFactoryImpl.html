52   * The Class ParliamentRiskPageModContentFactoryImpl.
53   */
54  @Component
55  public final class ParliamentRiskPageModContentFactoryImpl extends AbstractParliamentPageModContentFactoryImpl {
56  
57  	private static final ComplianceCheckPageItemRendererClickListener CLICK_LISTENER = new ComplianceCheckPageItemRendererClickListener();
58  	/** The Constant PARLIAMENT_DECISION_FLOW. */
59  	private static final String PARLIAMENT_RISK_SUMMARY = "Parliament Risk Summary";
60  
61  	/**
62  	 * Instantiates a new parliament risk page mod content factory impl.
63  	 */
64  	public ParliamentRiskPageModContentFactoryImpl() {
65  		super();
66  	}
67  
68  	@Secured({ "ROLE_ANONYMOUS", "ROLE_USER", "ROLE_ADMIN" })
69  	@Override
70  	public Layout createContent(final String parameters, final MenuBar menuBar, final Panel panel) {
71  		final VerticalLayout panelContent = createPanelContent();
72  		getParliamentMenuItemFactory().createParliamentTopicMenu(menuBar);
73  
74  		final String pageId = getPageId(parameters);
75  
76  		final HorizontalLayout horizontalLayout = new HorizontalLayout();
77  
78  		final DataContainer<RuleViolation, String> dataContainer = getApplicationManager()
79  				.getDataContainer(RuleViolation.class);
80  
81  		final List<RuleViolation> ruleViolations = dataContainer.getAll();
82  		final List<ComplianceCheckImpl> checks = new ArrayList<>();
83  
84  		for (final Entry<String, List<RuleViolation>> idMapViolations : ruleViolations.stream().collect(Collectors.groupingBy(RuleViolation::getReferenceId)).entrySet()) {
85  			checks.add(new ComplianceCheckImpl(idMapViolations.getValue()));
86  		}
87  
88  		Collections.sort(checks, (o1, o2) -> Integer.compare(o2.getRuleViolations().size(), o1.getRuleViolations().size()));
89  
90  		for (final Entry<Status, List<RuleViolation>> statusEntry : ruleViolations.stream().collect(Collectors.groupingBy(RuleViolation::getStatus)).entrySet()) {
91  			horizontalLayout.addComponent(new CounterStatisticsCard(
92  					VaadinIcons.WARNING,new CounterStatisticModel("ALL:" +statusEntry.getKey(),statusEntry.getValue().size()).withShow(StatisticShow.Sum)
93                      .withIconHidden().withShowOnlyStatistic(true),"ALL:" +statusEntry.getKey()));
94  		}
95  
96  		for (final Entry<ResourceType, List<RuleViolation>> statusEntry : ruleViolations.stream().collect(Collectors.groupingBy(RuleViolation::getResourceType)).entrySet()) {
97  			horizontalLayout.addComponent(new CounterStatisticsCard(
98  					VaadinIcons.WARNING,new CounterStatisticModel("ALL:" +statusEntry.getKey(),statusEntry.getValue().size()).withShow(StatisticShow.Sum)
99                      .withIconHidden().withShowOnlyStatistic(true),"ALL:" +statusEntry.getKey()));
100 		}
101 
102 
103 		panelContent.addComponent(horizontalLayout);
104 
105 		getGridFactory().createBasicBeanItemGrid(panelContent, ComplianceCheckImpl.class, checks, "Risk",
106 				new String[] { "name", "resourceType", "numberRuleViolations", "ruleSummary" }, new String[] { "id", "ruleViolations" }, CLICK_LISTENER, null, null);
107 
108 		getPageActionEventHelper().createPageEvent(ViewAction.VISIT_PARLIAMENT_RANKING_VIEW, ApplicationEventGroup.USER,
109 				NAME, parameters, pageId);
110 		panel.setCaption(new StringBuilder().append(NAME).append("::").append(PARLIAMENT_RISK_SUMMARY).toString());
111 
112 		return panelContent;
113 
114 	}
115 
116 	/**
117 	 * The Class ComplianceCheckImpl.
118 	 */
119 	public static class ComplianceCheckImpl implements ComplianceCheck {
120 
121 		private static final long serialVersionUID = 1L;
122 		private final List<RuleViolation> ruleViolations;
123 		private final ResourceType resourceType;
124 		private final String id;
125 		private final String name;
126 
127 		/**
128 		 * Instantiates a new compliance check impl.
129 		 *
130 		 * @param ruleViolations the rule violations
131 		 */
132 		public ComplianceCheckImpl(final List<RuleViolation> ruleViolations) {
133 			this.ruleViolations = new ArrayList<>(ruleViolations);
134 			this.resourceType = ruleViolations.get(0).getResourceType();
135 			this.id = ruleViolations.get(0).getReferenceId();
136 			this.name = ruleViolations.get(0).getName();
137 		}
138 
139 		@Override
140 		public ResourceType getResourceType() {
141 			return resourceType;
142 		}
143 
144 		@Override
145 		public List<RuleViolation> getRuleViolations() {
146 			return new ArrayList<>(ruleViolations);
147 		}
148 
149 
150 		/**
151 		 * Gets the number rule violations.
152 		 *
153 		 * @return the number rule violations
154 		 */
155 		public int getNumberRuleViolations() {
156 			return ruleViolations.size();
157 		}
158 
159 		/**
160 		 * Gets the rule summary.
161 		 *
162 		 * @return the rule summary
163 		 */
164 		public String getRuleSummary() {
165 			final StringBuilder builder = new StringBuilder();
166 			for (final RuleViolation ruleViolation : ruleViolations) {
167 				builder.append('[').append(ruleViolation.getRuleName()).append('/').append(ruleViolation.getStatus()) .append(']');
168 			}
169 			return builder.toString();
170 		}
171 
172 		@Override
173 		public String getId() {
174 			return id;
175 		}
176 
177 		@Override
178 		public String getName() {
179 			return name;
180 		}
181 	}
182 
183 
184 	@Override
185 	public boolean matches(final String page, final String parameters) {
186 		return NAME.equals(page) && StringUtils.contains(parameters, PageMode.RULES.toString())
187 				&& parameters.contains(RiskIndicators.RISK_SUMMARY.toString());
188 	}
