87  	 * Creates the page visit history.
88  	 *
89  	 * @param pageName
90  	 *            the page name
91  	 * @param pageId
92  	 *            the page id
93  	 * @param panelContent
94  	 *            the panel content
95  	 */
96  	protected final void createPageVisitHistory(final String pageName, final String pageId,
97  			final VerticalLayout panelContent) {
98  
99  		final TabSheet tabsheet = new TabSheet();
100 		tabsheet.setWidth(100, Unit.PERCENTAGE);
101 		tabsheet.setHeight(100, Unit.PERCENTAGE);
102 
103 		panelContent.addComponent(tabsheet);
104 		panelContent.setExpandRatio(tabsheet, ContentRatio.LARGE);
105 
106 		final HorizontalLayout tabContentPageItemRankHistory = new HorizontalLayout();
107 		tabContentPageItemRankHistory.setWidth(100, Unit.PERCENTAGE);
108 		tabContentPageItemRankHistory.setHeight(100, Unit.PERCENTAGE);
109 		final Tab tabPageItemRankHistory = tabsheet.addTab(tabContentPageItemRankHistory);
110 
111 		tabPageItemRankHistory.setCaption(CURRENT_PAGE_VISIT_HISTORY);
112 		adminChartDataManager.createApplicationActionEventPageElementDailySummaryChart(tabContentPageItemRankHistory,
113 				pageName, pageId);
114 
115 		final HorizontalLayout tabContentPageModeSummary = new HorizontalLayout();
116 		tabContentPageModeSummary.setWidth(100, Unit.PERCENTAGE);
117 		tabContentPageModeSummary.setHeight(100, Unit.PERCENTAGE);
118 		final Tab tabPageModeSummary = tabsheet.addTab(tabContentPageModeSummary);
119 		tabPageModeSummary.setCaption(GENERAL_PAGE_MODE_PAGE_VISIT);
120 
121 		adminChartDataManager.createApplicationActionEventPageModeDailySummaryChart(tabContentPageModeSummary,
122 				pageName);
123 
124 	}

189 	/**
190 	 * Gets the page id.
191 	 *
192 	 * @param parameters
193 	 *            the parameters
194 	 * @return the page id
195 	 */
196 	protected static final String getPageId(final String parameters) {
197 		if (parameters != null) {
198 			String cleanedString = parameters;
199 			if (parameters.contains("[")) {
200 				cleanedString = cleanedString.replace(cleanedString.substring(cleanedString.indexOf('[') , cleanedString.lastIndexOf(']')+1), "");
201 			}
202 
203 			return cleanedString.substring(cleanedString.lastIndexOf('/') + "/".length());
204 		} else {
205 			return "";
206 		}
207 	}

218 	/**
219 	 * Gets the page nr.
220 	 *
221 	 * @param parameters the parameters
222 	 * @return the page nr
223 	 */
224 	protected static final int getPageNr(final String parameters) {
225 		final String cleanedString = parameters;
226 		String pageNrValue;
227 		if (cleanedString != null && cleanedString.contains("[") && cleanedString.contains("]")) {
228 			pageNrValue = cleanedString.substring(cleanedString.indexOf('[') + 1, cleanedString.lastIndexOf(']'));
229 		} else {
230 			pageNrValue = "";
231 		}
232 
233 		int pageNr = 1;
234 
235 		if (pageNrValue.length() > 0 && !pageNrValue.contains("/")) {
236 			pageNr = Integer.parseInt(pageNrValue);
237 		}
238 		return pageNr;
239 	}
