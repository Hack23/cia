58  	/**
59  	 * Creates the content.
60  	 *
61  	 * @param parameters the parameters
62  	 * @param menuBar the menu bar
63  	 * @param panel the panel
64  	 * @return the layout
65  	 */
66  	@Secured({ "ROLE_ANONYMOUS", "ROLE_USER", "ROLE_ADMIN" })
67  	@Override
68  	public Layout createContent(final String parameters, final MenuBar menuBar, final Panel panel) {
69  		final VerticalLayout panelContent = createPanelContent();
70  		panel.setContent(panelContent);
71  
72  		final String pageId = getPageId(parameters);
73  		final ViewRiksdagenParty viewRiksdagenParty = getItem(parameters);
74  
75  		getPartyMenuItemFactory().createPartyMenuBar(menuBar, pageId);
76  		createPageHeader(panel, panelContent, "Party Overview " + viewRiksdagenParty.getPartyName(),
77  				"Party Details", "Explore detailed information about political parties and their activities.");
78  
79  		final Link addPartyPageLink = getPageLinkFactory().addPartyPageLink(viewRiksdagenParty);
80  		panelContent.addComponent(addPartyPageLink);
81  		panelContent.setExpandRatio(addPartyPageLink, ContentRatio.SMALL);
82  
83  		// Load summary if available
84  		final DataContainer<ViewRiksdagenPartySummary, String> partySummaryDataContainer = getApplicationManager()
85  				.getDataContainer(ViewRiksdagenPartySummary.class);
86  		final ViewRiksdagenPartySummary viewRiksdagenPartySummary = partySummaryDataContainer.load(pageId);
87  
88  		// Create a card panel similar to the politician overview
89  		final Panel cardPanel = new Panel();
90  		cardPanel.addStyleName("politician-overview-card");
91  		cardPanel.setWidth("100%");
92  		cardPanel.setHeightUndefined();
93  		Responsive.makeResponsive(cardPanel);
94  
95  		final VerticalLayout cardContent = new VerticalLayout();
96  		cardContent.setMargin(true);
97  		cardContent.setSpacing(true);
98  		cardContent.setWidth("100%");
99  		cardPanel.setContent(cardContent);
100 
101 		panelContent.addComponent(cardPanel);
102 		panelContent.setExpandRatio(cardPanel, ContentRatio.SMALL_GRID);
103 
104 		// Header layout
105 		final HorizontalLayout headerLayout = new HorizontalLayout();
106 		headerLayout.setSpacing(true);
107 		headerLayout.setWidth("100%");
108 		headerLayout.addStyleName("card-header-section");
109 
110 		final String titleText = viewRiksdagenParty.getPartyName();
111 		final Label titleLabel = new Label(titleText, ContentMode.HTML);
112 		titleLabel.addStyleName("card-title");
113 		titleLabel.setWidthUndefined();
114 		headerLayout.addComponent(titleLabel);
115 
116 		cardContent.addComponent(headerLayout);
117 
118 		// Divider line for better separation
119 		final Label divider = new Label("<hr/>", ContentMode.HTML);
120 		divider.addStyleName("card-divider");
121 		divider.setWidth("100%");
122 		cardContent.addComponent(divider);
123 
124 		// Create single row for four sections
125 		final HorizontalLayout sectionsLayout = new HorizontalLayout();
126 		sectionsLayout.setSpacing(true);
127 		sectionsLayout.setWidth("100%");
128 		cardContent.addComponent(sectionsLayout);
129 
130 		// 1. Political Influence & Position
131 		final VerticalLayout politicalInfluenceLayout = createSectionLayout("Political Influence & Position");
132 		addPoliticalInfluenceMetrics(politicalInfluenceLayout, viewRiksdagenParty, viewRiksdagenPartySummary);
133 		sectionsLayout.addComponent(politicalInfluenceLayout);
134 		sectionsLayout.setExpandRatio(politicalInfluenceLayout, 1.0f);
135 
136 		// 2. Parliamentary Engagement
137 		final VerticalLayout parliamentaryEngagementLayout = createSectionLayout("Parliamentary Engagement");
138 		addParliamentaryEngagementMetrics(parliamentaryEngagementLayout, viewRiksdagenParty, viewRiksdagenPartySummary);
139 		sectionsLayout.addComponent(parliamentaryEngagementLayout);
140 		sectionsLayout.setExpandRatio(parliamentaryEngagementLayout, 1.0f);
141 
142 		// 3. Legislative Impact
143 		final VerticalLayout legislativeImpactLayout = createSectionLayout("Legislative Impact");
144 		addLegislativeImpactMetrics(legislativeImpactLayout, viewRiksdagenParty, viewRiksdagenPartySummary);
145 		sectionsLayout.addComponent(legislativeImpactLayout);
146 		sectionsLayout.setExpandRatio(legislativeImpactLayout, 1.0f);
147 
148 		// 4. Member Performance
149 		final VerticalLayout memberPerformanceLayout = createSectionLayout("Member Performance");
150 		addMemberPerformanceMetrics(memberPerformanceLayout, viewRiksdagenParty, viewRiksdagenPartySummary);
151 		sectionsLayout.addComponent(memberPerformanceLayout);
152 		sectionsLayout.setExpandRatio(memberPerformanceLayout, 1.0f);
153 
154 		// After the card, add the overview layout
155 		final VerticalLayout overviewLayout = new VerticalLayout();
156 		overviewLayout.setSizeFull();
157 		panelContent.addComponent(overviewLayout);
158 		panelContent.setExpandRatio(overviewLayout, ContentRatio.LARGE_FORM);
159 
160 		getPartyMenuItemFactory().createOverviewPage(overviewLayout, pageId);
161 
162 		getPageActionEventHelper().createPageEvent(ViewAction.VISIT_PARTY_VIEW, ApplicationEventGroup.USER, NAME, parameters,
163 				pageId);
164 		return panelContent;
165 	}

169 	 * Creates a section layout with a title and consistent styling.
170 	 *
171 	 * @param title the section title
172 	 * @return the vertical layout configured for the section
173 	 */
174 	private VerticalLayout createSectionLayout(String title) {
175 	    final VerticalLayout layout = new VerticalLayout();
176 	    layout.setSpacing(true);
177 	    layout.setMargin(true);
178 	    layout.addStyleName("card-details-column");
179 	    layout.setWidth("100%");
180 
181 	    final Label header = new Label(title);
182 	    header.addStyleName("card-section-title");
183 	    layout.addComponent(header);
184 
185 	    // Add some vertical padding after the header
186 	    final Label padding = new Label();
187 	    padding.setHeight("10px");
188 	    layout.addComponent(padding);
189 
190 	    return layout;
191 	}

363 	/**
364 	 * Calculate percentage.
365 	 *
366 	 * @param value the value
367 	 * @param total the total
368 	 * @return the int
369 	 */
370 	// Helper method for calculating percentages
371 	private int calculatePercentage(long value, long total) {
372 	    return total > 0 ? Math.round((float) value * 100 / total) : 0;
373 	}

375 	/**
376 	 * Calculate activity rate.
377 	 *
378 	 * @param activeMembers the active members
379 	 * @param totalMembers the total members
380 	 * @return the double
381 	 */
382 	// Helper method for calculating activity rates
383 	private double calculateActivityRate(long activeMembers, long totalMembers) {
384 	    return totalMembers > 0 ? (double) activeMembers * 100 / totalMembers : 0;
385 	}

390 	/**
391 	 * Creates a row displaying a caption and value, with optional icon and tooltip.
392 	 *
393 	 * @param caption the field caption
394 	 * @param value   the field value
395 	 * @param icon    a VaadinIcons icon for better visual cue
396 	 * @param tooltip optional tooltip to provide more info
397 	 * @return a HorizontalLayout representing the info row
398 	 */
399 	private HorizontalLayout createInfoRow(final String caption, final String value, VaadinIcons icon, final String tooltip) {
400 		final HorizontalLayout layout = new HorizontalLayout();
401 		layout.setSpacing(true);
402 		layout.addStyleName("metric-label");
403 		layout.setWidthUndefined();
404 
405 		if (icon != null) {
406 			final Label iconLabel = new Label(icon.getHtml(), ContentMode.HTML);
407 			iconLabel.addStyleName("card-info-icon");
408 			if (tooltip != null && !tooltip.isEmpty()) {
409 				iconLabel.setDescription(tooltip);
410 			}
411 			layout.addComponent(iconLabel);
412 		}
413 
414 		final Label captionLabel = new Label(caption);
415 		captionLabel.addStyleName("card-info-caption");
416 		if (tooltip != null && !tooltip.isEmpty()) {
417 			captionLabel.setDescription(tooltip);
418 		}
419 
420 		final Label valueLabel = new Label(value != null ? value : "");
421 		valueLabel.addStyleName("card-info-value");
422 
423 		layout.addComponents(captionLabel, valueLabel);
424 		return layout;
425 	}
