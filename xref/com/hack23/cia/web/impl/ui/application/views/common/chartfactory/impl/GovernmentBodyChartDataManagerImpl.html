91      /**
92       * Adds a data point to the DataSeries if the year and value are valid and value > 0.
93       *
94       * @param dataSeries the data series
95       * @param year the year
96       * @param value the value
97       */
98      private static void addDataPoint(final DataSeries dataSeries, final Integer year, final Number value) {
99          if (dataSeries == null || year == null || value == null) {
100             return;
101         }
102         final double doubleValue = value.doubleValue();
103         if (doubleValue > 0) {
104             // Ensure consistent date format: " 01-JAN-YYYY"
105             final String formattedDate = String.format(Locale.ENGLISH," 01-JAN-%d", year);
106             dataSeries.add(formattedDate, doubleValue);
107         }
108     }

110     /**
111      * Creates a chart using the provided data/series objects, then appends it to the given layout.
112      *
113      * @param layout the layout
114      * @param label the label
115      * @param dataSeries the data series
116      * @param series the series
117      */
118     private void addChartToLayout(final AbstractOrderedLayout layout, final String label,
119                                   final DataSeries dataSeries, final Series series) {
120         Objects.requireNonNull(layout, "Layout cannot be null");
121         Objects.requireNonNull(label, "Label cannot be null");
122 
123         final DCharts chart = new DCharts()
124                 .setDataSeries(dataSeries)
125                 .setOptions(getChartOptions().createOptionsXYDateFloatLogYAxisLegendOutside(series))
126                 .show();
127 
128         ChartUtils.addChart(layout, label, chart, true);
129     }

131     /**
132      * Consolidates logic to retrieve, group, and process data by a descriptive field
133      * (e.g., 'Utgiftsområdesnamn' or 'Inkomsttitelgruppsnamn'), feeding the results into a DataSeries.
134      *
135      * @param dataSeries the data series
136      * @param series the series
137      * @param groupedData the grouped data
138      */
139     private void buildAnnualOutcomeDataSeriesByField(final DataSeries dataSeries, final Series series,
140             final Map<String, List<GovernmentBodyAnnualOutcomeSummary>> groupedData) {
141         Optional.ofNullable(groupedData)
142             .ifPresent(data -> data.entrySet().stream()
143                 .filter(entry -> entry.getValue() != null && !entry.getValue().isEmpty())
144                 .forEach(entry -> {
145                     series.addSeries(new XYseries().setLabel(entry.getKey()));
146                     dataSeries.newSeries();
147 
148                     entry.getValue().stream()
149                         .filter(Objects::nonNull)
150                         .forEach(summary -> {
151                             // Only process if we have a valid year and value map
152                             Optional.ofNullable(summary.getValueMap())
153                                 .ifPresent(valueMap -> {
154                                     // Get the year directly from the summary
155                                     final Integer year = summary.getYear();
156                                     // Get the total value for this year
157                                     final Double totalValue = valueMap.values().stream()
158                                         .filter(Objects::nonNull)
159                                         .mapToDouble(Number::doubleValue)
160                                         .sum();
161 
162                                     // Only add if we have valid year and value
163                                     if (year != null && totalValue > 0) {
164                                         addDataPoint(dataSeries, year, totalValue);
165                                     }
166                                 });
167                         });
168                 }));
169     }

171     /**
172      * Helper method to generate a headcount data series given a map of year -> list of summaries.
173      *
174      * @param dataSeries the data series
175      * @param series the series
176      * @param yearlyData the yearly data
177      * @param label the label
178      */
179     private void buildHeadcountDataSeries(final DataSeries dataSeries, final Series series,
180                                           final Map<Integer, List<GovernmentBodyAnnualSummary>> yearlyData,
181                                           final String label) {
182     	Optional.ofNullable(yearlyData)
183         .ifPresent(data -> {
184             series.addSeries(new XYseries().setLabel(label));
185             dataSeries.newSeries();
186 
187             data.entrySet().stream()
188                 .filter(entry -> Objects.nonNull(entry.getValue()) && !entry.getValue().isEmpty())
189                 .forEach(entry -> {
190                     final int totalHeadcount = entry.getValue().stream()
191                         .filter(Objects::nonNull)
192                         .mapToInt(GovernmentBodyAnnualSummary::getHeadCount)
193                         .sum();
194                     addDataPoint(dataSeries, entry.getKey(), totalHeadcount);
195                 });
196         });
197     }

199     /**
200      * Consolidates logic for creating a chart that is grouped by some string field in descriptionFields.
201      *
202      * @param layout the layout
203      * @param field the field
204      * @param chartLabel the chart label
205      */
206     private void createMinistryFieldSummary(final AbstractOrderedLayout layout, final String field, final String chartLabel) {
207         Objects.requireNonNull(layout, "Layout cannot be null");
208         Objects.requireNonNull(field, "Field cannot be null");
209         Objects.requireNonNull(chartLabel, "Chart label cannot be null");
210 
211         final DataSeries dataSeries = new DataSeries();
212         final Series series = new Series();
213 
214         Optional.ofNullable(esvApi.getGovernmentBodyReportByMinistry())
215             .ifPresent(reportByMinistry ->
216                 reportByMinistry.entrySet().stream()
217                     .filter(entry -> entry.getValue() != null && !entry.getValue().isEmpty())
218                     .forEach(entry -> {
219                         final Map<Integer, Double> annualTotals = entry.getValue().stream()
220                             .filter(Objects::nonNull)
221                             .filter(summary -> Optional.ofNullable(summary.getDescriptionFields())
222                                 .map(fields -> fields.get(field))
223                                 .isPresent())
224                             .collect(Collectors.groupingBy(
225                                 GovernmentBodyAnnualOutcomeSummary::getYear,
226                                 Collectors.summingDouble(GovernmentBodyAnnualOutcomeSummary::getYearTotal)
227                             ));
228 
229                         if (!annualTotals.isEmpty()) {
230                             series.addSeries(new XYseries().setLabel(entry.getKey()));
231                             dataSeries.newSeries();
232                             annualTotals.forEach((year, total) ->
233                                 addDataPoint(dataSeries, year + 1, total));
234                         }
235                     }));
236 
237         addChartToLayout(layout, chartLabel, dataSeries, series);
238     }

255     @Override
256     public void createGovernmentBodyExpenditureSummaryChart(final VerticalLayout layout,
257                                                             final String governmentBodyName) {
258         Objects.requireNonNull(layout, "Layout cannot be null");
259         Objects.requireNonNull(governmentBodyName, "Government body name cannot be null");
260 
261         final Map<String, List<GovernmentBodyAnnualOutcomeSummary>> allSummaries = esvApi.getGovernmentBodyReport();
262 
263         final List<GovernmentBodyAnnualOutcomeSummary> summariesForBody = Optional.ofNullable(allSummaries)
264                 .map(summaries -> summaries.get(governmentBodyName))
265                 .orElse(Collections.emptyList());
266 
267         final Map<String, List<GovernmentBodyAnnualOutcomeSummary>> grouped = summariesForBody.stream()
268                 .filter(Objects::nonNull)
269                 .filter(summary -> Optional.ofNullable(summary.getDescriptionFields())
270                     .map(fields -> fields.get(ANSLAGSPOSTSNAMN))
271                     .isPresent())
272                 .sorted(Comparator.comparing(GovernmentBodyAnnualOutcomeSummary::getYear))
273                 .collect(Collectors.groupingBy(
274                     summary -> summary.getDescriptionFields().get(ANSLAGSPOSTSNAMN)));
275 
276         final DataSeries dataSeries = new DataSeries();
277         final Series series = new Series();
278         buildAnnualOutcomeDataSeriesByField(dataSeries, series, grouped);
279 
280         addChartToLayout(layout, governmentBodyName + " " + ANNUAL_EXPENDITURE, dataSeries, series);
281     }

380     @Override
381     public void createMinistryGovernmentBodyHeadcountSummaryChart(final AbstractOrderedLayout layout) {
382         Objects.requireNonNull(layout, "Layout cannot be null");
383 
384         final Map<Integer, List<GovernmentBodyAnnualSummary>> map = esvApi.getData();
385         final List<String> ministryNames = esvApi.getMinistryNames();
386 
387         final DataSeries dataSeries = new DataSeries();
388         final Series series = new Series();
389 
390         Optional.ofNullable(ministryNames)
391             .filter(names -> map != null)
392             .ifPresent(names -> names.stream()
393                 .forEach(ministryName -> {
394                     series.addSeries(new XYseries().setLabel(ministryName));
395                     dataSeries.newSeries();
396 
397                     map.entrySet().stream()
398                         .filter(entry -> entry.getValue() != null)
399                         .forEach(entry -> {
400                             final int headcount = entry.getValue().stream()
401                                 .filter(Objects::nonNull)
402                                 .filter(summary -> summary.getMinistry() != null)
403                                 .filter(summary -> summary.getMinistry().equalsIgnoreCase(ministryName))
404                                 .mapToInt(GovernmentBodyAnnualSummary::getHeadCount)
405                                 .sum();
406                             addDataPoint(dataSeries, entry.getKey(), headcount);
407                         });
408                 }));
409 
410         addChartToLayout(layout, ANNUAL_HEADCOUNT_ALL_MINISTRIES, dataSeries, series);
411     }
