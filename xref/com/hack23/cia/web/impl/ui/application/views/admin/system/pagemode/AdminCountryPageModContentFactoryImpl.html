85  	@Override
86  	public Layout createContent(final String parameters, final MenuBar menuBar, final Panel panel) {
87  		final VerticalLayout content = createPanelContent();
88  		content.setSizeFull();
89  		content.addStyleName("admin-country-content");
90  
91  		final String pageId = getPageId(parameters);
92  		final int pageNr = getPageNr(parameters);
93  
94  		getMenuItemFactory().createMainPageMenuBar(menuBar);
95  
96  		createPageHeader(panel, content,
97  				"Admin Country Management",
98  				"Country Overview",
99  				"Manage and review country-specific data, including metrics and geopolitical information.");
100 
101 		final DataContainer<CountryElement, Long> dataContainer = getApplicationManager()
102 				.getDataContainer(CountryElement.class);
103 
104 		final List<CountryElement> pageOrderBy = dataContainer.getPageOrderBy(
105 				pageNr,
106 				DEFAULT_RESULTS_PER_PAGE,
107 				CountryElement_.countryName);
108 
109 		getPagingUtil().createPagingControls(
110 				content,
111 				NAME,
112 				pageId,
113 				dataContainer.getSize(),
114 				pageNr,
115 				DEFAULT_RESULTS_PER_PAGE);
116 
117 		getGridFactory()
118 				.createBasicBeanItemNestedPropertiesGrid(
119 						content,
120 						CountryElement.class,
121 						pageOrderBy,
122 						COUNTRY2,
123 						null,
124 						COLUMN_ORDER,
125 						HIDE_COLUMNS,
126 						LISTENER,
127 						null,
128 						null);
129 
130 		if (pageId != null && !pageId.isEmpty()) {
131 			final CountryElement country = dataContainer.load(Long.valueOf(pageId));
132 			if (country != null) {
133 
134 				// Create a card panel for country details
135 				final Panel cardPanel = new Panel();
136 				cardPanel.addStyleName("politician-overview-card");
137 				cardPanel.setWidth("100%");
138 				cardPanel.setHeightUndefined();
139 				Responsive.makeResponsive(cardPanel);
140 
141 				final VerticalLayout cardContent = new VerticalLayout();
142 				cardContent.setMargin(true);
143 				cardContent.setSpacing(true);
144 				cardContent.setWidth("100%");
145 				cardPanel.setContent(cardContent);
146 
147 				content.addComponent(cardPanel);
148 
149 				// Card Header
150 				final HorizontalLayout headerLayout = new HorizontalLayout();
151 				headerLayout.setSpacing(true);
152 				headerLayout.setWidth("100%");
153 				headerLayout.addStyleName("card-header-section");
154 
155 				final String titleText = "Country Details";
156 				final Label titleLabel = new Label(titleText, ContentMode.HTML);
157 				titleLabel.addStyleName("card-title");
158 				titleLabel.setWidthUndefined();
159 				headerLayout.addComponent(titleLabel);
160 
161 				cardContent.addComponent(headerLayout);
162 
163 				// Divider
164 				final Label divider = new Label("<hr/>", ContentMode.HTML);
165 				divider.addStyleName("card-divider");
166 				divider.setWidth("100%");
167 				cardContent.addComponent(divider);
168 
169 				// Attributes layout
170 				final VerticalLayout attributesLayout = new VerticalLayout();
171 				attributesLayout.setSpacing(true);
172 				attributesLayout.setWidth("100%");
173 				cardContent.addComponent(attributesLayout);
174 
175 				// Show fields if not null or empty
176 				addInfoRowIfNotNull(attributesLayout, "Country ID:", country.getId(), VaadinIcons.FLAG_O);
177 				addInfoRowIfNotNull(attributesLayout, "Name:", country.getCountryName(), VaadinIcons.GLOBE);
178 				addInfoRowIfNotNull(attributesLayout, "ISO2 Code:", country.getIso2Code(), VaadinIcons.CODE);
179 				addInfoRowIfNotNull(attributesLayout, "Capital:", country.getCapitalCity(), VaadinIcons.BUILDING);
180 				addInfoRowIfNotNull(attributesLayout, "Longitude:", country.getLongitude(), VaadinIcons.ARROWS_LONG_H);
181 				addInfoRowIfNotNull(attributesLayout, "Latitude:", country.getLatitude(), VaadinIcons.ARROWS_LONG_H);
182 			}
183 		}
184 
185 		getPageActionEventHelper().createPageEvent(
186 				ViewAction.VISIT_ADMIN_COUNTRY_VIEW,
187 				ApplicationEventGroup.ADMIN,
188 				NAME,
189 				null,
190 				pageId);
191 
192 		return content;
193 	}

195 	/**
196 	 * Adds an info row to the parent layout if value is not null or empty.
197 	 *
198 	 * @param parent the parent layout
199 	 * @param caption the field caption
200 	 * @param value   the field value
201 	 * @param icon    a VaadinIcons icon
202 	 */
203 	private void addInfoRowIfNotNull(final VerticalLayout parent, final String caption, final String value, final VaadinIcons icon) {
204 		if (value != null && !value.trim().isEmpty() && !"null".equalsIgnoreCase(value)) {
205 			parent.addComponent(createInfoRow(caption, value, icon));
206 		}
207 	}
208 
209 	/**
210 	 * Creates a simple info row (caption and value) with optional icon.
211 	 *
212 	 * @param caption the field caption
213 	 * @param value   the field value
214 	 * @param icon    a VaadinIcons icon
215 	 * @return a HorizontalLayout representing the info row
216 	 */
217 	private HorizontalLayout createInfoRow(final String caption, final String value, final VaadinIcons icon) {
218 		final HorizontalLayout layout = new HorizontalLayout();
219 		layout.setSpacing(true);
220 		layout.addStyleName("metric-label");
221 		layout.setWidthUndefined();
222 
223 		if (icon != null) {
224 			final Label iconLabel = new Label(icon.getHtml(), ContentMode.HTML);
225 			iconLabel.addStyleName("card-info-icon");
226 			layout.addComponent(iconLabel);
227 		}
228 
229 		final Label captionLabel = new Label(caption);
230 		captionLabel.addStyleName("card-info-caption");
231 
232 		final Label valueLabel = new Label(value);
233 		valueLabel.addStyleName("card-info-value");
234 
235 		layout.addComponents(captionLabel, valueLabel);
236 		return layout;
237 	}
