66      /**
67       * Load active politicians by person id.
68       *
69       * @return the map
70       */
71      @SuppressWarnings("unchecked")
72      public Map<String, List<ViewRiksdagenPolitician>> loadActivePoliticiansByPersonId() {
73          final DataContainer<ViewRiksdagenPolitician, String> politicianDataContainer = applicationManager
74                  .getDataContainer(ViewRiksdagenPolitician.class);
75          final List<ViewRiksdagenPolitician> activePoliticians = politicianDataContainer
76                  .findListByProperty(new Object[] { Boolean.TRUE }, ViewRiksdagenPolitician_.active);
77          return activePoliticians.stream().collect(Collectors.groupingBy(ViewRiksdagenPolitician::getPersonId));
78      }

132     /**
133      * Creates the baseball style card.
134      *
135      * @param govMember the gov member
136      * @param politician the politician
137      * @param ballotSummary the ballot summary
138      * @param governmentBodyByMinistry the government body by ministry
139      * @param reportByMinistry the report by ministry
140      * @param experienceSummary the experience summary
141      * @return the panel
142      */
143     public Panel createBaseballStyleCard(final ViewRiksdagenGovermentRoleMember govMember,
144             final ViewRiksdagenPolitician politician, final ViewRiksdagenPoliticianBallotSummary ballotSummary, final Map<String, List<GovernmentBodyAnnualSummary>> governmentBodyByMinistry,
145             final Map<String, List<GovernmentBodyAnnualOutcomeSummary>> reportByMinistry, final ViewRiksdagenPoliticianExperienceSummary experienceSummary) {
146 
147         final Panel cardPanel = createBaseCard();
148         final VerticalLayout cardContent = createCardContent();
149         cardPanel.setContent(cardContent);
150 
151         CardInfoRowUtil.createCardHeader(cardContent,govMember.getRoleCode() + " " + govMember.getFirstName() + " " + govMember.getLastName()
152             + " (" + govMember.getParty() + ")");
153         cardContent.addComponent(pageLinkFactory.createPoliticianPageLink(politician));
154 
155         final Link pageLink = new Link("Party " + politician.getParty(),
156                 new ExternalResource("#!" + UserViews.PARTY_VIEW_NAME + "/" + politician.getParty()));
157         pageLink.setId(ViewAction.VISIT_PARTY_VIEW.name() + "/" + politician.getParty());
158         pageLink.setIcon(VaadinIcons.GROUP);
159 
160         cardContent.addComponent(pageLink);
161 
162         final boolean isPartyLeader = PartyLeaderUtil.isPartyLeader(applicationManager, politician.getPersonId());
163         if (isPartyLeader) {
164             final ViewRiksdagenPartyRoleMember leaderRole = PartyLeaderUtil.getPartyLeaderRole(applicationManager, politician.getPersonId());
165             if (leaderRole != null) {
166                 final Label subHeader = new Label(
167                         "Partiledare (" + govMember.getParty() + ") since " + leaderRole.getFromDate());
168                 subHeader.addStyleName("card-subtitle");
169                 cardContent.addComponent(subHeader);
170             }
171         }
172 
173         // After creating the divider following the header/subtitle
174         // We create a vertical layout to hold Tenure and Experience on separate rows
175         final VerticalLayout statsContainer = CardInfoRowUtil.createStatsContainer();
176 
177         // Tenure Row
178         final HorizontalLayout tenureLayout = new HorizontalLayout();
179         tenureLayout.setSpacing(true);
180         tenureLayout.addStyleName("card-tenure");
181         final Label tenureIcon = new Label(VaadinIcons.CLOCK.getHtml(), ContentMode.HTML);
182         tenureIcon.setDescription("Tenure in days");
183 
184         final Label tenureLabel = new Label("Tenure:");
185         tenureLabel.addStyleName("card-tenure-text");
186 
187         final Label tenureValue = new Label(govMember.getTotalDaysServed() + " days");
188         tenureValue.addStyleName("card-tenure-value");
189 
190         tenureLayout.addComponents(tenureIcon, tenureLabel, tenureValue);
191         statsContainer.addComponent(tenureLayout);
192 
193         // Experience Row
194         final int govYears = (int) (politician.getTotalDaysServedGovernment() / 365);
195         final int partyYears = (int) (politician.getTotalDaysServedParty() / 365);
196         final int parliamentYears = (int) (politician.getTotalDaysServedParliament() / 365);
197         addExperienceRow(statsContainer, govYears, partyYears, parliamentYears);
198 
199         // Add the statsContainer to the cardContent
200         cardContent.addComponent(statsContainer);
201 
202         // Create grid for the four sections
203         final HorizontalLayout sectionsGrid = new HorizontalLayout();
204         sectionsGrid.setSpacing(true);
205         sectionsGrid.setWidth("100%");
206 
207         // Add the four main sections
208         final VerticalLayout politicalRoleLayout = CardInfoRowUtil.createSectionLayout("Political Role & Influence");
209         addPoliticalRoleMetrics(politicalRoleLayout, govMember, politician, ballotSummary, experienceSummary);
210         sectionsGrid.addComponent(politicalRoleLayout);
211 
212         final VerticalLayout performanceLayout = CardInfoRowUtil.createSectionLayout("Parliamentary Performance");
213         politicianLeaderboardUtil.addParliamentaryPerformanceMetrics(performanceLayout, politician, ballotSummary);
214         sectionsGrid.addComponent(performanceLayout);
215 
216         cardContent.addComponent(sectionsGrid);
217 
218         final HorizontalLayout sections2Grid = new HorizontalLayout();
219         sections2Grid.setSpacing(true);
220         sections2Grid.setWidth("100%");
221 
222         final VerticalLayout legislativeLayout = CardInfoRowUtil.createSectionLayout("Legislative Activity");
223         politicianLeaderboardUtil.addLegislativeMetrics(legislativeLayout, politician);
224         sections2Grid.addComponent(legislativeLayout);
225 
226         final VerticalLayout alignmentLayout = CardInfoRowUtil.createSectionLayout("Party Alignment");
227         politicianLeaderboardUtil.addPartyAlignmentMetrics(alignmentLayout, politician, ballotSummary);
228         sections2Grid.addComponent(alignmentLayout);
229         cardContent.addComponent(sections2Grid);
230 
231         politicianLeaderboardUtil.addMinistryRoleSummary(cardContent, govMember, governmentBodyByMinistry, reportByMinistry);
232 
233         return cardPanel;
234     }

236     /**
237      * Creates the leader card.
238      *
239      * @param leader the leader
240      * @param ballotSummary the ballot summary
241      * @param governmentBodyByMinistry the government body by ministry
242      * @param reportByMinistry the report by ministry
243      * @param experienceSummary the experience summary
244      * @return the panel
245      */
246     public Panel createLeaderCard(final ViewRiksdagenPolitician leader, final ViewRiksdagenPoliticianBallotSummary ballotSummary,
247             final Map<String, List<GovernmentBodyAnnualSummary>> governmentBodyByMinistry,
248             final Map<String, List<GovernmentBodyAnnualOutcomeSummary>> reportByMinistry, final ViewRiksdagenPoliticianExperienceSummary experienceSummary) {
249 
250         final Panel cardPanel = createBaseCard();
251         final VerticalLayout cardContent = createCardContent();
252         cardPanel.setContent(cardContent);
253 
254         CardInfoRowUtil.createCardHeader(cardContent,"Partiledare " + leader.getFirstName() + " " + leader.getLastName() + " ("
255                 + leader.getParty() + ")");
256 
257         // Politician detail link
258         cardContent.addComponent(pageLinkFactory.createPoliticianPageLink(leader));
259 
260         // Party link
261         final Link partyLink = new Link("Party " + leader.getParty(),
262                 new ExternalResource("#!" + UserViews.PARTY_VIEW_NAME + "/" + leader.getParty()));
263         partyLink.setIcon(VaadinIcons.GROUP);
264         cardContent.addComponent(partyLink);
265 
266         final boolean isPartyLeader = PartyLeaderUtil.isPartyLeader(applicationManager, leader.getPersonId());
267         if (isPartyLeader) {
268             final ViewRiksdagenPartyRoleMember leaderRole = PartyLeaderUtil.getPartyLeaderRole(applicationManager, leader.getPersonId());
269             if (leaderRole != null) {
270                 final Label subHeader = new Label("Partiledare (" + leader.getParty() + ") since " + leaderRole.getFromDate());
271                 subHeader.addStyleName("card-subtitle");
272                 cardContent.addComponent(subHeader);
273             }
274         }
275 
276         // Government or not
277         ViewRiksdagenGovermentRoleMember govMember = null;
278         if (leader.isActiveGovernment()) {
279             final Label govLabel = new Label("Currently in Government");
280             govLabel.addStyleName("card-subtitle");
281             cardContent.addComponent(govLabel);
282 
283             // Add ministry summary if we can identify their ministry
284             // The ministry detail is stored in the same structure as the ministry snippet:
285             // We need to find which ministry they belong to
286             // In the ministry snippet, "govMember.getDetail()" gives ministry detail key.
287             // Here we only have leader, not govMember. We must find a corresponding approach:
288 
289             // Let's assume we can identify the leader's ministry from active government roles data:
290             // we do similar approach: load active government role members and find the one matching this leader
291             final ViewRiksdagenPolitician pol = leader; // same as leader
292             govMember = findGovernmentRoleForLeader(pol);
293             if (govMember != null) {
294                 politicianLeaderboardUtil.addMinistryRoleSummary(cardContent, govMember, governmentBodyByMinistry, reportByMinistry);
295             }
296 
297         } else {
298             final Label nonGovLabel = new Label("Not in Government");
299             nonGovLabel.addStyleName("card-subtitle-nongov");
300             cardContent.addComponent(nonGovLabel);
301         }
302 
303         // Tenure and Experience rows
304         final VerticalLayout statsContainer = CardInfoRowUtil.createStatsContainer();
305 
306         // Tenure (assuming leader might have totalDaysServed property)
307         final Label tenureIcon = new Label(VaadinIcons.CLOCK.getHtml(), ContentMode.HTML);
308         tenureIcon.setDescription("Total Tenure");
309         final Label tenureLabel = new Label("Tenure:");
310         tenureLabel.addStyleName("card-tenure-text");
311         final Label tenureValue = new Label(leader.getTotalDaysServedParty() + " days");
312         tenureValue.addStyleName("card-tenure-value");
313         final HorizontalLayout tenureLayout = new HorizontalLayout(tenureIcon, tenureLabel, tenureValue);
314         tenureLayout.setSpacing(true);
315         tenureLayout.addStyleName("card-tenure");
316         statsContainer.addComponent(tenureLayout);
317 
318         // Experience
319         final int govYears = (int) (leader.getTotalDaysServedGovernment() / 365);
320         final int partyYears = (int) (leader.getTotalDaysServedParty() / 365);
321         final int parliamentYears = (int) (leader.getTotalDaysServedParliament() / 365);
322         addExperienceRow(statsContainer, govYears, partyYears, parliamentYears);
323 
324         cardContent.addComponent(statsContainer);
325 
326         // Create grid for the four sections
327         final HorizontalLayout sectionsGrid = new HorizontalLayout();
328         sectionsGrid.setSpacing(true);
329         sectionsGrid.setWidth("100%");
330 
331         // Add the four main sections
332         final VerticalLayout politicalRoleLayout = CardInfoRowUtil.createSectionLayout("Political Role & Influence");
333         addPoliticalRoleMetrics(politicalRoleLayout, PartyLeaderUtil.getPartyLeaderRole(applicationManager, leader.getPersonId()), govMember, leader, ballotSummary,experienceSummary);
334         sectionsGrid.addComponent(politicalRoleLayout);
335 
336         final VerticalLayout performanceLayout = CardInfoRowUtil.createSectionLayout("Parliamentary Performance");
337         politicianLeaderboardUtil.addParliamentaryPerformanceMetrics(performanceLayout, leader, ballotSummary);
338         sectionsGrid.addComponent(performanceLayout);
339 
340         cardContent.addComponent(sectionsGrid);
341 
342         final HorizontalLayout sections2Grid = new HorizontalLayout();
343         sections2Grid.setSpacing(true);
344         sections2Grid.setWidth("100%");
345 
346         final VerticalLayout legislativeLayout = CardInfoRowUtil.createSectionLayout("Legislative Activity");
347         politicianLeaderboardUtil.addLegislativeMetrics(legislativeLayout, leader);
348         sections2Grid.addComponent(legislativeLayout);
349 
350         final VerticalLayout alignmentLayout = CardInfoRowUtil.createSectionLayout("Party Alignment");
351         politicianLeaderboardUtil.addPartyAlignmentMetrics(alignmentLayout, leader, ballotSummary);
352         sections2Grid.addComponent(alignmentLayout);
353         cardContent.addComponent(sections2Grid);
354 
355         return cardPanel;
356     }

445     /**
446      * Find government role for leader.
447      *
448      * @param leader the leader
449      * @return the view riksdagen goverment role member
450      */
451     @SuppressWarnings("unchecked")
452     private ViewRiksdagenGovermentRoleMember findGovernmentRoleForLeader(final ViewRiksdagenPolitician leader) {
453         final DataContainer<ViewRiksdagenGovermentRoleMember, String> govermentRoleMemberDataContainer = applicationManager
454                 .getDataContainer(ViewRiksdagenGovermentRoleMember.class);
455         final List<ViewRiksdagenGovermentRoleMember> activeGovMembers = govermentRoleMemberDataContainer.findListByProperty(
456                 new Object[] { Boolean.TRUE }, ViewRiksdagenGovermentRoleMember_.active);
457 
458         return activeGovMembers.stream()
459                 .filter(govMember -> govMember.getPersonId().equals(leader.getPersonId()))
460                 .findFirst().orElse(null);
461     }
