19  package com.hack23.cia.web.impl.ui.application.views.common.paging;
20  
21  import org.springframework.beans.factory.annotation.Autowired;
22  import org.springframework.stereotype.Component;
23  
24  import com.hack23.cia.web.impl.ui.application.views.common.pagelinks.api.PageLinkFactory;
25  import com.hack23.cia.web.impl.ui.application.views.common.sizing.ContentRatio;
26  import com.vaadin.ui.AbstractOrderedLayout;
27  import com.vaadin.ui.HorizontalLayout;
28  import com.vaadin.ui.Label;
29  import com.vaadin.ui.Link;
30  
31  /**
32   * The Class PagingUtil.
33   */
34  @Component
35  public final class PagingUtilImpl implements PagingUtil {
36  
37  	/** The first page. */
38  	private static final String FIRST_PAGE = "first page";
39  
40  	/** The last page. */
41  	private static final String LAST_PAGE = "last page";
42  
43  	/** The limit for displaying start end links. */
44  	private static final int LIMIT_FOR_DISPLAYING_START_END_LINKS = 5;
45  
46  	/** The next page. */
47  	private static final String NEXT_PAGE = "next page";
48  
49  	/** The page header. */
50  	private static final String PAGE_HEADER = "Page: ";
51  
52  	/** The page one. */
53  	private static final int PAGE_ONE = 1;
54  
55  	/** The page separator. */
56  	private static final char PAGE_SEPARATOR = '/';
57  
58  	/** The pages total results. */
59  	private static final String PAGES_TOTAL_RESULTS = " pages. Total results:";
60  
61  	/** The previous page. */
62  	private static final String PREVIOUS_PAGE = "previous page";
63  
64  	/** The results per page. */
65  	private static final String RESULTS_PER_PAGE = " results per page:";
66  
67  	/** The show. */
68  	private static final String SHOW = " :: Show ";
69  
70  	/** The page link factory. */
71  	@Autowired
72  	private PageLinkFactory pageLinkFactory;
73  
74  	/**
75  	 * Adds the paging link.
76  	 *
77  	 * @param label          the label
78  	 * @param name           the name
79  	 * @param pageId         the page id
80  	 * @param maxPages       the max pages
81  	 * @param pagingControls the paging controls
82  	 */
83  	private void addPagingLink(final String label, final String name, final String pageId, final long maxPages, final HorizontalLayout pagingControls) {
84  		final Link previousPageLink = pageLinkFactory.createAdminPagingLink(label,name, pageId, String.valueOf(maxPages));
85  		pagingControls.addComponent(previousPageLink);
86  		pagingControls.setExpandRatio(previousPageLink, ContentRatio.SMALL);
87  	}
88  
89  	/**
90  	 * Creates the paging controls.
91  	 *
92  	 * @param content       the content
93  	 * @param name          the name
94  	 * @param pageId        the page id
95  	 * @param size          the size
96  	 * @param pageNr        the page nr
97  	 * @param resultPerPage the result per page
98  	 */
99  	@Override
100 	public void createPagingControls(final AbstractOrderedLayout content, final String name, final String pageId, final Long size, final int pageNr,
101 			final int resultPerPage) {
102 				final HorizontalLayout pagingControls = new HorizontalLayout();
103 				pagingControls.setSpacing(true);
104 				pagingControls.setMargin(true);
105 
106 				final long maxPages = (size +resultPerPage-1) / resultPerPage;
107 
108 				final StringBuilder stringBuilder = new StringBuilder();
109 				stringBuilder.append(PAGE_HEADER)
110 				.append(pageNr)
111 				.append(PAGE_SEPARATOR)
112 				.append(maxPages)
113 				.append(PAGES_TOTAL_RESULTS)
114 				.append(size)
115 				.append(RESULTS_PER_PAGE)
116 				.append(resultPerPage)
117 				.append(SHOW);
118 				final Label pageInfo = new Label(stringBuilder.toString());
119 				pagingControls.addComponent(pageInfo);
120 				pagingControls.setExpandRatio(pageInfo, ContentRatio.SMALL);
121 
122 
123 				if (pageNr > PAGE_ONE) {
124 					addPagingLink(PREVIOUS_PAGE,name, pageId, pageNr -1L,pagingControls);
125 				}
126 
127 				if (maxPages > PAGE_ONE && pageNr < maxPages) {
128 					addPagingLink(NEXT_PAGE,name, pageId, pageNr +1L,pagingControls);
129 				}
130 
131 				if (maxPages > LIMIT_FOR_DISPLAYING_START_END_LINKS && pageNr > PAGE_ONE) {
132 					addPagingLink(FIRST_PAGE,name, pageId, 1,pagingControls);
133 				}
134 
135 				if (maxPages > LIMIT_FOR_DISPLAYING_START_END_LINKS && pageNr < maxPages) {
136 					addPagingLink(LAST_PAGE,name, pageId, maxPages,pagingControls);
137 				}
138 
139 				content.addComponent(pagingControls);
140 				content.setExpandRatio(pagingControls, ContentRatio.SMALL2);
141 
142 			}
143 
144 }
