103 	@Override
104 	public void createCommitteeeMenuBar(final MenuBar menuBar, final String pageId) {
105 		initApplicationMenuBar(menuBar);
106 
107 		applicationMenuItemFactory.addRankingMenu(menuBar);
108 
109 		committeeRankingMenuItemFactory.createCommitteeRankingTopics(menuBar.addItem(COMMITTEE_RANKING_TEXT, VaadinIcons.GROUP, null));
110 
111 		final MenuItem committeeItem = menuBar.addItem("Committee "+ pageId, VaadinIcons.GROUP,null);
112 
113 
114 		committeeItem.addItem(OVERVIEW_TEXT, VaadinIcons.GROUP,
115 				new PageModeMenuCommand(UserViews.COMMITTEE_VIEW_NAME, PageMode.OVERVIEW, pageId));
116 
117 		final MenuItem rolesItem = committeeItem.addItem(ROLES_TEXT, VaadinIcons.GROUP, null);
118 
119 		rolesItem.addItem(CURRENT_MEMBERS_TEXT, VaadinIcons.GROUP, new PageModeMenuCommand(UserViews.COMMITTEE_VIEW_NAME,
120 				CommitteePageMode.CURRENT_MEMBERS.toString(), pageId));
121 
122 		rolesItem.addItem(MEMBER_HISTORY_TEXT, VaadinIcons.GROUP, new PageModeMenuCommand(UserViews.COMMITTEE_VIEW_NAME,
123 				CommitteePageMode.MEMBERHISTORY.toString(), pageId));
124 
125 		rolesItem.addItem(ROLE_GHANT_TEXT, VaadinIcons.GROUP,
126 				new PageModeMenuCommand(UserViews.COMMITTEE_VIEW_NAME, CommitteePageMode.ROLEGHANT.toString(), pageId));
127 
128 		final MenuItem documentItem = committeeItem.addItem(DOCUMENTS_TEXT, VaadinIcons.GROUP, null);
129 
130 		documentItem.addItem(DOCUMENT_ACTIVITY_TEXT, VaadinIcons.GROUP, new PageModeMenuCommand(UserViews.COMMITTEE_VIEW_NAME,
131 				CommitteePageMode.DOCUMENTACTIVITY.toString(), pageId));
132 
133 		documentItem.addItem(DOCUMENT_HISTORY_TEXT, VaadinIcons.GROUP, new PageModeMenuCommand(UserViews.COMMITTEE_VIEW_NAME,
134 				CommitteePageMode.DOCUMENT_HISTORY.toString(), pageId));
135 
136 		final MenuItem ballotItem = committeeItem.addItem(BALLOTS_TEXT, VaadinIcons.GROUP, null);
137 
138 		ballotItem.addItem(BALLOT_DECISION_SUMMARY_TEXT, VaadinIcons.GROUP, new PageModeMenuCommand(UserViews.COMMITTEE_VIEW_NAME,
139 				CommitteePageMode.BALLOTDECISIONSUMMARY.toString(), pageId));
140 
141 		ballotItem.addItem(DECISION_SUMMARY_TEXT, VaadinIcons.GROUP, new PageModeMenuCommand(UserViews.COMMITTEE_VIEW_NAME,
142 				CommitteePageMode.DECISIONSUMMARY.toString(), pageId));
143 
144 		ballotItem.addItem(DECISION_TYPE_DAILY_SUMMARY_TEXT, VaadinIcons.GROUP, new PageModeMenuCommand(
145 				UserViews.COMMITTEE_VIEW_NAME, CommitteePageMode.DECISIONTYPEDAILYSUMMARY.toString(), pageId));
146 
147 		ballotItem.addItem("Decision flow", VaadinIcons.GROUP, new PageModeMenuCommand(
148 				UserViews.COMMITTEE_VIEW_NAME, PageMode.CHARTS+"/"+ ChartIndicators.DECISION_FLOW_CHART, pageId));
149 
150 
151 		committeeItem.addItem(PAGE_VISIT_HISTORY_TEXT, VaadinIcons.GROUP,
152 				new PageModeMenuCommand(UserViews.COMMITTEE_VIEW_NAME, PageMode.PAGEVISITHISTORY, pageId));
153 
154 	}
155 
156 	@Override
157 	public void createOverviewPage(final VerticalLayout panelContent, final String pageId) {
158 		final ResponsiveRow grid = RowUtil.createGridLayout(panelContent);
159 
160 		createButtonLink(grid,CURRENT_MEMBERS_TEXT, VaadinIcons.GROUP, new PageModeMenuCommand(UserViews.COMMITTEE_VIEW_NAME,
161 				CommitteePageMode.CURRENT_MEMBERS.toString(), pageId), "Current roles and days served");
162 
163 		createButtonLink(grid,MEMBER_HISTORY_TEXT, VaadinIcons.GROUP, new PageModeMenuCommand(UserViews.COMMITTEE_VIEW_NAME,
164 				CommitteePageMode.MEMBERHISTORY.toString(), pageId), "History of all roles and days served");
165 
166 		createButtonLink(grid,ROLE_GHANT_TEXT, VaadinIcons.GROUP,
167 				new PageModeMenuCommand(UserViews.COMMITTEE_VIEW_NAME, CommitteePageMode.ROLEGHANT.toString(), pageId), "Gantt chart of all roles over time");
168 
169 
170 		createButtonLink(grid,DOCUMENT_ACTIVITY_TEXT, VaadinIcons.GROUP, new PageModeMenuCommand(UserViews.COMMITTEE_VIEW_NAME,
171 				CommitteePageMode.DOCUMENTACTIVITY.toString(), pageId), "Chart of document activity by document type.");
172 
173 		createButtonLink(grid,DOCUMENT_HISTORY_TEXT, VaadinIcons.GROUP, new PageModeMenuCommand(UserViews.COMMITTEE_VIEW_NAME,
174 				CommitteePageMode.DOCUMENT_HISTORY.toString(), pageId), "Document history list");
175 
176 
177 		createButtonLink(grid,BALLOT_DECISION_SUMMARY_TEXT, VaadinIcons.GROUP, new PageModeMenuCommand(UserViews.COMMITTEE_VIEW_NAME,
178 				CommitteePageMode.BALLOTDECISIONSUMMARY.toString(), pageId), "Summary of all ballot decisions");
179 
180 		createButtonLink(grid,DECISION_SUMMARY_TEXT, VaadinIcons.GROUP, new PageModeMenuCommand(UserViews.COMMITTEE_VIEW_NAME,
181 				CommitteePageMode.DECISIONSUMMARY.toString(), pageId), "Summary of all ballots");
182 
183 		createButtonLink(grid,DECISION_TYPE_DAILY_SUMMARY_TEXT, VaadinIcons.GROUP, new PageModeMenuCommand(
184 				UserViews.COMMITTEE_VIEW_NAME, CommitteePageMode.DECISIONTYPEDAILYSUMMARY.toString(), pageId), "Chart over decisions by decisions type, daily summary");
185 
186 		createButtonLink(grid,"Decision flow", VaadinIcons.GROUP, new PageModeMenuCommand(
187 				UserViews.COMMITTEE_VIEW_NAME, PageMode.CHARTS+"/"+ ChartIndicators.DECISION_FLOW_CHART, pageId),"desc");
188 
189 
190 		createButtonLink(grid,PAGE_VISIT_HISTORY_TEXT, VaadinIcons.GROUP,
191 				new PageModeMenuCommand(UserViews.COMMITTEE_VIEW_NAME, PageMode.PAGEVISITHISTORY, pageId), "View history of page visit for this page.");
192 
193 
194 	}
