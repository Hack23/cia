48  /**
49   * Coalition prediction page mod content factory implementation.
50   * 
51   * Provides UI for coalition formation predictions, party alignment analysis,
52   * and coalition stability assessment based on historical voting patterns.
53   */
54  @Component
55  public final class CoalitionPredictionPageModContentFactoryImpl extends AbstractParliamentPageModContentFactoryImpl {
56  
57  	@Autowired
58  	private CoalitionPredictionService coalitionPredictionService;
59  
60  	/**
61  	 * Instantiates a new coalition prediction page mod content factory impl.
62  	 */
63  	public CoalitionPredictionPageModContentFactoryImpl() {
64  		super();
65  	}
66  
67  	@Secured({ "ROLE_ANONYMOUS", "ROLE_USER", "ROLE_ADMIN" })
68  	@Override
69  	public Layout createContent(final String parameters, final MenuBar menuBar, final Panel panel) {
70  		final VerticalLayout panelContent = createPanelContent();
71  
72  		setupMenuAndHeader(menuBar, panel, panelContent);
73  		final String selectedYear = extractSelectedYear(parameters, ParliamentChartConstants.DEFAULT_YEAR);
74  
75  		addCoalitionScenarios(panelContent, selectedYear);
76  		addPartyAlignmentMatrix(panelContent, selectedYear);
77  
78  		recordPageVisit(parameters, selectedYear);
79  
80  		return panelContent;
81  	}
82  
83  	/**
84  	 * Setup menu and header.
85  	 *
86  	 * @param menuBar the menu bar
87  	 * @param panel the panel
88  	 * @param panelContent the panel content
89  	 */
90  	private void setupMenuAndHeader(final MenuBar menuBar, final Panel panel, final VerticalLayout panelContent) {
91  		getParliamentMenuItemFactory().createParliamentTopicMenu(menuBar);
92  		CardInfoRowUtil.createPageHeader(panel, panelContent,
93  			ParliamentPageTitleConstants.COALITION_PREDICTION_TITLE,
94  			ParliamentPageTitleConstants.COALITION_PREDICTION_SUBTITLE,
95  			ParliamentPageTitleConstants.COALITION_PREDICTION_DESC);
96  	}

98  	/**
99  	 * Adds the coalition scenarios section.
100 	 *
101 	 * @param panelContent the panel content
102 	 * @param selectedYear the selected year
103 	 */
104 	private void addCoalitionScenarios(final VerticalLayout panelContent, final String selectedYear) {
105 		final List<CoalitionScenario> scenarios = coalitionPredictionService.predictCoalitions(selectedYear);
106 
107 		final VerticalLayout scenariosLayout = new VerticalLayout();
108 		scenariosLayout.setSpacing(true);
109 		scenariosLayout.setMargin(true);
110 		scenariosLayout.setWidth(100, Unit.PERCENTAGE);
111 
112 		final Label header = new Label("Coalition Formation Scenarios (Ranked by Probability)");
113 		header.addStyleName(ValoTheme.LABEL_H2);
114 		scenariosLayout.addComponent(header);
115 
116 		if (scenarios == null || scenarios.isEmpty()) {
117 			final Label noScenariosLabel = new Label("No viable coalition scenarios found for the selected year.");
118 			noScenariosLabel.addStyleName(ValoTheme.LABEL_LIGHT);
119 			scenariosLayout.addComponent(noScenariosLabel);
120 		} else {
121 			int rank = 1;
122 			for (final CoalitionScenario scenario : scenarios) {
123 				scenariosLayout.addComponent(createScenarioRow(rank++, scenario));
124 			}
125 		}
126 
127 		panelContent.addComponent(scenariosLayout);
128 		panelContent.setExpandRatio(scenariosLayout, ContentRatio.LARGE);
129 	}
130 
131 	/**
132 	 * Creates a single coalition scenario row.
133 	 *
134 	 * @param rank the rank
135 	 * @param scenario the scenario
136 	 * @return the horizontal layout
137 	 */
138 	private HorizontalLayout createScenarioRow(final int rank, final CoalitionScenario scenario) {
139 		final HorizontalLayout row = new HorizontalLayout();
140 		row.setSpacing(true);
141 		row.setWidth(100, Unit.PERCENTAGE);
142 
143 		// Rank badge
144 		final Label rankLabel = new Label("#" + rank);
145 		rankLabel.addStyleName(ValoTheme.LABEL_H3);
146 		rankLabel.setWidth(50, Unit.PIXELS);
147 		row.addComponent(rankLabel);
148 
149 		// Coalition parties
150 		final Label partiesLabel = new Label(String.join(" + ", scenario.getParties()));
151 		partiesLabel.addStyleName(ValoTheme.LABEL_BOLD);
152 		partiesLabel.setWidth(200, Unit.PIXELS);
153 		row.addComponent(partiesLabel);
154 
155 		// Probability bar
156 		final ProgressBar probabilityBar = new ProgressBar();
157 		// Vaadin ProgressBar API requires float value; casting from double results in minimal precision loss.
158 		// This is acceptable for UI display purposes as the bar visualization does not require high precision.
159 		probabilityBar.setValue((float) scenario.getProbability());
160 		probabilityBar.setCaption(String.format(Locale.ENGLISH, "%.1f%% probability", scenario.getProbability() * 100));
161 		probabilityBar.setWidth(200, Unit.PIXELS);
162 		row.addComponent(probabilityBar);
163 
164 		// Stability index
165 		final Label stabilityLabel = new Label(
166 			String.format(Locale.ENGLISH, "Stability: %d/100", scenario.getStabilityIndex()));
167 		stabilityLabel.addStyleName(
168 			scenario.getStabilityIndex() > 70 ? ValoTheme.LABEL_SUCCESS : ValoTheme.LABEL_FAILURE);
169 		stabilityLabel.setWidth(150, Unit.PIXELS);
170 		row.addComponent(stabilityLabel);
171 
172 		// Seats
173 		final Label seatsLabel = new Label(String.format(Locale.ENGLISH, "%d seats", scenario.getTotalSeats()));
174 		seatsLabel.setWidth(100, Unit.PIXELS);
175 		row.addComponent(seatsLabel);
176 
177 		// Bloc relationship
178 		final Label blocLabel = new Label(scenario.getBlocRelationship());
179 		blocLabel.addStyleName(ValoTheme.LABEL_TINY);
180 		blocLabel.setWidth(150, Unit.PIXELS);
181 		row.addComponent(blocLabel);
182 
183 		return row;
184 	}

186 	/**
187 	 * Adds the party alignment matrix section.
188 	 *
189 	 * @param panelContent the panel content
190 	 * @param selectedYear the selected year
191 	 */
192 	private void addPartyAlignmentMatrix(final VerticalLayout panelContent, final String selectedYear) {
193 		final Map<String, Map<String, Double>> alignmentMatrix = 
194 			coalitionPredictionService.getAlignmentMatrix(selectedYear);
195 
196 		final VerticalLayout matrixLayout = new VerticalLayout();
197 		matrixLayout.setSpacing(true);
198 		matrixLayout.setMargin(true);
199 		matrixLayout.setWidth(100, Unit.PERCENTAGE);
200 
201 		final Label header = new Label("Party Alignment Matrix (Voting Pattern Compatibility)");
202 		header.addStyleName(ValoTheme.LABEL_H2);
203 		matrixLayout.addComponent(header);
204 
205 		final Label description = new Label(
206 			"Alignment rates show how often parties vote the same way. Higher rates indicate coalition compatibility.");
207 		description.addStyleName(ValoTheme.LABEL_SMALL);
208 		matrixLayout.addComponent(description);
209 
210 		// Create a simple text-based matrix
211 		for (final Map.Entry<String, Map<String, Double>> entry : alignmentMatrix.entrySet()) {
212 			final String party = entry.getKey();
213 			final Map<String, Double> alignments = entry.getValue();
214 
215 			final HorizontalLayout partyRow = new HorizontalLayout();
216 			partyRow.setSpacing(true);
217 
218 			final Label partyLabel = new Label(party);
219 			partyLabel.addStyleName(ValoTheme.LABEL_BOLD);
220 			partyLabel.setWidth(50, Unit.PIXELS);
221 			partyRow.addComponent(partyLabel);
222 
223 			for (final Map.Entry<String, Double> alignment : alignments.entrySet()) {
224 				final Label alignmentLabel = new Label(
225 					String.format(Locale.ENGLISH, "%s: %.1f%%", alignment.getKey(), alignment.getValue() * 100));
226 				alignmentLabel.setWidth(120, Unit.PIXELS);
227 				partyRow.addComponent(alignmentLabel);
228 			}
229 
230 			matrixLayout.addComponent(partyRow);
231 		}
232 
233 		panelContent.addComponent(matrixLayout);
234 		panelContent.setExpandRatio(matrixLayout, ContentRatio.SMALL_GRID);
235 	}

248 			parameters,
249 			selectedYear
250 		);
251 	}
