38  /**
39   * The Class DecisionDataFactoryImpl.
40   */
41  @Service
42  @Transactional(propagation = Propagation.REQUIRED, readOnly = true)
43  public final class DecisionDataFactoryImpl implements DecisionDataFactory {
44  
45      /** Common text constants for doc type. */
46      private static final String PROP = "Proposition";
47  
48      /** The Constant MOTION. */
49      private static final String MOTION = "Motion";
50  
51      /** Known min and max length constraints from original code. */
52      private static final int CHAMBER_MIN_LENGTH = "avslag".length(); // 6
53  
54      /** The Constant CHAMBER_MAX_LENGTH. */
55      private static final int CHAMBER_MAX_LENGTH = "återförvisning till utskottet".length(); // 29
56  
57      /**
58       * Pattern for cleaning up the 'chamber' string. Aggregates multiple .replace() calls
59       * into a single regex. This pattern will remove or replace these tokens:
60       *  (UTSKOTTET), ( or ), UTBSKOTTET, UBTSKOTTET, UTKOTTET
61       *
62       * We'll do it case-insensitively, but remember we do an uppercase pass anyway.
63       */
64      private static final Pattern CLEANUP_PATTERN = Pattern.compile(
65          "(\\(UTSKOTTET\\))|(\\()|(\\))|(UTBSKOTTET)|(UBTSKOTTET)|(UTKOTTET)",
66          Pattern.CASE_INSENSITIVE
67      );
68  
69      /** The application manager. */
70      @Autowired
71      private ApplicationManager applicationManager;
72  
73      /**
74       * Instantiates a new decision data factory impl.
75       */
76      public DecisionDataFactoryImpl() {
77          super();
78      }
79  
80      /**
81       * Creates the committee summary.
82       *
83       * @param processedIn the processed in
84       * @return the list
85       */
86      @Override
87      public List<ProposalCommitteeeSummary> createCommitteeSummary(final String processedIn) {
88          final List<ProposalCommitteeeSummary> summary = new ArrayList<>();
89  
90          // NOTE: Still loads ALL DocumentStatusContainer rows
91          final DataContainer<DocumentStatusContainer, Long> dataContainer =
92                  applicationManager.getDataContainer(DocumentStatusContainer.class);
93  
94          // For each doc, potentially add a summary
95          for (final DocumentStatusContainer document : dataContainer.getAll()) {
96              addProposalCommitteeeSummary(processedIn, summary, document);
97          }
98          return summary;
99      }
100 
101     /**
102      * Check and build a summary if the document qualifies based on 'processedIn' and
103      * the proposal/chamber constraints.
104      *
105      * @param processedIn the processed in
106      * @param summary the summary
107      * @param document the document
108      */
109     private static void addProposalCommitteeeSummary(final String processedIn,
110             final List<ProposalCommitteeeSummary> summary, final DocumentStatusContainer document) {
111 
112         // Null checks on 'documentProposal' and 'proposal'
113         if (document.getDocumentProposal() == null ||
114             document.getDocumentProposal().getProposal() == null) {
115             return;
116         }
117 
118         final DocumentProposalData proposal = document.getDocumentProposal().getProposal();
119         final String chamber = proposal.getChamber();
120         // Are we processed in the correct place?
121         if ((chamber == null) || proposal.getProcessedIn() == null || proposal.getProcessedIn().isEmpty()) {
122             return;
123         }
124         // Do we have a non-empty 'committee'?
125         // Does 'processedIn' match?
126         if (proposal.getCommittee() == null || proposal.getCommittee().isEmpty() || !proposal.getProcessedIn().contains(processedIn)) {
127             return;
128         }
129         // Is the chamber length within the specified min/max?
130         final int length = chamber.length();
131         if (length < CHAMBER_MIN_LENGTH || length > CHAMBER_MAX_LENGTH) {
132             return;
133         }
134 
135         // If all checks pass, add the summary
136         summary.add(new ProposalCommitteeeSummary(
137             getCommitteeShortName(proposal),
138             getDocumentName(document),
139             cleanupDecision(chamber),
140             document.getDocument().getHangarId(),
141             proposal.getWording(),
142             proposal.getWording2(),
143             proposal.getDecisionType()
144         ));
145     }
146 
147     /**
148      * Cleanup the 'chamber' string by uppercasing, removing tokens via regex,
149      * and normalizing "UTKOTTET" to "UTSKOTTET" if it appears alone.
150      *
151      * @param chamber the chamber
152      * @return the string
153      */
154     private static String cleanupDecision(final String chamber) {
155         // Convert to uppercase English
156         String upper = chamber.toUpperCase(Locale.ENGLISH);
157 
158         // Remove or replace known substrings with the pattern
159         upper = CLEANUP_PATTERN.matcher(upper).replaceAll("");
160 
161         // If we find "UTKOTTET" alone, replace it with "UTSKOTTET"
162         // (We already removed parentheses above).
163         if (upper.contains("UTKOTTET")) {
164             upper = upper.replace("UTKOTTET", "UTSKOTTET");
165         }
166         return upper.trim();
167     }
168 
169     /**
170      * Extract the committee short name from 'processedIn' by removing digits
171      * and certain punctuation, converting to uppercase, and truncating if comma found.
172      *
173      * @param proposal the proposal
174      * @return the committee short name
175      */
176     private static String getCommitteeShortName(final DocumentProposalData proposal) {
177         // e.g. "UU12" => "UU" then uppercase => "UU"
178         final String upperCase = proposal.getProcessedIn()
179                                    .replaceAll("\\d", "")
180                                    .replace("/:", "")
181                                    .toUpperCase(Locale.ENGLISH);
182         // If there's a comma, only take up to that comma
183         final int commaIndex = upperCase.indexOf(',');
184         return (commaIndex >= 0) ? upperCase.substring(0, commaIndex) : upperCase;
185     }
186 
187     /**
188      * Return a human-readable doc name: "Proposition" if docType="prop",
189      * or doc.getSubType() if subType is long enough, else "Motion".
190      *
191      * @param document the document
192      * @return the document name
193      */
194     private static String getDocumentName(final DocumentStatusContainer document) {
195         if ("prop".equalsIgnoreCase(document.getDocument().getDocumentType())) {
196             return PROP;
197         } else if (document.getDocument().getSubType() != null
198                    && document.getDocument().getSubType().length() > MOTION.length()) {
199             return document.getDocument().getSubType();
200         } else {
201             return MOTION;
202         }
203     }
