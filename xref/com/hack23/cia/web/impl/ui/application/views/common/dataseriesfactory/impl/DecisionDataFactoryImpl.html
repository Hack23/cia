55  	/**
56  	 * Adds the proposal committeee summary.
57  	 *
58  	 * @param processedIn
59  	 *            the processed in
60  	 * @param summary
61  	 *            the summary
62  	 * @param document
63  	 *            the document
64  	 */
65  	private static void addProposalCommitteeeSummary(final String processedIn, final List<ProposalCommitteeeSummary> summary,
66  			final DocumentStatusContainer document) {
67  		if (document.getDocumentProposal() != null && document.getDocumentProposal().getProposal() != null) {
68  
69  			final DocumentProposalData proposal = document.getDocumentProposal().getProposal();
70  
71  			if (proposal.getProcessedIn() != null && !proposal.getProcessedIn().isEmpty()
72  					&& proposal.getCommittee() != null && !proposal.getCommittee().isEmpty()
73  					&& proposal.getProcessedIn().contains(processedIn) && proposal.getChamber().length() <= "Ã¥terfÃ¶rvisning till utskottet".length() && proposal.getChamber().length() >= "avslag".length()) {
74  				
75  				summary.add(new ProposalCommitteeeSummary(getCommittteeShortName(proposal), getDocumentName(document) , cleanupDecision(proposal.getChamber()) , document.getDocument().getHangarId()));
76  
77  			}
78  		}
79  	}
80  
81  	/**
82  	 * Cleanup decision.
83  	 *
84  	 * @param chamber
85  	 *            the chamber
86  	 * @return the string
87  	 */
88  	private static String cleanupDecision(final String chamber) {
89  		return chamber.toUpperCase(Locale.ENGLISH).replace("(UTSKOTTET)", "").replace("UTKOTTET", "UTSKOTTET").replace("UTBSKOTTET", "UTSKOTTET").replace("UBTSKOTTET", "UTSKOTTET").replace("(", "").replace(")","");
90  	}
91  
92  	/**
93  	 * Gets the committtee short name.
94  	 *
95  	 * @param proposal
96  	 *            the proposal
97  	 * @return the committtee short name
98  	 */
99  	private static String getCommittteeShortName(final DocumentProposalData proposal) {
100 		final String upperCase = proposal.getProcessedIn().replaceAll("\\d","").replace("/:","").toUpperCase(Locale.ENGLISH);
101 				
102 		if (upperCase.contains(",")) {
103 			return upperCase.substring(0, upperCase.indexOf(','));
104 		} else {
105 			return upperCase;
106 		}
107 	}
108 
109 	/**
110 	 * Gets the document name.
111 	 *
112 	 * @param document
113 	 *            the document
114 	 * @return the document name
115 	 */
116 	private static String getDocumentName(final DocumentStatusContainer document) {
117 		if ("prop".equalsIgnoreCase(document.getDocument().getDocumentType())) {
118 			return "Proposition";
119 		} else if (document.getDocument().getSubType() != null && document.getDocument().getSubType().length() > "motion".length()) {
120 			return document.getDocument().getSubType();
121 		} else {
122 			return "Motion";
123 		}
124 		
125 	}
126 
127 	@Override
128 	public List<ProposalCommitteeeSummary> createCommitteeSummary(final String processedIn) {
129 		final List<ProposalCommitteeeSummary> summary = new ArrayList<>();
130 
131 		final DataContainer<DocumentStatusContainer, Long> dataContainer = applicationManager
132 				.getDataContainer(DocumentStatusContainer.class);
133 
134 		for (final DocumentStatusContainer document : dataContainer.getAll()) {
135 
136 			addProposalCommitteeeSummary(processedIn, summary, document);
137 		}
138 		return summary;
139 	}
