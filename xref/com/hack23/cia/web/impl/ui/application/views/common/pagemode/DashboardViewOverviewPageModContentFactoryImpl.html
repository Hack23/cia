531 	/**
532 	 * Creates the head count chart.
533 	 *
534 	 * @param panelContent the panel content
535 	 * @param orgId the org id
536 	 */
537 	private void createHeadCountCard(final Layout panelContent, final String orgId) {
538 		
539 
540 		final Map<String, List<GovernmentBodyAnnualSummary>> map = esvApi.getData().get(2022).stream().collect(Collectors.groupingBy(GovernmentBodyAnnualSummary::getOrgNumber));
541 		final List<GovernmentBodyAnnualSummary> list = map.get(getPageId(orgId));
542 		
543 		if (list != null && !list.isEmpty()) {
544 			final GovernmentBodyAnnualSummary findName = list.get(0);
545 			
546 			if (findName != null) {
547 				Map<Integer,GovernmentBodyAnnualSummary> dataPerGovernmentBody = esvApi.getDataPerGovernmentBody(findName.getName());
548 				Set<Integer> keySet = dataPerGovernmentBody.keySet();
549 				GovernmentBodyAnnualSummary governmentBodyAnnualSummary = dataPerGovernmentBody.get(Collections.max(keySet));
