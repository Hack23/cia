255 	/**
256 	 * Creates the admin menu.
257 	 *
258 	 * @param mainViewItem
259 	 *            the main view item
260 	 */
261 	private static void createAdminMenu(final MenuItem mainViewItem) {
262 		if (UserContextUtil.allowRoleInSecurityContext(ROLE_ADMIN)) {
263 			final MenuItem adminMenuItem = mainViewItem.addItem(ADMIN_TEXT, VaadinIcons.SERVER, null);
264 
265 			final MenuItem configurationMenuItem = adminMenuItem.addItem(CONFIGURATION, VaadinIcons.TOOLS, null);
266 			configurationMenuItem.addItem(APPLICATION_CONFIGURATION,VaadinIcons.TOOLS, COMMAND_APPLICATION_CONFIGURATION);
267 
268 			configurationMenuItem.addItem(AGENCY,VaadinIcons.SERVER, COMMAND_AGENCY);
269 			configurationMenuItem.addItem(PORTAL, VaadinIcons.SITEMAP,COMMAND_PORTAL);
270 			configurationMenuItem.addItem(COUNTRY,VaadinIcons.FLAG, COMMAND_COUNTRY);
271 			configurationMenuItem.addItem(LANGUAGE,VaadinIcons.ACCESSIBILITY, COMMAND_LANGUAGE);
272 
273 			final MenuItem managementMenuItem = adminMenuItem.addItem(MANAGEMENT, VaadinIcons.SERVER, null);
274 
275 			managementMenuItem.addItem(AGENT_OPERATIONS_TEXT,VaadinIcons.BRIEFCASE, COMMAND_AGENT_OPERATION);
276 
277 			managementMenuItem.addItem(DATA_SUMMARY_TEXT,VaadinIcons.DATABASE, COMMAND_DATASUMMARY);
278 			managementMenuItem.addItem(DATA_AUTHOR_SUMMARY,VaadinIcons.DATABASE, COMMAND_AUTHOR_DATASUMMARY);
279 
280 
281 			managementMenuItem.addItem(EMAIL,VaadinIcons.MAILBOX, COMMAND_EMAIL);
282 
283 
284 			managementMenuItem.addItem(SYSTEM_PERFORMANCE,VaadinIcons.DASHBOARD, COMMAND_MONITORING);
285 
286 			final MenuItem userActivityMenuItem = adminMenuItem.addItem(USER_ACTIVITY, VaadinIcons.DATABASE, null);
287 			userActivityMenuItem.addItem(APPLICATION_SESSION,VaadinIcons.LAPTOP, COMMAND_APPLICATION_SESSION);
288 			userActivityMenuItem.addItem(APPLICATION_SESSION_CHARTS,VaadinIcons.LAPTOP, COMMAND_APPLICATION_SESSION_CHARTS);
289 			userActivityMenuItem.addItem(APPLICATION_EVENT,VaadinIcons.ARROWS, COMMAND_APPLICATION_EVENTS);
290 			userActivityMenuItem.addItem(APPLICATION_EVENT_CHARTS,VaadinIcons.ARROWS, COMMAND_APPLICATION_EVENTS_CHARTS);
291 
292 			userActivityMenuItem.addItem(USERACCOUNT,VaadinIcons.GROUP, COMMAND_USERACCOUNT);
293 
294 		}

297 	/**
298 	 * Creates the button link.
299 	 *
300 	 * @param row
301 	 *            the panel content
302 	 * @param linkText
303 	 *            the link text
304 	 * @param icon
305 	 *            the icon
306 	 * @param command
307 	 *            the command
308 	 * @param description
309 	 *            the description
310 	 */
311 	protected static final void createButtonLink(final ResponsiveRow row,final String linkText,final Resource icon, final ClickListener command, final String description) {
312 		final CssLayout layout = new CssLayout();
313 		layout.addStyleName("v-layout-content-overview-panel-level2");
314 		Responsive.makeResponsive(layout);
315 		layout.setSizeUndefined();
316 
317 
318 		final Button button = new Button(linkText);
319 		Responsive.makeResponsive(button);
320 		button.setStyleName(LINK_STYLE_NAME);
321 		button.addStyleName("title");
322 		button.addClickListener(command);
323 		button.setIcon(icon);
324 		button.setWidth(100, Unit.PERCENTAGE);
325 
326 
327 		layout.addComponent(button);
328 
329 		final Label descriptionLabel = new Label(description);
330 		descriptionLabel.addStyleName("itembox");
331 		Responsive.makeResponsive(descriptionLabel);
332 		descriptionLabel.setWidth(100, Unit.PERCENTAGE);
333 		layout.addComponent(descriptionLabel);
334 
335 		row.addColumn().withDisplayRules(DISPLAY_SIZE_XS_DEVICE,DISPLAYS_SIZE_XM_DEVICE,DISPLAY_SIZE_MD_DEVICE,DISPLAY_SIZE_LG_DEVICE).withComponent(layout);
336 	}

338 	/**
339 	 * Inits the application menu bar.
340 	 *
341 	 * @param menuBar
342 	 *            the menu bar
343 	 */
344 	protected static final void initApplicationMenuBar(final MenuBar menuBar) {
345 		menuBar.removeItems();
346 		menuBar.setWidth(MENU_BAR_WIDTH);
347 		menuBar.setStyleName(HEADER_STYLE_NAME);
348 		final MenuItem mainViewItem = menuBar.addItem(APPLICATION, VaadinIcons.SERVER, null);
349 
350 		mainViewItem.addItem(START_TEXT, VaadinIcons.HOME, COMMAND_MAINVIEW_OVERVIEW);
351 
352 		mainViewItem.addItem(DASHBOARD, VaadinIcons.DASHBOARD, COMMAND_DASHBOARDVIEW_OVERVIEW);
353 
354 		final MenuItem mainItem = mainViewItem.addItem(MAIN, VaadinIcons.STAR, null);
355 
356 		mainItem.addItem(PAGE_VISIT_HISTORY_TEXT, VaadinIcons.CHART, COMMAND_MAINVIEW_PAGEVISITHISTORY);
357 
358 
359 		if (UserContextUtil.allowRoleInSecurityContext(ROLE_ADMIN) || UserContextUtil.allowRoleInSecurityContext(ROLE_USER)) {
360 			mainViewItem.addItem(USERHOME, VaadinIcons.USER,COMMAND_USERHOME);
361 			createAdminMenu(mainViewItem);
362 			mainViewItem.addItem(LOGOUT, VaadinIcons.SIGN_OUT, COMMAND_LOGOUT);
363 		} else {
364 			mainViewItem.addItem(LOGIN, VaadinIcons.SIGN_IN, COMMAND_LOGIN);
365 			mainViewItem.addItem(REGISTER, VaadinIcons.FILE_ADD, COMMAND_REGISTER);
366 		}
