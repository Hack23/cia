107 	@Override
108 	public void createDocumentMenuBar(final MenuBar menuBar, final String pageId) {
109 			createDocumentsMenuBar(menuBar);
110 
111 			final MenuItem documentItem = menuBar.addItem(DOCUMENT, VaadinIcons.FILE, null);
112 
113 			documentItem.addItem(OVERVIEW_TEXT, VaadinIcons.FILE,
114 					new PageModeMenuCommand(UserViews.DOCUMENT_VIEW_NAME, PageMode.OVERVIEW, pageId));
115 
116 			documentItem.addItem(DOCUMENT_ACTIVITY_TEXT, VaadinIcons.FILE, new PageModeMenuCommand(UserViews.DOCUMENT_VIEW_NAME,
117 					DocumentPageMode.DOCUMENTACTIVITY.toString(), pageId));
118 
119 			documentItem.addItem(PERSON_REFERENCES, VaadinIcons.FILE, new PageModeMenuCommand(UserViews.DOCUMENT_VIEW_NAME,
120 					DocumentPageMode.PERSONREFERENCES.toString(), pageId));
121 
122 			documentItem.addItem(DOCUMENT_DETAILS, VaadinIcons.FILE, new PageModeMenuCommand(UserViews.DOCUMENT_VIEW_NAME,
123 					DocumentPageMode.DOCUMENTDETAILS.toString(), pageId));
124 
125 			documentItem.addItem(DOCUMENT_DATA, VaadinIcons.FILE, new PageModeMenuCommand(UserViews.DOCUMENT_VIEW_NAME,
126 					DocumentPageMode.DOCUMENTDATA.toString(), pageId));
127 
128 			documentItem.addItem(DOCUMENT_REFERENCES, VaadinIcons.FILE, new PageModeMenuCommand(UserViews.DOCUMENT_VIEW_NAME,
129 					DocumentPageMode.DOCUMENTREFERENCES.toString(), pageId));
130 
131 			documentItem.addItem(DOCUMENT_DECISION, VaadinIcons.FILE, new PageModeMenuCommand(UserViews.DOCUMENT_VIEW_NAME,
132 					DocumentPageMode.DOCUMENTDECISION.toString(), pageId));
133 
134 			documentItem.addItem(DOCUMENT_ATTACHEMENTS, VaadinIcons.FILE, new PageModeMenuCommand(UserViews.DOCUMENT_VIEW_NAME,
135 					DocumentPageMode.DOCUMENTATTACHMENTS.toString(), pageId));
136 
137 
138 			documentItem.addItem(PAGE_VISIT_HISTORY_TEXT, null,
139 					new PageModeMenuCommand(UserViews.DOCUMENT_VIEW_NAME, PageMode.PAGEVISITHISTORY,pageId));
140 
141 	}

144 	@Override
145 	public void createDocumentsMenuBar(final MenuBar menuBar) {
146 		initApplicationMenuBar(menuBar);
147 
148 		applicationMenuItemFactory.addRankingMenu(menuBar);
149 		final MenuItem documentsItem = menuBar.addItem(DOCUMENTS, VaadinIcons.FILE, null);
150 		documentsItem.addItem("List all",VaadinIcons.GROUP, COMMAND_DOCUMENTS);
151 		documentsItem.addItem("Search Documents",VaadinIcons.GROUP, COMMAND_SEARCH_DOCUMENT);
152 
153 	}

156 	@Override

159 
160 		createButtonLink(grid,DOCUMENT_ACTIVITY_TEXT, VaadinIcons.FILE, new PageModeMenuCommand(UserViews.DOCUMENT_VIEW_NAME,

163 		createButtonLink(grid,PERSON_REFERENCES, VaadinIcons.FILE, new PageModeMenuCommand(UserViews.DOCUMENT_VIEW_NAME,
164 				DocumentPageMode.PERSONREFERENCES.toString(), pageId), PERSON_REFERENCES);
165 
166 		createButtonLink(grid,DOCUMENT_DETAILS, VaadinIcons.FILE, new PageModeMenuCommand(UserViews.DOCUMENT_VIEW_NAME,
167 				DocumentPageMode.DOCUMENTDETAILS.toString(), pageId), DOCUMENT_DETAILS);

169 		createButtonLink(grid,DOCUMENT_DATA, VaadinIcons.FILE, new PageModeMenuCommand(UserViews.DOCUMENT_VIEW_NAME,
