68  	@Secured({ "ROLE_ADMIN" })
69  	@Override
70  	public Layout createContent(final String parameters, final MenuBar menuBar, final Panel panel) {
71  		final VerticalLayout content = createPanelContent();
72  
73  		final String pageId = getPageId(parameters);
74  		final int pageNr = getPageNr(parameters);
75  
76  		getMenuItemFactory().createMainPageMenuBar(menuBar);
77  
78  		createPageHeader(panel, content, "Admin Application Events", "Event Details",
79  				"Review a comprehensive list of application events, including timestamps and statuses.");
80  
81  		final DataContainer<ApplicationActionEvent, Long> dataContainer = getApplicationManager()
82  				.getDataContainer(ApplicationActionEvent.class);
83  
84  		final List<ApplicationActionEvent> pageOrderBy = dataContainer.getPageOrderBy(pageNr, DEFAULT_RESULTS_PER_PAGE,
85  				ApplicationActionEvent_.createdDate);
86  
87  		getPagingUtil().createPagingControls(content, NAME, pageId, dataContainer.getSize(), pageNr,
88  				DEFAULT_RESULTS_PER_PAGE);
89  
90  		getGridFactory().createBasicBeanItemGrid(content, ApplicationActionEvent.class, pageOrderBy,
91  				APPLICATION_ACTION_EVENT, COLUMN_ORDER, HIDE_COLUMNS, LISTENER, null, null);
92  
93  		if (pageId != null && !pageId.isEmpty()) {
94  			final ApplicationActionEvent applicationActionEvent = dataContainer.load(Long.valueOf(pageId));
95  			if (applicationActionEvent != null) {
96  				// Instead of form fields, use a card layout similar to the party overview.
97  				final Panel cardPanel = new Panel();
98  				cardPanel.addStyleName("politician-overview-card"); // Reuse existing style
99  				cardPanel.setWidth("100%");
100 				cardPanel.setHeightUndefined();
101 				Responsive.makeResponsive(cardPanel);
102 
103 				final VerticalLayout cardContent = new VerticalLayout();
104 				cardContent.setMargin(true);
105 				cardContent.setSpacing(true);
106 				cardContent.setWidth("100%");
107 				cardPanel.setContent(cardContent);
108 
109 				content.addComponent(cardPanel);
110 
111 				// Card Header
112 				final HorizontalLayout headerLayout = new HorizontalLayout();
113 				headerLayout.setSpacing(true);
114 				headerLayout.setWidth("100%");
115 				headerLayout.addStyleName("card-header-section");
116 
117 				final String titleText = "Application Event Details";
118 				final Label titleLabel = new Label(titleText, ContentMode.HTML);
119 				titleLabel.addStyleName("card-title");
120 				titleLabel.setWidthUndefined();
121 				headerLayout.addComponent(titleLabel);
122 
123 				cardContent.addComponent(headerLayout);
124 
125 				// Divider line
126 				final Label divider = new Label("<hr/>", ContentMode.HTML);
127 				divider.addStyleName("card-divider");
128 				divider.setWidth("100%");
129 				cardContent.addComponent(divider);
130 
131 				// Attributes layout
132 				final VerticalLayout attributesLayout = new VerticalLayout();
133 				attributesLayout.setSpacing(true);
134 				attributesLayout.setWidth("100%");
135 				cardContent.addComponent(attributesLayout);
136 
137 				addInfoRowIfNotNull(attributesLayout, "Created Date:",
138 						String.valueOf(applicationActionEvent.getCreatedDate()), VaadinIcons.CALENDAR);
139 				addInfoRowIfNotNull(attributesLayout, "Event Group:",
140 						String.valueOf(applicationActionEvent.getEventGroup()), VaadinIcons.INFO);
141 				addInfoRowIfNotNull(attributesLayout, "Application Operation:",
142 						String.valueOf(applicationActionEvent.getApplicationOperation()), VaadinIcons.TOOLS);
143 				addInfoRowIfNotNull(attributesLayout, "Page:",
144 						String.valueOf(applicationActionEvent.getPage()), VaadinIcons.FILE_TEXT);
145 				addInfoRowIfNotNull(attributesLayout, "Page Mode:",
146 						String.valueOf(applicationActionEvent.getPageMode()), VaadinIcons.LIST);
147 				addInfoRowIfNotNull(attributesLayout, "Element Id:",
148 						String.valueOf(applicationActionEvent.getElementId()), VaadinIcons.POINTER);
149 				addInfoRowIfNotNull(attributesLayout, "Action Name:",
150 						String.valueOf(applicationActionEvent.getActionName()), VaadinIcons.PLAY_CIRCLE);
151 				addInfoRowIfNotNull(attributesLayout, "User Id:",
152 						String.valueOf(applicationActionEvent.getUserId()), VaadinIcons.USER);
153 				addInfoRowIfNotNull(attributesLayout, "Session Id:",
154 						String.valueOf(applicationActionEvent.getSessionId()), VaadinIcons.KEY);
155 				addInfoRowIfNotNull(attributesLayout, "Error Message:",
156 						String.valueOf(applicationActionEvent.getErrorMessage()), VaadinIcons.WARNING);
157 				addInfoRowIfNotNull(attributesLayout, "Application Message:",
158 						String.valueOf(applicationActionEvent.getApplicationMessage()), VaadinIcons.INFO_CIRCLE);
159 			}
160 		}
161 
162 		getPageActionEventHelper().createPageEvent(ViewAction.VISIT_ADMIN_APPLICATION_EVENTS_VIEW,
163 				ApplicationEventGroup.ADMIN, NAME, null, pageId);
164 
165 		content.addStyleName("v-layout-content-overview-panel-level1");
166 
167 		return content;
168 	}

170 	@Override
171 	public boolean matches(final String page, final String parameters) {
172 		return NAME.equals(page) && (parameters == null || !parameters.contains(PageMode.CHARTS.toString()));
173 	}

175 	/**
176 	 * Adds an info row to the parent layout if value is not null or empty.
177 	 *
178 	 * @param parent the parent layout
179 	 * @param caption the caption
180 	 * @param value the value
181 	 * @param icon the icon
182 	 */
183 	private void addInfoRowIfNotNull(final VerticalLayout parent, final String caption, final String value,
184 			final VaadinIcons icon) {
185 		if (value != null && !value.trim().isEmpty() && !"null".equalsIgnoreCase(value)) {
186 			parent.addComponent(createInfoRow(caption, value, icon));
187 		}
188 	}
189 
190 	/**
191 	 * Creates a simple info row (caption and value) with optional icon.
192 	 *
193 	 * @param caption the field caption
194 	 * @param value   the field value
195 	 * @param icon    a VaadinIcons icon
196 	 * @return a HorizontalLayout representing the info row
197 	 */
198 	private HorizontalLayout createInfoRow(final String caption, final String value, VaadinIcons icon) {
199 		final HorizontalLayout layout = new HorizontalLayout();
200 		layout.setSpacing(true);
201 		layout.addStyleName("metric-label");
202 		layout.setWidthUndefined();
203 
204 		if (icon != null) {
205 			final Label iconLabel = new Label(icon.getHtml(), ContentMode.HTML);
206 			iconLabel.addStyleName("card-info-icon");
207 			layout.addComponent(iconLabel);
208 		}
209 
210 		final Label captionLabel = new Label(caption);
211 		captionLabel.addStyleName("card-info-caption");
212 
213 		final Label valueLabel = new Label(value != null ? value : "");
214 		valueLabel.addStyleName("card-info-value");
215 
216 		layout.addComponents(captionLabel, valueLabel);
217 		return layout;
218 	}
