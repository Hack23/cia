39  /**
40   * The Class GovernmentBodyHeadcountPageModContentFactoryImpl.
41   */
42  @Component
43  public final class GovernmentBodyHeadcountPageModContentFactoryImpl extends AbstractGovernmentBodyPageModContentFactoryImpl {
44  
45  	/** The Constant GOVERNMENT_BODIES. */
46  	private static final String GOVERNMENT_BODIES = "Headcount:";
47  
48  	/** The Constant GOVERNMENT_BODY. */
49  	private static final String GOVERNMENT_BODY = "GovernmentBody:";
50  
51  	/** The government body chart data manager. */
52  	@Autowired
53  	private GovernmentBodyChartDataManager governmentBodyChartDataManager;
54  
55  	/**
56  	 * Instantiates a new government body headcount page mod content factory impl.
57  	 */
58  	public GovernmentBodyHeadcountPageModContentFactoryImpl() {
59  		super();
60  	}
61  
62  	@Secured({ "ROLE_ANONYMOUS", "ROLE_USER", "ROLE_ADMIN" })
63  	@Override
64  	public Layout createContent(final String parameters, final MenuBar menuBar, final Panel panel) {
65  		final VerticalLayout panelContent = createPanelContent();
66  
67  		final String pageId = getPageId(parameters);
68  		final List<GovernmentBodyAnnualSummary> list = getItem(parameters);
69  
70  		if (list != null && !list.isEmpty()) {
71  			final GovernmentBodyAnnualSummary governmentBodyAnnualSummary = list.get(0);
72  
73  			if (governmentBodyAnnualSummary != null) {
74  				getGovernmentBodyMenuItemFactory().createGovernmentBodyMenuBar(menuBar, pageId,governmentBodyAnnualSummary.getName());
75  				LabelFactory.createHeader2Label(panelContent,GOVERNMENT_BODIES + governmentBodyAnnualSummary.getName());
76  				governmentBodyChartDataManager.createGovernmentBodyHeadcountSummaryChart(panelContent, governmentBodyAnnualSummary.getName());
77  				panel.setCaption(GOVERNMENT_BODY + ":" + governmentBodyAnnualSummary.getName());
78  			}
79  			getPageActionEventHelper().createPageEvent(ViewAction.VISIT_MINISTRY_VIEW, ApplicationEventGroup.USER, NAME,
80  					parameters, pageId);
81  
82  		}
83  
84  		return panelContent;
85  
86  	}
87  
88  	@Override
89  	public boolean matches(final String page, final String parameters) {
90  		return NAME.equals(page) && StringUtils.contains(parameters, GovernmentBodyPageMode.HEADCOUNT.toString());
91  	}
92  
93  }
