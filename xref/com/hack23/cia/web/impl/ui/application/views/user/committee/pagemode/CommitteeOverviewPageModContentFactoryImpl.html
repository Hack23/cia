25

38  import com.vaadin.ui.Panel;

41  /**
42   * The Class CommitteeOverviewPageModContentFactoryImpl.
43   */
44  @Component
45  public final class CommitteeOverviewPageModContentFactoryImpl extends AbstractCommitteePageModContentFactoryImpl {
46  
47  	/**
48  	 * Instantiates a new committee overview page mod content factory impl.
49  	 */
50  	public CommitteeOverviewPageModContentFactoryImpl() {
51  		super();
52  	}
53  
54  	/**
55  	 * Creates the content.
56  	 *
57  	 * @param parameters the parameters
58  	 * @param menuBar the menu bar
59  	 * @param panel the panel
60  	 * @return the layout
61  	 */
62  	@Secured({ "ROLE_ANONYMOUS", "ROLE_USER", "ROLE_ADMIN" })
63  	@Override
64  	public Layout createContent(final String parameters, final MenuBar menuBar, final Panel panel) {
65  		final VerticalLayout panelContent = createPanelContent();
66  		panel.setContent(panelContent);
67  
68  		final String pageId = getPageId(parameters);
69  
70  		final ViewRiksdagenCommittee viewRiksdagenCommittee = getItem(parameters);
71  		getCommitteeMenuItemFactory().createCommitteeeMenuBar(menuBar, pageId);
72  
73  		CardInfoRowUtil.createPageHeader(panel, panelContent,
74  				CommitteeViewConstants.OV_TITLE_HEADER + viewRiksdagenCommittee.getEmbeddedId().getDetail(),
75  				CommitteeViewConstants.OV_TITLE,
76  				CommitteeViewConstants.OV_DESCRIPTION);
77  
78  		final Link addCommitteePageLink = getPageLinkFactory().addCommitteePageLink(viewRiksdagenCommittee);
79  		panelContent.addComponent(addCommitteePageLink);
80  		panelContent.setExpandRatio(addCommitteePageLink, ContentRatio.SMALL);
81  
82  		// Create a card panel for the committee overview
83  		final Panel cardPanel = new Panel();
84  		cardPanel.addStyleName("politician-overview-card");
85  		cardPanel.setWidth("100%");
86  		cardPanel.setHeightUndefined();
87  		Responsive.makeResponsive(cardPanel);
88  
89  		final VerticalLayout cardContent = new VerticalLayout();
90  		cardContent.setMargin(true);
91  		cardContent.setSpacing(true);
92  		cardContent.setWidth("100%");
93  		cardPanel.setContent(cardContent);
94  
95  		panelContent.addComponent(cardPanel);
96  		panelContent.setExpandRatio(cardPanel, ContentRatio.SMALL_GRID);
97  
98  		CardInfoRowUtil.createCardHeader(cardContent,"Committee: " + viewRiksdagenCommittee.getEmbeddedId().getDetail());
99  
100 		// Two-column layout for committee attributes
101 		final HorizontalLayout attributesLayout = new HorizontalLayout();
102 		attributesLayout.setSpacing(true);
103 		attributesLayout.setWidth("100%");
104 		cardContent.addComponent(attributesLayout);
105 
106 		// First column: Committee Profile (keep existing, but add activity level)
107 		final VerticalLayout profileDetailsLayout = CardInfoRowUtil.createSectionLayout(CommitteeViewConstants.OV_SECTION_PROFILE);
108 
109 		profileDetailsLayout.addComponent(CardInfoRowUtil.createInfoRow("Detail:", viewRiksdagenCommittee.getEmbeddedId().getDetail(),
110 		        VaadinIcons.INFO_CIRCLE, "Internal identifier detail for the committee"));
111 		profileDetailsLayout.addComponent(CardInfoRowUtil.createInfoRow("Status:", viewRiksdagenCommittee.isActive() ? "Active" : "Inactive",
112 		        VaadinIcons.FLAG, "Current committee status"));
113 		profileDetailsLayout.addComponent(CardInfoRowUtil.createInfoRow("Activity Level:", viewRiksdagenCommittee.getActivityLevel(),
114 		        VaadinIcons.CHART, "Committee's current activity level"));
115 		profileDetailsLayout.addComponent(CardInfoRowUtil.createInfoRow("First Assignment:", String.valueOf(viewRiksdagenCommittee.getFirstAssignmentDate()),
116 		        VaadinIcons.CALENDAR, "Date the committee's first assignment started"));
117 		profileDetailsLayout.addComponent(CardInfoRowUtil.createInfoRow("Last Assignment:", String.valueOf(viewRiksdagenCommittee.getLastAssignmentDate()),
118 		        VaadinIcons.CALENDAR_CLOCK, "Date of the committee's most recent assignment"));
119 
120 		// Second column: Membership Statistics
121 		final VerticalLayout membershipStatsLayout = CardInfoRowUtil.createSectionLayout(CommitteeViewConstants.OV_SECTION_MEMBERSHIP);
122 
123 		membershipStatsLayout.addComponent(CardInfoRowUtil.createInfoRow("Current Members:", String.valueOf(viewRiksdagenCommittee.getCurrentMemberSize()),
124 		        VaadinIcons.GROUP, "Total current committee members"));
125 		membershipStatsLayout.addComponent(CardInfoRowUtil.createInfoRow("Regular Members:", String.valueOf(viewRiksdagenCommittee.getCurrentRegularMembers()),
126 		        VaadinIcons.USER, "Number of current regular members"));
127 		membershipStatsLayout.addComponent(CardInfoRowUtil.createInfoRow("Leadership Positions:", String.valueOf(viewRiksdagenCommittee.getCurrentLeadershipPositions()),
128 		        VaadinIcons.STAR, "Current number of leadership positions"));
129 		membershipStatsLayout.addComponent(CardInfoRowUtil.createInfoRow("Substitute Positions:", String.valueOf(viewRiksdagenCommittee.getCurrentSubstitutePositions()),
130 		        VaadinIcons.USER_CLOCK, "Current number of substitute positions"));
131 		membershipStatsLayout.addComponent(CardInfoRowUtil.createInfoRow("Total Leadership Roles:", String.valueOf(viewRiksdagenCommittee.getTotalLeadershipPositions()),
132 		        VaadinIcons.USERS, "Historical total of leadership positions"));
133 		membershipStatsLayout.addComponent(CardInfoRowUtil.createInfoRow("Total Substitute Roles:", String.valueOf(viewRiksdagenCommittee.getTotalSubstitutePositions()),
134 		        VaadinIcons.USERS, "Historical total of substitute positions"));
135 
136 		// Third column: Document Statistics
137 		final VerticalLayout documentStatsLayout = CardInfoRowUtil.createSectionLayout(CommitteeViewConstants.OV_SECTION_DOCUMENT);
138 
139 		documentStatsLayout.addComponent(CardInfoRowUtil.createInfoRow("Total Documents:", String.valueOf(viewRiksdagenCommittee.getTotalDocuments()),
140 		        VaadinIcons.FILE_TEXT, "Total number of documents produced"));
141 		documentStatsLayout.addComponent(CardInfoRowUtil.createInfoRow("Documents Last Year:", String.valueOf(viewRiksdagenCommittee.getDocumentsLastYear()),
142 		        VaadinIcons.FILE_O, "Documents produced in the last year"));
143 		documentStatsLayout.addComponent(CardInfoRowUtil.createInfoRow("Avg Documents/Member:", String.format(Locale.ENGLISH,"%.1f", viewRiksdagenCommittee.getAvgDocumentsPerMember()),
144 		        VaadinIcons.CHART_LINE, "Average documents per committee member"));
145 		documentStatsLayout.addComponent(CardInfoRowUtil.createInfoRow("Committee Motions:", String.valueOf(viewRiksdagenCommittee.getTotalCommitteeMotions()),
146 		        VaadinIcons.FILE_PRESENTATION, "Total number of committee motions"));
147 		documentStatsLayout.addComponent(CardInfoRowUtil.createInfoRow("Total Assignments:", String.valueOf(viewRiksdagenCommittee.getTotalAssignments()),
148 		        VaadinIcons.TASKS, "Total number of assignments"));
149 		documentStatsLayout.addComponent(CardInfoRowUtil.createInfoRow("Days Served:", String.valueOf(viewRiksdagenCommittee.getTotalDaysServed()),
150 		        VaadinIcons.CLOCK, "Total days of committee service"));
151 
152 		// Clear existing components and add all three columns
153 		attributesLayout.removeAllComponents();
154 		attributesLayout.addComponents(profileDetailsLayout, membershipStatsLayout, documentStatsLayout);
155 
156 		// Optional: Add some styling to make the columns more responsive
157 		attributesLayout.setWidth("100%");
158 		attributesLayout.setSpacing(true);
159 
160 
161 		// After the card, add the overview layout
162 		final VerticalLayout overviewLayout = new VerticalLayout();
163 		overviewLayout.setSizeFull();
164 		panelContent.addComponent(overviewLayout);
165 		panelContent.setExpandRatio(overviewLayout, ContentRatio.LARGE_FORM);
166 
167 		getCommitteeMenuItemFactory().createOverviewPage(overviewLayout, pageId);
168 
169 		getPageActionEventHelper().createPageEvent(ViewAction.VISIT_COMMITTEE_VIEW, ApplicationEventGroup.USER, NAME,
170 				parameters, pageId);
171 
172 		return panelContent;
173 	}
174 
175 	/**
176 	 * Matches.
177 	 *
178 	 * @param page the page
179 	 * @param parameters the parameters
180 	 * @return true, if successful
181 	 */
182 	@Override
183 	public boolean matches(final String page, final String parameters) {
184 		return PageCommandCommitteeConstants.COMMAND_COMMITTEE_OVERVIEW.matches(page, parameters);
185 	}
186 
187 }
