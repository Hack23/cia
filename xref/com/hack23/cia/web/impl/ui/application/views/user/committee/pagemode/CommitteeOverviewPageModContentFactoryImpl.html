216 	/**
217 	 * Creates a row displaying a caption and value, with optional icon and tooltip.
218 	 *
219 	 * @param caption the field caption
220 	 * @param value   the field value
221 	 * @param icon    a VaadinIcons icon for better visual cue
222 	 * @param tooltip optional tooltip to provide more info
223 	 * @return a HorizontalLayout representing the info row
224 	 */
225 	private HorizontalLayout createInfoRow(final String caption, final String value, VaadinIcons icon, final String tooltip) {
226 		final HorizontalLayout layout = new HorizontalLayout();
227 		layout.setSpacing(true);
228 		layout.addStyleName("metric-label");
229 		layout.setWidthUndefined();
230 
231 		if (icon != null) {
232 			final Label iconLabel = new Label(icon.getHtml(), ContentMode.HTML);
233 			iconLabel.addStyleName("card-info-icon");
234 			if (tooltip != null && !tooltip.isEmpty()) {
235 				iconLabel.setDescription(tooltip);
236 			}
237 			layout.addComponent(iconLabel);
238 		}
239 
240 		final Label captionLabel = new Label(caption);
241 		captionLabel.addStyleName("card-info-caption");
242 		if (tooltip != null && !tooltip.isEmpty()) {
243 			captionLabel.setDescription(tooltip);
244 		}
245 
246 		final Label valueLabel = new Label(value != null ? value : "");
247 		valueLabel.addStyleName("card-info-value");
248 
249 		layout.addComponents(captionLabel, valueLabel);
250 		return layout;
251 	}

253 	/**
254 	 * Matches.
255 	 *
256 	 * @param page the page
257 	 * @param parameters the parameters
258 	 * @return true, if successful
259 	 */
260 	@Override
261 	public boolean matches(final String page, final String parameters) {
262 		final String pageId = getPageId(parameters);
263 		return NAME.equals(page) && (StringUtils.isEmpty(parameters) || parameters.equals(pageId)
264 				|| parameters.contains(PageMode.OVERVIEW.toString()));
265 	}
