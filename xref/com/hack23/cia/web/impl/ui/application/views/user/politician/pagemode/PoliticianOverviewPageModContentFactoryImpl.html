59  	@Secured({ "ROLE_ANONYMOUS", "ROLE_USER", "ROLE_ADMIN" })
60  	@Override
61  	public Layout createContent(final String parameters, final MenuBar menuBar, final Panel panel) {
62  		final VerticalLayout panelContent = createPanelContent();
63  		panel.setContent(panelContent);
64  
65  		final String pageId = getPageId(parameters);
66  		final ViewRiksdagenPolitician viewRiksdagenPolitician = getItem(parameters);
67  
68  		getPoliticianMenuItemFactory().createPoliticianMenuBar(menuBar, pageId);
69  
70  		createPageHeader(
71  				panel,
72  				panelContent,
73  				"Politician Overview " + viewRiksdagenPolitician.getFirstName() + ' ' + viewRiksdagenPolitician.getLastName()
74  						+ " (" + viewRiksdagenPolitician.getParty() + ')',
75  				"Politician Details",
76  				"Detailed profiles and activities of politicians."
77  		);
78  
79  		final PersonData personData = getApplicationManager().getDataContainer(PersonData.class)
80  				.load(viewRiksdagenPolitician.getPersonId());
81  
82  		createOverviewContent(panelContent, personData, viewRiksdagenPolitician, pageId);
83  
84  		getPageActionEventHelper().createPageEvent(
85  				ViewAction.VISIT_POLITICIAN_VIEW,
86  				ApplicationEventGroup.USER,
87  				UserViews.POLITICIAN_VIEW_NAME,
88  				parameters,
89  				pageId
90  		);
91  
92  		return panelContent;
93  	}

95  	/**
96  	 * Creates the overview content in a card style similar to the scoreboard snippet.
97  	 *
98  	 * @param panelContent            the panel content
99  	 * @param personData              the person data
100 	 * @param viewRiksdagenPolitician the view riksdagen politician
101 	 * @param pageId                  the page id
102 	 */
103 	private void createOverviewContent(final VerticalLayout panelContent, final PersonData personData,
104 			final ViewRiksdagenPolitician viewRiksdagenPolitician, final String pageId) {
105 
106 		// Link to politician detail page
107 		final Link createPoliticianPageLink = getPageLinkFactory().createPoliticianPageLink(personData);
108 		createPoliticianPageLink.addStyleName("card-subtitle");
109 		panelContent.addComponent(createPoliticianPageLink);
110 		panelContent.setExpandRatio(createPoliticianPageLink, ContentRatio.SMALL);
111 
112 		// Create a panel (card) for the politician
113 		final Panel cardPanel = new Panel();
114 		cardPanel.addStyleName("politician-overview-card");
115 		cardPanel.setWidth("100%");
116 		cardPanel.setHeightUndefined();
117 		Responsive.makeResponsive(cardPanel);
118 
119 		final VerticalLayout cardContent = new VerticalLayout();
120 		cardContent.setMargin(true);
121 		cardContent.setSpacing(true);
122 		cardContent.setWidth("100%");
123 		cardPanel.setContent(cardContent);
124 
125 		panelContent.addComponent(cardPanel);
126 		panelContent.setExpandRatio(cardPanel, ContentRatio.SMALL_GRID);
127 
128 		// Header layout similar to scoreboard snippet
129 		final HorizontalLayout headerLayout = new HorizontalLayout();
130 		headerLayout.setSpacing(true);
131 		headerLayout.setWidth("100%");
132 		headerLayout.addStyleName("card-header-section");
133 
134 		final String titleText = viewRiksdagenPolitician.getFirstName() + " " + viewRiksdagenPolitician.getLastName()
135 				+ " (" + viewRiksdagenPolitician.getParty() + ")";
136 		final Label titleLabel = new Label(titleText, ContentMode.HTML);
137 		titleLabel.addStyleName("card-title");
138 		titleLabel.setWidthUndefined();
139 		headerLayout.addComponent(titleLabel);
140 
141 		// Party link
142 		final Link partyLink = new Link("Party " + viewRiksdagenPolitician.getParty(),
143 				new ExternalResource("#!" + UserViews.PARTY_VIEW_NAME + "/" + viewRiksdagenPolitician.getParty()));
144 		partyLink.setIcon(VaadinIcons.GROUP);
145 		partyLink.addStyleName("card-title");
146 		headerLayout.addComponent(partyLink);
147 
148 		cardContent.addComponent(headerLayout);
149 
150 		// Divider line for better separation
151 		final Label divider = new Label("<hr/>", ContentMode.HTML);
152 		divider.addStyleName("card-divider");
153 		divider.setWidth("100%");
154 		cardContent.addComponent(divider);
155 
156 		// Image and details layout
157 		final HorizontalLayout imageAndDetailsLayout = new HorizontalLayout();
158 		imageAndDetailsLayout.setSpacing(true);
159 		imageAndDetailsLayout.setWidth("100%");
160 		cardContent.addComponent(imageAndDetailsLayout);
161 
162 		// Politician image (smaller)
163 		final String imageUrl = personData.getImageUrl192().replace("http://", "https://");
164 		final Image image = new Image("", new ExternalResource(imageUrl));
165 		image.setDescription("Picture of " + viewRiksdagenPolitician.getFirstName() + " "
166 				+ viewRiksdagenPolitician.getLastName());
167 		image.setWidth(100, Unit.PIXELS);
168 		image.addStyleName("politician-image");
169 
170 		imageAndDetailsLayout.addComponent(image);
171 
172 		// Two-column layout for attributes
173 		final HorizontalLayout attributesLayout = new HorizontalLayout();
174 		attributesLayout.setSpacing(true);
175 		attributesLayout.setWidth("100%");
176 		imageAndDetailsLayout.addComponent(attributesLayout);
177 		imageAndDetailsLayout.setExpandRatio(attributesLayout, 1.0f);
178 
179 		// First column: Profile Details
180 		final VerticalLayout profileDetailsLayout = new VerticalLayout();
181 		profileDetailsLayout.setSpacing(true);
182 		profileDetailsLayout.addStyleName("card-details-column");
183 		profileDetailsLayout.setWidthUndefined();
184 
185 		final Label profileDetailsHeader = new Label("Profile Details");
186 		profileDetailsHeader.addStyleName("card-section-title");
187 		profileDetailsLayout.addComponent(profileDetailsHeader);
188 
189 		profileDetailsLayout.addComponent(createInfoRow("Gender:", viewRiksdagenPolitician.getGender(), VaadinIcons.USER, "Politician's gender"));
190 		profileDetailsLayout.addComponent(createInfoRow("Born Year:", String.valueOf(viewRiksdagenPolitician.getBornYear()), VaadinIcons.CALENDAR, "Year the politician was born"));
191 		profileDetailsLayout.addComponent(createInfoRow("Total Days Served:", String.valueOf(viewRiksdagenPolitician.getTotalDaysServed()), VaadinIcons.CLOCK, "Total number of days in public service"));
192 		profileDetailsLayout.addComponent(createInfoRow("Ministry Days Served:", String.valueOf(viewRiksdagenPolitician.getTotalDaysServedGovernment()), VaadinIcons.CLOCK, "Total number of days as Minister"));
193 		profileDetailsLayout.addComponent(createInfoRow("Parliament Days Served:", String.valueOf(viewRiksdagenPolitician.getTotalDaysServedParliament()), VaadinIcons.CLOCK, "Total number of days in Parliament"));
194 
195 		attributesLayout.addComponent(profileDetailsLayout);
196 
197 		// Second column: Service Statistics
198 		final VerticalLayout serviceStatsLayout = new VerticalLayout();
199 		serviceStatsLayout.setSpacing(true);
200 		serviceStatsLayout.addStyleName("card-details-column");
201 		serviceStatsLayout.setWidthUndefined();
202 
203 		final Label serviceStatsHeader = new Label("Service Statistics");
204 		serviceStatsHeader.addStyleName("card-section-title");
205 		serviceStatsLayout.addComponent(serviceStatsHeader);
206 
207 		// Additional stats (Assuming these methods exist)
208 		serviceStatsLayout.addComponent(createInfoRow("Total Assignments:", String.valueOf(viewRiksdagenPolitician.getTotalAssignments()), VaadinIcons.BAR_CHART, "Total number of assignments held"));
209 		serviceStatsLayout.addComponent(createInfoRow("First Assignment Date:", String.valueOf(viewRiksdagenPolitician.getFirstAssignmentDate().toString()), VaadinIcons.FLASH, "First assignment in public service"));
210 		serviceStatsLayout.addComponent(createInfoRow("Last Assignment Date:", String.valueOf(viewRiksdagenPolitician.getLastAssignmentDate().toString()), VaadinIcons.FLASH, "Last assignment in public service"));
211 		serviceStatsLayout.addComponent(createInfoRow("Ministry Assignments:", String.valueOf(viewRiksdagenPolitician.getTotalMinistryAssignments()), VaadinIcons.INSTITUTION, "Number of ministry level assignments"));
212 		serviceStatsLayout.addComponent(createInfoRow("Speaker Assignments:", String.valueOf(viewRiksdagenPolitician.getTotalSpeakerAssignments()), VaadinIcons.MICROPHONE, "Number of speaker assignments"));
213 
214 
215 		attributesLayout.addComponent(serviceStatsLayout);
216 
217 		// After the card, add the overview layout for extended details
218 		final VerticalLayout overviewLayout = new VerticalLayout();
219 		overviewLayout.setSizeFull();
220 		panelContent.addComponent(overviewLayout);
221 		panelContent.setExpandRatio(overviewLayout, ContentRatio.LARGE_FORM);
222 
223 		getPoliticianMenuItemFactory().createOverviewPage(overviewLayout, pageId);
224 	}

226 	/**
227 	 * Creates a row displaying a caption and value, with optional icon and tooltip.
228 	 *
229 	 * @param caption the field caption
230 	 * @param value   the field value
231 	 * @param icon    a VaadinIcons icon for better visual cue
232 	 * @param tooltip optional tooltip to provide more info
233 	 * @return a HorizontalLayout representing the info row
234 	 */
235 	private HorizontalLayout createInfoRow(final String caption, final String value, VaadinIcons icon, final String tooltip) {
236 		final HorizontalLayout layout = new HorizontalLayout();
237 		layout.setSpacing(true);
238 		layout.addStyleName("metric-label");
239 		layout.setWidthUndefined();
240 
241 		Label iconLabel = null;
242 		if (icon != null) {
243 			iconLabel = new Label(icon.getHtml(), ContentMode.HTML);
244 			iconLabel.setDescription(tooltip);
245 			iconLabel.addStyleName("card-info-icon");
246 			layout.addComponent(iconLabel);
247 		}
248 
249 		final Label captionLabel = new Label(caption);
250 		captionLabel.addStyleName("card-info-caption");
251 		if (tooltip != null && !tooltip.isEmpty()) {
252 			captionLabel.setDescription(tooltip);
253 		}
254 
255 		final Label valueLabel = new Label(value != null ? value : "");
256 		valueLabel.addStyleName("card-info-value");
257 
258 		layout.addComponents(captionLabel, valueLabel);
259 		return layout;
260 	}

262 	@Override
263 	public boolean matches(final String page, final String parameters) {
264 		final String pageId = getPageId(parameters);
265 		return NAME.equals(page) && (StringUtils.isEmpty(parameters) || parameters.equals(pageId)
266 				|| parameters.contains(PageMode.OVERVIEW.toString()));
267 	}
