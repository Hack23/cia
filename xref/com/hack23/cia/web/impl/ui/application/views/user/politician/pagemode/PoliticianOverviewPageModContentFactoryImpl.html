63  	/**
64  	 * Creates the content.
65  	 *
66  	 * @param parameters the parameters
67  	 * @param menuBar the menu bar
68  	 * @param panel the panel
69  	 * @return the layout
70  	 */
71  	@Secured({ "ROLE_ANONYMOUS", "ROLE_USER", "ROLE_ADMIN" })
72  	@Override
73  	public Layout createContent(final String parameters, final MenuBar menuBar, final Panel panel) {
74  		final VerticalLayout panelContent = createPanelContent();
75  		panel.setContent(panelContent);
76  
77  		final String pageId = getPageId(parameters);
78  		final ViewRiksdagenPolitician viewRiksdagenPolitician = getItem(parameters);
79  		final ViewRiksdagenPoliticianBallotSummary viewRiksdagenPoliticianBallotSummary = getViewRiksdagenPoliticianBallotSummary(
80  				parameters);
81  
82  		getPoliticianMenuItemFactory().createPoliticianMenuBar(menuBar, pageId);
83  
84  		createPageHeader(panel, panelContent,
85  				"Politician Overview " + viewRiksdagenPolitician.getFirstName() + ' '
86  						+ viewRiksdagenPolitician.getLastName() + " (" + viewRiksdagenPolitician.getParty() + ')',
87  				"Politician Details", "Detailed profiles and activities of politicians.");
88  
89  		final PersonData personData = getApplicationManager().getDataContainer(PersonData.class)
90  				.load(viewRiksdagenPolitician.getPersonId());
91  
92  		createOverviewContent(panelContent, personData, viewRiksdagenPolitician, viewRiksdagenPoliticianBallotSummary,
93  				pageId);
94  
95  		getPageActionEventHelper().createPageEvent(ViewAction.VISIT_POLITICIAN_VIEW, ApplicationEventGroup.USER,
96  				UserViews.POLITICIAN_VIEW_NAME, parameters, pageId);
97  
98  		return panelContent;
99  	}

342 	/**
343 	 * Calculate service years.
344 	 *
345 	 * @param start the start
346 	 * @param end the end
347 	 * @return the string
348 	 */
349 	private String calculateServiceYears(Date start, Date end) {
350 		try {
351 			if (start == null || end == null) {
352 				return "N/A";
353 			}
354 
355 			// Convert java.util.Date to milliseconds and calculate years
356 			final long diffInMillies = Math.abs(end.getTime() - start.getTime());
357 			final long years = diffInMillies / (1000L * 60 * 60 * 24 * 365);
358 
359 			return years + " years";
360 		} catch (final Exception e) {
361 			return "N/A";
362 		}
363 	}

365 	/**
366 	 * Creates a row displaying a caption and value, with optional icon and tooltip.
367 	 *
368 	 * @param caption the field caption
369 	 * @param value   the field value
370 	 * @param icon    a VaadinIcons icon for better visual cue
371 	 * @param tooltip optional tooltip to provide more info
372 	 * @return a HorizontalLayout representing the info row
373 	 */
374 	private HorizontalLayout createInfoRow(final String caption, final String value, VaadinIcons icon,
375 			final String tooltip) {
376 		final HorizontalLayout layout = new HorizontalLayout();
377 		layout.setSpacing(true);
378 		layout.addStyleName("metric-label");
379 		layout.setWidthUndefined();
380 
381 		Label iconLabel = null;
382 		if (icon != null) {
383 			iconLabel = new Label(icon.getHtml(), ContentMode.HTML);
384 			iconLabel.setDescription(tooltip);
385 			iconLabel.addStyleName("card-info-icon");
386 			layout.addComponent(iconLabel);
387 		}
388 
389 		final Label captionLabel = new Label(caption);
390 		captionLabel.addStyleName("card-info-caption");
391 		if (tooltip != null && !tooltip.isEmpty()) {
392 			captionLabel.setDescription(tooltip);
393 		}
394 
395 		final Label valueLabel = new Label(value != null ? value : "");
396 		valueLabel.addStyleName("card-info-value");
397 
398 		layout.addComponents(captionLabel, valueLabel);
399 		return layout;
400 	}
