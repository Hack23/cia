39  /**
40   * The Class MinistryRankingDataGridPageModContentFactoryImpl.
41   */
42  @Service
43  public final class MinistryRankingDataGridPageModContentFactoryImpl
44  		extends AbstractMinistryRankingPageModContentFactoryImpl {
45  
46  	private static final String[] COLUMN_ORDER = { "nameId", "totalDaysServed", "currentMemberSize", "totalAssignments",
47  			"firstAssignmentDate", "lastAssignmentDate", "active" };
48  
49  	/** The Constant DATAGRID. */
50  	private static final String[] HIDE_COLUMNS = {"active"};
51  
52  	/** The Constant LISTENER. */
53  	private static final PageItemPropertyClickListener LISTENER = new PageItemPropertyClickListener(UserViews.MINISTRY_VIEW_NAME, "nameId");
54  
55  	/** The Constant MINISTRIES. */
56  	private static final String MINISTRIES = "Ministries";
57  
58  	/**
59  	 * Instantiates a new ministry ranking data grid page mod content factory
60  	 * impl.
61  	 */
62  	public MinistryRankingDataGridPageModContentFactoryImpl() {
63  		super();
64  	}
65  
66  	@Secured({ "ROLE_ANONYMOUS", "ROLE_USER", "ROLE_ADMIN" })
67  	@Override
68  	public Layout createContent(final String parameters, final MenuBar menuBar, final Panel panel) {
69  		final VerticalLayout panelContent = createPanelContent();
70  
71  		getMinistryRankingMenuItemFactory().createMinistryRankingMenuBar(menuBar);
72  		createPageHeader(panel, panelContent, "Ministry Ranking Dashboard", "Ministry Rankings", "Compare ministry rankings to assess their assignments, performance, and overall impact.");
73  
74  		final String pageId = getPageId(parameters);
75  
76  		final DataContainer<ViewRiksdagenMinistry, String> dataContainer = getApplicationManager()
77  				.getDataContainer(ViewRiksdagenMinistry.class);
78  
79  		getGridFactory().createBasicBeanItemGrid(panelContent, ViewRiksdagenMinistry.class, dataContainer.getAllOrderBy(ViewRiksdagenMinistry_.currentMemberSize),
80  				MINISTRIES,
81  				COLUMN_ORDER, HIDE_COLUMNS, LISTENER, null, null);
82  
83  
84  		getPageActionEventHelper().createPageEvent(ViewAction.VISIT_MINISTRY_RANKING_VIEW, ApplicationEventGroup.USER,
85  				NAME, parameters, pageId);
86  
87  		return panelContent;
88  
89  	}
90  
91  	@Override
92  	public boolean matches(final String page, final String parameters) {
93  		return NAME.equals(page)
94  				&& StringUtils.contains(parameters, PageMode.DATAGRID.toString());
95  	}
96  
97  }
