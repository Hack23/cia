55  	/**
56  	 * Adds the commitee summary.
57  	 *
58  	 * @param stringBuilder         the string builder
59  	 * @param entry                 the entry
60  	 * @param vewRiksdagenCommittee the vew riksdagen committee
61  	 */
62  	private static void addCommiteeSummary(final StringBuilder stringBuilder,
63  			final Entry<String, List<ProposalCommitteeeSummary>> entry,
64  			final Optional<ViewRiksdagenCommittee> vewRiksdagenCommittee) {
65  		if (vewRiksdagenCommittee.isPresent()) {
66  
67  			final Map<String, List<ProposalCommitteeeSummary>> docTypeMap = entry.getValue().stream()
68  					.collect(Collectors.groupingBy(ProposalCommitteeeSummary::getDocType));
69  
70  			stringBuilder.append('\n').append(vewRiksdagenCommittee.get().getEmbeddedId().getDetail());
71  			for (final Entry<String, List<ProposalCommitteeeSummary>> docEntry : docTypeMap.entrySet()) {
72  				if (docEntry.getKey().length() > 0 && entry.getKey().length() > 0) {
73  					addEntry(stringBuilder, docEntry);
74  				}
75  			}
76  		}
77  	}

79  	/**
80  	 * Adds the decision data rows.
81  	 *
82  	 * @param chart
83  	 *            the chart
84  	 * @param entry
85  	 *            the entry
86  	 * @param vewRiksdagenCommittee
87  	 *            the vew riksdagen committee
88  	 */
89  	private static void addDecisionDataRows(final SankeyChart chart,
90  			final Entry<String, List<ProposalCommitteeeSummary>> entry,
91  			final ViewRiksdagenCommittee vewRiksdagenCommittee) {
92  		final Map<String, List<ProposalCommitteeeSummary>> decisionMap = entry.getValue().stream()
93  				.collect(Collectors.groupingBy(ProposalCommitteeeSummary::getDecision));
94  
95  		for (final Entry<String, List<ProposalCommitteeeSummary>> decisionEntry : decisionMap.entrySet()) {
96  			if (decisionEntry.getKey().length() > 0 && entry.getKey().length() > 0) {
97  				chart.addDataRow(vewRiksdagenCommittee.getEmbeddedId().getDetail(), decisionEntry.getKey(),
98  						decisionEntry.getValue().size());
99  			}
100 		}
101 	}

128 	/**
129 	 * Adds the doc type decision data rows.
130 	 *
131 	 * @param chart
132 	 *            the chart
133 	 * @param entry
134 	 *            the entry
135 	 */
136 	private static void addDocTypeDecisionDataRows(final SankeyChart chart,
137 			final Entry<String, List<ProposalCommitteeeSummary>> entry) {
138 		final Map<String, List<ProposalCommitteeeSummary>> docTypeMap = entry.getValue().stream()
139 				.collect(Collectors.groupingBy(ProposalCommitteeeSummary::getDocType));
140 
141 		for (final Entry<String, List<ProposalCommitteeeSummary>> docEntry : docTypeMap.entrySet()) {
142 
143 			final Map<String, List<ProposalCommitteeeSummary>> decisionMap = docEntry.getValue().stream()
144 					.collect(Collectors.groupingBy(ProposalCommitteeeSummary::getDecision));
145 
146 			for (final Entry<String, List<ProposalCommitteeeSummary>> decisionEntry : decisionMap.entrySet()) {
147 				if (decisionEntry.getKey().length() > 0 && entry.getKey().length() > 0) {
148 					chart.addDataRow(docEntry.getKey(), decisionEntry.getKey(), decisionEntry.getValue().size());
149 				}
150 			}
151 		}
152 	}

154 	/**
155 	 * Adds the entry.
156 	 *
157 	 * @param stringBuilder the string builder
158 	 * @param entry         the entry
159 	 * @param docEntry      the doc entry
160 	 */
161 	private static void addEntry(final StringBuilder stringBuilder,
162 			final Entry<String, List<ProposalCommitteeeSummary>> docEntry) {
163 		stringBuilder.append("\n ( ").append(docEntry.getValue().size()).append(' ').append(docEntry.getKey()).append(" -> ");
164 
165 		final Map<String, List<ProposalCommitteeeSummary>> decisionMap = docEntry.getValue().stream()
166 				.collect(Collectors.groupingBy(ProposalCommitteeeSummary::getDecision));
167 
168 		for (final Entry<String, List<ProposalCommitteeeSummary>> decisionEntry : decisionMap.entrySet()) {
169 			if (decisionEntry.getKey().length() > 0) {
170 				stringBuilder.append("\n   ").append(decisionEntry.getValue().size()).append(' ').append(decisionEntry.getKey()).append(' ');
171 
172 				final List<ProposalCommitteeeSummary> decisionSummaryList = decisionEntry.getValue();
173 
174 				for (final ProposalCommitteeeSummary proposalCommitteeeSummary : decisionSummaryList) {
175 					stringBuilder.append("\n    ").append(proposalCommitteeeSummary.getDecision()).append(':').append(proposalCommitteeeSummary.getWording()).append(' ').append(proposalCommitteeeSummary.getWording2()).append(' ');
176 
177 				}
178 
179 			}
180 		}
181 		stringBuilder.append(')');
182 	}

184 	@Override
185 	public SankeyChart createAllDecisionFlow(final Map<String, List<ViewRiksdagenCommittee>> committeeMap,final String rm) {
186 		final List<ProposalCommitteeeSummary> createCommitteeSummary = decisionDataFactory.createCommitteeSummary(rm);
187 
188 
189 		final SankeyChart chart = new SankeyChart();
190 
191 		final Map<String, List<ProposalCommitteeeSummary>> orgProposalMap = createCommitteeSummary.stream()
192 				.collect(Collectors.groupingBy(ProposalCommitteeeSummary::getOrg));
193 
194 
195 		for (final Entry<String, List<ProposalCommitteeeSummary>> entry : orgProposalMap.entrySet()) {
196 
197 			if (committeeMap.containsKey(entry.getKey())) {
198 
199 				final Optional<ViewRiksdagenCommittee> vewRiksdagenCommittee = committeeMap.get(entry.getKey()).stream().findFirst();
200 				if (vewRiksdagenCommittee.isPresent()) {
201 					addDocTypeDataRows(chart, entry, vewRiksdagenCommittee.get());
202 					addDecisionDataRows(chart, entry, vewRiksdagenCommittee.get());
203 				}
204 			}
205 		}
206 
207 		chart.drawChart();
208 		return chart;
209 	}

211 	@Override
212 	public SankeyChart createCommitteeDecisionFlow(final ViewRiksdagenCommittee viewRiksdagenCommittee,
213 			final Map<String, List<ViewRiksdagenCommittee>> committeeMap, final String rm) {
214 		final List<ProposalCommitteeeSummary> createCommitteeSummary = decisionDataFactory.createCommitteeSummary(rm);
215 
216 		final SankeyChart chart = new SankeyChart();
217 
218 		final Map<String, List<ProposalCommitteeeSummary>> orgProposalMap = createCommitteeSummary.stream()
219 				.collect(Collectors.groupingBy(ProposalCommitteeeSummary::getOrg));
220 
221 		for (final Entry<String, List<ProposalCommitteeeSummary>> entry : orgProposalMap.entrySet()) {
222 			if (committeeMap.containsKey(entry.getKey().toUpperCase(Locale.ENGLISH))
223 					&& viewRiksdagenCommittee.getEmbeddedId().getOrgCode().toUpperCase(Locale.ENGLISH).equals(entry.getKey().toUpperCase(Locale.ENGLISH))) {
224 				addDocTypeDecisionDataRows(chart, entry);
225 			}
226 		}
227 
228 		chart.drawChart();
229 		return chart;
230 	}

239 				.collect(Collectors.groupingBy(ProposalCommitteeeSummary::getOrg));
240 
241 		for (final Entry<String, List<ProposalCommitteeeSummary>> entry : orgProposalMap.entrySet()) {

246 		area.setValue(stringBuilder.toString());
