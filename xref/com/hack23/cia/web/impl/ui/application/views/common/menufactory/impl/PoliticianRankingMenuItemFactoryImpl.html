19  package com.hack23.cia.web.impl.ui.application.views.common.menufactory.impl;
20  
21  import org.springframework.beans.factory.annotation.Autowired;
22  import org.springframework.stereotype.Service;
23  
24  import com.hack23.cia.web.impl.ui.application.views.common.menufactory.api.ApplicationMenuItemFactory;
25  import com.hack23.cia.web.impl.ui.application.views.common.menufactory.api.PoliticianRankingMenuItemFactory;
26  import com.hack23.cia.web.impl.ui.application.views.common.pagelinks.api.PageModeMenuCommand;
27  import com.hack23.cia.web.impl.ui.application.views.common.rows.RowUtil;
28  import com.hack23.cia.web.impl.ui.application.views.common.viewnames.ChartIndicators;
29  import com.hack23.cia.web.impl.ui.application.views.common.viewnames.PageMode;
30  import com.hack23.cia.web.impl.ui.application.views.common.viewnames.UserViews;
31  import com.jarektoro.responsivelayout.ResponsiveRow;
32  import com.vaadin.icons.VaadinIcons;
33  import com.vaadin.ui.MenuBar;
34  import com.vaadin.ui.MenuBar.MenuItem;
35  import com.vaadin.ui.VerticalLayout;
36  
37  /**
38   * The Class PoliticianRankingMenuItemFactoryImpl.
39   *
40   * <p>
41   * Builds menus and overviews for politician rankings, enabling analysis of
42   * individual politicians' influence, roles, and longevity in various political
43   * arenas such as parliament, committees, and parties. Emphasizes political
44   * context, strategic influence, and institutional footprints rather than
45   * technical details.
46   * </p>
47   */
48  @Service
49  public final class PoliticianRankingMenuItemFactoryImpl extends AbstractMenuItemFactoryImpl
50          implements PoliticianRankingMenuItemFactory {
51  
52      // Label constants
53      private static final String ALL_PARTIES = "All parties";
54      private static final String CHART_BY_TOPIC_TEXT = "Chart by topic";
55      private static final String CURRENT_PARTIES = "Current parties";
56      private static final String OVERVIEW_TEXT = "Overview";
57      private static final String PAGE_VISIT_HISTORY_TEXT = "Page Visit History";
58      private static final String POLITICAL_EXPERIENCE_SUMMARY = "Political Experience Summary";
59      private static final String POLITICIAN_RANKING = "Politician Ranking";
60  
61      // Description for total experience detail (tooltip)
62      private static final String CURRENT_AND_PAST_ASSIGNMENTS_DESCRIPTION =
63              "Summarized roles, responsibilities, and tenure influence";
64  
65      // Politically focused descriptions (~50 chars)
66      private static final String DESC_POLITICAL_EXPERIENCE = "Experience metrics: roles shaping political influence.";
67      private static final String DESC_ALL_PARTIES = "All parties: mapping legislative engagements.";
68      private static final String DESC_CURRENT_PARTIES = "Current parties: active parliamentary presence.";
69      private static final String DESC_PAGE_HISTORY = "Visit history: tracing engagement over time.";
70  
71      // Page mode commands
72      private static final PageModeMenuCommand COMMAND_ALL_PARTIES = new PageModeMenuCommand(
73              UserViews.POLITICIAN_RANKING_VIEW_NAME, PageMode.CHARTS, ChartIndicators.ALLPARTIES.toString());
74      private static final PageModeMenuCommand COMMAND_CURRENT_PARTIES = new PageModeMenuCommand(
75              UserViews.POLITICIAN_RANKING_VIEW_NAME, PageMode.CHARTS, ChartIndicators.CURRENTPARTIES.toString());
76      private static final PageModeMenuCommand COMMAND_DATAGRID = new PageModeMenuCommand(
77              UserViews.POLITICIAN_RANKING_VIEW_NAME, PageMode.DATAGRID);
78      private static final PageModeMenuCommand COMMAND_OVERVIEW = new PageModeMenuCommand(
79              UserViews.POLITICIAN_RANKING_VIEW_NAME, PageMode.OVERVIEW);
80      private static final PageModeMenuCommand COMMAND_PAGEVISITHISTORY = new PageModeMenuCommand(
81              UserViews.POLITICIAN_RANKING_VIEW_NAME, PageMode.PAGEVISITHISTORY);
82  
83      /** The application menu item factory. */
84      @Autowired
85      private ApplicationMenuItemFactory applicationMenuItemFactory;
86  
87      /**
88       * Instantiates a new politician ranking menu item factory implementation.
89       */
90      public PoliticianRankingMenuItemFactoryImpl() {
91          super();
92      }
93  
94      /**
95       * Creates the overview page with concise, politically focused descriptions.
96       * Each button link offers a ~50 char description emphasizing political dynamics.
97       *
98       * @param panelContent the vertical layout to hold the overview content
99       */
100     @Override
101     public void createOverviewPage(final VerticalLayout panelContent) {
102         final ResponsiveRow grid = RowUtil.createGridLayout(panelContent);
103 
104         // Politician experience summary: USER_CLOCK to indicate roles over time
105         createButtonLink(grid, POLITICAL_EXPERIENCE_SUMMARY, VaadinIcons.USER_CLOCK,
106                 COMMAND_DATAGRID, DESC_POLITICAL_EXPERIENCE);
107 
108         // All parties: analyzing roles, use GROUP icon for multiple parties
109         createButtonLink(grid, ALL_PARTIES, VaadinIcons.GROUP,
110                 COMMAND_ALL_PARTIES, DESC_ALL_PARTIES);
111 
112         // Current parties: similarly GROUP icon for active parties
113         createButtonLink(grid, CURRENT_PARTIES, VaadinIcons.GROUP,
114                 COMMAND_CURRENT_PARTIES, DESC_CURRENT_PARTIES);
115 
116         // Page visit history: use HISTORY icon for historical data
117         createButtonLink(grid, PAGE_VISIT_HISTORY_TEXT, VaadinIcons.CHART,
118                 COMMAND_PAGEVISITHISTORY, DESC_PAGE_HISTORY);
119     }
120 
121     /**
122      * Creates the politician ranking menu bar and integrates it into the main
123      * application menu. Ensures logical grouping and context-appropriate icons.
124      *
125      * @param menuBar the main menu bar to which the politician ranking items are added
126      */
127     @Override
128     public void createPoliticianRankingMenuBar(final MenuBar menuBar) {
129         initApplicationMenuBar(menuBar);
130         applicationMenuItemFactory.addRankingMenu(menuBar);
131 
132         // Politician Ranking: USER icon for individual-centric analysis
133         createPoliticianRankingTopics(menuBar.addItem(POLITICIAN_RANKING, VaadinIcons.USER, null));
134     }
135 
136     /**
137      * Adds politician ranking topics to the specified menu item. Uses icons to convey
138      * context—e.g., LIST for lists, CHART for data visualization, GROUP for parties,
139      * HISTORY for historical data.
140      *
141      * @param politicianMenuItem the parent menu item for politician ranking topics
142      */
143     @Override
144     public void createPoliticianRankingTopics(final MenuItem politicianMenuItem) {
145         // Overview: DASHBOARD for a general overview panel
146         politicianMenuItem.addItem(OVERVIEW_TEXT, VaadinIcons.DASHBOARD, COMMAND_OVERVIEW);
147 
148         // Political experience summary: USER_CLOCK indicating experience over time
149         final MenuItem listItem = politicianMenuItem.addItem(POLITICAL_EXPERIENCE_SUMMARY, VaadinIcons.USER_CLOCK, COMMAND_DATAGRID);
150         listItem.setDescription(CURRENT_AND_PAST_ASSIGNMENTS_DESCRIPTION);
151 
152         // Chart by topic: CHART icon for visual data representation
153         final MenuItem chartByTopic = politicianMenuItem.addItem(CHART_BY_TOPIC_TEXT, VaadinIcons.CHART, null);
154 
155         chartByTopic.addItem(ALL_PARTIES, VaadinIcons.GROUP, COMMAND_ALL_PARTIES);
156         chartByTopic.addItem(CURRENT_PARTIES, VaadinIcons.GROUP, COMMAND_CURRENT_PARTIES);
157 
158         politicianMenuItem.addItem(PAGE_VISIT_HISTORY_TEXT, VaadinIcons.CHART, COMMAND_PAGEVISITHISTORY);
159     }
160 }
