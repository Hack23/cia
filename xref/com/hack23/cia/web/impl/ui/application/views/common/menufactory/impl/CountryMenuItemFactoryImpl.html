97  	/**
98  	 * Adds the sources and indicators to menu.
99  	 *
100 	 * @param countryIndicators
101 	 *            the country indicators
102 	 * @param sourceIndicatorMap
103 	 *            the source indicator map
104 	 */
105 	private static void addSourcesAndIndicatorsToMenu(final MenuItem countryIndicators,
106 			final Map<String, List<ViewWorldbankIndicatorDataCountrySummary>> sourceIndicatorMap) {
107 
108 		final Map<String, List<ViewWorldbankIndicatorDataCountrySummary>> sortedIndicatorMap = sourceIndicatorMap
109 				.entrySet().stream().sorted((e1, e2) -> e1.getKey().compareTo(e2.getKey()))
110 				.collect(Collectors.toMap(Map.Entry::getKey, Map.Entry::getValue, (e1, e2) -> e1, LinkedHashMap::new));
111 
112 		for (final Entry<String, List<ViewWorldbankIndicatorDataCountrySummary>> entry : sortedIndicatorMap
113 				.entrySet()) {
114 
115 			final MenuItem sourceItems = countryIndicators.addItem(entry.getKey(), null, null);
116 
117 			final List<ViewWorldbankIndicatorDataCountrySummary> sortedEntries = entry.getValue().stream()
118 					.sorted((e1, e2) -> e1.getIndicatorName().compareTo(e2.getIndicatorName()))
119 					.toList();
120 
121 			for (final ViewWorldbankIndicatorDataCountrySummary indciatorSummary : sortedEntries) {
122 				final MenuItem addItem = sourceItems.addItem(indciatorSummary.getIndicatorName(),
123 						new PageModeMenuCommand(UserViews.COUNTRY_RANKING_VIEW_NAME, PageMode.INDICATORS,
124 								indciatorSummary.getEmbeddedId().getIndicatorId()));
125 				addItem.setStyleName("RestrictedHeader");
126 			}
127 		}
128 
129 	}

143 	public void createCountryTopicMenu(final MenuItem charts) {
144 		charts.addItem(OVERVIEW_TEXT, VaadinIcons.LINE_CHART,
145 				COMMAND_OVERVIEW);
146 
147 		final MenuItem countryIndicators = charts.addItem(COUNTRY_INDICATORS_SWEDEN, VaadinIcons.LINE_CHART, null);
148 
149 		addSourcesAndIndicatorsToMenu(countryIndicators, getTopicIndicatorMap());
150 
151 		charts.addItem(PAGE_VISIT_HISTORY_TEXT, VaadinIcons.LINE_CHART, COMMAND_PAGEVISITHISTORY);
152 
153 	}

155 	@Override
156 	public void createOverviewPage(final VerticalLayout panelContent) {
157 		final MenuBar menuBar = new MenuBar();
158 		panelContent.addComponent(menuBar);
159 		panelContent.setComponentAlignment(menuBar, Alignment.TOP_LEFT);
160 		panelContent.setExpandRatio(menuBar, ContentRatio.LARGE);
161 
162 		addSourcesAndIndicatorsToMenu(menuBar.addItem("By Topic",VaadinIcons.LINE_CHART, null), getTopicIndicatorMap());
163 		menuBar.setAutoOpen(true);
164 	}

166 	/**
167 	 * Gets the topic indicator map.
168 	 *
169 	 * @return the topic indicator map
170 	 */
171 	private Map<String, List<ViewWorldbankIndicatorDataCountrySummary>> getTopicIndicatorMap() {
172 		final DataContainer<ViewWorldbankIndicatorDataCountrySummary, WorldbankIndicatorDataCountrySummaryEmbeddedId> indicatorDataCountrSummaryDailyDataContainer = applicationManager
173 				.getDataContainer(ViewWorldbankIndicatorDataCountrySummary.class);
174 
175 		return indicatorDataCountrSummaryDailyDataContainer
176 				.findListByEmbeddedProperty(ViewWorldbankIndicatorDataCountrySummary.class,ViewWorldbankIndicatorDataCountrySummary_.embeddedId,WorldbankIndicatorDataCountrySummaryEmbeddedId.class,WorldbankIndicatorDataCountrySummaryEmbeddedId_.countryId,"SE").parallelStream()
177 				.filter(t -> t != null && t.getSourceValue() != null && t.getEndYear() > DATA_POINTS_FOR_YEAR_ABOVE && t.getDataPoint() > MINIMUM_NUMBER_DATA_POINTS)
178 				.flatMap(t -> Arrays.asList(t.getTopics().split(";")).stream()
179 						.map(topic -> new AbstractMap.SimpleEntry<>(topic, t)))
180 
181 				.collect(Collectors.groupingBy(SimpleEntry::getKey,
182 						Collectors.mapping(SimpleEntry::getValue, Collectors.toList())));
183 	}
