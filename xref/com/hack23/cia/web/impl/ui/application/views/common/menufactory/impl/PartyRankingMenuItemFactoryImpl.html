37  /**
38   * The Class PartyRankingMenuItemFactoryImpl.
39   *
40   * <p>
41   * Responsible for creating and organizing party ranking menu items, charts, and
42   * overview pages. Enhances user navigation to analyze party influence, roles,
43   * and performance within EU, government, committees, and parliament. Leverages
44   * VaadinIcons to visually communicate context and concise descriptions focusing
45   * on political significance.
46   * </p>
47   */
48  @Service
49  public final class PartyRankingMenuItemFactoryImpl extends AbstractMenuItemFactoryImpl implements PartyRankingMenuItemFactory {
50  
51      // Label constants
52      private static final String PART_LEADERS_SCOREBOARD = "Party leaders scoreboard";
53      private static final String ALL_PARTIES_TOTAL_DAYS_SERVED_IN_PARLIAMENT = "All parties total days served in parliament";
54      private static final String CHART_BY_TOPIC_TEXT = "Chart by topic";
55      private static final String CURRENT_PARTIES_ACTIVE_IN_COMMITTEES_HEAD_COUNT = "Current parties in committees";
56      private static final String CURRENT_PARTIES_ACTIVE_IN_MINISTRIES_HEAD_COUNT_TEXT = "Current parties in government";
57      private static final String CURRENT_PARTIES_ACTIVE_IN_PARLIAMENT_HEAD_COUNT = "Current parties in parliament";
58      private static final String OVERVIEW_TEXT = "Overview";
59      private static final String PAGE_VISIT_HISTORY_TEXT = "Page Visit History";
60      private static final String PARTY_BY_TOTAL_MEMBERS_BASED_ON_ROLES_IN_DEPARTMENTS_COMMITTEES_AND_PARLIAMENT =
61              "Party by total members across EU/gov/committees/parliament";
62      private static final String PARTY_RANKING = "Party Ranking";
63      private static final String TOTAL_MEMBERS = "Total members";
64  
65      // Political analyst perspective descriptions (~50 chars)
66      private static final String DESC_ALL_PARTIES_ROLES = "All parties: influence across key institutions.";
67      private static final String DESC_GOVERNMENT_HEADCOUNT = "Governing parties: evaluating institutional strength.";
68      private static final String DESC_COMMITTEES_HEADCOUNT = "Committees: parties' agenda-setting influence.";
69      private static final String DESC_PARLIAMENT_HEADCOUNT = "Parliament: mapping party legislative leverage.";
70      private static final String DESC_LEADERS_SCOREBOARD = "Leaders: comparing party leadership impact.";
71      private static final String DESC_DAYS_SERVED_PARLIAMENT = "All parties: experience shaping policy timelines.";
72      private static final String DESC_PAGE_VISIT_HISTORY = "Visit history: tracking public engagement patterns.";
73  
74      // Page mode commands
75      private static final PageModeMenuCommand COMMAND_CHARTS_ALL_PARTIES = new PageModeMenuCommand(UserViews.PARTY_RANKING_VIEW_NAME, PageMode.CHARTS, ChartIndicators.ALLPARTIES.toString());
76      private static final PageModeMenuCommand COMMAND_CHARTS_CURRENT_COMMITTEES = new PageModeMenuCommand(UserViews.PARTY_RANKING_VIEW_NAME, PageMode.CHARTS, ChartIndicators.CURRENTCOMMITTEES.toString());
77      private static final PageModeMenuCommand COMMAND_CHARTS_CURRENT_GOVERNMENT_PARTIES = new PageModeMenuCommand(UserViews.PARTY_RANKING_VIEW_NAME, PageMode.CHARTS, ChartIndicators.CURRENTGOVERMENTPARTIES.toString());
78      private static final PageModeMenuCommand COMMAND_CHARTS_CURRENT_PARTIES = new PageModeMenuCommand(UserViews.PARTY_RANKING_VIEW_NAME, PageMode.CHARTS, ChartIndicators.CURRENTPARTIES.toString());
79      private static final PageModeMenuCommand COMMAND_CHARTS_CURRENT_PARTIES_LEADER_SCOREBOARD = new PageModeMenuCommand(UserViews.PARTY_RANKING_VIEW_NAME, PageMode.CHARTS, ChartIndicators.CURRENTPARTYLEADERSCORECARD.toString());
80      private static final PageModeMenuCommand COMMAND_DATAGRID = new PageModeMenuCommand(UserViews.PARTY_RANKING_VIEW_NAME, PageMode.DATAGRID);
81      private static final PageModeMenuCommand COMMAND_OVERVIEW = new PageModeMenuCommand(UserViews.PARTY_RANKING_VIEW_NAME, PageMode.OVERVIEW);
82      private static final PageModeMenuCommand COMMAND_PAGEVISITHISTORY = new PageModeMenuCommand(UserViews.PARTY_RANKING_VIEW_NAME, PageMode.PAGEVISITHISTORY);
83  
84      /** Application menu item factory for integrating ranking menus. */
85      @Autowired
86      private ApplicationMenuItemFactory applicationMenuItemFactory;
87  
88      /**
89       * Instantiates a new party ranking menu item factory implementation.
90       */
91      public PartyRankingMenuItemFactoryImpl() {
92          super();
93      }
94  
95      /**
96       * Creates the overview page with concise, politically focused descriptions.
97       * Each button link is accompanied by ~50 character descriptions highlighting
98       * political context rather than technical details.
99       *
100      * @param panelContent the vertical layout to hold the overview content
101      */
102     @Override
103     public void createOverviewPage(final VerticalLayout panelContent) {
104         final ResponsiveRow grid = RowUtil.createGridLayout(panelContent);
105 
106         createButtonLink(grid, PART_LEADERS_SCOREBOARD, VaadinIcons.TROPHY,
107                 COMMAND_CHARTS_CURRENT_PARTIES_LEADER_SCOREBOARD, DESC_LEADERS_SCOREBOARD);
108 
109         createButtonLink(grid, TOTAL_MEMBERS, VaadinIcons.USERS,
110                 COMMAND_DATAGRID, DESC_ALL_PARTIES_ROLES);
111 
112         createButtonLink(grid, CURRENT_PARTIES_ACTIVE_IN_MINISTRIES_HEAD_COUNT_TEXT, VaadinIcons.INSTITUTION,
113                 COMMAND_CHARTS_CURRENT_GOVERNMENT_PARTIES, DESC_GOVERNMENT_HEADCOUNT);
114 
115         createButtonLink(grid, CURRENT_PARTIES_ACTIVE_IN_COMMITTEES_HEAD_COUNT, VaadinIcons.GROUP,
116                 COMMAND_CHARTS_CURRENT_COMMITTEES, DESC_COMMITTEES_HEADCOUNT);
117 
118         createButtonLink(grid, CURRENT_PARTIES_ACTIVE_IN_PARLIAMENT_HEAD_COUNT, VaadinIcons.INSTITUTION,
119                 COMMAND_CHARTS_CURRENT_PARTIES, DESC_PARLIAMENT_HEADCOUNT);
120 
121 
122         createButtonLink(grid, ALL_PARTIES_TOTAL_DAYS_SERVED_IN_PARLIAMENT, VaadinIcons.CLOCK,
123                 COMMAND_CHARTS_ALL_PARTIES, DESC_DAYS_SERVED_PARLIAMENT);
124 
125         createButtonLink(grid, PAGE_VISIT_HISTORY_TEXT, VaadinIcons.CHART,
126                 COMMAND_PAGEVISITHISTORY, DESC_PAGE_VISIT_HISTORY);
127     }
128 
129     /**
130      * Creates the party ranking menu bar and integrates it into the main application
131      * menu. Ensures consistent icon usage and logical grouping of party ranking
132      * topics.
133      *
134      * @param menuBar the main menu bar to which the party ranking items will be added
135      */
136     @Override
137     public void createPartyRankingMenuBar(final MenuBar menuBar) {
138         initApplicationMenuBar(menuBar);
139         applicationMenuItemFactory.addRankingMenu(menuBar);
140 
141         // Suitable icon (e.g., LINE_CHART) for ranking visualization
142         createPartyRankingTopics(menuBar.addItem(PARTY_RANKING, VaadinIcons.LINE_CHART, null));
143     }
144 
145     /**
146      * Adds party ranking topics to the specified menu item. Uses icons that reflect
147      * the content context, such as charts for data visualization, groups or
148      * institutions for organizational structures, and history for past activity.
149      *
150      * @param partynMenuItem the parent menu item to which party ranking topics are added
151      */
152     @Override
153     public void createPartyRankingTopics(final MenuItem partynMenuItem) {
154         partynMenuItem.addItem(OVERVIEW_TEXT, VaadinIcons.DASHBOARD, COMMAND_OVERVIEW);
155 
156         partynMenuItem.addItem(PART_LEADERS_SCOREBOARD, VaadinIcons.TROPHY,COMMAND_CHARTS_CURRENT_PARTIES_LEADER_SCOREBOARD);
157 
158         // Total members using USERS icon for multiple people
159         final MenuItem listItem = partynMenuItem.addItem(TOTAL_MEMBERS, VaadinIcons.USERS, COMMAND_DATAGRID);
160         listItem.setDescription(PARTY_BY_TOTAL_MEMBERS_BASED_ON_ROLES_IN_DEPARTMENTS_COMMITTEES_AND_PARLIAMENT);
161 
162         final MenuItem chartByTopic = partynMenuItem.addItem(CHART_BY_TOPIC_TEXT, VaadinIcons.CHART, null);
163 
164         chartByTopic.addItem(CURRENT_PARTIES_ACTIVE_IN_MINISTRIES_HEAD_COUNT_TEXT, VaadinIcons.INSTITUTION,
165                 COMMAND_CHARTS_CURRENT_GOVERNMENT_PARTIES);
166 
167         chartByTopic.addItem(CURRENT_PARTIES_ACTIVE_IN_COMMITTEES_HEAD_COUNT, VaadinIcons.GROUP,
168                 COMMAND_CHARTS_CURRENT_COMMITTEES);
169 
170         chartByTopic.addItem(CURRENT_PARTIES_ACTIVE_IN_PARLIAMENT_HEAD_COUNT, VaadinIcons.INSTITUTION,
171                 COMMAND_CHARTS_CURRENT_PARTIES);
172 
173         chartByTopic.addItem(ALL_PARTIES_TOTAL_DAYS_SERVED_IN_PARLIAMENT, VaadinIcons.CLOCK,
174                 COMMAND_CHARTS_ALL_PARTIES);
175 
176         partynMenuItem.addItem(PAGE_VISIT_HISTORY_TEXT, VaadinIcons.CHART, COMMAND_PAGEVISITHISTORY);
177     }
