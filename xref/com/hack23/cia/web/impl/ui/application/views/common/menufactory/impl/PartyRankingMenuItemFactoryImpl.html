19  package com.hack23.cia.web.impl.ui.application.views.common.menufactory.impl;
20  
21  import org.springframework.beans.factory.annotation.Autowired;
22  import org.springframework.stereotype.Service;
23  
24  import com.hack23.cia.web.impl.ui.application.views.common.menufactory.api.ApplicationMenuItemFactory;
25  import com.hack23.cia.web.impl.ui.application.views.common.menufactory.api.PartyRankingMenuItemFactory;
26  import com.hack23.cia.web.impl.ui.application.views.common.menufactory.api.pagecommands.PageCommandPartyRankingConstants;
27  import com.hack23.cia.web.impl.ui.application.views.common.rows.RowUtil;
28  import com.jarektoro.responsivelayout.ResponsiveRow;
29  import com.vaadin.icons.VaadinIcons;
30  import com.vaadin.ui.MenuBar;
31  import com.vaadin.ui.MenuBar.MenuItem;
32  import com.vaadin.ui.VerticalLayout;
33  
34  /**
35   * The Class PartyRankingMenuItemFactoryImpl.
36   *
37   * <p>
38   * Responsible for creating and organizing party ranking menu items, charts, and
39   * overview pages. Enhances user navigation to analyze party influence, roles,
40   * and performance within EU, government, committees, and parliament. Leverages
41   * VaadinIcons to visually communicate context and concise descriptions focusing
42   * on political significance.
43   * </p>
44   */
45  @Service
46  public final class PartyRankingMenuItemFactoryImpl extends AbstractMenuItemFactoryImpl
47                  implements PartyRankingMenuItemFactory {
48  
49          /** Application menu item factory for integrating ranking menus. */
50          @Autowired
51          private ApplicationMenuItemFactory applicationMenuItemFactory;
52  
53          /**
54           * Instantiates a new party ranking menu item factory implementation.
55           */
56          public PartyRankingMenuItemFactoryImpl() {
57                  super();
58          }
59  
60          /**
61           * Creates the overview page with concise, politically focused descriptions.
62           * Each button link is accompanied by ~50 character descriptions highlighting
63           * political context rather than technical details.
64           *
65           * @param panelContent the vertical layout to hold the overview content
66           */
67          @Override
68          public void createOverviewPage(final VerticalLayout panelContent) {
69                  final ResponsiveRow grid = RowUtil.createGridLayout(panelContent);
70  
71                  createButtonLink(grid, PART_LEADERS_SCOREBOARD, VaadinIcons.TROPHY,
72                  		COMMAND_PARTY_LEADER_SCOREBOARD, DESC_LEADERS_SCOREBOARD);
73  
74                  createButtonLink(grid, TOTAL_MEMBERS, VaadinIcons.USERS,
75                  		PageCommandPartyRankingConstants.COMMAND_PARTY_RANKING_DATAGRID, DESC_ALL_PARTIES_ROLES);
76  
77                  createButtonLink(grid, CURRENT_PARTIES_ACTIVE_IN_MINISTRIES_HEAD_COUNT_TEXT, VaadinIcons.INSTITUTION,
78                                  COMMAND_CHARTS_CURRENT_GOVERNMENT_PARTIES, DESC_GOVERNMENT_HEADCOUNT);
79  
80                  createButtonLink(grid, CURRENT_PARTIES_ACTIVE_IN_COMMITTEES_HEAD_COUNT, VaadinIcons.GROUP,
81                  		PageCommandPartyRankingConstants.COMMAND_PARTY_RANKING_CHARTS_CURRENT_PARTIES, DESC_COMMITTEES_HEADCOUNT);
82  
83                  createButtonLink(grid, CURRENT_PARTIES_ACTIVE_IN_PARLIAMENT_HEAD_COUNT, VaadinIcons.INSTITUTION,
84                                  COMMAND_CHARTS_CURRENT_GOVERNMENT_PARTIES, DESC_PARLIAMENT_HEADCOUNT);
85  
86                  createButtonLink(grid, ALL_PARTIES_TOTAL_DAYS_SERVED_IN_PARLIAMENT, VaadinIcons.CLOCK,
87                  		PageCommandPartyRankingConstants.COMMAND_PARTY_RANKING_CHARTS_ALL_PARTIES, DESC_DAYS_SERVED_PARLIAMENT);
88  
89                  createButtonLink(grid, RANKING_PAGE_VISIT_TEXT, VaadinIcons.CHART,
90                                  PARTY_RANKING_COMMAND_PAGEVISIT_HISTORY, RANKING_PAGE_VISIT_DESC);
91          }
92  
93          /**
94           * Creates the party ranking menu bar and integrates it into the main
95           * application
96           * menu. Ensures consistent icon usage and logical grouping of party ranking
97           * topics.
98           *
99           * @param menuBar the main menu bar to which the party ranking items will be
100          *                added
101          */
102         @Override
103         public void createPartyRankingMenuBar(final MenuBar menuBar) {
104                 initApplicationMenuBar(menuBar);
105                 applicationMenuItemFactory.addRankingMenu(menuBar);
106 
107                 // Suitable icon (e.g., LINE_CHART) for ranking visualization
108                 createPartyRankingTopics(menuBar.addItem(PARTY_RANKING, VaadinIcons.LINE_CHART, null));
109         }
110 
111         /**
112          * Adds party ranking topics to the specified menu item. Uses icons that reflect
113          * the content context, such as charts for data visualization, groups or
114          * institutions for organizational structures, and history for past activity.
115          *
116          * @param partynMenuItem the parent menu item to which party ranking topics are
117          *                       added
118          */
119         @Override
120         public void createPartyRankingTopics(final MenuItem partynMenuItem) {
121                 // Use specific ranking commands instead of generic ones
122                 partynMenuItem.addItem(RANKING_OVERVIEW_TEXT, VaadinIcons.DASHBOARD,
123                 		PageCommandPartyRankingConstants.COMMAND_PARTY_RANKING_OVERVIEW);
124 
125                 partynMenuItem.addItem(OVERVIEW_TEXT, VaadinIcons.DASHBOARD, PageCommandPartyRankingConstants.COMMAND_PARTY_RANKING_OVERVIEW);
126 
127                 partynMenuItem.addItem(PART_LEADERS_SCOREBOARD, VaadinIcons.TROPHY,
128                 		COMMAND_PARTY_LEADER_SCOREBOARD);
129 
130                 // Total members using USERS icon for multiple people
131                 final MenuItem listItem = partynMenuItem.addItem(TOTAL_MEMBERS, VaadinIcons.USERS, PageCommandPartyRankingConstants.COMMAND_PARTY_RANKING_DATAGRID);
132                 listItem.setDescription(PARTY_BY_TOTAL_MEMBERS_BASED_ON_ROLES_IN_DEPARTMENTS_COMMITTEES_AND_PARLIAMENT);
133 
134                 final MenuItem chartByTopic = partynMenuItem.addItem(CHART_BY_TOPIC_TEXT, VaadinIcons.CHART, null);
135 
136                 chartByTopic.addItem(CURRENT_PARTIES_ACTIVE_IN_MINISTRIES_HEAD_COUNT_TEXT, VaadinIcons.INSTITUTION,
137                 		PageCommandPartyRankingConstants.COMMAND_CHARTS_CURRENT_GOVERNMENT_PARTIES);
138 
139                 chartByTopic.addItem(CURRENT_PARTIES_ACTIVE_IN_COMMITTEES_HEAD_COUNT, VaadinIcons.GROUP,
140                 		PageCommandPartyRankingConstants.COMMAND_PARTY_RANKING_CHARTS_CURRENT_PARTIES);
141 
142                 chartByTopic.addItem(CURRENT_PARTIES_ACTIVE_IN_PARLIAMENT_HEAD_COUNT, VaadinIcons.INSTITUTION,
143                                 COMMAND_CHARTS_CURRENT_GOVERNMENT_PARTIES);
144 
145                 chartByTopic.addItem(ALL_PARTIES_TOTAL_DAYS_SERVED_IN_PARLIAMENT, VaadinIcons.CLOCK,
146                 		PageCommandPartyRankingConstants.COMMAND_PARTY_RANKING_CHARTS_ALL_PARTIES);
147 
148                 partynMenuItem.addItem(RANKING_PAGE_VISIT_TEXT, VaadinIcons.CHART,
149                                 PARTY_RANKING_COMMAND_PAGEVISIT_HISTORY);
150 
151                 partynMenuItem.addItem(RANKING_OVERVIEW_TEXT, VaadinIcons.DASHBOARD,
152                 		PageCommandPartyRankingConstants.COMMAND_PARTY_RANKING_OVERVIEW);
153 
154                 partynMenuItem.addItem(RANKING_PAGE_VISIT_TEXT, VaadinIcons.CHART,
155                                 PARTY_RANKING_COMMAND_PAGEVISIT_HISTORY);
156         }
157 }
