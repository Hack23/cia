119 	@Override
120 	public void createOverviewPage(final VerticalLayout panelContent) {
121 		final ResponsiveRow grid = RowUtil.createGridLayout(panelContent);
122 
123 		createButtonLink(grid,TOTAL_MEMBERS,VaadinIcons.GROUP,
124 				COMMAND_DATAGRID, "Scoreboard all parties current assignments and roles in eu,government,committes and parliament");
125 		createButtonLink(grid,CURRENT_PARTIES_ACTIVE_IN_MINISTRIES_HEAD_COUNT_TEXT,VaadinIcons.GROUP,
126 				COMMAND_CHARTS_CURRENT_GOVERNMENT_PARTIES, "Chart over the headcount by party in current goverment");
127 
128 		createButtonLink(grid,CURRENT_PARTIES_ACTIVE_IN_COMMITTEES_HEAD_COUNT,VaadinIcons.GROUP,
129 				COMMAND_CHARTS_CURRENT_COMMITTEES, "Chart over the headcount by party in current committees");
130 		createButtonLink(grid,CURRENT_PARTIES_ACTIVE_IN_PARLIAMENT_HEAD_COUNT,VaadinIcons.GROUP,
131 				COMMAND_CHARTS_CURRENT_PARTIES, "Chart over the headcount by party in current parliament");
132 
133 		createButtonLink(grid,PART_LEADERS_SCOREBOARD,VaadinIcons.GROUP,
134 				COMMAND_CHARTS_CURRENT_PARTIES_LEADER_SCOREBOARD, "Party leaders scoreboard");
135 
136 
137 		createButtonLink(grid,ALL_PARTIES_TOTAL_DAYS_SERVED_IN_PARLIAMENT,VaadinIcons.GROUP,
138 				COMMAND_CHARTS_ALL_PARTIES, "Chart all parties total politician days serverd in parliament");
139 
140 		createButtonLink(grid,PAGE_VISIT_HISTORY_TEXT, VaadinIcons.GROUP,
141 				COMMAND_PAGEVISITHISTORY, "View history of page visit for this page.");

146 	@Override
147 	public void createPartyRankingMenuBar(final MenuBar menuBar) {
148 		initApplicationMenuBar(menuBar);
149 
150 		applicationMenuItemFactory.addRankingMenu(menuBar);
151 
152 		createPartyRankingTopics(menuBar.addItem(PARTY_RANKING, VaadinIcons.GROUP,null));
153 
154 	}

157 	@Override
158 	public void createPartyRankingTopics(final MenuItem partynMenuItem) {
159 
160 		partynMenuItem.addItem(OVERVIEW_TEXT, VaadinIcons.GROUP,
161 				COMMAND_OVERVIEW);
162 
163 		final MenuItem listByTopic = partynMenuItem.addItem(RANKING_LIST_BY_TOPIC_TEXT, VaadinIcons.GROUP, null);
164 
165 		final MenuItem listItem = listByTopic.addItem(TOTAL_MEMBERS,VaadinIcons.GROUP,
166 				COMMAND_DATAGRID);
167 		listItem.setDescription(PARTY_BY_TOTAL_MEMBERS_BASED_ON_ROLES_IN_DEPARTMENTS_COMMITTEES_AND_PARLIAMENT);
168 
169 		final MenuItem chartByTopic = partynMenuItem.addItem(CHART_BY_TOPIC_TEXT, VaadinIcons.GROUP, null);
170 
171 
172 		chartByTopic.addItem(CURRENT_PARTIES_ACTIVE_IN_MINISTRIES_HEAD_COUNT_TEXT,VaadinIcons.GROUP,
173 				COMMAND_CHARTS_CURRENT_GOVERNMENT_PARTIES);
174 
175 		chartByTopic.addItem(CURRENT_PARTIES_ACTIVE_IN_COMMITTEES_HEAD_COUNT,VaadinIcons.GROUP,
176 				COMMAND_CHARTS_CURRENT_COMMITTEES);
177 
178 		chartByTopic.addItem(CURRENT_PARTIES_ACTIVE_IN_PARLIAMENT_HEAD_COUNT,VaadinIcons.GROUP,
179 				COMMAND_CHARTS_CURRENT_PARTIES);
180 		chartByTopic.addItem(ALL_PARTIES_TOTAL_DAYS_SERVED_IN_PARLIAMENT,VaadinIcons.GROUP,
181 				COMMAND_CHARTS_ALL_PARTIES);
182 
183 		partynMenuItem.addItem(PAGE_VISIT_HISTORY_TEXT, VaadinIcons.GROUP,
184 				COMMAND_PAGEVISITHISTORY);
185 	}
