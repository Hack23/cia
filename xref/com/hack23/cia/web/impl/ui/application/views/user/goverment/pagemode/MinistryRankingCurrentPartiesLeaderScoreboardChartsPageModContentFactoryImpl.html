62  	@Secured({ "ROLE_ANONYMOUS", "ROLE_USER", "ROLE_ADMIN" })
63  	@Override
64  	public Layout createContent(final String parameters, final MenuBar menuBar, final Panel panel) {
65  		panel.setSizeFull();
66  
67  		final VerticalLayout panelContent = createPanelContent();
68  		panelContent.setSizeFull();
69  		panel.setContent(panelContent);
70  
71  		getMinistryRankingMenuItemFactory().createMinistryRankingMenuBar(menuBar);
72  
73  		final String pageId = getPageId(parameters);
74  
75  		createPageHeader(panel, panelContent, "Ministry Rankings", "Leader Scoreboard",
76  				"Visual representation of ministry leaders and their performance.");
77  
78  		final ResponsiveRow row = RowUtil.createGridLayout(panelContent);
79  		row.setSizeFull();
80  
81  		final List<ViewRiksdagenGovermentRoleMember> activeGovMembers = loadActiveGovernmentRoleMembers();
82  		final Map<String, List<ViewRiksdagenPolitician>> activePoliticianMap = loadActivePoliticiansByPersonId();
83  
84  		final Map<String, Boolean> partyLeaderMap = computePartyLeaders(activePoliticianMap.keySet());
85  
86  		// Sort roles
87  		activeGovMembers.sort((a, b) -> {
88  			final boolean aLeader = partyLeaderMap.getOrDefault(a.getPersonId(), false);
89  			final boolean bLeader = partyLeaderMap.getOrDefault(b.getPersonId(), false);
90  			final int aPriority = getRolePriority(a.getRoleCode(), aLeader);
91  			final int bPriority = getRolePriority(b.getRoleCode(), bLeader);
92  			return Integer.compare(aPriority, bPriority);
93  		});
94  
95  		final Map<Integer, List<GovernmentBodyAnnualSummary>> dataMap = esvApi.getData();
96  		final List<GovernmentBodyAnnualSummary> currentYearGovernmentBodies = dataMap.get(CURRENT_YEAR);
97  		final Map<String, List<GovernmentBodyAnnualSummary>> governmentBodyByMinistry = currentYearGovernmentBodies
98  				.stream().collect(Collectors.groupingBy(GovernmentBodyAnnualSummary::getMinistry));
99  
100 		final Map<String, List<GovernmentBodyAnnualOutcomeSummary>> reportByMinistry = esvApi
101 				.getGovernmentBodyReportByMinistry();
102 
103 		for (final ViewRiksdagenGovermentRoleMember govMember : activeGovMembers) {
104 			final ViewRiksdagenPolitician politician = activePoliticianMap.get(govMember.getPersonId()).get(0);
105 			final Panel cardPanel = createBaseballStyleCard(govMember, politician, governmentBodyByMinistry,
106 					reportByMinistry);
107 
108 			// Responsive column rules
109 			row.addColumn().withDisplayRules(DISPLAY_SIZE_XS_DEVICE, DISPLAYS_SIZE_XM_DEVICE, DISPLAY_SIZE_MD_DEVICE,
110 					DISPLAY_SIZE_LG_DEVICE).withComponent(cardPanel);
111 		}
112 
113 		getPageActionEventHelper().createPageEvent(ViewAction.VISIT_MINISTRY_RANKING_VIEW, ApplicationEventGroup.USER,
114 				NAME, parameters, pageId);
115 
116 		return panelContent;
117 	}

134 	private Map<String, Boolean> computePartyLeaders(Iterable<String> personIds) {
135 		final DataContainer<ViewRiksdagenPartyRoleMember, String> partyRoleMemberDataContainer = getApplicationManager()
136 				.getDataContainer(ViewRiksdagenPartyRoleMember.class);
137 
138 		final Map<String, Boolean> result = new HashMap<>();
139 		for (final String personId : personIds) {
140 			final List<ViewRiksdagenPartyRoleMember> roles = partyRoleMemberDataContainer.findListByProperty(
141 					new Object[] { personId, Boolean.TRUE }, ViewRiksdagenPartyRoleMember_.personId,
142 					ViewRiksdagenPartyRoleMember_.active);
143 
144 			final boolean isLeader = roles.stream().anyMatch(
145 					role -> role.getRoleCode() != null && "Partiledare".equalsIgnoreCase(role.getRoleCode().trim()));
146 
147 			result.put(personId, isLeader);
148 		}
149 		return result;
150 	}

152 	private int getRolePriority(String role, boolean isPartyLeader) {
153 		final String roleNormalized = role.toLowerCase(Locale.ROOT).trim();
154 		if ("statsminister".equals(roleNormalized)) {
155 			return 1;
156 		} else if (isPartyLeader) {
157 			return 2;
158 		} else if (roleNormalized.endsWith("minister")) {
159 			return 3;
160 		} else if ("statsr√•d".equalsIgnoreCase(roleNormalized)) {
161 			return 4;
162 		} else {
163 			return 5;
164 		}
165 	}

296 	private void addMinistryRoleSummary(final VerticalLayout cardLayout,
297 			final ViewRiksdagenGovermentRoleMember govMember,
298 			final Map<String, List<GovernmentBodyAnnualSummary>> governmentBodyByMinistry,
299 			final Map<String, List<GovernmentBodyAnnualOutcomeSummary>> reportByMinistry) {
300 
301 		cardLayout.addComponent(getPageLinkFactory().addMinistryPageLink(govMember.getDetail()));
302 
303 		final List<GovernmentBodyAnnualSummary> ministryBodies = governmentBodyByMinistry.get(govMember.getDetail());
304 		final int totalHeadCount = ministryBodies.stream().mapToInt(GovernmentBodyAnnualSummary::getAnnualWorkHeadCount)
305 				.sum();
306 
307 		// Bodies count
308 		final int bodyCount = ministryBodies.size();
309 		cardLayout.addComponent(createMetricRow(VaadinIcons.GROUP,
310 				getPageLinkFactory().addMinistryGovermentBodiesPageLink(govMember.getDetail()),
311 				"Number of government bodies", String.valueOf(bodyCount) // display the count separately
312 		));
313 
314 		// Headcount
315 		cardLayout.addComponent(createMetricRow(VaadinIcons.USER,
316 				getPageLinkFactory().addMinistryGovermentBodiesHeadcountPageLink(govMember.getDetail()),
317 				"Total headcount of government bodies", String.valueOf(totalHeadCount) // display the headcount
318 																						// separately
319 		));
320 
321 		final List<GovernmentBodyAnnualOutcomeSummary> outcomeSummaries = reportByMinistry.get(govMember.getDetail());
322 		double currentYearIncome = 0;
323 		double currentYearSpending = 0;
324 		if (outcomeSummaries != null) {
325 			final Map<Integer, Double> annualIncome = outcomeSummaries.stream()
326 					.filter(t -> t.getDescriptionFields().get(INKOMSTTITELGRUPPSNAMN) != null)
327 					.collect(Collectors.groupingBy(GovernmentBodyAnnualOutcomeSummary::getYear,
328 							Collectors.summingDouble(GovernmentBodyAnnualOutcomeSummary::getYearTotal)));
329 
330 			final Map<Integer, Double> annualSpending = outcomeSummaries.stream()
331 					.filter(t -> t.getDescriptionFields().get(EXPENDITURE_GROUP_NAME) != null)
332 					.collect(Collectors.groupingBy(GovernmentBodyAnnualOutcomeSummary::getYear,
333 							Collectors.summingDouble(GovernmentBodyAnnualOutcomeSummary::getYearTotal)));
334 
335 			if (annualIncome.get(CURRENT_YEAR) != null) {
336 				currentYearIncome = annualIncome.get(CURRENT_YEAR) / 1000;
337 			}
338 
339 			if (annualSpending.get(CURRENT_YEAR) != null) {
340 				currentYearSpending = annualSpending.get(CURRENT_YEAR) / 1000;
341 			}
342 		}
343 
344 		// Income
345 		final String incomeStr = String.format(Locale.ROOT,"%.2f B SEK", currentYearIncome);
346 		cardLayout.addComponent(createMetricRow(VaadinIcons.ARROW_UP,
347 				getPageLinkFactory().addMinistryGovermentBodiesIncomePageLink(govMember.getDetail()),
348 				"Yearly Income (B SEK)", incomeStr // display income outside link
349 		));
350 
351 		// Spending
352 		final String spendingStr = String.format(Locale.ROOT,"%.2f B SEK", currentYearSpending);
353 		cardLayout.addComponent(
354 				createMetricRow(VaadinIcons.ARROW_DOWN, getPageLinkFactory().addMinistrGovermentBodiesSpendingPageLink(
355 						govMember.getDetail()), "Yearly Spending (B SEK)", spendingStr // display
356 																											// spending
357 																											// outside
358 																											// link
359 				));
360 	}

362 	private HorizontalLayout createMetricRow(VaadinIcons icon, com.vaadin.ui.Component linkComponent,
363 			String description, String valueText) {
364 		final HorizontalLayout layout = new HorizontalLayout();
365 		layout.setSpacing(true);
366 		layout.addStyleName("metric-label");
367 		layout.setWidthUndefined();
368 
369 		final Label iconLabel = new Label(icon.getHtml(), ContentMode.HTML);
370 		iconLabel.setDescription(description);
371 
372 		// Value displayed outside of the link
373 		Label valueLabel = null;
374 		if (valueText != null && !valueText.isEmpty()) {
375 			valueLabel = new Label(valueText);
376 			valueLabel.addStyleName("metric-value");
377 		}
378 
379 		layout.addComponent(iconLabel);
380 		layout.addComponent(linkComponent);
381 		if (valueLabel != null) {
382 			layout.addComponent(valueLabel);
383 		}
384 
385 		return layout;
386 	}
