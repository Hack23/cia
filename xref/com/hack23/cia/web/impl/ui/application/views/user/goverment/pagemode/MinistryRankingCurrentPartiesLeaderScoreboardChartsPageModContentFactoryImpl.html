95  		panel.setSizeFull();
96  
97  		final VerticalLayout panelContent = createPanelContent();
98  		panelContent.setSizeFull();
99  		panel.setContent(panelContent);
100 
101 		getMinistryRankingMenuItemFactory().createMinistryRankingMenuBar(menuBar);
102 
103 		final String pageId = getPageId(parameters);
104 
105 		createPageHeader(panel, panelContent, "Ministry Rankings", "Leader Scoreboard",
106 				"Visual representation of ministry leaders and their performance.");
107 
108 		final ResponsiveRow row = RowUtil.createGridLayout(panelContent);
109 		row.setSizeFull();
110 
111 		final List<ViewRiksdagenGovermentRoleMember> activeGovMembers = loadActiveGovernmentRoleMembers();
112 		final Map<String, List<ViewRiksdagenPolitician>> activePoliticianMap = loadActivePoliticiansByPersonId();
113 
114 		final Map<String, Boolean> partyLeaderMap = computePartyLeaders(activePoliticianMap.keySet());
115 
116 		// Sort roles
117 		activeGovMembers.sort((a, b) -> {
118 			final boolean aLeader = partyLeaderMap.getOrDefault(a.getPersonId(), false);
119 			final boolean bLeader = partyLeaderMap.getOrDefault(b.getPersonId(), false);
120 			final int aPriority = getRolePriority(a.getRoleCode(), aLeader);
121 			final int bPriority = getRolePriority(b.getRoleCode(), bLeader);
122 			return Integer.compare(aPriority, bPriority);
123 		});
124 
125 		final Map<Integer, List<GovernmentBodyAnnualSummary>> dataMap = esvApi.getData();
126 		final List<GovernmentBodyAnnualSummary> currentYearGovernmentBodies = dataMap.get(CURRENT_YEAR);
127 		final Map<String, List<GovernmentBodyAnnualSummary>> governmentBodyByMinistry = currentYearGovernmentBodies
128 				.stream().collect(Collectors.groupingBy(GovernmentBodyAnnualSummary::getMinistry));
129 
130 		final Map<String, List<GovernmentBodyAnnualOutcomeSummary>> reportByMinistry = esvApi
131 				.getGovernmentBodyReportByMinistry();
132 
133 		for (final ViewRiksdagenGovermentRoleMember govMember : activeGovMembers) {
134 			final ViewRiksdagenPolitician politician = activePoliticianMap.get(govMember.getPersonId()).get(0);
135 			final ViewRiksdagenPoliticianBallotSummary ballotSummary = getApplicationManager()
136 					.getDataContainer(ViewRiksdagenPoliticianBallotSummary.class).load(govMember.getPersonId());

143 		}
144 
145 		getPageActionEventHelper().createPageEvent(ViewAction.VISIT_MINISTRY_RANKING_VIEW, ApplicationEventGroup.USER,
146 				NAME, parameters, pageId);
147 
148 		return panelContent;
149 	}

176 	/**
177 	 * Compute party leaders.
178 	 *
179 	 * @param personIds the person ids
180 	 * @return the map
181 	 */
182 	private Map<String, Boolean> computePartyLeaders(Iterable<String> personIds) {
183 		final DataContainer<ViewRiksdagenPartyRoleMember, String> partyRoleMemberDataContainer = getApplicationManager()
184 				.getDataContainer(ViewRiksdagenPartyRoleMember.class);
185 
186 		final Map<String, Boolean> result = new HashMap<>();
187 		for (final String personId : personIds) {
188 			final List<ViewRiksdagenPartyRoleMember> roles = partyRoleMemberDataContainer.findListByProperty(
189 					new Object[] { personId, Boolean.TRUE }, ViewRiksdagenPartyRoleMember_.personId,
190 					ViewRiksdagenPartyRoleMember_.active);
191 
192 			final boolean isLeader = roles.stream().anyMatch(
193 					role -> role.getRoleCode() != null && "Partiledare".equalsIgnoreCase(role.getRoleCode().trim()));
194 
195 			result.put(personId, isLeader);
196 		}
197 		return result;
198 	}

201 	 * Gets the role priority.
202 	 *
203 	 * @param role the role
204 	 * @param isPartyLeader the is party leader
205 	 * @return the role priority
206 	 */
207 	private int getRolePriority(String role, boolean isPartyLeader) {
208 		final String roleNormalized = role.toLowerCase(Locale.ROOT).trim();
209 		if ("statsminister".equals(roleNormalized)) {
210 			return 1;
211 		} else if (isPartyLeader) {
212 			return 2;
213 		} else if (roleNormalized.endsWith("minister")) {
214 			return 3;
215 		} else if ("statsråd".equalsIgnoreCase(roleNormalized)) {
216 			return 4;
217 		} else {
218 			return 5;
219 		}
220 	}

397 	/**
398 	 * Adds the ministry role summary.
399 	 *
400 	 * @param cardLayout the card layout
401 	 * @param govMember the gov member
402 	 * @param governmentBodyByMinistry the government body by ministry
403 	 * @param reportByMinistry the report by ministry
404 	 */
405 	private void addMinistryRoleSummary(final VerticalLayout cardLayout,
406 			final ViewRiksdagenGovermentRoleMember govMember,
407 			final Map<String, List<GovernmentBodyAnnualSummary>> governmentBodyByMinistry,
408 			final Map<String, List<GovernmentBodyAnnualOutcomeSummary>> reportByMinistry) {
409 
410 		cardLayout.addComponent(getPageLinkFactory().addMinistryPageLink(govMember.getDetail()));
411 
412 		final List<GovernmentBodyAnnualSummary> ministryBodies = governmentBodyByMinistry.get(govMember.getDetail());
413 		final int totalHeadCount = ministryBodies.stream().mapToInt(GovernmentBodyAnnualSummary::getAnnualWorkHeadCount)
414 				.sum();
415 
416 		// Bodies count
417 		final int bodyCount = ministryBodies.size();
418 		cardLayout.addComponent(createMetricRow(VaadinIcons.GROUP,
419 				getPageLinkFactory().addMinistryGovermentBodiesPageLink(govMember.getDetail()),
420 				"Number of government bodies", String.valueOf(bodyCount) // display the count separately
421 		));
422 
423 		// Headcount
424 		cardLayout.addComponent(createMetricRow(VaadinIcons.USER,
425 				getPageLinkFactory().addMinistryGovermentBodiesHeadcountPageLink(govMember.getDetail()),
426 				"Total headcount of government bodies", String.valueOf(totalHeadCount) // display the headcount
427 																						// separately
428 		));
429 
430 		final List<GovernmentBodyAnnualOutcomeSummary> outcomeSummaries = reportByMinistry.get(govMember.getDetail());
431 		double currentYearIncome = 0;
432 		double currentYearSpending = 0;
433 		if (outcomeSummaries != null) {
434 			final Map<Integer, Double> annualIncome = outcomeSummaries.stream()
435 					.filter(t -> t.getDescriptionFields().get(INKOMSTTITELGRUPPSNAMN) != null)
436 					.collect(Collectors.groupingBy(GovernmentBodyAnnualOutcomeSummary::getYear,
437 							Collectors.summingDouble(GovernmentBodyAnnualOutcomeSummary::getYearTotal)));
438 
439 			final Map<Integer, Double> annualSpending = outcomeSummaries.stream()
440 					.filter(t -> t.getDescriptionFields().get(EXPENDITURE_GROUP_NAME) != null)
441 					.collect(Collectors.groupingBy(GovernmentBodyAnnualOutcomeSummary::getYear,
442 							Collectors.summingDouble(GovernmentBodyAnnualOutcomeSummary::getYearTotal)));
443 
444 			if (annualIncome.get(CURRENT_YEAR) != null) {
445 				currentYearIncome = annualIncome.get(CURRENT_YEAR) / 1000;
446 			}
447 
448 			if (annualSpending.get(CURRENT_YEAR) != null) {
449 				currentYearSpending = annualSpending.get(CURRENT_YEAR) / 1000;
450 			}
451 		}
452 
453 		// Income
454 		final String incomeStr = String.format(Locale.ENGLISH,"%.2f B SEK", currentYearIncome);
455 		cardLayout.addComponent(createMetricRow(VaadinIcons.ARROW_UP,
456 				getPageLinkFactory().addMinistryGovermentBodiesIncomePageLink(govMember.getDetail()),
457 				"Yearly Income (B SEK)", incomeStr // display income outside link
458 		));
459 
460 		// Spending
461 		final String spendingStr = String.format(Locale.ENGLISH,"%.2f B SEK", currentYearSpending);
462 		cardLayout.addComponent(
463 				createMetricRow(VaadinIcons.ARROW_DOWN, getPageLinkFactory().addMinistrGovermentBodiesSpendingPageLink(
464 						govMember.getDetail()), "Yearly Spending (B SEK)", spendingStr // display
465 																											// spending
466 																											// outside
467 																											// link
468 				));
469 	}

612 	/**
613 	 * Creates the info row.
614 	 *
615 	 * @param caption the caption
616 	 * @param value the value
617 	 * @param icon the icon
618 	 * @param tooltip the tooltip
619 	 * @return the horizontal layout
620 	 */
621 	private HorizontalLayout createInfoRow(final String caption, final String value, VaadinIcons icon,
622 			final String tooltip) {
623 		final HorizontalLayout layout = new HorizontalLayout();
624 		layout.setSpacing(true);
625 		layout.addStyleName("metric-label");
626 		layout.setWidthUndefined();
627 
628 		Label iconLabel = null;
629 		if (icon != null) {
630 			iconLabel = new Label(icon.getHtml(), ContentMode.HTML);
631 			iconLabel.setDescription(tooltip);
632 			iconLabel.addStyleName("card-info-icon");
633 			layout.addComponent(iconLabel);
634 		}
635 
636 		final Label captionLabel = new Label(caption);
637 		captionLabel.addStyleName("card-info-caption");
638 		if (tooltip != null && !tooltip.isEmpty()) {
639 			captionLabel.setDescription(tooltip);
640 		}
641 
642 		final Label valueLabel = new Label(value != null ? value : "");
643 		valueLabel.addStyleName("card-info-value");
644 
645 		layout.addComponents(captionLabel, valueLabel);
646 		return layout;
647 	}
