97  	@Secured({ "ROLE_ANONYMOUS", "ROLE_USER", "ROLE_ADMIN" })
98  	@Override
99  	public Layout createContent(final String parameters, final MenuBar menuBar, final Panel panel) {
100 		final VerticalLayout panelContent = createPanelContent();
101 
102 		getMinistryRankingMenuItemFactory().createMinistryRankingMenuBar(menuBar);
103 
104 		final String pageId = getPageId(parameters);
105 
106 		LabelFactory.createHeader2Label(panelContent, "Dashboard Government members");
107 
108 		final Label descriptionLabel = new Label(
109 				"Visualize political activity in Sweden, present key performance indicators and metadata for the actors on national level");
110 		descriptionLabel.addStyleName("itembox");
111 		Responsive.makeResponsive(descriptionLabel);
112 		descriptionLabel.setWidth(100, Unit.PERCENTAGE);
113 		panelContent.addComponent(descriptionLabel);
114 
115 		final ResponsiveRow row = RowUtil.createGridLayout(panelContent);
116 
117 		final DataContainer<ViewRiksdagenGovermentRoleMember, String> govermentRoleMemberDataContainer = getApplicationManager()
118 				.getDataContainer(ViewRiksdagenGovermentRoleMember.class);
119 
120 		final List<ViewRiksdagenGovermentRoleMember> listMinistryMembers = govermentRoleMemberDataContainer
121 				.findListByProperty(new Object[] { Boolean.TRUE }, ViewRiksdagenGovermentRoleMember_.active);
122 
123 		final DataContainer<ViewRiksdagenPolitician, String> politicianDataContainer = getApplicationManager()
124 				.getDataContainer(ViewRiksdagenPolitician.class);
125 
126 		final List<ViewRiksdagenPolitician> parliamentDataList = politicianDataContainer
127 				.findListByProperty(new Object[] { Boolean.TRUE }, ViewRiksdagenPolitician_.activeGovernment);
128 		final Map<String, List<ViewRiksdagenPolitician>> parliamentMap = parliamentDataList.stream()
129 				.collect(Collectors.groupingBy(ViewRiksdagenPolitician::getPersonId));
130 
131 		final Map<Integer, List<GovernmentBodyAnnualSummary>> dataMap = esvApi.getData();
132 		final List<GovernmentBodyAnnualSummary> headCountGovermentBodies = dataMap.get(2024);
133 		final Map<String, List<GovernmentBodyAnnualSummary>> governmentBodyMinistryMap = headCountGovermentBodies
134 				.stream().collect(Collectors.groupingBy(GovernmentBodyAnnualSummary::getMinistry));
135 
136 		for (final ViewRiksdagenGovermentRoleMember viewRiksdagenGovermentRoleMember : listMinistryMembers) {
137 			createDashboardGovernmentMember(row, governmentBodyMinistryMap, viewRiksdagenGovermentRoleMember,
138 					parliamentMap.get(viewRiksdagenGovermentRoleMember.getPersonId()).get(0));
139 
140 		}
141 
142 		panel.setCaption(NAME + "::" + CHARTS + parameters);
143 
144 		getPageActionEventHelper().createPageEvent(ViewAction.VISIT_MINISTRY_RANKING_VIEW, ApplicationEventGroup.USER,
145 				NAME, parameters, pageId);
146 
147 		return panelContent;
148 
149 	}

151 	private void createDashboardGovernmentMember(final ResponsiveRow row,
152 			final Map<String, List<GovernmentBodyAnnualSummary>> governmentBodyMinistryMap,
153 			final ViewRiksdagenGovermentRoleMember viewRiksdagenGovermentRoleMember,
154 			final ViewRiksdagenPolitician viewRiksdagenPolitician) {
155 
156 		final CssLayout layout = new CssLayout();
157 		layout.addStyleName("v-layout-content-overview-dashboard-panel-level2");
158 		Responsive.makeResponsive(layout);
159 		layout.setSizeUndefined();
160 
161 
162 		final DataContainer<ViewRiksdagenPartyRoleMember, String> partyRoleMemberDataContainer = getApplicationManager()
163 				.getDataContainer(ViewRiksdagenPartyRoleMember.class);
164 
165 
166 		final Label titleLabel = new Label(viewRiksdagenGovermentRoleMember.getRoleCode() + " "
167 				+ viewRiksdagenGovermentRoleMember.getFirstName() + " " + viewRiksdagenGovermentRoleMember.getLastName()
168 				+ " (" + viewRiksdagenGovermentRoleMember.getParty() +")");
169 		layout.addComponent(getPageLinkFactory().createPoliticianPageLink(viewRiksdagenPolitician));
170 		Responsive.makeResponsive(titleLabel);
171 		titleLabel.addStyleName("title");
172 		titleLabel.setWidth(100, Unit.PERCENTAGE);
173 		layout.addComponent(titleLabel);
174 
175 
176 		final List<ViewRiksdagenPartyRoleMember> partyRole = partyRoleMemberDataContainer.findListByProperty(new Object[] { viewRiksdagenPolitician.getPersonId(), Boolean.TRUE },ViewRiksdagenPartyRoleMember_.personId, ViewRiksdagenPartyRoleMember_.active);
177 
178 		if (partyRole.size()> 0) {
179 			final Label partyRoleLabel = new Label(partyRole.get(0).getRoleCode() + " (" + viewRiksdagenGovermentRoleMember.getParty() +")  since " +partyRole.get(0).getFromDate() );
180 			Responsive.makeResponsive(partyRoleLabel);
181 			partyRoleLabel.addStyleName("title");
182 			partyRoleLabel.setWidth(100, Unit.PERCENTAGE);
183 			layout.addComponent(partyRoleLabel);
184 
185 		}
186 
187 
188 		final Map<String, List<GovernmentBodyAnnualOutcomeSummary>> reportByMinistry = esvApi.getGovernmentBodyReportByMinistry();
189 
190 		createPoliticaExperienceSummary(governmentBodyMinistryMap, viewRiksdagenGovermentRoleMember, viewRiksdagenPolitician, layout);
191 		createMinistryRoleSummary(reportByMinistry, governmentBodyMinistryMap, viewRiksdagenGovermentRoleMember, viewRiksdagenPolitician, layout);
192 
193 
194 		row.addColumn().withDisplayRules(DISPLAY_SIZE_XS_DEVICE, DISPLAYS_SIZE_XM_DEVICE, DISPLAY_SIZE_MD_DEVICE,
195 				DISPLAY_SIZE_LG_DEVICE).withComponent(layout);
196 	}

198 	private void createMinistryRoleSummary(final Map<String, List<GovernmentBodyAnnualOutcomeSummary>> reportByMinistry, final Map<String, List<GovernmentBodyAnnualSummary>> governmentBodyMinistryMap,
199 			final ViewRiksdagenGovermentRoleMember viewRiksdagenGovermentRoleMember,
200 			final ViewRiksdagenPolitician viewRiksdagenPolitician, final CssLayout layout) {
201 
202 
203 		layout.addComponent(getPageLinkFactory().addMinistryPageLink(viewRiksdagenGovermentRoleMember.getDetail()));
204 
205 		final List<GovernmentBodyAnnualSummary> governentBodies = governmentBodyMinistryMap
206 				.get(viewRiksdagenGovermentRoleMember.getDetail());
207 
208 		layout.addComponent(getPageLinkFactory().addMinistryGovermentBodiesPageLink(viewRiksdagenGovermentRoleMember.getDetail(),governentBodies.size()));
209 		layout.addComponent(getPageLinkFactory().addMinistryGovermentBodiesHeadcountPageLink(viewRiksdagenGovermentRoleMember.getDetail(),governentBodies.stream().mapToInt(GovernmentBodyAnnualSummary::getAnnualWorkHeadCount).sum()));
210 
211 		final List<GovernmentBodyAnnualOutcomeSummary> listOutCome = reportByMinistry.get(viewRiksdagenGovermentRoleMember.getDetail());
212 		final Map<Integer, Double> annualIncomeSummaryMap = listOutCome.stream().filter(t -> t.getDescriptionFields().get(INKOMSTTITELGRUPPSNAMN) != null).collect(Collectors.groupingBy(GovernmentBodyAnnualOutcomeSummary::getYear,Collectors.summingDouble(GovernmentBodyAnnualOutcomeSummary::getYearTotal)));
213 		final Map<Integer, Double> annualSpendingSummaryMap = listOutCome.stream().filter(t -> t.getDescriptionFields().get(EXPENDITURE_GROUP_NAME) != null).collect(Collectors.groupingBy(GovernmentBodyAnnualOutcomeSummary::getYear,Collectors.summingDouble(GovernmentBodyAnnualOutcomeSummary::getYearTotal)));
214 
215 		layout.addComponent(getPageLinkFactory().addMinistryGovermentBodiesIncomePageLink(viewRiksdagenGovermentRoleMember.getDetail(),annualIncomeSummaryMap.get(2024) /1000));
216 		layout.addComponent(getPageLinkFactory().addMinistrGovermentBodiesSpendingPageLink(viewRiksdagenGovermentRoleMember.getDetail(),annualSpendingSummaryMap.get(2024)/1000));
217 
218 
219 	}

222 	private void createPoliticaExperienceSummary(final Map<String, List<GovernmentBodyAnnualSummary>> governmentBodyMinistryMap,
223 			final ViewRiksdagenGovermentRoleMember viewRiksdagenGovermentRoleMember,
224 			final ViewRiksdagenPolitician viewRiksdagenPolitician, final CssLayout layout) {
225 
226 		final Label ministryDaysServedLabel = new Label("Hold position for " + viewRiksdagenGovermentRoleMember.getTotalDaysServed() + " (days) in " + viewRiksdagenGovermentRoleMember.getDetail());
227 		Responsive.makeResponsive(ministryDaysServedLabel);
228 		ministryDaysServedLabel.setWidth(100, Unit.PERCENTAGE);
229 		layout.addComponent(ministryDaysServedLabel);
230 
231 		{
232 		final Label experienceLabel = new Label("Political Experience:" + (viewRiksdagenPolitician.getTotalDaysServedGovernment() / 365) + " year Government, " + (viewRiksdagenPolitician.getTotalDaysServedParty() / 365) + " year Party ," + (viewRiksdagenPolitician.getTotalDaysServedParliament() / 365) +" year parliament");
233 		Responsive.makeResponsive(experienceLabel);
234 		experienceLabel.setWidth(100, Unit.PERCENTAGE);
235 		layout.addComponent(experienceLabel);
236 		}
237 	}

242 		return NAME.equals(page) && StringUtils.contains(parameters, PageMode.CHARTS.toString())
243 				&& parameters.contains(ChartIndicators.CURRENTMINISTRIESLEADERSCORECARD.toString());
