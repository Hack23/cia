102 	@Secured({ "ROLE_ANONYMOUS", "ROLE_USER", "ROLE_ADMIN" })
103 	@Override
104 	public Layout createContent(final String parameters, final MenuBar menuBar, final Panel panel) {
105 		final VerticalLayout panelContent = createPanelContent();
106 
107 		getMinistryRankingMenuItemFactory().createMinistryRankingMenuBar(menuBar);
108 
109 		final String pageId = getPageId(parameters);
110 
111 		LabelFactory.createHeader2Label(panelContent, "Dashboard Government members");
112 
113 		final Label descriptionLabel = new Label(
114 				"Visualize political activity in Sweden, present key performance indicators and metadata for the actors on national level");
115 		descriptionLabel.addStyleName("itembox");
116 		Responsive.makeResponsive(descriptionLabel);
117 		descriptionLabel.setWidth(100, Unit.PERCENTAGE);
118 		panelContent.addComponent(descriptionLabel);
119 
120 		final ResponsiveRow row = RowUtil.createGridLayout(panelContent);
121 
122 		final DataContainer<ViewRiksdagenGovermentRoleMember, String> govermentRoleMemberDataContainer = getApplicationManager()
123 				.getDataContainer(ViewRiksdagenGovermentRoleMember.class);
124 
125 		final List<ViewRiksdagenGovermentRoleMember> listMinistryMembers = govermentRoleMemberDataContainer
126 				.findListByProperty(new Object[] { Boolean.TRUE }, ViewRiksdagenGovermentRoleMember_.active);
127 
128 		final DataContainer<ViewRiksdagenPolitician, String> politicianDataContainer = getApplicationManager()
129 				.getDataContainer(ViewRiksdagenPolitician.class);
130 
131 		final List<ViewRiksdagenPolitician> parliamentDataList = politicianDataContainer
132 				.findListByProperty(new Object[] { Boolean.TRUE }, ViewRiksdagenPolitician_.activeGovernment);
133 		final Map<String, List<ViewRiksdagenPolitician>> parliamentMap = parliamentDataList.stream()
134 				.collect(Collectors.groupingBy(ViewRiksdagenPolitician::getPersonId));
135 
136 		final Map<Integer, List<GovernmentBodyAnnualSummary>> dataMap = esvApi.getData();
137 		final List<GovernmentBodyAnnualSummary> headCountGovermentBodies = dataMap.get(2022);
138 		final Map<String, List<GovernmentBodyAnnualSummary>> governmentBodyMinistryMap = headCountGovermentBodies
139 				.stream().collect(Collectors.groupingBy(GovernmentBodyAnnualSummary::getMinistry));
140 
141 		final DataContainer<ViewRiksdagenPartyRoleMember, String> partyRoleMemberDataContainer = getApplicationManager()
142 				.getDataContainer(ViewRiksdagenPartyRoleMember.class);
143 
144 		final List<ViewRiksdagenPartyRoleMember> partyDataList = partyRoleMemberDataContainer
145 				.findListByProperty(new Object[] { Boolean.TRUE }, ViewRiksdagenPartyRoleMember_.active);
146 
147 		for (final ViewRiksdagenGovermentRoleMember viewRiksdagenGovermentRoleMember : listMinistryMembers) {
148 			createDashboardGovernmentMember(row, governmentBodyMinistryMap, viewRiksdagenGovermentRoleMember,
149 					parliamentMap.get(viewRiksdagenGovermentRoleMember.getPersonId()).get(0));
150 
151 		}
152 
153 		panel.setCaption(NAME + "::" + CHARTS + parameters);
154 
155 		getPageActionEventHelper().createPageEvent(ViewAction.VISIT_MINISTRY_RANKING_VIEW, ApplicationEventGroup.USER,
156 				NAME, parameters, pageId);
157 
158 		return panelContent;
159 
160 	}

162 	private void createDashboardGovernmentMember(final ResponsiveRow row,
163 			final Map<String, List<GovernmentBodyAnnualSummary>> governmentBodyMinistryMap,
164 			final ViewRiksdagenGovermentRoleMember viewRiksdagenGovermentRoleMember,
165 			final ViewRiksdagenPolitician viewRiksdagenPolitician) {
166 
167 		final CssLayout layout = new CssLayout();
168 		layout.addStyleName("v-layout-content-overview-dashboard-panel-level2");
169 		Responsive.makeResponsive(layout);
170 		layout.setSizeUndefined();
171 
172 
173 		final DataContainer<ViewRiksdagenPartyRoleMember, String> partyRoleMemberDataContainer = getApplicationManager()
174 				.getDataContainer(ViewRiksdagenPartyRoleMember.class);
175 
176 
177 		final Label titleLabel = new Label(viewRiksdagenGovermentRoleMember.getRoleCode() + " "
178 				+ viewRiksdagenGovermentRoleMember.getFirstName() + " " + viewRiksdagenGovermentRoleMember.getLastName()
179 				+ " (" + viewRiksdagenGovermentRoleMember.getParty() +")");
180 		Responsive.makeResponsive(titleLabel);
181 		titleLabel.addStyleName("title");
182 		titleLabel.setWidth(100, Unit.PERCENTAGE);
183 		layout.addComponent(titleLabel);
184 
185 
186 		final List<ViewRiksdagenPartyRoleMember> partyRole = partyRoleMemberDataContainer.findListByProperty(new Object[] { viewRiksdagenPolitician.getPersonId(), Boolean.TRUE },ViewRiksdagenPartyRoleMember_.personId, ViewRiksdagenPartyRoleMember_.active);
187 
188 		if (partyRole.size()> 0) {
189 			final Label partyRoleLabel = new Label(partyRole.get(0).getRoleCode() + " (" + viewRiksdagenGovermentRoleMember.getParty() +")  since " +partyRole.get(0).getFromDate() );
190 			Responsive.makeResponsive(partyRoleLabel);
191 			partyRoleLabel.addStyleName("title");
192 			partyRoleLabel.setWidth(100, Unit.PERCENTAGE);
193 			layout.addComponent(partyRoleLabel);
194 
195 		}
196 
197 
198 		final Map<String, List<GovernmentBodyAnnualOutcomeSummary>> reportByMinistry = esvApi.getGovernmentBodyReportByMinistry();
199 
200 		createMinistryRoleSummary(reportByMinistry, governmentBodyMinistryMap, viewRiksdagenGovermentRoleMember, viewRiksdagenPolitician, layout);
201 		createPoliticaExperienceSummary(governmentBodyMinistryMap, viewRiksdagenGovermentRoleMember, viewRiksdagenPolitician, layout);
202 
203 
204 		row.addColumn().withDisplayRules(DISPLAY_SIZE_XS_DEVICE, DISPLAYS_SIZE_XM_DEVICE, DISPLAY_SIZE_MD_DEVICE,
205 				DISPLAY_SIZE_LG_DEVICE).withComponent(layout);
206 	}

208 	private void createMinistryRoleSummary(final Map<String, List<GovernmentBodyAnnualOutcomeSummary>> reportByMinistry, final Map<String, List<GovernmentBodyAnnualSummary>> governmentBodyMinistryMap,
209 			final ViewRiksdagenGovermentRoleMember viewRiksdagenGovermentRoleMember,
210 			final ViewRiksdagenPolitician viewRiksdagenPolitician, final CssLayout layout) {
211 		String ministryTitle = "";
212 		if (viewRiksdagenGovermentRoleMember.getRoleCode().contains("minister")) {
213 			ministryTitle = "Head of " + viewRiksdagenGovermentRoleMember.getDetail() + " for " + viewRiksdagenGovermentRoleMember.getTotalDaysServed() + "(days)";
214 		} else {
215 			ministryTitle = "Supports head of " + viewRiksdagenGovermentRoleMember.getDetail()  + " for " + viewRiksdagenGovermentRoleMember.getTotalDaysServed() + "(days)";
216 		}
217 		final Label ministryTitleLabel = new Label(ministryTitle);
218 		Responsive.makeResponsive(ministryTitleLabel);
219 		ministryTitleLabel.setWidth(100, Unit.PERCENTAGE);
220 		layout.addComponent(ministryTitleLabel);
221 		final HorizontalLayout horizontalLayout = new HorizontalLayout();
222 		Responsive.makeResponsive(horizontalLayout);
223 		final List<GovernmentBodyAnnualSummary> governentBodies = governmentBodyMinistryMap
224 				.get(viewRiksdagenGovermentRoleMember.getDetail());
225 		horizontalLayout
226 				.addComponent(
227 						new CounterStatisticsCard(VaadinIcons.WARNING,
228 								new CounterStatisticModel("Government bodies", governentBodies.size())
229 										.withShow(StatisticShow.Sum).withIconHidden().withShowOnlyStatistic(true),
230 								"Government bodies"));
231 		horizontalLayout
232 				.addComponent(new CounterStatisticsCard(
233 						VaadinIcons.WARNING, new CounterStatisticModel("Headcount", governentBodies.stream().mapToInt(GovernmentBodyAnnualSummary::getAnnualWorkHeadCount).sum())
234 								.withShow(StatisticShow.Sum).withIconHidden().withShowOnlyStatistic(true),
235 						"Headcount"));
236 
237 
238 
239 		final List<GovernmentBodyAnnualOutcomeSummary> listOutCome = reportByMinistry.get(viewRiksdagenGovermentRoleMember.getDetail());
240 
241 
242 		final Map<Integer, Double> annualIncomeSummaryMap = listOutCome.stream().filter(t -> t.getDescriptionFields().get(INKOMSTTITELGRUPPSNAMN) != null).collect(Collectors.groupingBy(GovernmentBodyAnnualOutcomeSummary::getYear,Collectors.summingDouble(GovernmentBodyAnnualOutcomeSummary::getYearTotal)));
243 		final Map<Integer, Double> annualSpendingSummaryMap = listOutCome.stream().filter(t -> t.getDescriptionFields().get(EXPENDITURE_GROUP_NAME) != null).collect(Collectors.groupingBy(GovernmentBodyAnnualOutcomeSummary::getYear,Collectors.summingDouble(GovernmentBodyAnnualOutcomeSummary::getYearTotal)));
244 
245 		horizontalLayout
246 		.addComponent(new CounterStatisticsCard(
247 				VaadinIcons.WARNING, new CounterStatisticModel("Income(B SEK)", annualIncomeSummaryMap.get(2021) /1000)
248 						.withShow(StatisticShow.Sum).withIconHidden().withShowOnlyStatistic(true),
249 				"Income"));
250 
251 
252 		horizontalLayout
253 		.addComponent(new CounterStatisticsCard(
254 				VaadinIcons.WARNING, new CounterStatisticModel("Spending(B SEK)", annualSpendingSummaryMap.get(2021)/1000)
255 						.withShow(StatisticShow.Sum).withIconHidden().withShowOnlyStatistic(true),
256 				"Spending"));
257 
258 
259 		layout.addComponent(horizontalLayout);
260 	}

271 		final HorizontalLayout horizontalLayout = new HorizontalLayout();

284 						"Government(year)"));
285

288 						new CounterStatisticsCard(VaadinIcons.WARNING,
289 								new CounterStatisticModel("Party(year)", viewRiksdagenPolitician.getTotalDaysServedParty() / 365)
290 										.withShow(StatisticShow.Sum).withIconHidden().withShowOnlyStatistic(true),
291 								"Party(year)"));
292 		horizontalLayout
293 				.addComponent(new CounterStatisticsCard(
294 						VaadinIcons.WARNING, new CounterStatisticModel("Speaker(year)", viewRiksdagenPolitician.getTotalDaysServedSpeaker() / 365)
295 								.withShow(StatisticShow.Sum).withIconHidden().withShowOnlyStatistic(true),
296 						"Speaker(year)"));
297 		layout.addComponent(horizontalLayout);
298 	}
