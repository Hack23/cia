19  package com.hack23.cia.web.impl.ui.application.views.common.rows;
20  
21  import com.hack23.cia.web.impl.ui.application.views.common.sizing.ContentRatio;
22  import com.jarektoro.responsivelayout.ResponsiveLayout;
23  import com.jarektoro.responsivelayout.ResponsiveRow;
24  import com.vaadin.server.Responsive;
25  import com.vaadin.server.Sizeable.Unit;
26  import com.vaadin.ui.AbstractOrderedLayout;
27  import com.vaadin.ui.Button;
28  import com.vaadin.ui.Component;
29  import com.vaadin.ui.CssLayout;
30  import com.vaadin.ui.Label;
31  
32  /**
33   * The Class RowUtil.
34   */
35  public final class RowUtil {
36  
37  	/** The Constant DISPLAY_SIZE_LG_DEVICE. */
38  	private static final int DISPLAY_SIZE_LG_DEVICE = 4;
39  
40  	/** The Constant DISPLAY_SIZE_MD_DEVICE. */
41  	private static final int DISPLAY_SIZE_MD_DEVICE = 4;
42  
43  	/** The Constant DISPLAY_SIZE_XS_DEVICE. */
44  	private static final int DISPLAY_SIZE_XS_DEVICE = 12;
45  
46  	/** The Constant DISPLAYS_SIZE_XM_DEVICE. */
47  	private static final int DISPLAYS_SIZE_XM_DEVICE = 6;
48  
49  	/** The Constant ITEMBOX. */
50  	private static final String ITEMBOX = "itembox";
51  
52  	/** The Constant TITLE. */
53  	private static final String TITLE = "title";
54  
55  	/**
56  	 * Creates the grid layout.
57  	 *
58  	 * @param panelContent the panel content
59  	 * @return the responsive row
60  	 */
61  	public static ResponsiveRow createGridLayout(final AbstractOrderedLayout panelContent) {
62  		final ResponsiveLayout layout = new ResponsiveLayout();
63  		Responsive.makeResponsive(layout);
64  		layout.addStyleName("v-layout-content-overview-panel-level1");
65  		layout.setWidth(100, Unit.PERCENTAGE);
66  		layout.setHeight(100, Unit.PERCENTAGE);
67  		panelContent.addComponent(layout);
68  		panelContent.setExpandRatio(layout, ContentRatio.LARGE);
69  		return layout.addRow();
70  	}
71  
72  	/**
73  	 * Creates the row component.
74  	 *
75  	 * @param row         the row
76  	 * @param component   the component
77  	 * @param description the description
78  	 */
79  	public static void createRowComponent(final ResponsiveRow row, final Component component,
80  			final String description) {
81  		final CssLayout layout = new CssLayout();
82  		layout.addStyleName(".v-layout-content-pagemode-panel-level2");
83  		Responsive.makeResponsive(layout);
84  		layout.setSizeUndefined();
85  
86  		final Label descriptionLabel = new Label(description);
87  		descriptionLabel.addStyleName(ITEMBOX);
88  		Responsive.makeResponsive(descriptionLabel);
89  		descriptionLabel.setWidth(100, Unit.PERCENTAGE);
90  		layout.addComponent(descriptionLabel);
91  
92  		component.addStyleName(ITEMBOX);
93  		component.addStyleName(TITLE);
94  		Responsive.makeResponsive(component);
95  		component.setWidth(100, Unit.PERCENTAGE);
96  		layout.addComponent(component);
97  
98  		row.addColumn().withDisplayRules(DISPLAY_SIZE_XS_DEVICE, DISPLAYS_SIZE_XM_DEVICE, DISPLAY_SIZE_MD_DEVICE,
99  				DISPLAY_SIZE_LG_DEVICE).withComponent(layout);
100 	}
101 
102 	/**
103 	 * Creates the row item.
104 	 *
105 	 * @param row         the row
106 	 * @param button      the button
107 	 * @param description the description
108 	 */
109 	public static void createRowItem(final ResponsiveRow row, final Button button, final String description) {
110 		final CssLayout layout = new CssLayout();
111 		layout.addStyleName("v-layout-content-overview-panel-level2");
112 		Responsive.makeResponsive(layout);
113 		layout.setSizeUndefined();
114 
115 		button.addStyleName(ITEMBOX);
116 		button.addStyleName(TITLE);
117 		Responsive.makeResponsive(button);
118 		button.setWidth(100, Unit.PERCENTAGE);
119 		layout.addComponent(button);
120 
121 		final Label descriptionLabel = new Label(description);
122 		descriptionLabel.addStyleName(ITEMBOX);
123 		Responsive.makeResponsive(descriptionLabel);
124 		descriptionLabel.setWidth(100, Unit.PERCENTAGE);
125 		layout.addComponent(descriptionLabel);
126 
127 		row.addColumn().withDisplayRules(DISPLAY_SIZE_XS_DEVICE, DISPLAYS_SIZE_XM_DEVICE, DISPLAY_SIZE_MD_DEVICE,
128 				DISPLAY_SIZE_LG_DEVICE).withComponent(layout);
129 	}
130 
131 }
