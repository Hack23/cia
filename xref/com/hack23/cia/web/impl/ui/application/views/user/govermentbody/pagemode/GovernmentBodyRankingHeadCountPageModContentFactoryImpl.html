24  import org.springframework.stereotype.Component;
25  
26  import com.hack23.cia.model.internal.application.system.impl.ApplicationEventGroup;
27  import com.hack23.cia.web.impl.ui.application.action.ViewAction;
28  import com.hack23.cia.web.impl.ui.application.views.common.chartfactory.api.GovernmentBodyChartDataManager;
29  import com.hack23.cia.web.impl.ui.application.views.common.viewnames.GovernmentBodyPageMode;
30  import com.vaadin.ui.Layout;
31  import com.vaadin.ui.MenuBar;
32  import com.vaadin.ui.Panel;
33  import com.vaadin.ui.VerticalLayout;
34  
35  /**
36   * The Class GovernmentBodyRankingHeadCountPageModContentFactoryImpl.
37   */
38  @Component
39  public final class GovernmentBodyRankingHeadCountPageModContentFactoryImpl extends AbstractGovernmentBodyRankingPageModContentFactoryImpl {
40  
41  	/** The Constant GOVERNMENT_BODIES. */
42  	private static final String GOVERNMENT_BODIES = "Government body headcount";
43  
44  	/** The government body chart data manager. */
45  	@Autowired
46  	private GovernmentBodyChartDataManager governmentBodyChartDataManager;
47  
48  	/**
49  	 * Instantiates a new government body ranking head count page mod content
50  	 * factory impl.
51  	 */
52  	public GovernmentBodyRankingHeadCountPageModContentFactoryImpl() {
53  		super();
54  	}
55  
56  	@Secured({ "ROLE_ANONYMOUS", "ROLE_USER", "ROLE_ADMIN" })
57  	@Override
58  	public Layout createContent(final String parameters, final MenuBar menuBar, final Panel panel) {
59  		final VerticalLayout panelContent = createPanelContent();
60  
61  		getGovernmentBodyRankingMenuItemFactory().createGovernmentBodyRankingMenuBar(menuBar);
62  
63  		final String pageId = getPageId(parameters);
64  
65  		panel.setCaption(NAME + "::" + GOVERNMENT_BODIES + parameters);
66  
67  		governmentBodyChartDataManager.createGovernmentBodyHeadcountSummaryChart(panelContent);
68  
69  		getPageActionEventHelper().createPageEvent(ViewAction.VISIT_MINISTRY_RANKING_VIEW, ApplicationEventGroup.USER, NAME,
70  				parameters, pageId);
71  
72  		return panelContent;
73  
74  	}
75  
76  	@Override
77  	public boolean matches(final String page, final String parameters) {
78  		return NAME.equals(page) && StringUtils.contains(parameters, GovernmentBodyPageMode.HEADCOUNT.toString());
79  	}
80  
81  }
