45  /**
46   * The Class GridFactoryImpl.
47   */
48  @Service
49  public final class GridFactoryImpl implements GridFactory {
50  
51  	/** The Constant LOGGER. */
52  	private static final Logger LOGGER = LoggerFactory.getLogger(GridFactoryImpl.class);
53  
54  	/** The Constant serialVersionUID. */
55  	private static final long serialVersionUID = 1L;
56  
57  	/**
58  	 * Instantiates a new grid factory impl.
59  	 */
60  	public GridFactoryImpl() {
61  		super();
62  	}
63  
64  	/**
65  	 * Configure column orders and hidden fields.
66  	 *
67  	 * @param columnOrder
68  	 *            the column order
69  	 * @param hideColumns
70  	 *            the hide columns
71  	 * @param grid
72  	 *            the grid
73  	 */
74  	@SuppressWarnings("rawtypes")
75  	private static void configureColumnOrdersAndHiddenFields(final String[] columnOrder, final String[] hideColumns,
76  			final Grid grid) {
77  		if (columnOrder != null) {
78  			grid.setColumnOrder(columnOrder);
79  		}
80  
81  		if (hideColumns != null) {
82  			for (final String o : hideColumns) {
83  				grid.removeColumn(o);
84  			}
85  		}
86  	}
87  
88  	/**
89  	 * Configure listeners.
90  	 *
91  	 * @param listener
92  	 *            the listener
93  	 * @param grid
94  	 *            the grid
95  	 */
96  	@SuppressWarnings({ "unchecked", "rawtypes" })
97  	private static void configureListeners(final SelectionListener listener, final Grid grid) {
98  
99  		if (listener != null) {
100 			grid.addSelectionListener(listener);
101 		}
102 	}
103 
104 	/**
105 	 * Creates the grid cell filter.
106 	 *
107 	 * @param columnOrder
108 	 *            the column order
109 	 * @param grid
110 	 *            the grid
111 	 * @param dataType
112 	 *            the data type
113 	 */
114 	@SuppressWarnings({ "unchecked", "rawtypes" })
115 	private static void createGridCellFilter(final String[] columnOrder, final Grid grid, final Class dataType) {
116 		if (columnOrder != null) {
117 			final GridCellFilter filter = new GridCellFilter(grid, dataType);
118 			for (final String column : columnOrder) {
119 				if (grid.getColumn(column) != null) {
120 					filter.setTextFilter(column, true, true);
121 				}
122 			}
123 		}
124 	}
125 
126 	/**
127 	 * Creates the nested properties.
128 	 *
129 	 * @param <T> the generic type
130 	 * @param grid the grid
131 	 * @param nestedProperties the nested properties
132 	 */
133 	private static <T extends Serializable> void createNestedProperties(final Grid<T> grid,
134 			final String[] nestedProperties) {
135 		if (nestedProperties != null) {
136 			for (final String property : nestedProperties) {
137 				final Column<T, ?> addColumn = grid.addColumn(new BeanNestedPropertyValueProvider<T>(property));
138 				addColumn.setId(property);
139 			}
140 		}
141 	}
142 
143 	/**
144 	 * Sets the column converters.
145 	 *
146 	 * @param collectionPropertyConverter
147 	 *            the collection property converter
148 	 * @param grid
149 	 *            the grid
150 	 */
151 	@SuppressWarnings({ "unchecked", "rawtypes" })
152 	private static void setColumnConverters(final ListPropertyConverter[] collectionPropertyConverter,
153 			final Grid grid) {
154 		if (collectionPropertyConverter != null) {
155 			for (final ListPropertyConverter converter : collectionPropertyConverter) {
156 				grid.removeColumn(converter.getColumn());
157 				final Column column = grid.addColumn(converter);
158 				column.setCaption(WordUtils.capitalize(converter.getColumn()));
159 				column.setId(converter.getColumn());
160 			}
161 		}
162 	}

164 	@Override
165 	public <T extends Serializable> void createBasicBeanItemGrid(final AbstractOrderedLayout panelContent,
166 			final Class<T> dataType, final List<T> datasource, final String caption, final String[] columnOrder,
167 			final String[] hideColumns, final PageItemRendererClickListener<?> listener, final String actionId,
168 			final ListPropertyConverter[] collectionPropertyConverters) {
169 		createBasicBeanItemNestedPropertiesGrid(panelContent, dataType, datasource, caption, null, columnOrder,
170 				hideColumns, listener, actionId, collectionPropertyConverters);
171 
172 	}
173 
174 	@Override
175 	public <T extends Serializable> void createBasicBeanItemNestedPropertiesGrid(
176 			final AbstractOrderedLayout panelContent, final Class<T> dataType, final List<T> datasource,
177 			final String caption, final String[] nestedProperties, final String[] columnOrder,
178 			final String[] hideColumns, final PageItemRendererClickListener<?> listener, final String actionId,
179 			final ListPropertyConverter[] collectionPropertyConverters) {
180 
181 		final Grid<T> grid = Grid.withPropertySet(BeanPropertySet.get(dataType));
182 		grid.setCaption(caption);
183 
184 		grid.setItems(datasource.stream().filter(Objects::nonNull).toList());
185 
186 		grid.setSelectionMode(SelectionMode.SINGLE);
187 
188 		createNestedProperties(grid, nestedProperties);
189 
190 		setColumnConverters(collectionPropertyConverters, grid);
191 
192 		configureColumnOrdersAndHiddenFields(columnOrder, hideColumns, grid);
193 
194 		configureListeners(listener, grid);
195 
196 
197 		grid.setSizeFull();
198 
199 		grid.setStyleName("Level2Header");
200 
201 		createGridCellFilter(columnOrder, grid, dataType);
202 
203 		grid.setResponsive(true);
204 
205 		panelContent.addComponent(grid);
206 		panelContent.setExpandRatio(grid, ContentRatio.GRID);
207 	}

209 	/**
210 	 * The Class BeanNestedPropertyValueProvider.
211 	 *
212 	 * @param <T>
213 	 *            the generic type
214 	 */
215 	public static final class BeanNestedPropertyValueProvider<T> implements ValueProvider<T, String> {
216 
217 		/** The Constant serialVersionUID. */
218 		private static final long serialVersionUID = 1L;
219 
220 		/** The property. */
221 		private final String property;
222 
223 		/**
224 		 * Instantiates a new bean nested property value provider.
225 		 *
226 		 * @param property
227 		 *            the property
228 		 */
229 		public BeanNestedPropertyValueProvider(final String property) {
230 			super();
231 			this.property = property;
232 		}
233 
234 		@Override
235 		public String apply(final T source) {
236 			try {
237 				return BeanUtils.getProperty(source, property);
238 			} catch (IllegalAccessException | InvocationTargetException | NoSuchMethodException e) {
239 				LOGGER.warn("Problem getting property : {} from source : {} , exception: {}", property, source, e);
240 				return "";
241 			}
242 		}
243 
244 	}
