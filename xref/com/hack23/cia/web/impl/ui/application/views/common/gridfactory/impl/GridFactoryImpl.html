47   * The Class GridFactoryImpl.
48   */
49  @Service
50  public final class GridFactoryImpl implements GridFactory {
51  
52  	/** The Constant LOGGER. */
53  	private static final Logger LOGGER = LoggerFactory.getLogger(GridFactoryImpl.class);
54  
55  	/** The Constant serialVersionUID. */
56  	private static final long serialVersionUID = 1L;
57  
58  	/**
59  	 * Instantiates a new grid factory impl.
60  	 */
61  	public GridFactoryImpl() {
62  		super();
63  	}
64  
65  	/**
66  	 * Configure column orders and hidden fields.
67  	 *
68  	 * @param columnOrder
69  	 *            the column order
70  	 * @param hideColumns
71  	 *            the hide columns
72  	 * @param grid
73  	 *            the grid
74  	 */
75  	private static void configureColumnOrdersAndHiddenFields(final String[] columnOrder, final String[] hideColumns,
76  			final Grid grid) {
77  		if (columnOrder != null) {
78  			grid.setColumnOrder(columnOrder);
79  		}
80  
81  		if (hideColumns != null) {
82  			for (final String o : hideColumns) {
83  				grid.removeColumn(o);
84  			}
85  		}
86  	}
87  
88  	/**
89  	 * Configure listeners.
90  	 *
91  	 * @param listener
92  	 *            the listener
93  	 * @param grid
94  	 *            the grid
95  	 */
96  	private static void configureListeners(final SelectionListener listener, final Grid grid) {
97  
98  		if (listener != null) {
99  			grid.addSelectionListener(listener);
100 		}
101 	}
102 
103 	/**
104 	 * Creates the grid cell filter.
105 	 *
106 	 * @param columnOrder
107 	 *            the column order
108 	 * @param grid
109 	 *            the grid
110 	 * @param dataType
111 	 *            the data type
112 	 */
113 	private static void createGridCellFilter(final String[] columnOrder, final Grid grid, final Class dataType) {
114 		if (columnOrder != null) {
115 			final GridCellFilter filter = new GridCellFilter(grid, dataType);
116 			for (final String column : columnOrder) {
117 				if (grid.getColumn(column) != null) {
118 					filter.setTextFilter(column, true, true);
119 				}
120 			}
121 		}
122 	}
123 
124 	private static <T extends Serializable> void createNestedProperties(final Grid<T> grid,
125 			final String[] nestedProperties) {
126 		if (nestedProperties != null) {
127 			for (final String property : nestedProperties) {
128 				final Column<T, ?> addColumn = grid.addColumn(new BeanNestedPropertyValueProvider<T>(property));
129 				addColumn.setId(property);
130 			}
131 		}
132 	}
133 
134 	/**
135 	 * Sets the column converters.
136 	 *
137 	 * @param collectionPropertyConverter
138 	 *            the collection property converter
139 	 * @param grid
140 	 *            the grid
141 	 */
142 	private static void setColumnConverters(final ListPropertyConverter[] collectionPropertyConverter,
143 			final Grid grid) {
144 		if (collectionPropertyConverter != null) {
145 			for (final ListPropertyConverter converter : collectionPropertyConverter) {
146 				grid.removeColumn(converter.getColumn());
147 				final Column column = grid.addColumn(converter);
148 				column.setCaption(WordUtils.capitalize(converter.getColumn()));
149 				column.setId(converter.getColumn());
150 			}
151 		}
152 	}

154 	@Override
155 	public <T extends Serializable> void createBasicBeanItemGrid(final AbstractOrderedLayout panelContent,
156 			final Class<T> dataType, final List<T> datasource, final String caption, final String[] columnOrder,
157 			final String[] hideColumns, final PageItemRendererClickListener<?> listener, final String actionId,
158 			final ListPropertyConverter[] collectionPropertyConverters) {
159 		createBasicBeanItemNestedPropertiesGrid(panelContent, dataType, datasource, caption, null, columnOrder,
160 				hideColumns, listener, actionId, collectionPropertyConverters);
161 
162 	}
163 
164 	@Override
165 	public <T extends Serializable> void createBasicBeanItemNestedPropertiesGrid(
166 			final AbstractOrderedLayout panelContent, final Class<T> dataType, final List<T> datasource,
167 			final String caption, final String[] nestedProperties, final String[] columnOrder,
168 			final String[] hideColumns, final PageItemRendererClickListener<?> listener, final String actionId,
169 			final ListPropertyConverter[] collectionPropertyConverters) {
170 
171 		final Grid<T> grid = Grid.withPropertySet(BeanPropertySet.get(dataType));
172 		grid.setCaption(caption);
173 
174 		grid.setItems(datasource.stream().filter(Objects::nonNull).collect(Collectors.toList()));
175 
176 		grid.setSelectionMode(SelectionMode.SINGLE);
177 
178 		createNestedProperties(grid, nestedProperties);
179 
180 		setColumnConverters(collectionPropertyConverters, grid);
181 
182 		configureColumnOrdersAndHiddenFields(columnOrder, hideColumns, grid);
183 
184 		configureListeners(listener, grid);
185 
186 
187 		grid.setSizeFull();
188 
189 		grid.setStyleName("Level2Header");
190 
191 		createGridCellFilter(columnOrder, grid, dataType);
192 
193 		grid.setResponsive(true);
194 
195 		panelContent.addComponent(grid);
196 		panelContent.setExpandRatio(grid, ContentRatio.GRID);
197 	}

199 	/**
200 	 * The Class BeanNestedPropertyValueProvider.
201 	 *
202 	 * @param <T>
203 	 *            the generic type
204 	 */
205 	public static final class BeanNestedPropertyValueProvider<T> implements ValueProvider<T, String> {
206 
207 		/** The Constant serialVersionUID. */
208 		private static final long serialVersionUID = 1L;
209 
210 		/** The property. */
211 		private final String property;
212 
213 		/**
214 		 * Instantiates a new bean nested property value provider.
215 		 *
216 		 * @param property
217 		 *            the property
218 		 */
219 		public BeanNestedPropertyValueProvider(final String property) {
220 			super();
221 			this.property = property;
222 		}
223 
224 		@Override
225 		public String apply(final T source) {
226 			try {
227 				return BeanUtils.getProperty(source, property);
228 			} catch (IllegalAccessException | InvocationTargetException | NoSuchMethodException e) {
229 				LOGGER.warn("Problem getting property : {} from source : {} , exception: {}", property, source, e);
230 				return "";
231 			}
232 		}
233 
234 	}
