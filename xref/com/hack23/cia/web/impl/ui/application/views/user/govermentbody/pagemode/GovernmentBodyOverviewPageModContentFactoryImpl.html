56  	@Secured({ "ROLE_ANONYMOUS", "ROLE_USER", "ROLE_ADMIN" })
57  	@Override
58  	public Layout createContent(final String parameters, final MenuBar menuBar, final Panel panel) {
59  		final VerticalLayout panelContent = createPanelContent();
60  		panel.setContent(panelContent);
61  
62  		final String pageId = getPageId(parameters);
63  
64  		final List<GovernmentBodyAnnualSummary> item = getItem(parameters);
65  
66  		if (item != null && !item.isEmpty()) {
67  			final GovernmentBodyAnnualSummary governmentBodyAnnualSummary = item.get(0);
68  
69  			getGovernmentBodyMenuItemFactory().createGovernmentBodyMenuBar(menuBar, pageId, governmentBodyAnnualSummary.getName());
70  
71  			createPageHeader(panel, panelContent,
72  					"Government Body Overview " + governmentBodyAnnualSummary.getName(),
73  					"Government Details",
74  					"Explore detailed information about government bodies and their functions.");
75  
76  			// Create a card-style panel
77  			final Panel cardPanel = new Panel();
78  			cardPanel.addStyleName("politician-overview-card");
79  			cardPanel.setWidth("100%");
80  			cardPanel.setHeightUndefined();
81  			Responsive.makeResponsive(cardPanel);
82  
83  			final VerticalLayout cardContent = new VerticalLayout();
84  			cardContent.setMargin(true);
85  			cardContent.setSpacing(true);
86  			cardContent.setWidth("100%");
87  			cardPanel.setContent(cardContent);
88  
89  			panelContent.addComponent(cardPanel);
90  			panelContent.setExpandRatio(cardPanel, ContentRatio.SMALL_GRID);
91  
92  			// Header layout for the card
93  			final HorizontalLayout headerLayout = new HorizontalLayout();
94  			headerLayout.setSpacing(true);
95  			headerLayout.setWidth("100%");
96  			headerLayout.addStyleName("card-header-section");
97  
98  			final Label titleLabel = new Label("Government Body Information", ContentMode.HTML);
99  			titleLabel.addStyleName("card-title");
100 			titleLabel.setWidthUndefined();
101 			headerLayout.addComponent(titleLabel);
102 			cardContent.addComponent(headerLayout);
103 
104 			// Divider line
105 			final Label divider = new Label("<hr/>", ContentMode.HTML);
106 			divider.addStyleName("card-divider");
107 			divider.setWidth("100%");
108 			cardContent.addComponent(divider);
109 
110 			// Two-column layout
111 			final HorizontalLayout attributesLayout = new HorizontalLayout();
112 			attributesLayout.setSpacing(true);
113 			attributesLayout.setWidth("100%");
114 			cardContent.addComponent(attributesLayout);
115 
116 			// Left column: Organization Profile
117 			final VerticalLayout profileDetailsLayout = new VerticalLayout();
118 			profileDetailsLayout.setSpacing(true);
119 			profileDetailsLayout.addStyleName("card-details-column");
120 			profileDetailsLayout.setWidthUndefined();
121 
122 			final Label profileDetailsHeader = new Label("Organization Profile");
123 			profileDetailsHeader.addStyleName("card-section-title");
124 			profileDetailsLayout.addComponent(profileDetailsHeader);
125 
126 			// Add fields for Organization Profile
127 			profileDetailsLayout.addComponent(createInfoRow("Name:", governmentBodyAnnualSummary.getName(), VaadinIcons.INFO_CIRCLE, "Government body name"));
128 			profileDetailsLayout.addComponent(createInfoRow("ID:", governmentBodyAnnualSummary.getGovermentBodyId(), VaadinIcons.CLIPBOARD_USER, "Unique government body identifier"));
129 			profileDetailsLayout.addComponent(createInfoRow("Ministry:", governmentBodyAnnualSummary.getMinistry(), VaadinIcons.INSTITUTION, "Ministry overseeing the body"));
130 			profileDetailsLayout.addComponent(createInfoRow("Org Number:", governmentBodyAnnualSummary.getOrgNumber(), VaadinIcons.BOOK, "Official organization number"));
131 			profileDetailsLayout.addComponent(createInfoRow("M Code:", governmentBodyAnnualSummary.getmCode(), VaadinIcons.CODE, "Internal M code identifier"));
132 			profileDetailsLayout.addComponent(createInfoRow("VAT:", governmentBodyAnnualSummary.getVat(), VaadinIcons.MONEY, "VAT number"));
133 
134 			// Right column: Annual Statistics
135 			final VerticalLayout serviceStatsLayout = new VerticalLayout();
136 			serviceStatsLayout.setSpacing(true);
137 			serviceStatsLayout.addStyleName("card-details-column");
138 			serviceStatsLayout.setWidthUndefined();
139 
140 			final Label serviceStatsHeader = new Label("Annual Statistics");
141 			serviceStatsHeader.addStyleName("card-section-title");
142 			serviceStatsLayout.addComponent(serviceStatsHeader);
143 
144 			// Add fields for Annual Statistics
145 			serviceStatsLayout.addComponent(createInfoRow("Year:", String.valueOf(governmentBodyAnnualSummary.getYear()), VaadinIcons.CALENDAR, "The reporting year"));
146 			serviceStatsLayout.addComponent(createInfoRow("Annual Work Head Count:", String.valueOf(governmentBodyAnnualSummary.getAnnualWorkHeadCount()), VaadinIcons.USER_CHECK, "Annual average number of full-time equivalents"));
147 			serviceStatsLayout.addComponent(createInfoRow("Head Count:", String.valueOf(governmentBodyAnnualSummary.getHeadCount()), VaadinIcons.GROUP, "Total number of staff members"));
148 			serviceStatsLayout.addComponent(createInfoRow("Consecutive Number:", String.valueOf(governmentBodyAnnualSummary.getConsecutiveNumber()), VaadinIcons.LINES_LIST, "Internal consecutive reference number"));
149 
150 			// If comment is short and informative, display it
151 			if (governmentBodyAnnualSummary.getComment() != null && !governmentBodyAnnualSummary.getComment().isEmpty()) {
152 				serviceStatsLayout.addComponent(createInfoRow("Comment:", governmentBodyAnnualSummary.getComment(), VaadinIcons.COMMENT_ELLIPSIS, "Additional remarks or notes"));
153 			}
154 
155 			attributesLayout.addComponents(profileDetailsLayout, serviceStatsLayout);
156 
157 			// After the card, add the overview layout
158 			final VerticalLayout overviewLayout = new VerticalLayout();
159 			overviewLayout.setSizeFull();
160 			panelContent.addComponent(overviewLayout);
161 			panelContent.setExpandRatio(overviewLayout, ContentRatio.LARGE_FORM);
162 
163 			getGovernmentBodyMenuItemFactory().createOverviewPage(overviewLayout, pageId);
164 
165 			getPageActionEventHelper().createPageEvent(ViewAction.VISIT_GOVERNMENT_BODY_VIEW,
166 					ApplicationEventGroup.USER, NAME, parameters, pageId);
167 		}
168 
169 		return panelContent;
170 	}

172 	/**
173 	 * Creates a row displaying a caption and value, with optional icon and tooltip.
174 	 *
175 	 * @param caption the field caption
176 	 * @param value   the field value
177 	 * @param icon    a VaadinIcons icon for better visual cue
178 	 * @param tooltip optional tooltip to provide more info
179 	 * @return a HorizontalLayout representing the info row
180 	 */
181 	private HorizontalLayout createInfoRow(final String caption, final String value, VaadinIcons icon,
182 			final String tooltip) {
183 		final HorizontalLayout layout = new HorizontalLayout();
184 		layout.setSpacing(true);
185 		layout.addStyleName("metric-label");
186 		layout.setWidthUndefined();
187 
188 		if (icon != null) {
189 			final Label iconLabel = new Label(icon.getHtml(), ContentMode.HTML);
190 			iconLabel.addStyleName("card-info-icon");
191 			if (tooltip != null && !tooltip.isEmpty()) {
192 				iconLabel.setDescription(tooltip);
193 			}
194 			layout.addComponent(iconLabel);
195 		}
196 
197 		final Label captionLabel = new Label(caption);
198 		captionLabel.addStyleName("card-info-caption");
199 		if (tooltip != null && !tooltip.isEmpty()) {
200 			captionLabel.setDescription(tooltip);
201 		}
202 
203 		final Label valueLabel = new Label(value != null ? value : "");
204 		valueLabel.addStyleName("card-info-value");
205 
206 		layout.addComponents(captionLabel, valueLabel);
207 		return layout;
208 	}

211 	public boolean matches(final String page, final String parameters) {
212 		final String pageId = getPageId(parameters);
213 		return NAME.equals(page) && (StringUtils.isEmpty(parameters) || parameters.equals(pageId)
214 				|| parameters.contains(PageMode.OVERVIEW.toString()));
215 	}
