170 	/**
171 	 * Creates a row displaying a caption and value, with optional icon and tooltip.
172 	 *
173 	 * @param caption the field caption
174 	 * @param value   the field value
175 	 * @param icon    a VaadinIcons icon for better visual cue
176 	 * @param tooltip optional tooltip to provide more info
177 	 * @return a HorizontalLayout representing the info row
178 	 */
179 	private HorizontalLayout createInfoRow(final String caption, final String value, VaadinIcons icon, final String tooltip) {
180 		final HorizontalLayout layout = new HorizontalLayout();
181 		layout.setSpacing(true);
182 		layout.addStyleName("metric-label");
183 		layout.setWidthUndefined();
184 
185 		if (icon != null) {
186 			final Label iconLabel = new Label(icon.getHtml(), ContentMode.HTML);
187 			iconLabel.addStyleName("card-info-icon");
188 			if (tooltip != null && !tooltip.isEmpty()) {
189 				iconLabel.setDescription(tooltip);
190 			}
191 			layout.addComponent(iconLabel);
192 		}
193 
194 		final Label captionLabel = new Label(caption);
195 		captionLabel.addStyleName("card-info-caption");
196 		if (tooltip != null && !tooltip.isEmpty()) {
197 			captionLabel.setDescription(tooltip);
198 		}
199 
200 		final Label valueLabel = new Label(value != null ? value : "");
201 		valueLabel.addStyleName("card-info-value");
202 
203 		layout.addComponents(captionLabel, valueLabel);
204 		return layout;
205 	}

212 	}
