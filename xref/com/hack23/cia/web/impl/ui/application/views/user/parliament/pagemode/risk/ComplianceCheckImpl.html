3   import java.util.ArrayList;
4   import java.util.List;
5   
6   import com.hack23.cia.model.internal.application.data.rules.impl.ResourceType;
7   import com.hack23.cia.model.internal.application.data.rules.impl.RuleViolation;
8   import com.hack23.cia.service.api.action.kpi.ComplianceCheck;
9   
10  /**
11   * The Class ComplianceCheckImpl.
12   */
13  public class ComplianceCheckImpl implements ComplianceCheck {
14  
15  	private static final long serialVersionUID = 1L;
16  	private final List<RuleViolation> ruleViolations;
17  	private final ResourceType resourceType;
18  	private final String id;
19  	private final String name;
20  
21  	/**
22  	 * Instantiates a new compliance check impl.
23  	 *
24  	 * @param ruleViolations the rule violations
25  	 */
26  	public ComplianceCheckImpl(final List<RuleViolation> ruleViolations) {
27  		this.ruleViolations = new ArrayList<>(ruleViolations);
28  		this.resourceType = ruleViolations.get(0).getResourceType();
29  		this.id = ruleViolations.get(0).getReferenceId();
30  		this.name = ruleViolations.get(0).getName();
31  	}
32  
33  	@Override
34  	public ResourceType getResourceType() {
35  		return resourceType;
36  	}
37  
38  	@Override
39  	public List<RuleViolation> getRuleViolations() {
40  		return new ArrayList<>(ruleViolations);
41  	}
42  
43  
44  	/**
45  	 * Gets the number rule violations.
46  	 *
47  	 * @return the number rule violations
48  	 */
49  	public int getNumberRuleViolations() {
50  		return ruleViolations.size();
51  	}
52  
53  	/**
54  	 * Gets the rule summary.
55  	 *
56  	 * @return the rule summary
57  	 */
58  	public String getRuleSummary() {
59  		final StringBuilder builder = new StringBuilder();
60  		for (final RuleViolation ruleViolation : ruleViolations) {
61  			builder.append('[').append(ruleViolation.getRuleName()).append('/').append(ruleViolation.getStatus()) .append(']');
62  		}
63  		return builder.toString();
64  	}
65  
66  	@Override
67  	public String getId() {
68  		return id;
69  	}
70  
71  	@Override
72  	public String getName() {
73  		return name;
74  	}
75  }
