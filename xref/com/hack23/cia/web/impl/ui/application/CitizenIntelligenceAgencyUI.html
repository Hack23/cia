48  /**
49   * The Class CitizenIntelligenceAgencyUI.
50   */
51  @SpringUI
52  @SpringViewDisplay
53  @Theme("cia")
54  @Push(transport = Transport.WEBSOCKET_XHR)
55  @Viewport("width=device-width, initial-scale=1")
56  public final class CitizenIntelligenceAgencyUI extends UI {
57  
58  	/** The Constant CRLF. */
59  	private static final String CRLF = "[\r\n]";
60  
61  	/** The Constant CRLF_REPLACEMENT. */
62  	private static final String CRLF_REPLACEMENT = "";
63  
64  	/** The Constant LOG_INFO_BROWSER_ADDRESS_APPLICATION_SESSION_ID_RESULT. */
65  	private static final String LOG_INFO_BROWSER_ADDRESS_APPLICATION_SESSION_ID_RESULT = "Browser url: {} , lang: {} , address: {} , application:{}, sessionId:{}, result:{}";
66  
67  	/** The Constant LOGGER. */
68  	private static final Logger LOGGER = LoggerFactory
69  			.getLogger(CitizenIntelligenceAgencyUI.class);
70  
71  	/** The Constant serialVersionUID. */
72  	private static final long serialVersionUID = 1L;
73  
74  	/** The application manager. */
75  	@Autowired
76  	private transient ApplicationManager applicationManager;
77  
78  	@Autowired
79  	private transient ConfigurationManager configurationManager;
80  
81  	/** The main view. */
82  	@Autowired
83  	private MainView mainView;
84  
85  	@Autowired
86  	private SpringNavigator springNavigator;
87  
88  	/**
89  	 * Instantiates a new citizen intelligence agency ui.
90  	 */
91  	public CitizenIntelligenceAgencyUI() {
92  		super();
93  	}
94  
95  
96  	@Override
97  	protected void init(final VaadinRequest request) {
98  		VaadinSession.getCurrent().setErrorHandler(new UiInstanceErrorHandler(this));
99  		setSizeFull();
100 		springNavigator.addView(CRLF_REPLACEMENT, mainView);
101 		setNavigator(springNavigator);
102 
103 
104 		final Page currentPage = Page.getCurrent();
105 		final String requestUrl = currentPage.getLocation().toString();
106 		final String language = request.getLocale().getLanguage();
107 		final UserConfiguration userConfiguration = configurationManager.getUserConfiguration(requestUrl, language);
108 
109 		currentPage.setTitle(userConfiguration.getAgency().getAgencyName() + ":" + userConfiguration.getPortal().getPortalName() + ":" + userConfiguration.getLanguage().getLanguageName());
110 
111 		if (getSession().getUIs().isEmpty()) {
112 			final WebBrowser webBrowser = currentPage.getWebBrowser();
113 
114 			final CreateApplicationSessionRequest serviceRequest = new CreateApplicationSessionRequest();
115 			serviceRequest.setSessionId(RequestContextHolder.currentRequestAttributes().getSessionId());
116 
117 			final String ipInformation = WebBrowserUtil.getIpInformation(webBrowser);
118 			serviceRequest.setIpInformation(ipInformation);
119 			serviceRequest.setTimeZone(webBrowser.getTimeZoneId());
120 			serviceRequest.setScreenSize(webBrowser.getScreenWidth() + "x" + webBrowser.getScreenHeight());
121 			serviceRequest.setUserAgentInformation(webBrowser.getBrowserApplication());
122 			serviceRequest.setLocale(webBrowser.getLocale().toString());
123 			serviceRequest.setOperatingSystem(WebBrowserUtil.getOperatingSystem(webBrowser));
124 			serviceRequest.setSessionType(ApplicationSessionType.ANONYMOUS);
125 
126 			final ServiceResponse serviceResponse = applicationManager.service(serviceRequest);
127 			LOGGER.info(LOG_INFO_BROWSER_ADDRESS_APPLICATION_SESSION_ID_RESULT,requestUrl.replaceAll(CRLF,CRLF_REPLACEMENT),language.replaceAll(CRLF,CRLF_REPLACEMENT),ipInformation.replaceAll(CRLF,CRLF_REPLACEMENT),webBrowser.getBrowserApplication().replaceAll(CRLF,CRLF_REPLACEMENT),serviceRequest.getSessionId().replaceAll(CRLF,CRLF_REPLACEMENT),serviceResponse.getResult().toString().replaceAll(CRLF,CRLF_REPLACEMENT));
128 		}
129 	}
