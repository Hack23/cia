49  /**
50   * The Class CitizenIntelligenceAgencyUI.
51   */
52  @SpringUI
53  @SpringViewDisplay
54  @Theme("cia")
55  @Push(transport = Transport.WEBSOCKET_XHR)
56  @StyleSheet({"https://fonts.googleapis.com/css?family=Inconsolata"})
57  @Viewport("width=device-width, initial-scale=1")
58  public final class CitizenIntelligenceAgencyUI extends UI {
59  
60  	/** The Constant CRLF. */
61  	private static final String CRLF = "[\r\n]";
62  
63  	/** The Constant CRLF_REPLACEMENT. */
64  	private static final String CRLF_REPLACEMENT = "";
65  	
66  	/** The Constant LOG_INFO_BROWSER_ADDRESS_APPLICATION_SESSION_ID_RESULT. */
67  	private static final String LOG_INFO_BROWSER_ADDRESS_APPLICATION_SESSION_ID_RESULT = "Browser url: {} , lang: {} , address: {} , application:{}, sessionId:{}, result:{}";
68  	
69  	/** The Constant LOGGER. */
70  	private static final Logger LOGGER = LoggerFactory
71  			.getLogger(CitizenIntelligenceAgencyUI.class);
72  
73  	/** The Constant serialVersionUID. */
74  	private static final long serialVersionUID = 1L;
75  
76  	/** The application manager. */
77  	@Autowired
78  	private transient ApplicationManager applicationManager;
79  
80  	@Autowired
81  	private transient ConfigurationManager configurationManager;
82  
83  	/** The main view. */
84  	@Autowired
85  	private MainView mainView;
86  
87  	@Autowired
88  	private SpringNavigator springNavigator;
89  
90  	/**
91  	 * Instantiates a new citizen intelligence agency ui.
92  	 */
93  	public CitizenIntelligenceAgencyUI() {
94  		super();
95  	}
96  
97  
98  	@Override
99  	protected void init(final VaadinRequest request) {
100 		VaadinSession.getCurrent().setErrorHandler(new UiInstanceErrorHandler(this));
101 		setSizeFull();
102 		springNavigator.addView(CRLF_REPLACEMENT, mainView);
103 		setNavigator(springNavigator);
104 
105 
106 		final Page currentPage = Page.getCurrent();
107 		final String requestUrl = currentPage.getLocation().toString();
108 		final String language = request.getLocale().getLanguage();
109 		final UserConfiguration userConfiguration = configurationManager.getUserConfiguration(requestUrl, language);
110 
111 		currentPage.setTitle(userConfiguration.getAgency().getAgencyName() + ":" + userConfiguration.getPortal().getPortalName() + ":" + userConfiguration.getLanguage().getLanguageName());
112 
113 		if (getSession().getUIs().isEmpty()) {
114 			final WebBrowser webBrowser = currentPage.getWebBrowser();
115 
116 			final CreateApplicationSessionRequest serviceRequest = new CreateApplicationSessionRequest();
117 			serviceRequest.setSessionId(RequestContextHolder.currentRequestAttributes().getSessionId());
118 
119 			final String ipInformation = WebBrowserUtil.getIpInformation(webBrowser);
120 			serviceRequest.setIpInformation(ipInformation);
121 			serviceRequest.setTimeZone(webBrowser.getTimeZoneId());
122 			serviceRequest.setScreenSize(webBrowser.getScreenWidth() + "x" + webBrowser.getScreenHeight());
123 			serviceRequest.setUserAgentInformation(webBrowser.getBrowserApplication());
124 			serviceRequest.setLocale(webBrowser.getLocale().toString());
125 			serviceRequest.setOperatingSystem(WebBrowserUtil.getOperatingSystem(webBrowser));
126 			serviceRequest.setSessionType(ApplicationSessionType.ANONYMOUS);
127 
128 			final ServiceResponse serviceResponse = applicationManager.service(serviceRequest);
129 			LOGGER.info(LOG_INFO_BROWSER_ADDRESS_APPLICATION_SESSION_ID_RESULT,requestUrl.replaceAll(CRLF,CRLF_REPLACEMENT),language.replaceAll(CRLF,CRLF_REPLACEMENT),ipInformation.replaceAll(CRLF,CRLF_REPLACEMENT),webBrowser.getBrowserApplication().replaceAll(CRLF,CRLF_REPLACEMENT),serviceRequest.getSessionId().replaceAll(CRLF,CRLF_REPLACEMENT),serviceResponse.getResult().toString().replaceAll(CRLF,CRLF_REPLACEMENT));
130 		}
131 	}
