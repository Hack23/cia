85  	 * Creates the content.
86  	 *
87  	 * @param parameters the parameters
88  	 * @param menuBar the menu bar
89  	 * @param panel the panel
90  	 * @return the layout
91  	 */
92  	@Secured({ "ROLE_ANONYMOUS", "ROLE_USER", "ROLE_ADMIN" })
93  	@Override
94  	public Layout createContent(final String parameters, final MenuBar menuBar, final Panel panel) {
95  		final VerticalLayout panelContent = createPanelContent();
96  		panel.setContent(panelContent);
97  
98  		final String pageId = getPageId(parameters);
99  		getPartyRankingMenuItemFactory().createPartyRankingMenuBar(menuBar);
100 
101 		createPageHeader(panel, panelContent, "Current Party Leaders Scoreboard", "Leader Performance",
102 				"Evaluate the performance of current party leaders including those not in government.");
103 
104 		final HorizontalLayout chartLayout = new HorizontalLayout();
105 		chartLayout.setSizeFull();
106 		panelContent.addComponent(chartLayout);
107 		panelContent.setExpandRatio(chartLayout, ContentRatio.LARGE_FORM);
108 
109 		final VerticalLayout wrapper = new VerticalLayout();
110 		wrapper.setSizeFull();
111 		chartLayout.addComponent(wrapper);
112 
113 		final ResponsiveRow row = RowUtil.createGridLayout(wrapper);
114 		row.setSizeFull();
115 
116 		final Map<String, ViewRiksdagenPolitician> politicianMap = loadPoliticiansByPersonId();
117 		final Map<String, Boolean> partyLeaderMap = computePartyLeaders(politicianMap.keySet());
118 
119 		final List<ViewRiksdagenPolitician> partyLeaders = politicianMap.values().stream()
120 				.filter(p -> partyLeaderMap.getOrDefault(p.getPersonId(), false))
121 				.collect(Collectors.toList());
122 
123 		// Sort: in government first, then alphabetical by last name
124 		partyLeaders.sort((a, b) -> {
125 			final boolean aInGov = a.isActiveGovernment();
126 			final boolean bInGov = b.isActiveGovernment();
127 			if (aInGov == bInGov) {
128 				return a.getLastName().compareToIgnoreCase(b.getLastName());
129 			}
130 			// government first
131 			return Boolean.compare(!aInGov, !bInGov);
132 		});
133 
134 		// Load ESV data for ministries
135 		final Map<Integer, List<GovernmentBodyAnnualSummary>> dataMap = esvApi.getData();
136 		final int CURRENT_YEAR = 2024;
137 		final List<GovernmentBodyAnnualSummary> currentYearGovernmentBodies = dataMap.get(CURRENT_YEAR);
138 		final Map<String, List<GovernmentBodyAnnualSummary>> governmentBodyByMinistry = currentYearGovernmentBodies
139 				.stream().collect(Collectors.groupingBy(GovernmentBodyAnnualSummary::getMinistry));
140 
141 		final Map<String, List<GovernmentBodyAnnualOutcomeSummary>> reportByMinistry = esvApi
142 				.getGovernmentBodyReportByMinistry();
143 
144 		for (final ViewRiksdagenPolitician leader : partyLeaders) {
145 			final Panel cardPanel = createLeaderCard(leader, governmentBodyByMinistry, reportByMinistry);
146 			row.addColumn().withDisplayRules(12, 6, 4, 4).withComponent(cardPanel);
147 		}
148 
149 		getPageActionEventHelper().createPageEvent(ViewAction.VISIT_PARTY_RANKING_VIEW, ApplicationEventGroup.USER, NAME,
150 				parameters, pageId);
151 
152 		return panelContent;
153 	}

170 	/**
171 	 * Compute party leaders.
172 	 *
173 	 * @param personIds the person ids
174 	 * @return the map
175 	 */
176 	private Map<String, Boolean> computePartyLeaders(Iterable<String> personIds) {
177 		final DataContainer<ViewRiksdagenPartyRoleMember, String> partyRoleMemberDataContainer = getApplicationManager()
178 				.getDataContainer(ViewRiksdagenPartyRoleMember.class);
179 
180 		final Map<String, Boolean> result = new HashMap<>();
181 		for (final String personId : personIds) {
182 			final List<ViewRiksdagenPartyRoleMember> roles = partyRoleMemberDataContainer.findListByProperty(
183 					new Object[] { personId, Boolean.TRUE }, ViewRiksdagenPartyRoleMember_.personId,
184 					ViewRiksdagenPartyRoleMember_.active);
185 
186 			final boolean isLeader = roles.stream()
187 					.anyMatch(role -> role.getRoleCode() != null && "Partiledare".equalsIgnoreCase(role.getRoleCode().trim()));
188 			result.put(personId, isLeader);
189 		}
190 		return result;
191 	}

235 	private Panel createLeaderCard(final ViewRiksdagenPolitician leader,
236 			final Map<String, List<GovernmentBodyAnnualSummary>> governmentBodyByMinistry,
237 			final Map<String, List<GovernmentBodyAnnualOutcomeSummary>> reportByMinistry) {
238 
239 		final Panel cardPanel = new Panel();
240 		cardPanel.addStyleName("leader-baseball-card");
241 		cardPanel.setSizeFull();
242 		Responsive.makeResponsive(cardPanel);
243 
244 		final VerticalLayout cardContent = new VerticalLayout();
245 		cardContent.setMargin(true);
246 		cardContent.setSpacing(true);
247 		cardContent.setSizeFull();
248 		cardPanel.setContent(cardContent);
249 
250 		// Header
251 		final HorizontalLayout headerLayout = new HorizontalLayout();
252 		headerLayout.setSpacing(true);

355 	 *
356 	 * @param leader the leader
357 	 * @return the view riksdagen goverment role member
358 	 */
359 	private ViewRiksdagenGovermentRoleMember findGovernmentRoleForLeader(ViewRiksdagenPolitician leader) {
360 		// Similar to ministry snippet loadActiveGovernmentRoleMembers:
361 		final DataContainer<ViewRiksdagenGovermentRoleMember, String> govermentRoleMemberDataContainer = getApplicationManager()
362 				.getDataContainer(ViewRiksdagenGovermentRoleMember.class);
363 		final List<ViewRiksdagenGovermentRoleMember> activeGovMembers = govermentRoleMemberDataContainer.findListByProperty(
364 				new Object[] { Boolean.TRUE }, ViewRiksdagenGovermentRoleMember_.active);
365 
366 		// Find the government role that matches this leader's personId
367 		return activeGovMembers.stream()
368 				.filter(govMember -> govMember.getPersonId().equals(leader.getPersonId()))
369 				.findFirst().orElse(null);
370 	}

372 	/**
373 	 * Adds the ministry role summary.
374 	 *
375 	 * @param cardLayout the card layout
376 	 * @param govMember the gov member
377 	 * @param governmentBodyByMinistry the government body by ministry
378 	 * @param reportByMinistry the report by ministry
379 	 */
380 	private void addMinistryRoleSummary(final VerticalLayout cardLayout,
381 			final ViewRiksdagenGovermentRoleMember govMember,
382 			final Map<String, List<GovernmentBodyAnnualSummary>> governmentBodyByMinistry,
383 			final Map<String, List<GovernmentBodyAnnualOutcomeSummary>> reportByMinistry) {
384 
385 		cardLayout.addComponent(getPageLinkFactory().addMinistryPageLink(govMember.getDetail()));
386 
387 		final List<GovernmentBodyAnnualSummary> ministryBodies = governmentBodyByMinistry.get(govMember.getDetail());
388 		if (ministryBodies == null || ministryBodies.isEmpty()) {
389 			return;
390 		}
391 
392 		final int totalHeadCount = ministryBodies.stream().mapToInt(GovernmentBodyAnnualSummary::getAnnualWorkHeadCount)
393 				.sum();
394 		final int bodyCount = ministryBodies.size();
395 
396 		cardLayout.addComponent(createMetricRow(VaadinIcons.GROUP,
397 				getPageLinkFactory().addMinistryGovermentBodiesPageLink(govMember.getDetail()),
398 				"Number of government bodies", String.valueOf(bodyCount)));
399 
400 		cardLayout.addComponent(createMetricRow(VaadinIcons.USER,
401 				getPageLinkFactory().addMinistryGovermentBodiesHeadcountPageLink(govMember.getDetail()),
402 				"Total headcount of government bodies", String.valueOf(totalHeadCount)));
403 
404 		final List<GovernmentBodyAnnualOutcomeSummary> outcomeSummaries = reportByMinistry.get(govMember.getDetail());
405 		double currentYearIncome = 0;
406 		double currentYearSpending = 0;
407 		final int CURRENT_YEAR = 2024;
408 		final String INKOMSTTITELGRUPPSNAMN = "Inkomsttitelgruppsnamn";
409 		final String EXPENDITURE_GROUP_NAME = "UtgiftsomrÃ¥desnamn";
410 
411 		if (outcomeSummaries != null) {
412 			final Map<Integer, Double> annualIncome = outcomeSummaries.stream()
413 					.filter(t -> t.getDescriptionFields().get(INKOMSTTITELGRUPPSNAMN) != null)
414 					.collect(Collectors.groupingBy(GovernmentBodyAnnualOutcomeSummary::getYear,
415 							Collectors.summingDouble(GovernmentBodyAnnualOutcomeSummary::getYearTotal)));
416 
417 			final Map<Integer, Double> annualSpending = outcomeSummaries.stream()
418 					.filter(t -> t.getDescriptionFields().get(EXPENDITURE_GROUP_NAME) != null)
419 					.collect(Collectors.groupingBy(GovernmentBodyAnnualOutcomeSummary::getYear,
420 							Collectors.summingDouble(GovernmentBodyAnnualOutcomeSummary::getYearTotal)));
421 
422 			if (annualIncome.get(CURRENT_YEAR) != null) {
423 				currentYearIncome = annualIncome.get(CURRENT_YEAR) / 1000;
424 			}
425 
426 			if (annualSpending.get(CURRENT_YEAR) != null) {
427 				currentYearSpending = annualSpending.get(CURRENT_YEAR) / 1000;
428 			}
429 		}
430 
431 		final String incomeStr = String.format(Locale.ROOT, "%.2f B SEK", currentYearIncome);
432 		cardLayout.addComponent(createMetricRow(VaadinIcons.ARROW_UP,
433 				getPageLinkFactory().addMinistryGovermentBodiesIncomePageLink(govMember.getDetail()),
434 				"Yearly Income (B SEK)", incomeStr));
435 
436 		final String spendingStr = String.format(Locale.ROOT, "%.2f B SEK", currentYearSpending);
437 		cardLayout.addComponent(createMetricRow(VaadinIcons.ARROW_DOWN,
438 				getPageLinkFactory().addMinistrGovermentBodiesSpendingPageLink(govMember.getDetail()),
439 				"Yearly Spending (B SEK)", spendingStr));
440 	}
