43  /**
44   * The Class CommitteeDecisionSummaryPageModContentFactoryImpl.
45   */
46  @Component
47  public final class CommitteeDecisionSummaryPageModContentFactoryImpl
48  		extends AbstractCommitteePageModContentFactoryImpl {
49  
50  	private static final String BALLOT_ID = "ballotId";
51  
52  	private static final String[] COLUMN_ORDER = { "createdDate", "publicDate", "committeeReport",
53  			"embeddedId.hangarId", "embeddedId.id", "embeddedId.issueNummer", "rm", "decisionType", "winner", "title",
54  			"header", "endNumber", "org", "committeeProposalUrlXml", BALLOT_ID, "againstProposalParties",
55  			"againstProposalNumber" };
56  
57  	/** The Constant COMMITTEE. */
58  	private static final String COMMITTEE = "Committee:";
59  
60  	/** The Constant DECISION_SUMMARY. */
61  	private static final String DECISION_SUMMARY = "Decision Summary";
62  
63  	private static final String[] HIDE_COLUMNS = { "embeddedId", "embeddedId.hangarId", "embeddedId.id",
64  			"endNumber", "org", "committeeProposalUrlXml", BALLOT_ID, "againstProposalParties", "againstProposalNumber",
65  			"createdDate" };
66  
67  	private static final PageItemPropertyClickListener LISTENER = new PageItemPropertyClickListener(
68  			UserViews.BALLOT_VIEW_NAME, BALLOT_ID);
69  
70  	private static final String[] NESTED_PROPERTIES = { "embeddedId.hangarId", "embeddedId.id",
71  			"embeddedId.issueNummer" };
72  
73  	/**
74  	 * Instantiates a new committee decision summary page mod content factory
75  	 * impl.
76  	 */
77  	public CommitteeDecisionSummaryPageModContentFactoryImpl() {
78  		super();
79  	}
80  
81  	@Secured({ "ROLE_ANONYMOUS", "ROLE_USER", "ROLE_ADMIN" })
82  	@Override
83  	public Layout createContent(final String parameters, final MenuBar menuBar, final Panel panel) {
84  		final VerticalLayout panelContent = createPanelContent();
85  
86  		final String pageId = getPageId(parameters);
87  
88  		final ViewRiksdagenCommittee viewRiksdagenCommittee = getItem(parameters);
89  		getCommitteeMenuItemFactory().createCommitteeeMenuBar(menuBar, pageId);
90  
91  		LabelFactory.createHeader2Label(panelContent, DECISION_SUMMARY);
92  
93  		final DataContainer<ViewRiksdagenCommitteeDecisions, ViewRiksdagenCommitteeDecisionsEmbeddedId> committeeDecisionDataContainer = getApplicationManager()
94  				.getDataContainer(ViewRiksdagenCommitteeDecisions.class);
95  
96  		final List<ViewRiksdagenCommitteeDecisions> decisionPartySummaryList = committeeDecisionDataContainer
97  				.findOrderedListByProperty(ViewRiksdagenCommitteeDecisions_.org, pageId,
98  						ViewRiksdagenCommitteeDecisions_.createdDate);
99  
100 		getGridFactory().createBasicBeanItemNestedPropertiesGrid(panelContent, ViewRiksdagenCommitteeDecisions.class,
101 				decisionPartySummaryList, DECISION_SUMMARY, NESTED_PROPERTIES, COLUMN_ORDER, HIDE_COLUMNS, LISTENER,
102 				BALLOT_ID, null);
103 
104 		panel.setCaption(NAME + "::" + COMMITTEE + viewRiksdagenCommittee.getEmbeddedId().getDetail());
105 		getPageActionEventHelper().createPageEvent(ViewAction.VISIT_COMMITTEE_VIEW, ApplicationEventGroup.USER, NAME,
106 				parameters, pageId);
107 		return panelContent;
108 
109 	}
110 
111 	@Override
112 	public boolean matches(final String page, final String parameters) {
113 		return NAME.equals(page) && StringUtils.contains(parameters, CommitteePageMode.DECISIONSUMMARY.toString());
114 	}
115 
116 }
