206 	 * Adds the ministry role summary.
207 	 *
208 	 * @param cardLayout the card layout
209 	 * @param govMember the gov member
210 	 * @param governmentBodyByMinistry the government body by ministry
211 	 * @param reportByMinistry the report by ministry
212 	 */
213 	public final void addMinistryRoleSummary(final VerticalLayout cardLayout,
214 			final ViewRiksdagenGovermentRoleMember govMember,
215 			final Map<String, List<GovernmentBodyAnnualSummary>> governmentBodyByMinistry,
216 			final Map<String, List<GovernmentBodyAnnualOutcomeSummary>> reportByMinistry) {
217 
218 		cardLayout.addComponent(pageLinkFactory.addMinistryPageLink(govMember.getDetail()));
219 
220 		final List<GovernmentBodyAnnualSummary> ministryBodies = governmentBodyByMinistry.get(govMember.getDetail());
221 		if (ministryBodies == null || ministryBodies.isEmpty()) {
222 			return;
223 		}
224 
225 		final int totalHeadCount = ministryBodies.stream().mapToInt(GovernmentBodyAnnualSummary::getAnnualWorkHeadCount)
226 				.sum();
227 		final int bodyCount = ministryBodies.size();
228 
229 		cardLayout.addComponent(createMetricRow(VaadinIcons.GROUP,
230 				pageLinkFactory.addMinistryGovermentBodiesPageLink(govMember.getDetail()),
231 				"Number of government bodies", String.valueOf(bodyCount)));
232 
233 		cardLayout.addComponent(createMetricRow(VaadinIcons.USER,
234 				pageLinkFactory.addMinistryGovermentBodiesHeadcountPageLink(govMember.getDetail()),
235 				"Total headcount of government bodies", String.valueOf(totalHeadCount)));
236 
237 		final List<GovernmentBodyAnnualOutcomeSummary> outcomeSummaries = reportByMinistry.get(govMember.getDetail());
238 		double currentYearIncome = 0;
239 		double currentYearSpending = 0;
240 		final int CURRENT_YEAR = 2024;
241 		final String INKOMSTTITELGRUPPSNAMN = "Inkomsttitelgruppsnamn";
242 		final String EXPENDITURE_GROUP_NAME = "UtgiftsomrÃ¥desnamn";
243 
244 		if (outcomeSummaries != null) {
245 			final Map<Integer, Double> annualIncome = outcomeSummaries.stream()
246 					.filter(t -> t.getDescriptionFields().get(INKOMSTTITELGRUPPSNAMN) != null)
247 					.collect(Collectors.groupingBy(GovernmentBodyAnnualOutcomeSummary::getYear,
248 							Collectors.summingDouble(GovernmentBodyAnnualOutcomeSummary::getYearTotal)));
249 
250 			final Map<Integer, Double> annualSpending = outcomeSummaries.stream()
251 					.filter(t -> t.getDescriptionFields().get(EXPENDITURE_GROUP_NAME) != null)
252 					.collect(Collectors.groupingBy(GovernmentBodyAnnualOutcomeSummary::getYear,
253 							Collectors.summingDouble(GovernmentBodyAnnualOutcomeSummary::getYearTotal)));
254 
255 			if (annualIncome.get(CURRENT_YEAR) != null) {
256 				currentYearIncome = annualIncome.get(CURRENT_YEAR) / 1000;
257 			}
258 
259 			if (annualSpending.get(CURRENT_YEAR) != null) {
260 				currentYearSpending = annualSpending.get(CURRENT_YEAR) / 1000;
261 			}
262 		}
263 
264 		final String incomeStr = String.format(Locale.ENGLISH, "%.2f B SEK", currentYearIncome);
265 		cardLayout.addComponent(createMetricRow(VaadinIcons.ARROW_UP,
266 				pageLinkFactory.addMinistryGovermentBodiesIncomePageLink(govMember.getDetail()),
267 				"Yearly Income (B SEK)", incomeStr));
268 
269 		final String spendingStr = String.format(Locale.ENGLISH, "%.2f B SEK", currentYearSpending);
270 		cardLayout.addComponent(createMetricRow(VaadinIcons.ARROW_DOWN,
271 				pageLinkFactory.addMinistrGovermentBodiesSpendingPageLink(govMember.getDetail()),
272 				"Yearly Spending (B SEK)", spendingStr));
273 	}

275 	/**
276 	 * Builds the top string.
277 	 *
278 	 * @param <T> the generic type
279 	 * @param items the items
280 	 * @param nameFunction the name function
281 	 * @param weightFunction the weight function
282 	 * @return the string
283 	 */
284 	private <T> String buildTopString(final List<T> items,
285 	                                  final java.util.function.Function<T, String> nameFunction,
286 	                                  final java.util.function.Function<T, Long> weightFunction) {
287 	    return items.stream()
288 	        .filter(i -> nameFunction.apply(i) != null && !"Other".equals(nameFunction.apply(i)))
289 	        .sorted((o1, o2) -> weightFunction.apply(o2).compareTo(weightFunction.apply(o1)))
290 	        .limit(3)
291 	        .map(nameFunction)
292 	        .collect(Collectors.joining(", "));
293 	}
