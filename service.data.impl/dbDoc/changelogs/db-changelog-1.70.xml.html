    <changeSet id="create-mv-annual-voting-metrics-1.70-001" author="intelligence-operative">
        <comment>
            Create materialized view for annual voting metrics (pre-aggregated from vote_data).
            
            Purpose: Eliminate cartesian product by aggregating vote_data by year FIRST,
            before joining with document metrics.
            
            Impact: Reduces annual_metrics CTE from 3.5M * 109K row join to 24 pre-aggregated rows.
            
            Refresh: Daily via cron job (see MATERIALIZED_VIEW_REFRESH_SCHEDULE.md)
        </comment>
        <sql><![CDATA[
CREATE MATERIALIZED VIEW mv_annual_voting_metrics AS
            WITH election_years AS (
                SELECT unnest(ARRAY[2002, 2006, 2010, 2014, 2018, 2022, 2026]) AS election_year
            )
            SELECT
                EXTRACT(year FROM vd.vote_date)::INTEGER AS year,
                CASE 
                    WHEN ey.election_year IS NOT NULL THEN TRUE 
                    ELSE FALSE 
                END AS is_election_year,
                COUNT(DISTINCT vd.embedded_id_ballot_id) AS total_ballots,
                COUNT(DISTINCT vd.embedded_id_intressent_id) AS active_politicians,
                ROUND(AVG(CASE WHEN vd.vote::TEXT != 'Frånvarande' THEN 1 ELSE 0 END) * 100, 2) AS attendance_rate,
                ROUND(AVG(CASE WHEN vd.vote::TEXT = 'Ja' THEN 1 ELSE 0 END) * 100, 2) AS avg_yes_rate,
                ROUND(AVG(CASE WHEN vd.vote::TEXT = 'Nej' THEN 1 ELSE 0 END) * 100, 2) AS avg_no_rate,
                ROUND(AVG(CASE WHEN vd.vote::TEXT = 'Avstår' THEN 1 ELSE 0 END) * 100, 2) AS avg_abstain_rate
            FROM public.vote_data vd
            LEFT JOIN election_years ey ON ey.election_year::NUMERIC = EXTRACT(year FROM vd.vote_date)
            WHERE vd.vote_date IS NOT NULL
              AND EXTRACT(year FROM vd.vote_date) BETWEEN 2002 AND 2026
            GROUP BY 
                EXTRACT(year FROM vd.vote_date)::INTEGER,
                CASE WHEN ey.election_year IS NOT NULL THEN TRUE ELSE FALSE END;

            CREATE UNIQUE INDEX idx_mv_annual_voting_metrics_year 
                ON mv_annual_voting_metrics (year);
            
            COMMENT ON MATERIALIZED VIEW mv_annual_voting_metrics IS 
                'Pre-aggregated annual voting metrics from vote_data (2002-2026). 
                 Replaces expensive vote_data aggregation in annual_metrics CTE.
                 Refresh: Daily. Performance: Reduces 3.5M row scan to 24 pre-computed rows.';
        ]]></sql>
        <rollback>
            DROP MATERIALIZED VIEW IF EXISTS mv_annual_voting_metrics CASCADE;
        </rollback>
    </changeSet>

    <changeSet id="create-mv-annual-document-metrics-1.70-002" author="intelligence-operative">
        <comment>
            Create materialized view for annual document metrics (pre-aggregated from document_data).
            
            Purpose: Separate document aggregation from voting aggregation to eliminate 
            cartesian product when joining on year.
            
            Impact: Reduces document_data contribution to annual_metrics from 109K rows to 24 rows.
            
            Refresh: Daily via cron job (see MATERIALIZED_VIEW_REFRESH_SCHEDULE.md)
        </comment>
        <sql><![CDATA[
CREATE MATERIALIZED VIEW mv_annual_document_metrics AS
            SELECT
                EXTRACT(year FROM dd.made_public_date)::INTEGER AS year,
                COUNT(DISTINCT dd.id) AS documents_produced,
                COUNT(DISTINCT CASE 
                    WHEN dd.document_type::TEXT = 'mot' THEN dd.id 
                    ELSE NULL 
                END) AS motions_filed,
                COUNT(DISTINCT CASE 
                    WHEN dd.document_type::TEXT = 'prop' THEN dd.id 
                    ELSE NULL 
                END) AS proposals_filed
            FROM public.document_data dd
            WHERE dd.made_public_date IS NOT NULL
              AND EXTRACT(year FROM dd.made_public_date) BETWEEN 2002 AND 2026
            GROUP BY EXTRACT(year FROM dd.made_public_date)::INTEGER;

            CREATE UNIQUE INDEX idx_mv_annual_document_metrics_year 
                ON mv_annual_document_metrics (year);
            
            COMMENT ON MATERIALIZED VIEW mv_annual_document_metrics IS 
                'Pre-aggregated annual document metrics from document_data (2002-2026). 
                 Replaces expensive document_data aggregation in annual_metrics CTE.
                 Refresh: Daily. Performance: Reduces 109K row scan to 24 pre-computed rows.';
        ]]></sql>
        <rollback>
            DROP MATERIALIZED VIEW IF EXISTS mv_annual_document_metrics CASCADE;
        </rollback>
    </changeSet>

    <changeSet id="recreate-view-riksdagen-election-year-behavioral-patterns-1.70-003" author="intelligence-operative">
        <comment>
            Recreate view_riksdagen_election_year_behavioral_patterns using materialized views.
            
            Problem: Original view had cartesian product:
            - vote_data (3.5M rows) LEFT JOIN document_data (109K rows) ON year
            - For a single year (e.g., 2020): 150K votes * 5K documents = 750M intermediate rows
            - Total temp file usage: 50GB+ (exceeds limit)
            
            Solution: Join pre-aggregated materialized views instead:
            - mv_annual_voting_metrics (24 rows) JOIN mv_annual_document_metrics (24 rows)
            - Total intermediate rows: 24 (one per year)
            - Temp file usage: less than 1MB
            
            Performance improvement: 99.99% reduction in intermediate result size.
        </comment>
        <sql><![CDATA[
DROP VIEW IF EXISTS public.view_riksdagen_election_year_behavioral_patterns CASCADE;
            
            CREATE VIEW public.view_riksdagen_election_year_behavioral_patterns AS
            WITH election_years AS (
                SELECT unnest(ARRAY[2002, 2006, 2010, 2014, 2018, 2022, 2026]) AS election_year
            ),
            annual_metrics AS (
                -- Join pre-aggregated metrics instead of raw tables
                SELECT
                    vm.year,
                    vm.is_election_year,
                    vm.total_ballots,
                    vm.active_politicians,
                    vm.attendance_rate,
                    vm.avg_yes_rate,
                    vm.avg_no_rate,
                    vm.avg_abstain_rate,
                    COALESCE(dm.documents_produced, 0) AS documents_produced,
                    COALESCE(dm.motions_filed, 0) AS motions_filed,
                    COALESCE(dm.proposals_filed, 0) AS proposals_filed
                FROM mv_annual_voting_metrics vm
                LEFT JOIN mv_annual_document_metrics dm ON vm.year = dm.year
            ),
            election_baseline AS (
                SELECT
                    PERCENTILE_CONT(0.5) WITHIN GROUP (ORDER BY total_ballots::DOUBLE PRECISION) 
                        FILTER (WHERE is_election_year) AS election_median_ballots,
                    AVG(total_ballots) FILTER (WHERE is_election_year) AS election_avg_ballots,
                    STDDEV(total_ballots) FILTER (WHERE is_election_year) AS election_stddev_ballots,
                    AVG(total_ballots) FILTER (WHERE NOT is_election_year) AS midterm_avg_ballots,
                    STDDEV(total_ballots) FILTER (WHERE NOT is_election_year) AS midterm_stddev_ballots,
                    
                    PERCENTILE_CONT(0.5) WITHIN GROUP (ORDER BY documents_produced::DOUBLE PRECISION) 
                        FILTER (WHERE is_election_year) AS election_median_docs,
                    AVG(documents_produced) FILTER (WHERE is_election_year) AS election_avg_docs,
                    STDDEV(documents_produced) FILTER (WHERE is_election_year) AS election_stddev_docs,
                    AVG(documents_produced) FILTER (WHERE NOT is_election_year) AS midterm_avg_docs,
                    
                    PERCENTILE_CONT(0.5) WITHIN GROUP (ORDER BY motions_filed::DOUBLE PRECISION) 
                        FILTER (WHERE is_election_year) AS election_median_motions,
                    AVG(motions_filed) FILTER (WHERE is_election_year) AS election_avg_motions,
                    STDDEV(motions_filed) FILTER (WHERE is_election_year) AS election_stddev_motions,
                    AVG(motions_filed) FILTER (WHERE NOT is_election_year) AS midterm_avg_motions,
                    
                    AVG(attendance_rate) FILTER (WHERE is_election_year) AS election_avg_attendance,
                    STDDEV(attendance_rate) FILTER (WHERE is_election_year) AS election_stddev_attendance,
                    AVG(attendance_rate) FILTER (WHERE NOT is_election_year) AS midterm_avg_attendance
                FROM annual_metrics
            )
            SELECT
                am.year,
                am.is_election_year,
                am.total_ballots,
                am.active_politicians,
                am.attendance_rate,
                am.avg_yes_rate,
                am.avg_no_rate,
                am.avg_abstain_rate,
                am.documents_produced,
                am.motions_filed,
                am.proposals_filed,
                
                -- Election baseline metrics
                ROUND(eb.election_median_ballots::NUMERIC, 2) AS election_median_ballots,
                ROUND(eb.election_avg_ballots, 2) AS election_avg_ballots,
                ROUND(eb.election_stddev_ballots, 2) AS election_stddev_ballots,
                ROUND(eb.midterm_avg_ballots, 2) AS midterm_avg_ballots,
                ROUND(eb.midterm_stddev_ballots, 2) AS midterm_stddev_ballots,
                ROUND(am.total_ballots::NUMERIC / NULLIF(eb.midterm_avg_ballots, 0), 2) AS ballot_ratio_vs_midterm,
                ROUND(am.total_ballots::NUMERIC / NULLIF(eb.election_avg_ballots, 0), 2) AS ballot_ratio_vs_election_avg,
                
                ROUND(eb.election_median_docs::NUMERIC, 2) AS election_median_docs,
                ROUND(eb.election_avg_docs, 2) AS election_avg_docs,
                ROUND(eb.election_stddev_docs, 2) AS election_stddev_docs,
                ROUND(eb.midterm_avg_docs, 2) AS midterm_avg_docs,
                ROUND(am.documents_produced::NUMERIC / NULLIF(eb.midterm_avg_docs, 0), 2) AS doc_ratio_vs_midterm,
                ROUND(am.documents_produced::NUMERIC / NULLIF(eb.election_avg_docs, 0), 2) AS doc_ratio_vs_election_avg,
                
                ROUND(eb.election_median_motions::NUMERIC, 2) AS election_median_motions,
                ROUND(eb.election_avg_motions, 2) AS election_avg_motions,
                ROUND(eb.election_stddev_motions, 2) AS election_stddev_motions,
                ROUND(eb.midterm_avg_motions, 2) AS midterm_avg_motions,
                ROUND(am.motions_filed::NUMERIC / NULLIF(eb.midterm_avg_motions, 0), 2) AS motion_ratio_vs_midterm,
                
                ROUND(eb.election_avg_attendance, 2) AS election_avg_attendance,
                ROUND(eb.midterm_avg_attendance, 2) AS midterm_avg_attendance,
                
                -- Z-scores
                CASE
                    WHEN eb.election_stddev_ballots > 0 AND am.is_election_year 
                    THEN ROUND((am.total_ballots::NUMERIC - eb.election_avg_ballots) / eb.election_stddev_ballots, 2)
                    ELSE 0
                END AS ballot_z_score_vs_election_avg,
                
                CASE
                    WHEN eb.election_stddev_docs > 0 AND am.is_election_year 
                    THEN ROUND((am.documents_produced::NUMERIC - eb.election_avg_docs) / eb.election_stddev_docs, 2)
                    ELSE 0
                END AS doc_z_score_vs_election_avg,
                
                -- Classifications
                CASE
                    WHEN am.is_election_year AND am.total_ballots::NUMERIC > (eb.election_avg_ballots + eb.election_stddev_ballots) 
                        THEN 'HIGH_ELECTION_ACTIVITY'
                    WHEN am.is_election_year AND am.total_ballots::NUMERIC < (eb.election_avg_ballots - eb.election_stddev_ballots) 
                        THEN 'LOW_ELECTION_ACTIVITY'
                    WHEN am.is_election_year THEN 'NORMAL_ELECTION_ACTIVITY'
                    ELSE 'MIDTERM_YEAR'
                END AS year_classification,
                
                CASE
                    WHEN am.is_election_year THEN
                        CASE
                            WHEN (am.total_ballots::NUMERIC > (eb.election_avg_ballots + eb.election_stddev_ballots))
                              OR (am.documents_produced::NUMERIC > (eb.election_avg_docs + eb.election_stddev_docs))
                              OR (am.motions_filed::NUMERIC > (eb.election_avg_motions + eb.election_stddev_motions))
                                THEN 'HIGH_ACTIVITY_ELECTION'
                            WHEN (am.total_ballots::NUMERIC < (eb.election_avg_ballots - eb.election_stddev_ballots))
                              AND (am.documents_produced::NUMERIC < (eb.election_avg_docs - eb.election_stddev_docs))
                                THEN 'LOW_ACTIVITY_ELECTION'
                            ELSE 'TYPICAL_ELECTION'
                        END
                    ELSE
                        CASE
                            WHEN am.total_ballots::NUMERIC > (eb.midterm_avg_ballots + COALESCE(eb.midterm_stddev_ballots, 0))
                                THEN 'HIGH_ACTIVITY_MIDTERM'
                            WHEN am.total_ballots::NUMERIC < (eb.midterm_avg_ballots - COALESCE(eb.midterm_stddev_ballots, 0))
                                THEN 'LOW_ACTIVITY_MIDTERM'
                            ELSE 'TYPICAL_MIDTERM'
                        END
                END AS composite_classification,
                
                LAG(am.total_ballots) OVER (ORDER BY am.year) AS prev_year_ballots,
                
                CASE
                    WHEN LAG(am.total_ballots) OVER (ORDER BY am.year) IS NOT NULL 
                      AND LAG(am.total_ballots) OVER (ORDER BY am.year) > 0
                    THEN ROUND(((am.total_ballots - LAG(am.total_ballots) OVER (ORDER BY am.year))::NUMERIC 
                        / LAG(am.total_ballots) OVER (ORDER BY am.year)::NUMERIC) * 100, 2)
                    ELSE NULL
                END AS yoy_ballot_change_pct
                
            FROM annual_metrics am
            CROSS JOIN election_baseline eb
            ORDER BY am.year;
            
            COMMENT ON VIEW public.view_riksdagen_election_year_behavioral_patterns IS 
                'Annual behavioral pattern comparison across all election years (2002, 2006, 2010, 2014, 2018, 2022, 2026) vs midterm years. 
                 OPTIMIZED: Uses materialized views (mv_annual_voting_metrics, mv_annual_document_metrics) to eliminate cartesian product. 
                 Features: multi-dimensional metrics (voting, documents, motions, attendance), election year baseline calculation using median and average, 
                 activity ratios vs both election and midterm baselines, z-score anomaly detection, year classification (HIGH/NORMAL/LOW election activity), 
                 composite classification considering multiple dimensions. Data Sources: vote_data (voting patterns), document_data (productivity). 
                 Election cycles: 7 election years, 17 midterm years (2002-2026). Framework 1: Temporal Analysis - Annual comparison. 
                 Use Case: Identify systematic election year behavioral shifts, detect unusual election patterns, predict future election behavior based on historical patterns. 
                 Performance: Reduced from 50GB+ temp file usage to less than 1MB by using pre-aggregated materialized views.';
        ]]></sql>
        <rollback>
            -- Rollback would require recreating original view, which is complex
            -- Better to handle via new changeset if needed
        </rollback>
    </changeSet>
