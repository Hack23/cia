84  			final ViewRiksdagenCommittee viewRiksdagenCommittee = committees

86  			final ViewRiksdagenCommittee viewRiksdagenCommitteeLoaded = dataViewer
87  					.load(ViewRiksdagenCommittee.class,
88  							viewRiksdagenCommittee.getEmbeddedId());
89  			assertNotNull(EXPECT_VALUE_IN_DATABASE,viewRiksdagenCommitteeLoaded);
90  
91  			final ViewRiksdagenCommittee viewRiksdagenCommitteeFound = dataViewer
92  					.findFirstByProperty(ViewRiksdagenCommittee.class,
93  							ViewRiksdagenCommittee_.embeddedId,
94  							viewRiksdagenCommitteeLoaded.getEmbeddedId());
95  
96  			assertEquals(EXPECT_SAME_OBJECT_LOADED,viewRiksdagenCommitteeLoaded,
97  					viewRiksdagenCommitteeFound);
98  		}

187 	/**
188 	 * View riksdagen party summary.
189 	 *
190 	 * @throws Exception
191 	 *             the exception
192 	 */
193 	@Test
194 	public void viewRiksdagenPartySummary() throws Exception {
195 
196 		final List<ViewRiksdagenPartySummary> committees = dataViewer
197 				.getAll(ViewRiksdagenPartySummary.class);
198 		assertNotNull(EXPECT_VALUE_IN_DATABASE,committees);
199 		if (committees.size() > 0) {
200 			final ViewRiksdagenPartySummary viewRiksdagenPartySummary = committees
201 					.get(4);
202 			final ViewRiksdagenPartySummary viewRiksdagenPartySummaryLoaded = dataViewer
203 					.load(ViewRiksdagenPartySummary.class,
204 							viewRiksdagenPartySummary.getParty());
205 			assertNotNull(EXPECT_VALUE_IN_DATABASE,viewRiksdagenPartySummaryLoaded);
206 
207 			final ViewRiksdagenPartySummary viewRiksdagenPartySummaryFound = dataViewer
208 					.findFirstByProperty(ViewRiksdagenPartySummary.class,
209 							ViewRiksdagenPartySummary_.party,
210 							viewRiksdagenPartySummaryLoaded.getParty());
211 
212 			assertEquals(EXPECT_SAME_OBJECT_LOADED,viewRiksdagenPartySummaryLoaded,
213 					viewRiksdagenPartySummaryFound);
214 		}
215 
216 		long parliamentSum = 0;
217 		long euSum = 0;
218 		long governmentSum = 0;
219 
220 		for (final ViewRiksdagenPartySummary viewRiksdagenPartySummary : committees) {
221 			if (viewRiksdagenPartySummary != null) {
222 				parliamentSum = parliamentSum
223 						+ viewRiksdagenPartySummary.getTotalActiveParliament();
224 				euSum = euSum + viewRiksdagenPartySummary.getTotalActiveEu();
225 				governmentSum = governmentSum
226 						+ viewRiksdagenPartySummary.getTotalActiveGovernment();
227 			}
228 		}
229 
230 		final List<ViewRiksdagenPolitician> activeWithNoParty = dataViewer.findOrderedListByProperty(ViewRiksdagenPolitician.class,ViewRiksdagenPolitician_.firstAssignmentDate, new Object[]  {true,null},ViewRiksdagenPolitician_.activeParliament,ViewRiksdagenPolitician_.party);
231 
232 		assertTrue(PARLIAMENT_DATA, ( parliamentSum + activeWithNoParty.size()) >= 349);
233 //		assertEquals(
234 //				"Should always be 20 in eu, but riksdagen data contains only 15",
235 //				15, euSum);
236 //		assertEquals("Should always be 23 in current government", 23,
237 //				governmentSum);
238 
239 	}
