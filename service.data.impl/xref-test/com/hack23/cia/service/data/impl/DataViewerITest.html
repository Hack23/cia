78  				.getAll(ViewRiksdagenCommittee.class);
79  		assertNotNull(EXPECT_VALUE_IN_DATABASE,committees);
80  		if (committees.size() > 0) {
81  			final ViewRiksdagenCommittee viewRiksdagenCommittee = committees
82  					.get(0);
83  			final ViewRiksdagenCommittee viewRiksdagenCommitteeLoaded = dataViewer
84  					.load(ViewRiksdagenCommittee.class,
85  							viewRiksdagenCommittee.getEmbeddedId());
86  			assertNotNull(EXPECT_VALUE_IN_DATABASE,viewRiksdagenCommitteeLoaded);
87  
88  			final ViewRiksdagenCommittee viewRiksdagenCommitteeFound = dataViewer
89  					.findFirstByProperty(ViewRiksdagenCommittee.class,
90  							ViewRiksdagenCommittee_.embeddedId,
91  							viewRiksdagenCommitteeLoaded.getEmbeddedId());
92  
93  			assertEquals(EXPECT_SAME_OBJECT_LOADED,viewRiksdagenCommitteeLoaded,
94  					viewRiksdagenCommitteeFound);
95  		}
96  	}

184 	/**
185 	 * View riksdagen party summary.
186 	 *
187 	 * @throws Exception
188 	 *             the exception
189 	 */
190 	@Test
191 	public void viewRiksdagenPartySummary() throws Exception {
192 
193 		final List<ViewRiksdagenPartySummary> committees = dataViewer
194 				.getAll(ViewRiksdagenPartySummary.class);
195 		assertNotNull(EXPECT_VALUE_IN_DATABASE,committees);
196 		if (committees.size() > 0) {
197 			final ViewRiksdagenPartySummary viewRiksdagenPartySummary = committees
198 					.get(4);
199 			final ViewRiksdagenPartySummary viewRiksdagenPartySummaryLoaded = dataViewer
200 					.load(ViewRiksdagenPartySummary.class,
201 							viewRiksdagenPartySummary.getParty());
202 			assertNotNull(EXPECT_VALUE_IN_DATABASE,viewRiksdagenPartySummaryLoaded);
203 
204 			final ViewRiksdagenPartySummary viewRiksdagenPartySummaryFound = dataViewer
205 					.findFirstByProperty(ViewRiksdagenPartySummary.class,
206 							ViewRiksdagenPartySummary_.party,
207 							viewRiksdagenPartySummaryLoaded.getParty());
208 
209 			assertEquals(EXPECT_SAME_OBJECT_LOADED,viewRiksdagenPartySummaryLoaded,
210 					viewRiksdagenPartySummaryFound);
211 		}
212 
213 		long parliamentSum = 0;
214 		long euSum = 0;
215 		long governmentSum = 0;
216 
217 		for (final ViewRiksdagenPartySummary viewRiksdagenPartySummary : committees) {
218 			if (viewRiksdagenPartySummary != null) {
219 				parliamentSum = parliamentSum
220 						+ viewRiksdagenPartySummary.getTotalActiveParliament();
221 				euSum = euSum + viewRiksdagenPartySummary.getTotalActiveEu();
222 				governmentSum = governmentSum
223 						+ viewRiksdagenPartySummary.getTotalActiveGovernment();
224 			}
225 		}
226 
227 		final List<ViewRiksdagenPolitician> activeWithNoParty = dataViewer.findOrderedListByProperty(ViewRiksdagenPolitician.class,ViewRiksdagenPolitician_.firstAssignmentDate, new Object[]  {true,null},ViewRiksdagenPolitician_.activeParliament,ViewRiksdagenPolitician_.party);
228 
229 		assertEquals(SHOULD_ALWAYS_BE_349_IN_PARLIAMENT, 349, parliamentSum + activeWithNoParty.size());
230 //		assertEquals(
231 //				"Should always be 20 in eu, but riksdagen data contains only 15",
232 //				15, euSum);
233 //		assertEquals("Should always be 23 in current government", 23,
234 //				governmentSum);
235 
236 	}
