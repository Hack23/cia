82  	 * Adds the cache hints.
83  	 *
84  	 * @param typedQuery
85  	 *            the typed query
86  	 * @param comment
87  	 *            the comment
88  	 */
89  	protected final void addCacheHints(final TypedQuery<?> typedQuery, final String comment) {
90  		typedQuery.setHint("org.hibernate.cacheMode", CacheMode.NORMAL);
91  		typedQuery.setHint("org.hibernate.cacheable", Boolean.TRUE);
92  		typedQuery.setHint("org.hibernate.comment", comment);

103 		final CriteriaQuery<String> criteria = getCriteriaBuilder().createQuery(String.class);

116 	@Override
117 	public final T findFirstByProperty(final SingularAttribute<T, ? extends Object> property, final Object value) {
118 		final CriteriaQuery<T> criteriaQuery = criteriaBuilder.createQuery(getPersistentClass());
119 		final Root<T> root = criteriaQuery.from(getPersistentClass());
120 		criteriaQuery.select(root);
121 		final Predicate condition = criteriaBuilder.equal(root.get(property), value);
122 		criteriaQuery.where(condition);
123 		final TypedQuery<T> typedQuery = getEntityManager().createQuery(criteriaQuery);
124 		addCacheHints(typedQuery, "findFirstByProperty");
125 
126 		final List<T> resultList = typedQuery.getResultList();
127 
128 		if (resultList.isEmpty()) {
129 			return null;
130 		} else {
131 			return LoadHelper.recursiveInitialize(resultList.get(0));
132 		}
133 	}

135 	@Override
136 	public final List<T> findListByProperty(final Object[] values,
137 			final SingularAttribute<T, ? extends Object>... properties) {
138 		final CriteriaQuery<T> criteriaQuery = criteriaBuilder.createQuery(persistentClass);
139 		final Root<T> root = criteriaQuery.from(persistentClass);
140 		criteriaQuery.select(root);
141 
142 		final Object value = values[0];
143 		final SingularAttribute<T, ? extends Object> property = properties[0];
144 		Predicate condition;
145 
146 		condition = QueryHelper.equalsIgnoreCaseIfStringPredicate(criteriaBuilder, root, value, property);
147 
148 		if (values.length > 1) {
149 			for (int i = 1; i < properties.length; i++) {
150 				final SingularAttribute<T, ? extends Object> property2 = properties[i];
151 				final Object value2 = values[i];
152 				final Predicate condition2 = QueryHelper.equalsIgnoreCaseIfStringPredicate(criteriaBuilder, root,
153 						value2, property2);
154 
155 				condition = criteriaBuilder.and(condition, condition2);
156 			}
157 		}
158 
159 		criteriaQuery.where(condition);
160 
161 		final TypedQuery<T> typedQuery = getEntityManager().createQuery(criteriaQuery);
162 		addCacheHints(typedQuery, "findListByProperty");
163 
164 		return typedQuery.getResultList();
165 	}

180 	public final List<T> findListByPropertyInList(final SingularAttribute<T, ? extends Object> property, final Object[] values) {
181 		final CriteriaQuery<T> criteriaQuery = criteriaBuilder.createQuery(getPersistentClass());
182 		final Root<T> root = criteriaQuery.from(getPersistentClass());
183 		criteriaQuery.select(root);
184 		final In<Object> in = criteriaBuilder.in((Path<Object>) root.get(property));
185 		for (final Object object : values) {
186 			in.value(object);
187 		}
188 		criteriaQuery.where(in);
189 		final TypedQuery<T> typedQuery = getEntityManager().createQuery(criteriaQuery);
190 		addCacheHints(typedQuery, "findListByPropertyInList");
191 		return typedQuery.getResultList();

202 	@Override
203 	public <V> List<T> findOrderedByPropertyListByEmbeddedProperty(final SingularAttribute<T, V> property,
204 			final Class<V> clazz2, final SingularAttribute<V, ? extends Object> property2, final Object value,
205 			final SingularAttribute<T, ? extends Object> orderByProperty) {
206 		final CriteriaQuery<T> criteriaQuery = criteriaBuilder
207 				.createQuery(persistentClass);
208 		final Root<T> root = criteriaQuery.from(persistentClass);
209 		criteriaQuery.select(root);
210 
211 		if (orderByProperty != null) {
212 			criteriaQuery.orderBy(criteriaBuilder.desc(root.get(orderByProperty)));
213 		}
214 
215 
216 		final Join<T, V> join = root.join(property);
217 
218 		final Path<? extends Object> path = join.get(property2);
219 
220 		final Predicate condition = criteriaBuilder.equal(path, value);
221 
222 		criteriaQuery.where(condition);
223 
224 		final TypedQuery<T> typedQuery = entityManager
225 				.createQuery(criteriaQuery);
226 
227 		addCacheHints(typedQuery, "findListByEmbeddedProperty." + persistentClass.getSimpleName());
228 
229 
230 		return typedQuery.getResultList();
231 	}

290 	private List<T> getPageOrderBy(final Integer pageNr, final Integer resultPerPage,
291 			final SingularAttribute<T, ? extends Object> orderBy) {
292 		final CriteriaQuery<T> criteriaQuery = criteriaBuilder.createQuery(getPersistentClass());
293 		final Root<T> root = criteriaQuery.from(getPersistentClass());
294 
295 		criteriaQuery.select(root);
296 
297 		if (orderBy != null) {
298 			criteriaQuery.orderBy(criteriaBuilder.desc(root.get(orderBy)));
299 		}
300 
301 		final TypedQuery<T> typedQuery = getEntityManager().createQuery(criteriaQuery);
302 		addCacheHints(typedQuery, "getAll");
303 
304 		if (pageNr != null && resultPerPage != null) {
305 			typedQuery.setFirstResult((pageNr - 1) * resultPerPage);
306 			typedQuery.setMaxResults(resultPerPage);
307 
308 		}
309 
310 		return typedQuery.getResultList();
311 
312 	}
