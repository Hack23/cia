337 	private List<T> getPageOrderBy(final Integer pageNr, final Integer resultPerPage,
338 			final SingularAttribute<T, ? extends Object> orderBy) {
339 		final CriteriaQuery<T> criteriaQuery = criteriaBuilder.createQuery(getPersistentClass());
340 		final Root<T> root = criteriaQuery.from(getPersistentClass());
341 
342 		criteriaQuery.select(root);
343

345 			criteriaQuery.orderBy(criteriaBuilder.desc(root.get(orderBy)));
346 		}
347 
348 		final TypedQuery<T> typedQuery = getEntityManager().createQuery(criteriaQuery);
349 		addCacheHints(typedQuery, "getAll");
350 
351 		if (pageNr != null && resultPerPage != null) {
352 			typedQuery.setFirstResult((pageNr - 1) * resultPerPage);
353 			typedQuery.setMaxResults(resultPerPage);
