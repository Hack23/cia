19  package com.hack23.cia.service.data.impl;
20  
21  import java.sql.Connection;
22  import java.util.HashMap;
23  import java.util.Map;
24  
25  import javax.persistence.EntityManager;
26  import javax.persistence.PersistenceContext;
27  
28  import org.hibernate.engine.spi.SharedSessionContractImplementor;
29  import org.javers.core.Javers;
30  import org.javers.core.MappingStyle;
31  import org.javers.hibernate.integration.HibernateUnproxyObjectAccessHook;
32  import org.javers.repository.sql.ConnectionProvider;
33  import org.javers.repository.sql.DialectName;
34  import org.javers.repository.sql.JaversSqlRepository;
35  import org.javers.repository.sql.SqlRepositoryBuilder;
36  import org.javers.spring.auditable.AuthorProvider;
37  import org.javers.spring.auditable.CommitPropertiesProvider;
38  import org.javers.spring.auditable.aspect.JaversAuditableAspect;
39  import org.javers.spring.jpa.TransactionalJpaJaversBuilder;
40  import org.springframework.context.annotation.Bean;
41  import org.springframework.context.annotation.Configuration;
42  import org.springframework.security.core.context.SecurityContext;
43  import org.springframework.security.core.context.SecurityContextHolder;
44  import org.springframework.transaction.PlatformTransactionManager;
45  
46  /**
47   * The Class AuditableAspectConfiguration.
48   */
49  @Configuration
50  public class AuditableAspectConfiguration {
51  
52  	/** The entity manager. */
53  	@PersistenceContext
54  	private EntityManager entityManager;
55  
56  	/**
57  	 * Instantiates a new auditable aspect configuration.
58  	 */
59  	public AuditableAspectConfiguration() {
60  		super();
61  	}
62  
63  	/**
64  	 * Gets the javers.
65  	 *
66  	 * @param txManager the tx manager
67  	 * @return the javers
68  	 */
69  	@Bean
70  	public Javers getJavers(final PlatformTransactionManager txManager) {
71  		final JaversSqlRepository sqlRepository = SqlRepositoryBuilder.sqlRepository()
72  				.withConnectionProvider(new ConnectionProvider() {
73  
74  					@Override
75  					public Connection getConnection() {
76  						final SharedSessionContractImplementor session = entityManager.unwrap(SharedSessionContractImplementor.class);
77  						return session.connection();
78  					}
79  				}).withDialect(DialectName.POSTGRES).build();
80  
81  		return TransactionalJpaJaversBuilder.javers().withTxManager(txManager)
82  				.withObjectAccessHook(new HibernateUnproxyObjectAccessHook()).registerJaversRepository(sqlRepository)
83  				.withMappingStyle(MappingStyle.BEAN).build();
84  	}
85  
86  	/**
87  	 * Javers auditable aspect.
88  	 *
89  	 * @param javers                   the javers
90  	 * @param authorProvider           the author provider
91  	 * @param commitPropertiesProvider the commit properties provider
92  	 * @return the javers auditable aspect
93  	 */
94  	@Bean
95  	public JaversAuditableAspect javersAuditableAspect(final Javers javers, final AuthorProvider authorProvider,
96  			final CommitPropertiesProvider commitPropertiesProvider) {
97  		return new JaversAuditableAspect(javers, authorProvider, commitPropertiesProvider);
98  	}
99  
100 	/**
101 	 * Author provider.
102 	 *
103 	 * @return the author provider
104 	 */
105 	@Bean
106 	public AuthorProvider authorProvider() {
107 		return () -> {
108 			final SecurityContext context = SecurityContextHolder.getContext();
109 			if (context != null && context.getAuthentication() != null) {
110 				return context.getAuthentication().getPrincipal().toString();
111 			} else {
112 				return "system";
113 			}
114 		};
115 	}
116 
117 	/**
118 	 * Commit properties provider.
119 	 *
120 	 * @return the commit properties provider
121 	 */
122 	@Bean
123 	public CommitPropertiesProvider commitPropertiesProvider() {
124 		return new CommitPropertiesProvider() {
125             @Override
126             public Map<String, String>  provideForCommittedObject(final Object domainObject) {
127                 final Map<String, String> props = new HashMap<>();
128                 props.put("key", "ok");
129                 return props;
130             }
131         };
132 	}
133 }
