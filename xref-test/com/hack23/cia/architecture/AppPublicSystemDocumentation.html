318 	/**
319 	 * Prints the plant uml.
320 	 *
321 	 * @param workspace
322 	 *            the workspace
323 	 * @throws WorkspaceWriterException
324 	 *             the workspace writer exception
325 	 * @throws IOException
326 	 *             Signals that an I/O exception has occurred.
327 	 * @throws InterruptedException
328 	 *             the interrupted exception
329 	 */
330 	private static void printPlantUml(final Workspace workspace)
331 			throws WorkspaceWriterException, IOException, InterruptedException {
332 		final StringWriter stringWriter = new StringWriter();
333 		final PlantUMLWriter plantUMLWriter = new PlantUMLWriter();
334 		//plantUMLWriter.setSizeLimit(16384);
335 		plantUMLWriter.write(workspace, stringWriter);
336 		String allPlantUmlsString = stringWriter.toString();
337 
338 		final String systemUml2 = allPlantUmlsString.substring(allPlantUmlsString.lastIndexOf("@startuml"),
339 				allPlantUmlsString.lastIndexOf("@enduml") + "@enduml".length());
340 		allPlantUmlsString = allPlantUmlsString.replace(systemUml2, "");
341 		writePlantUml("Citizen-Intelligence-Agency-System-System-Deployment", systemUml2);
342 
343 		final String componentUml = allPlantUmlsString.substring(allPlantUmlsString.lastIndexOf("@startuml"),
344 				allPlantUmlsString.lastIndexOf("@enduml") + "@enduml".length());
345 		allPlantUmlsString = allPlantUmlsString.replace(componentUml, "");
346 		writePlantUml("Citizen-Intelligence-Agency-System-Web-Application-Components", componentUml);
347 
348 		final String containersUml = allPlantUmlsString.substring(allPlantUmlsString.lastIndexOf("@startuml"),
349 				allPlantUmlsString.lastIndexOf("@enduml") + "@enduml".length());
350 		allPlantUmlsString = allPlantUmlsString.replace(containersUml, "");
351 		writePlantUml("Citizen-Intelligence-Agency-System-Containers", containersUml);
352 
353 		final String systemUml = allPlantUmlsString.substring(allPlantUmlsString.lastIndexOf("@startuml"),
354 				allPlantUmlsString.lastIndexOf("@enduml") + "@enduml".length());
355 		allPlantUmlsString = allPlantUmlsString.replace(systemUml, "");
356 		writePlantUml("Citizen-Intelligence-Agency-System-System-Context", systemUml);
357 
358 		final String enterpriseUml = allPlantUmlsString.substring(allPlantUmlsString.lastIndexOf("@startuml"),
359 				allPlantUmlsString.lastIndexOf("@enduml") + "@enduml".length());
360 		allPlantUmlsString = allPlantUmlsString.replace(enterpriseUml, "");
361 		writePlantUml("Enterprise-Context-for-Hack23", enterpriseUml);
362 	}

364 	/**
365 	 * Write plant uml.
366 	 *
367 	 * @param filename
368 	 *            the filename
369 	 * @param content
370 	 *            the content
371 	 * @throws IOException
372 	 *             Signals that an I/O exception has occurred.
373 	 * @throws InterruptedException
374 	 *             the interrupted exception
375 	 */
376 	private static void writePlantUml(final String filename, final String content)
377 			throws IOException, InterruptedException {
378 		final String fullFilePathPlantUmlFile = Paths.get(".").toAbsolutePath().normalize().toString() + File.separator
379 				+ "target" + File.separator + "site" + File.separator + "architecture" + File.separator + filename
380 				+ ".pu";
381 		FileUtils.writeStringToFile(new File(fullFilePathPlantUmlFile), content, Charset.defaultCharset());
382 	}
