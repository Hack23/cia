134     /**
135      * Find button by text.
136      *
137      * @param text the text
138      * @return the web element
139      */
140     public WebElement findButtonByText(final String text) {
141         return findButtons().stream()
142             .filter(button -> matchesButtonText(button, text))
143             .findFirst()
144             .orElse(null);
145     }

147     /**
148      * Matches button text.
149      *
150      * @param button the button
151      * @param text the text
152      * @return true, if successful
153      */
154     private boolean matchesButtonText(final WebElement button, final String text) {
155         final String buttonText = button.getText().trim();
156         return text.equalsIgnoreCase(buttonText) || buttonText.endsWith(text);
157     }

288 	 * Find button.
289 	 *
290 	 * @param buttonLabel the button label
291 	 * @return the web element
292 	 */
293 	public WebElement findButton(final String buttonLabel) {
294 
295 		wait.until(helper.containsButton(buttonLabel));
296 
297 		for (final WebElement webElement : helper.getButtons()) {
298 			final WebElement refreshElement = StaleElementUtils.refreshElement(webElement, driver);
299 			if (ExpectedConditions.not(ExpectedConditions.stalenessOf(refreshElement)).apply(driver)
300 					&& (buttonLabel.equalsIgnoreCase(refreshElement.getText().trim())
301 							|| refreshElement.getText().trim().endsWith(buttonLabel))) {
302 				return refreshElement;
303 			}
304 		}
305 		return null;
306 	}

409 	/**
410 	 * Gets the menu item.
411 	 *
412 	 * @param element the element
413 	 * @param level   the level
414 	 * @param caption the caption
415 	 * @return the menu item
416 	 */
417 	WebElement getMenuItem(final WebElement element, final int level, final String... caption) {
418 		final List<WebElement> findElements = element.findElements(By.className("v-menubar-menuitem-caption"));
419 		if (caption.length == level) {
420 			for (final WebElement webElement : findElements) {
421 				if (!ExpectedConditions.stalenessOf(element).apply(driver) && StaleElementUtils
422 						.refreshElement(webElement, driver).getText().contains(caption[level - 1])) {
423 					return StaleElementUtils.refreshElement(webElement, driver);
424 				}
425 			}
426 		} else {
427 			for (final WebElement webElement : findElements) {
428 				if (!ExpectedConditions.stalenessOf(element).apply(driver)
429 						&& caption[level - 1].equals(StaleElementUtils.refreshElement(webElement, driver).getText())) {
430 					return getMenuItem(StaleElementUtils.refreshElement(webElement, driver), level + 1, caption);
431 				}
432 			}
433 		}
434 
435 		final List<WebElement> findElements2 = driver.findElements(By.className("v-menubar-menuitem"));
436 		if (caption.length == level) {
437 			for (final WebElement webElement : findElements2) {
438 				if (!ExpectedConditions.stalenessOf(element).apply(driver) && StaleElementUtils
439 						.refreshElement(webElement, driver).getText().contains(caption[level - 1])) {
440 					return StaleElementUtils.refreshElement(webElement, driver);
441 				}
442 			}
443 		}
444 
445 		return null;
446 	}

523 	/**
524 	 * Logout user.
525 	 */
526 	public void logoutUser() {
527 
528 		final WebElement logoutButton = findButton("Logout");
529 		assertNotNull("Expect to find a Logout Button", logoutButton);
530 
531 		final Cookie cookie = driver.manage().getCookieNamed("JSESSIONID");
532 		final String sessionId = cookie.getValue();
533 
534 		performClickAction(logoutButton);
535 
536 		final WebElement body = driver.findElement(By.tagName("body"));
537 		body.sendKeys(Keys.ESCAPE);
538 
539 		wait.until(helper.containsViewAction(ViewAction.VISIT_MAIN_VIEW));
540 
541 		wait.until(ExpectedConditions.visibilityOfAllElementsLocatedBy(By.tagName("body")));
542 		driver.navigate().refresh();
543 		action.pause(500L).perform();
544 
545 		wait.until(webDriver -> "complete"
546 				.equals(((JavascriptExecutor) webDriver).executeScript("return document.readyState")));
547 
548 		wait.until(ExpectedConditions.visibilityOfAllElementsLocatedBy(By.tagName("body")));
549 
550 		final Cookie newCookie = driver.manage().getCookieNamed("JSESSIONID");
551 
552 		final String newSessionId = newCookie.getValue();
553 
554 		assertNotEquals(sessionId, newSessionId);
555 
556 		final String url = systemTestTargetUrl + "#!" + CommonsViews.MAIN_VIEW_NAME;
557 
558 		assertEquals(url, driver.getCurrentUrl());
559 		cleanBrowser();
560 	}
