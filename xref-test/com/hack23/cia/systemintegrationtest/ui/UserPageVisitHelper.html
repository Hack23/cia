63      /**
64       * Grab screenshot.
65       */
66      public void grabScreenshot() {
67          try {
68              final File scrFile = ((TakesScreenshot) driver).getScreenshotAs(OutputType.FILE);
69              screenShotNumber++;
70              final Path targetPath = Path.of("target", "screenshot_" + screenShotNumber + ".png");
71              Files.copy(scrFile.toPath(), targetPath, StandardCopyOption.REPLACE_EXISTING);
72          } catch (final IOException e) {
73              LOG.error("Failed to save screenshot", e);
74          }
75      }

107 	/**
108 	 * Contains button.
109 	 *
110 	 * @param value the value
111 	 * @return the expected condition
112 	 */
113 	public ExpectedCondition<Boolean> containsButton(final String value) {
114 		return new ExpectedCondition<>() {
115 
116 			@Override
117 			public Boolean apply(final WebDriver driver) {
118 				for (final WebElement webElement : getButtons()) {
119 					final WebElement refreshElement = StaleElementUtils.refreshElement(webElement, driver);
120 
121 					if (ExpectedConditions.not(ExpectedConditions.stalenessOf(refreshElement)).apply(driver)
122 							&& (value.equalsIgnoreCase(refreshElement.getText().trim())
123 									|| refreshElement.getText().trim().endsWith(value))) {
124 						return true;
125 					}
126 				}
127 
128 				return false;
129 			}
130 
131 			@Override
132 			public String toString() {
133 				return String.format("Button \"%s\". ", value);
134 			}
135 		};
136 	}

139 	 * Contains menu item.
140 	 *
141 	 * @param userPageVisit the user page visit
142 	 * @param element the element
143 	 * @param caption the caption
144 	 * @return the expected condition
145 	 */
146 	public ExpectedCondition<Boolean> containsMenuItem(final UserPageVisit userPageVisit, final WebElement element, final String... caption) {
147 		return new ExpectedCondition<>() {
148 			@Override
149 			public Boolean apply(final WebDriver driver) {
150 				return !ExpectedConditions.stalenessOf(element).apply(driver)
151 						&& userPageVisit.getMenuItem(element, 1, caption) != null;
152 			}
153 		};
154 	}

199 	/**
200 	 * Gets the button elements.
201 	 *
202 	 * @return the button elements
203 	 */
204 	List<WebElement> getButtonElements() {
205 		final List<WebElement> result = new ArrayList<>(driver.findElements(By.className("v-nativebutton")));
206 		result.addAll(driver.findElements(By.className("v-button")));
207 		result.addAll(driver.findElements(By.className("v-button-caption")));
208 		result.addAll(driver.findElements(By.tagName("a")));
209 
210 		return result;
211 	}

214 	/**
215 	 * Gets the buttons.
216 	 *
217 	 * @return the buttons
218 	 */
219 	public List<WebElement> getButtons() {
220 		final List<WebElement> result = getButtonElements();
221 		final WebDriverWait wait = new WebDriverWait(driver, TestConstants.WAIT_FOR_PAGE_ELEMENT,DEFAULT_WAIT);
222 		wait.until(ExpectedConditions.refreshed(ExpectedConditions.visibilityOfAllElements(result)));
223 
224 		return getButtonElements();
225 	}

227 	/**
228 	 * Gets the actions available.
229 	 *
230 	 * @return the actions available
231 	 */
232 	public Set<ViewAction> getActionsAvailable() {
233 		final Set<ViewAction> actions = new HashSet<>();
234 
235 		for (final ViewAction actionValue : ViewAction.values()) {
236 			if (driver.findElements(By.id(actionValue.name())).size() > 0) {
237 				actions.add(actionValue);
238 			}
239 		}
240 		return actions;
241 	}
