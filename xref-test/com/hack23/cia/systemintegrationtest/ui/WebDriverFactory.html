1   package com.hack23.cia.systemintegrationtest.ui;
2   
3   import org.openqa.selenium.WebDriver;
4   import org.openqa.selenium.chrome.ChromeDriver;
5   import org.openqa.selenium.chrome.ChromeOptions;
6   
7   import io.github.bonigarcia.wdm.WebDriverManager;
8   import io.github.bonigarcia.wdm.config.DriverManagerType;
9   
10  /**
11   * A factory for creating WebDriver objects.
12   */
13  public class WebDriverFactory {
14  
15  	/** The driver. */
16  	private static WebDriver driver;
17  
18      /**
19       * Instantiates a new web driver factory.
20       */
21      private WebDriverFactory() {
22      }
23  
24      /**
25       * Creates a new WebDriver object.
26       *
27       * @return the web driver
28       */
29      public static WebDriver createDriver() {
30         if (driver == null ) {
31      	   WebDriverManager.getInstance(DriverManagerType.CHROME).setup();
32      	   driver = createChromeDriver();
33      	   configureDriver(driver);
34         }
35          return driver;
36      }
37  
38      /**
39       * Creates a new WebDriver object.
40       *
41       * @return the web driver
42       */
43      private static WebDriver createChromeDriver() {
44          final ChromeOptions options = new ChromeOptions();
45          options.addArguments(
46                  "--allow-insecure-localhost",
47                  "--start-maximized");
48          return new ChromeDriver(options);
49      }
50  
51      /**
52       * Configure driver.
53       *
54       * @param driver the driver
55       */
56      private static void configureDriver(final WebDriver driver) {
57          driver.manage().timeouts().pageLoadTimeout(TestConstants.DEFAULT_BROWSER_TIMEOUT);
58          driver.manage().timeouts().scriptTimeout(TestConstants.DEFAULT_BROWSER_TIMEOUT);
59      }
60  }
