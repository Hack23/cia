7   import java.lang.reflect.InvocationTargetException;
8   import java.lang.reflect.Method;

13  import org.openqa.selenium.By;
14  import org.openqa.selenium.StaleElementReferenceException;

16  import org.openqa.selenium.WebElement;

18  import org.slf4j.LoggerFactory;
19  
20  public class StaleElementUtils {
21  
22  	/** The Constant LOGGER. */
23  	private static final Logger LOGGER = LoggerFactory.getLogger(StaleElementUtils.class);
24  
25  	/**
26  	 * Gets the by.
27  	 *
28  	 * @param key
29  	 *            the key
30  	 * @param value
31  	 *            the value
32  	 * @return the by
33  	 * @throws InvocationTargetException
34  	 *             the invocation target exception
35  	 * @throws IllegalAccessException
36  	 *             the illegal access exception
37  	 */
38  	private static By getBy(final String key, final String value) throws InvocationTargetException, IllegalAccessException {
39  		final By by = null;
40  		final Class clazz = By.class;
41  		final String methodName = key.replace(" ", "");
42  		final Method m = getCaseInsensitiveStaticDeclaredMethod(clazz, methodName);
43  		return (By) m.invoke(null, value);
44  	}
45  
46  	/**
47  	 * Gets the case insensitive declared method.
48  	 *
49  	 * @param obj
50  	 *            the obj
51  	 * @param methodName
52  	 *            the method name
53  	 * @return the case insensitive declared method
54  	 */
55  	private static Method getCaseInsensitiveDeclaredMethod(final Object obj, final String methodName) {
56  		final Method[] methods = obj.getClass().getMethods();
57  		Method method = null;
58  		for (final Method m : methods) {
59  			if (m.getName().equalsIgnoreCase(methodName)) {
60  				method = m;
61  				break;
62  			}
63  		}
64  		if (method == null) {
65  			throw new IllegalStateException(String.format(Locale.ENGLISH,"%s Method name is not found for this Class %s", methodName,
66  					obj.getClass().toString()));
67  		}
68  		return method;
69  	}
70  
71  	/**
72  	 * Gets the case insensitive static declared method.
73  	 *
74  	 * @param clazz
75  	 *            the clazz
76  	 * @param methodName
77  	 *            the method name
78  	 * @return the case insensitive static declared method
79  	 */
80  	private static Method getCaseInsensitiveStaticDeclaredMethod(final Class clazz, final String methodName) {
81  		final Method[] methods = clazz.getMethods();
82  		Method method = null;
83  		for (final Method m : methods) {
84  			if (m.getName().equalsIgnoreCase(methodName)) {
85  				method = m;
86  				break;
87  			}
88  		}
89  		if (method == null) {
90  			throw new IllegalStateException(
91  					String.format(Locale.ENGLISH,"%s Method name is not found for this Class %s", methodName, clazz.toString()));
92  		}
93  		return method;
94  	}
95  
96  	/**
97  	 * Gets the web element.
98  	 *
99  	 * @param lastObject
100 	 *            the last object
101 	 * @param key
102 	 *            the key
103 	 * @param value
104 	 *            the value
105 	 * @return the web element
106 	 */
107 	private static WebElement getWebElement(final Object lastObject, final String key, final String value) {
108 		WebElement element = null;
109 		try {
110 			final By by = getBy(key, value);
111 			final Method m = getCaseInsensitiveDeclaredMethod(lastObject, "findElement");
112 			element = (WebElement) m.invoke(lastObject, by);
113 		} catch (final InvocationTargetException | IllegalAccessException e) {
114 			e.printStackTrace();
115 		}
116 		return element;
117 	}
118 
119 	/**
120 	 * Checks if is element stale.
121 	 *
122 	 * @param e
123 	 *            the e
124 	 * @return true, if is element stale
125 	 */
126 	public static boolean isElementStale(final WebElement e) {
127 		try {
128 			e.isDisplayed();
129 			return false;
130 		} catch (final StaleElementReferenceException ex) {
131 			return true;
132 		}
133 	}
134 
135 	/**
136 	 * Refresh element.
137 	 *
138 	 * @param elem
139 	 *            the elem
140 	 * @param driver
141 	 *            the driver
142 	 * @return the web element
143 	 */
144 	public static WebElement refreshElement(final WebElement elem, final WebDriver driver) {
145 		if (!isElementStale(elem)) {
146 			return elem;
147 		}
148 		Object lastObject = null;
149 		try {
150 			final String[] arr = elem.toString().split("->");
151 			final List<String> newStr = new ArrayList<>();
152 			for (final String s : arr) {
153 				final String newstr = s.trim().replaceAll("^\\[+", "").replaceAll("\\]+$", "");
154 				final String[] parts = newstr.split(": ");
155 				final String key = parts[0];
156 				String value = parts[1];
157 				final int leftBracketsCount = value.length() - value.replace("[", "").length();
158 				final int rightBracketscount = value.length() - value.replace("]", "").length();
159 				if (leftBracketsCount - rightBracketscount == 1) {
160 					value = value + "]";
161 				}
162 				if (lastObject == null) {
163 					lastObject = driver;
164 				} else {
165 					lastObject = getWebElement(lastObject, key, value);
166 				}
167 			}
168 		} catch (final Exception e) {
169 			LOGGER.error("Error in Refreshing the stale Element.");
170 		}
171 		return (WebElement) lastObject;
172 	}
