179 	public ExpectedCondition<Boolean> containsButton(final String value) {
180 		return new ExpectedCondition<Boolean>() {
181 
182 			@Override
183 			public Boolean apply(final WebDriver driver) {
184 				for (final WebElement webElement : getButtons()) {
185 
186 					final WebElement refreshElement = StaleElementUtils.refreshElement(webElement, driver);
187 					if (ExpectedConditions.not(ExpectedConditions.stalenessOf(refreshElement)).apply(driver) && (value.equalsIgnoreCase(refreshElement.getText().trim()) || refreshElement.getText().trim().endsWith(value))) {
188 						return true;
189 					}
190 				}
191 
192 				return false;
193 			}
194 
195 			@Override
196 			public String toString() {
197 				return String.format("Button \"%s\". ", value);
198 			}
199 		};
200 	}

439 	/**
440 	 * Gets the menu item.
441 	 *
442 	 * @param element
443 	 *            the element
444 	 * @param level
445 	 *            the level
446 	 * @param caption
447 	 *            the caption
448 	 * @return the menu item
449 	 */
450 	private WebElement getMenuItem(final WebElement element,final int level,final String... caption) {
451 		final List<WebElement> findElements = element.findElements(By.className("v-menubar-menuitem-caption"));
452 		if (caption.length == level) {
453 			for (final WebElement webElement : findElements) {
454 				if (!ExpectedConditions.stalenessOf(element).apply(driver) && StaleElementUtils.refreshElement(webElement, driver).getText().contains(caption[level -1])) {
455 					return StaleElementUtils.refreshElement(webElement, driver);
456 				}
457 			}
458 		} else {
459 			for (final WebElement webElement : findElements) {
460 				if (!ExpectedConditions.stalenessOf(element).apply(driver) && caption[level -1].equals(StaleElementUtils.refreshElement(webElement, driver).getText())) {
461 					return getMenuItem(StaleElementUtils.refreshElement(webElement, driver), level +1 ,caption);
462 				}
463 			}
464 		}
465 
466 			final List<WebElement> findElements2 = driver.findElements(By.className("v-menubar-menuitem"));
467 			if (caption.length == level) {
468 				for (final WebElement webElement : findElements2) {
469 						if (!ExpectedConditions.stalenessOf(element).apply(driver) && StaleElementUtils.refreshElement(webElement, driver).getText().contains(caption[level -1])) {
470 						return StaleElementUtils.refreshElement(webElement, driver);
471 					}
472 				}
473 			}
474 
475 		return null;
476 	}

558

560 		final String sessionId = cookie.getValue();

562

575 		wait.until(webDriver -> ((JavascriptExecutor) webDriver).executeScript("return document.readyState").equals("complete"));
576 		
577 		wait.until(ExpectedConditions.visibilityOfAllElementsLocatedBy(By.tagName("body")));
578 
579 		final Cookie newCookie= driver.manage().getCookieNamed("JSESSIONID");
580 
581 		final String newSessionId = newCookie.getValue();
582 
583 		assertNotEquals(sessionId,newSessionId);

609 	 *
610 	 * @param clickElement
611 	 *            the click element
612 	 * @param waitDelay
613 	 *            the wait delay
614 	 * @throws InterruptedException
615 	 *             the interrupted exception
616 	 */
617 	private void performClickAction(final WebElement clickElement, final int waitDelay)
618 			throws InterruptedException {
619 		assertNotNull(clickElement);
620 		final WebDriverWait wait = new WebDriverWait(driver, waitDelay);
621 		wait.until(containsViewAction(ViewAction.VISIT_MAIN_VIEW));
622 		wait.until(ExpectedConditions.elementToBeClickable(StaleElementUtils.refreshElement(clickElement,driver)));
623 
624 		if (browser.contains("htmlunit")) {
625 			StaleElementUtils.refreshElement(clickElement,driver).click();
626 		} else {
627 			action.pause(250L).perform();
628 			action.clickAndHold(StaleElementUtils.refreshElement(clickElement,driver)).release().perform();
629 			action.pause(250L).perform();
630 		}
631 
632 		wait.until(webDriver -> ((JavascriptExecutor) webDriver).executeScript("return document.readyState").equals("complete"));
633 		wait.until(containsViewAction(ViewAction.VISIT_MAIN_VIEW));
634 
635 		grabScreenshot(driver);
636 
637 	}
