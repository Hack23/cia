93  	/**
94  	 * Sets the env.
95  	 *
96  	 * @param key the key
97  	 * @param value the value
98  	 */
99  	public static void setEnv(final String key, final String value) {
100 		try {
101 			final Map<String, String> env = System.getenv();
102 			final Class<?> cl = env.getClass();
103 			final Field field = cl.getDeclaredField("m");
104 			field.setAccessible(true);
105 			final Map<String, String> writableEnv = (Map<String, String>) field.get(env);
106 			writableEnv.put(key, value);
107 		} catch (final Exception e) {
108 			throw new IllegalStateException("Failed to set environment variable", e);
109 		}
110 	}

154 	/**
155 	 * Inits the logger.
156 	 */
157 	private static void initLogger() {
158 		System.setProperty("logback.configurationFile", "src/main/resources/logback.xml");
159 		System.setProperty("slf4j", "true");
160 		System.setProperty("org.eclipse.jetty.util.log.class", "org.eclipse.jetty.util.log.Slf4jLog");
161 		// System.setProperty("javax.net.debug", "all");
162 
163 		LogManager.getLogManager().reset();
164 		SLF4JBridgeHandler.install();
165 		java.util.logging.Logger.getLogger("global").setLevel(Level.FINEST);
166 	}

168 	/**
169 	 * Inits the.
170 	 *
171 	 * @throws Exception the exception
172 	 */
173 	public final void init() throws Exception {
174 		initialised = true;
175 		server = new Server();
176 		Security.addProvider(new BouncyCastleProvider());
177 		// Setup JMX
178 		final MBeanContainer mbContainer = new MBeanContainer(ManagementFactory.getPlatformMBeanServer());
179 		server.addBean(mbContainer);
180 
181 		// Enable parsing of jndi-related parts of web.xml and jetty-env.xml
182 		final org.eclipse.jetty.webapp.Configuration.ClassList classlist = org.eclipse.jetty.webapp.Configuration.ClassList
183 				.setServerDefault(server);
184 		classlist.addAfter("org.eclipse.jetty.webapp.FragmentConfiguration",
185 				"org.eclipse.jetty.plus.webapp.EnvConfiguration", "org.eclipse.jetty.plus.webapp.PlusConfiguration");
186 		classlist.addBefore("org.eclipse.jetty.webapp.JettyWebXmlConfiguration",
187 				"org.eclipse.jetty.annotations.AnnotationConfiguration");
188 
189 		final HttpConfiguration http_config = new HttpConfiguration();
190 		http_config.setSecureScheme("https");
191 		http_config.setSecurePort(28443);
192 		http_config.setSendServerVersion(false);
193 
194 		final HttpConfiguration https_config = new HttpConfiguration(http_config);
195 		https_config.addCustomizer(new SecureRequestCustomizer());
196 
197 		final SslContextFactory sslContextFactory = new SslContextFactory();
198 		sslContextFactory.setKeyStoreType("PKCS12");
199 		sslContextFactory.setKeyStorePath("target/keystore.p12");
200 		sslContextFactory.setTrustStorePath("target/keystore.p12");
201 		sslContextFactory.setTrustStoreType("PKCS12");
202 
203 		sslContextFactory.setKeyStorePassword("changeit");
204 		sslContextFactory.setTrustStorePassword("changeit");
205 		sslContextFactory.setKeyManagerPassword("changeit");
206 		sslContextFactory.setCertAlias("jetty");
207 		sslContextFactory.setIncludeCipherSuites("TLS_AES_256_GCM_SHA384", "TLS_CHACHA20_POLY1305_SHA256",
208 				"TLS_ECDHE_RSA_WITH_AES_256_GCM_SHA384", "TLS_ECDHE_RSA_WITH_CHACHA20_POLY1305_SHA256",
209 				"TLS_DHE_RSA_WITH_CHACHA20_POLY1305_SHA256");
210 		sslContextFactory.setExcludeProtocols("SSL", "SSLv2", "SSLv2Hello", "SSLv3", "TLSv1", "TLSv1.1");
211 		sslContextFactory.setIncludeProtocols("TLSv1.2", "TLSv1.3");
212 
213 		final ServerConnector sslConnector = new ServerConnector(server,
214 				new SslConnectionFactory(sslContextFactory, "http/1.1"), new HttpConnectionFactory(https_config),
215 				new HTTP2CServerConnectionFactory(https_config));
216 		sslConnector.setPort(PORT);
217 
218 		server.setConnectors(new ServerConnector[] { sslConnector });
219 		final WebAppContext handler = new WebAppContext("src/main/webapp", "/");
220 		handler.setExtraClasspath("target/classes");
221 		handler.setParentLoaderPriority(true);
222 		handler.setConfigurationDiscovered(true);
223 		handler.setClassLoader(Thread.currentThread().getContextClassLoader());
224 		final HandlerList handlers = new HandlerList();
225 
226 		handlers.setHandlers(new Handler[] { handler, new DefaultHandler() });
227 
228 		server.setHandler(handlers);
229 	}

236 	public final void start() throws Exception {
237 		if (!initialised) {
238 			LOGGER.info("Exiting init not called before start");
239 			System.exit(-1);
240 		}
241 
242 		try {
243 			server.start();
244 		} catch (final Exception e) {
245 			LOGGER.error("Problem starting server", e);
246 		}
247 	}
