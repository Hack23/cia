108 	/**
109 	 * Sets the env.
110 	 *
111 	 * @param key the key
112 	 * @param value the value
113 	 */
114 	public static void setEnv(final String key, final String value) {
115 		try {
116 			final Map<String, String> env = System.getenv();
117 			final Class<?> cl = env.getClass();
118 			final Field field = cl.getDeclaredField("m");
119 			field.setAccessible(true);
120 			@SuppressWarnings("unchecked")
121 			final Map<String, String> writableEnv = (Map<String, String>) field.get(env);
122 			writableEnv.put(key, value);
123 		} catch (final Exception e) {
124 			throw new IllegalStateException("Failed to set environment variable", e);
125 		}
126 	}

154 	/**
155 	 * Inits the.
156 	 *
157 	 * @throws Exception the exception
158 	 */
159 	public void init() throws Exception {
160 		initialised = true;
161 		server = new Server();
162 		Security.addProvider(new BouncyCastleProvider());
163 		// Setup JMX
164 		final MBeanContainer mbContainer = new MBeanContainer(ManagementFactory.getPlatformMBeanServer());
165 		server.addBean(mbContainer);
166 
167 		final HttpConfiguration http_config = new HttpConfiguration();
168 		http_config.setSecureScheme("https");
169 		http_config.setSecurePort(28443);
170 		http_config.setSendServerVersion(false);
171 
172 		final HttpConfiguration https_config = new HttpConfiguration(http_config);
173 		https_config.addCustomizer(new SecureRequestCustomizer(false));
174 
175 		final SslContextFactory.Server sslContextFactory = new SslContextFactory.Server();
176 		sslContextFactory.setKeyStoreType("PKCS12");
177 		sslContextFactory.setKeyStorePath("target/keystore.p12");
178 		sslContextFactory.setTrustStorePath("target/keystore.p12");
179 		sslContextFactory.setTrustStoreType("PKCS12");
180 
181 		sslContextFactory.setKeyStorePassword("changeit");
182 		sslContextFactory.setTrustStorePassword("changeit");
183 		sslContextFactory.setKeyManagerPassword("changeit");
184 		sslContextFactory.setCertAlias("jetty");
185 		sslContextFactory.setIncludeCipherSuites("TLS_AES_256_GCM_SHA384", "TLS_CHACHA20_POLY1305_SHA256",
186 				"TLS_ECDHE_RSA_WITH_AES_256_GCM_SHA384", "TLS_ECDHE_RSA_WITH_CHACHA20_POLY1305_SHA256",
187 				"TLS_DHE_RSA_WITH_CHACHA20_POLY1305_SHA256");
188 		sslContextFactory.setExcludeProtocols("SSL", "SSLv2", "SSLv2Hello", "SSLv3", "TLSv1", "TLSv1.1");
189 		sslContextFactory.setIncludeProtocols("TLSv1.2", "TLSv1.3");
190 
191 		final ServerConnector sslConnector = new ServerConnector(server,
192 				new SslConnectionFactory(sslContextFactory, "http/1.1"), new HttpConnectionFactory(https_config),
193 				new HTTP2CServerConnectionFactory(https_config));
194 		sslConnector.setPort(PORT);
195 
196 		server.setConnectors(new ServerConnector[] { sslConnector });
197 		final WebAppContext handler = new WebAppContext("src/main/webapp", "/");
198 		handler.setExtraClasspath("target/classes");
199 		handler.setParentLoaderPriority(true);
200 		handler.setConfigurationDiscovered(true);
201 		handler.setClassLoader(Thread.currentThread().getContextClassLoader());
202 		final HandlerList handlers = new HandlerList();
203 
204 		handlers.setHandlers(new Handler[] { handler, new DefaultHandler() });
205 
206 		server.setHandler(handlers);
207 	}

214 	public void start() throws Exception {
215 		if (!initialised) {
216 			LOGGER.info("Exiting init not called before start");
217 			System.exit(-1);
218 		}
219 
220 		try {
221 			server.start();
222 		} catch (final Exception e) {
223 			LOGGER.error("Problem starting server", e);
224 		}
225 	}
