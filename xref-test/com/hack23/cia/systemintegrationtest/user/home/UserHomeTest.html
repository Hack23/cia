27  	 *
28  	 * @throws Exception the exception
29  	 */
30  	@Test(timeout = DEFAULT_TIMEOUT)
31  	public void siteLoginUserChangePasswordFailureTest() throws Exception {
32  		pageVisit.visitDirectPage(PageCommandUserConstants.COMMAND_REGISTER);
33  		final String username = UUID.randomUUID().toString();
34  		final String password = TestUtils.generatePassword();
35  
36  		pageVisit.registerNewUser(username, password);
37  
38  		final WebElement userAccountMenuItem = pageVisit.getMenuItem(MenuItemConstants.USERACCOUNT);
39  		assertNotNull(userAccountMenuItem);
40  		pageVisit.performClickAction(userAccountMenuItem);
41  
42  		final WebElement securitySettingMenuItem = pageVisit.getMenuItem(MenuItemConstants.SECURITY_SETTING_TEXT);
43  		assertNotNull(securitySettingMenuItem);
44  		pageVisit.performClickAction(securitySettingMenuItem);
45  
46  		pageVisit.changePassword("wrong" + password,"new"+ password,"new" + password);
47  
48  		pageVisit.checkNotificationMessage(ChangePasswordClickListener.PROBLEM_CHANGING_PASSWORD);
49  
50  	}
