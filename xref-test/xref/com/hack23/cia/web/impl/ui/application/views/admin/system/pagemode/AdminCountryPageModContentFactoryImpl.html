40  import com.vaadin.ui.VerticalLayout;

42  /**
43   * The Class AdminCountryPageModContentFactoryImpl.
44   */
45  @Component
46  public final class AdminCountryPageModContentFactoryImpl extends AbstractAdminSystemPageModContentFactoryImpl {
47  
48  	/** The Constant COLUMN_ORDER. */
49  	private static final String[] COLUMN_ORDER = {
50  			"hjid",
51  			"id",
52  			"countryName",
53  			"iso2Code",
54  			"capitalCity",
55  			"longitude",
56  			"latitude"
57  	};
58  
59  	/** The Constant COUNTRY2. */
60  	private static final String COUNTRY2 = "Country";
61  
62  	/** The Constant HIDE_COLUMNS. */
63  	private static final String[] HIDE_COLUMNS = {
64  			"hjid",
65  			"id",
66  			"region",
67  			"adminregion",
68  			"incomeLevel",
69  			"lendingType",
70  			"longitude",
71  			"latitude"
72  	};
73  
74  	/** The Constant LISTENER. */
75  	private static final PageItemPropertyClickListener LISTENER = new PageItemPropertyClickListener(AdminViews.ADMIN_COUNTRY_VIEW_NAME, "hjid");
76  
77  	/** The Constant NAME. */
78  	public static final String NAME = AdminViews.ADMIN_COUNTRY_VIEW_NAME;
79  
80  	/**
81  	 * Instantiates a new admin country page mod content factory impl.
82  	 */
83  	public AdminCountryPageModContentFactoryImpl() {
84  		super(NAME);
85  	}
86  
87  	@Secured({ "ROLE_ADMIN" })
88  	@Override
89  	public Layout createContent(final String parameters, final MenuBar menuBar, final Panel panel) {
90  		final VerticalLayout content = createPanelContent();
91  		content.setSizeFull();
92  		content.addStyleName("admin-country-content");
93  
94  		final String pageId = getPageId(parameters);
95  		final int pageNr = getPageNr(parameters);
96  
97  		getMenuItemFactory().createMainPageMenuBar(menuBar);
98  
99  		CardInfoRowUtil.createPageHeader(panel, content,
100 				 AdminViewConstants.ADMIN_COUNTRY_MANAGEMENT,
101 				 AdminViewConstants.COUNTRY_OVERVIEW,
102 				 AdminViewConstants.COUNTRY_OVERVIEW_DESCRIPTION);
103 
104 		final DataContainer<CountryElement, Long> dataContainer = getApplicationManager()
105 				.getDataContainer(CountryElement.class);
106 
107 		final List<CountryElement> pageOrderBy = dataContainer.getPageOrderBy(
108 				pageNr,
109 				DEFAULT_RESULTS_PER_PAGE,
110 				CountryElement_.countryName);
111 
112 		getPagingUtil().createPagingControls(
113 				content,
114 				NAME,
115 				pageId,
116 				dataContainer.getSize(),
117 				pageNr,
118 				DEFAULT_RESULTS_PER_PAGE);
119 
120 		getGridFactory()
121 				.createBasicBeanItemNestedPropertiesGrid(
122 						content,
123 						CountryElement.class,
124 						pageOrderBy,
125 						COUNTRY2,
126 						null,
127 						COLUMN_ORDER,
128 						HIDE_COLUMNS,
129 						LISTENER,
130 						null,
131 						null);
132 
133 		if (pageId != null && !pageId.isEmpty()) {
134 			final CountryElement country = dataContainer.load(Long.valueOf(pageId));
135 			if (country != null) {
136 
137 				// Create a card panel for country details
138 				final Panel cardPanel = new Panel();
139 				cardPanel.addStyleName("politician-overview-card");
140 				cardPanel.setWidth("100%");
141 				cardPanel.setHeightUndefined();
142 				Responsive.makeResponsive(cardPanel);
143 
144 				final VerticalLayout cardContent = new VerticalLayout();
145 				cardContent.setMargin(true);
146 				cardContent.setSpacing(true);
147 				cardContent.setWidth("100%");
148 				cardPanel.setContent(cardContent);
149 
150 				content.addComponent(cardPanel);
151 
152 				CardInfoRowUtil.createCardHeader(cardContent,"Country Details");
153 
154 				// Attributes layout
155 				final VerticalLayout attributesLayout = new VerticalLayout();
156 				attributesLayout.setSpacing(true);
157 				attributesLayout.setWidth("100%");
158 				cardContent.addComponent(attributesLayout);
159 
160 				// Show fields if not null or empty
161 				CardInfoRowUtil.addInfoRowIfNotNull(attributesLayout, "Country ID:", country.getId(), VaadinIcons.FLAG_O);
162 				CardInfoRowUtil.addInfoRowIfNotNull(attributesLayout, "Name:", country.getCountryName(), VaadinIcons.GLOBE);
163 				CardInfoRowUtil.addInfoRowIfNotNull(attributesLayout, "ISO2 Code:", country.getIso2Code(), VaadinIcons.CODE);
164 				CardInfoRowUtil.addInfoRowIfNotNull(attributesLayout, "Capital:", country.getCapitalCity(), VaadinIcons.BUILDING);
165 				CardInfoRowUtil.addInfoRowIfNotNull(attributesLayout, "Longitude:", country.getLongitude(), VaadinIcons.ARROWS_LONG_H);
166 				CardInfoRowUtil.addInfoRowIfNotNull(attributesLayout, "Latitude:", country.getLatitude(), VaadinIcons.ARROWS_LONG_H);
167 			}
168 		}
169 
170 		getPageActionEventHelper().createPageEvent(
171 				ViewAction.VISIT_ADMIN_COUNTRY_VIEW,
172 				ApplicationEventGroup.ADMIN,
173 				NAME,
174 				null,
175 				pageId);
176 
177 		return content;
178 	}
179 
180 }
