56  	/**
57  	 * Creates the content.
58  	 *
59  	 * @param parameters the parameters
60  	 * @param menuBar the menu bar
61  	 * @param panel the panel
62  	 * @return the layout
63  	 */
64  	@Secured({ "ROLE_ANONYMOUS", "ROLE_USER", "ROLE_ADMIN" })
65  	@Override
66  	public Layout createContent(final String parameters, final MenuBar menuBar, final Panel panel) {
67  		final VerticalLayout panelContent = createPanelContent();
68  		final ViewRiksdagenParty viewRiksdagenParty = getItem(parameters);
69  		final String pageId = getPageId(parameters);
70  
71  		getPartyMenuItemFactory().createPartyMenuBar(menuBar, pageId);
72  
73  		CardInfoRowUtil.createPageHeader(panel, panelContent,
74  		    PartyViewConstants.OVERVIEW_HEADER + " " + viewRiksdagenParty.getPartyName(),
75  		    PartyViewConstants.GENERAL_SUBTITLE,
76  		    PartyViewConstants.OVERVIEW_DESC);
77  
78  		panel.setContent(panelContent);
79  
80  		final Link addPartyPageLink = getPageLinkFactory().addPartyPageLink(viewRiksdagenParty);
81  		panelContent.addComponent(addPartyPageLink);
82  		panelContent.setExpandRatio(addPartyPageLink, ContentRatio.SMALL);
83  
84  		// Load summary if available
85  		final DataContainer<ViewRiksdagenPartySummary, String> partySummaryDataContainer = getApplicationManager()
86  				.getDataContainer(ViewRiksdagenPartySummary.class);
87  		final ViewRiksdagenPartySummary viewRiksdagenPartySummary = partySummaryDataContainer.load(pageId);
88  
89  		// Create a card panel similar to the politician overview
90  		final Panel cardPanel = new Panel();
91  		cardPanel.addStyleName("politician-overview-card");
92  		cardPanel.setWidth("100%");
93  		cardPanel.setHeightUndefined();
94  		Responsive.makeResponsive(cardPanel);
95  
96  		final VerticalLayout cardContent = new VerticalLayout();
97  		cardContent.setMargin(true);
98  		cardContent.setSpacing(true);
99  		cardContent.setWidth("100%");
100 		cardPanel.setContent(cardContent);
101 
102 		panelContent.addComponent(cardPanel);
103 		panelContent.setExpandRatio(cardPanel, ContentRatio.SMALL_GRID);
104 
105 		CardInfoRowUtil.createCardHeader(cardContent,viewRiksdagenParty.getPartyName());
106 
107 		// Create single row for four sections
108 		final HorizontalLayout sectionsLayout = new HorizontalLayout();
109 		sectionsLayout.setSpacing(true);
110 		sectionsLayout.setWidth("100%");
111 		cardContent.addComponent(sectionsLayout);
112 
113 		// 1. Political Influence & Position
114 		final VerticalLayout politicalInfluenceLayout = CardInfoRowUtil.createSectionLayout("Political Influence & Position");
115 		addPoliticalInfluenceMetrics(politicalInfluenceLayout, viewRiksdagenParty, viewRiksdagenPartySummary);
116 		sectionsLayout.addComponent(politicalInfluenceLayout);
117 		sectionsLayout.setExpandRatio(politicalInfluenceLayout, 1.0f);
118 
119 		// 2. Parliamentary Engagement
120 		final VerticalLayout parliamentaryEngagementLayout = CardInfoRowUtil.createSectionLayout("Parliamentary Engagement");
121 		addParliamentaryEngagementMetrics(parliamentaryEngagementLayout, viewRiksdagenParty, viewRiksdagenPartySummary);
122 		sectionsLayout.addComponent(parliamentaryEngagementLayout);
123 		sectionsLayout.setExpandRatio(parliamentaryEngagementLayout, 1.0f);
124 
125 		// 3. Legislative Impact
126 		final VerticalLayout legislativeImpactLayout = CardInfoRowUtil.createSectionLayout("Legislative Impact");
127 		addLegislativeImpactMetrics(legislativeImpactLayout, viewRiksdagenParty, viewRiksdagenPartySummary);
128 		sectionsLayout.addComponent(legislativeImpactLayout);
129 		sectionsLayout.setExpandRatio(legislativeImpactLayout, 1.0f);
130 
131 		// 4. Member Performance
132 		final VerticalLayout memberPerformanceLayout = CardInfoRowUtil.createSectionLayout("Member Performance");
133 		addMemberPerformanceMetrics(memberPerformanceLayout, viewRiksdagenParty, viewRiksdagenPartySummary);
134 		sectionsLayout.addComponent(memberPerformanceLayout);
135 		sectionsLayout.setExpandRatio(memberPerformanceLayout, 1.0f);
136 
137 		// After the card, add the overview layout
138 		final VerticalLayout overviewLayout = new VerticalLayout();
139 		overviewLayout.setSizeFull();
140 		panelContent.addComponent(overviewLayout);
141 		panelContent.setExpandRatio(overviewLayout, ContentRatio.LARGE_FORM);
142 
143 		getPartyMenuItemFactory().createOverviewPage(overviewLayout, pageId);
144 
145 		getPageActionEventHelper().createPageEvent(ViewAction.VISIT_PARTY_VIEW, ApplicationEventGroup.USER, NAME, parameters,
146 				pageId);
147 		return panelContent;
148 	}

151 	/**
152 	 * Adds the political influence metrics.
153 	 *
154 	 * @param layout the layout
155 	 * @param party the party
156 	 * @param summary the summary
157 	 */
158 	// 1. Political Influence & Position
159 	private void addPoliticalInfluenceMetrics(final VerticalLayout layout,
160 	        final ViewRiksdagenParty party,
161 	        final ViewRiksdagenPartySummary summary) {
162 
163 	    if (summary != null) {
164 	        // Government Influence
165 	        layout.addComponent(CardInfoRowUtil.createInfoRow("Government Position:",
166 	            summary.isActiveGovernment() ? "In Government" : "Opposition",
167 	            VaadinIcons.INSTITUTION,
168 	            "Current position in government"));
169 
170 	        layout.addComponent(CardInfoRowUtil.createInfoRow("Ministers:",
171 	            String.valueOf(summary.getCurrentMinistryAssignments()),
172 	            VaadinIcons.GROUP,
173 	            "Current ministerial positions"));
174 
175 	        // Parliamentary Strength
176 	        layout.addComponent(CardInfoRowUtil.createInfoRow("Parliament Members:",
177 	            String.valueOf(party.getHeadCount()),
178 	            VaadinIcons.USERS,
179 	            "Total number of parliament members"));
180 
181 	        layout.addComponent(CardInfoRowUtil.createInfoRow("Committee Positions:",
182 	            String.valueOf(summary.getCurrentCommitteeAssignments()),
183 	            VaadinIcons.CLIPBOARD_USER,
184 	            "Current committee assignments"));
185 
186 	        // Leadership Roles
187 	        layout.addComponent(CardInfoRowUtil.createInfoRow("Leadership Positions:",
188 	            String.valueOf(summary.getCurrentCommitteeLeadershipAssignments()),
189 	            VaadinIcons.STAR,
190 	            "Current committee leadership roles"));
191 	    }
192 	}

194 	/**
195 	 * Adds the parliamentary engagement metrics.
196 	 *
197 	 * @param layout the layout
198 	 * @param party the party
199 	 * @param summary the summary
200 	 */
201 	// 2. Parliamentary Engagement
202 	private void addParliamentaryEngagementMetrics(final VerticalLayout layout,
203 	        final ViewRiksdagenParty party,
204 	        final ViewRiksdagenPartySummary summary) {
205 
206 	    if (summary != null) {
207 	        // Active Participation
208 	        layout.addComponent(CardInfoRowUtil.createInfoRow("Parliament Activity:",
209 	            String.format(Locale.ENGLISH,"%.1f%%", calculateActivityRate(summary.getTotalActiveParliament(), party.getHeadCount())),
210 	            VaadinIcons.CHART_LINE,
211 	            "Percentage of active members in parliament"));
212 
213 	        // Committee Engagement
214 	        layout.addComponent(CardInfoRowUtil.createInfoRow("Committee Involvement:",
215 	            String.valueOf(summary.getTotalActiveCommittee()),
216 	            VaadinIcons.USERS,
217 	            "Members active in committees"));
218 
219 	        // Historical Presence
220 	        layout.addComponent(CardInfoRowUtil.createInfoRow("Days in Government:",
221 	            String.format(Locale.ENGLISH,"%,d", summary.getTotalDaysServedGovernment()),
222 	            VaadinIcons.CLOCK,
223 	            "Total days served in government"));
224 
225 	        layout.addComponent(CardInfoRowUtil.createInfoRow("Parliamentary Experience:",
226 	            String.format(Locale.ENGLISH,"%,d", summary.getTotalDaysServedParliament()),
227 	            VaadinIcons.CALENDAR_CLOCK,
228 	            "Total days served in parliament"));
229 	    }
230 	}

282 	// 4. Member Performance

296 	        // Member Focus Areas

301 	            VaadinIcons.FLAG,

304 	        layout.addComponent(CardInfoRowUtil.createInfoRow("Committee Focus:",
305 	            String.format(Locale.ENGLISH,"%d (%d%%)",
306 	                summary.getCommitteeFocusedMembers(),
307 	                calculatePercentage(summary.getCommitteeFocusedMembers(), party.getHeadCount())),
308 	            VaadinIcons.CLIPBOARD_USER,

311 	        // Collaboration Metrics

313 	            String.format(Locale.ENGLISH,"%d (%d%%)",

321 	/**
322 	 * Calculate percentage.
323 	 *
324 	 * @param value the value
325 	 * @param total the total
326 	 * @return the int
327 	 */
328 	// Helper method for calculating percentages
329 	private int calculatePercentage(final long value, final long total) {
330 	    return total > 0 ? Math.round((float) value * 100 / total) : 0;
331 	}
332 
333 	/**
334 	 * Calculate activity rate.
335 	 *
336 	 * @param activeMembers the active members
337 	 * @param totalMembers the total members
338 	 * @return the double
339 	 */
340 	// Helper method for calculating activity rates
341 	private double calculateActivityRate(final long activeMembers, final long totalMembers) {
342 	    return totalMembers > 0 ? (double) activeMembers * 100 / totalMembers : 0;
343 	}
