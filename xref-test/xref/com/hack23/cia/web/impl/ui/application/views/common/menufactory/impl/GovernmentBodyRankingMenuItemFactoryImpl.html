19  package com.hack23.cia.web.impl.ui.application.views.common.menufactory.impl;
20  
21  import org.springframework.beans.factory.annotation.Autowired;
22  import org.springframework.stereotype.Service;
23  
24  import com.hack23.cia.web.impl.ui.application.views.common.menufactory.api.ApplicationMenuItemFactory;
25  import com.hack23.cia.web.impl.ui.application.views.common.menufactory.api.GovernmentBodyRankingMenuItemFactory;
26  import com.hack23.cia.web.impl.ui.application.views.common.menufactory.api.pagecommands.PageCommandGovernmentBodyRankingConstants;
27  import com.hack23.cia.web.impl.ui.application.views.common.rows.RowUtil;
28  import com.jarektoro.responsivelayout.ResponsiveRow;
29  import com.vaadin.icons.VaadinIcons;
30  import com.vaadin.ui.MenuBar;
31  import com.vaadin.ui.VerticalLayout;
32  
33  /**
34   * The Class GovernmentBodyRankingMenuItemFactoryImpl.
35   *
36   * <p>
37   * This class builds menus and overview pages for government body rankings,
38   * enabling users to navigate through institutional staffing, revenue streams,
39   * expenditure focus, and historical engagement patterns. Through integration
40   * with {@link ApplicationMenuItemFactory}, it ensures a coherent experience
41   * within the main menu structure.
42   * </p>
43   */
44  @Service
45  public final class GovernmentBodyRankingMenuItemFactoryImpl extends AbstractMenuItemFactoryImpl
46                  implements GovernmentBodyRankingMenuItemFactory {
47  
48          /** The application menu item factory, used to integrate into the main menu. */
49          @Autowired
50          private ApplicationMenuItemFactory applicationMenuItemFactory;
51  
52          /**
53           * Instantiates a new government body ranking menu item factory implementation.
54           */
55          public GovernmentBodyRankingMenuItemFactoryImpl() {
56                  super();
57          }
58  
59          /**
60           * Creates the government body ranking menu bar and integrates it into the main
61           * application menu. Adds government body ranking topics as sub-items.
62           *
63           * @param menuBar the main application {@link MenuBar} to which the government
64           *                body ranking menu will be added.
65           */
66          @Override
67          public void createGovernmentBodyRankingMenuBar(final MenuBar menuBar) {
68                  initApplicationMenuBar(menuBar);
69                  applicationMenuItemFactory.addRankingMenu(menuBar);
70                  createGovernmentBodyRankingTopics(
71                                  menuBar.addItem(GOVERNMENT_BODY_RANKING, VaadinIcons.BUILDING_O, null));
72          }
73  
74          /**
75           * Adds menu items for various government body ranking topics:
76           * <ul>
77           * <li>Overview: General dashboard icon</li>
78           * <li>Government Bodies: Building icon for institutions</li>
79           * <li>Headcount: Multiple users for staffing</li>
80           * <li>Income: Money deposit icon for revenues</li>
81           * <li>Expenditure: Money withdraw icon for spending</li>
82           * <li>Page Visit History: History icon for past engagement</li>
83           * </ul>
84           *
85           * @param menuItem the {@link MenuBar.MenuItem} representing "GovernmentBody
86           *                 Ranking"
87           *                 under which these topics will be added.
88           */
89          @Override
90          public void createGovernmentBodyRankingTopics(final MenuBar.MenuItem menuItem) {
91                  menuItem.addItem(OVERVIEW_TEXT, VaadinIcons.DASHBOARD, COMMAND_GOVERNMENT_BODY_RANKING_OVERVIEW);
92                  menuItem.addItem(GOVERNMENT_BODIES, VaadinIcons.BUILDING_O, PageCommandGovernmentBodyRankingConstants.COMMAND_GOVERNMENT_BODY_RANKING_DATAGRID);
93                  menuItem.addItem(GOVERNMENT_BODIES_HEADCOUNT, VaadinIcons.USER_CHECK,
94                  PageCommandGovernmentBodyRankingConstants.COMMAND_GOVERNMENT_BODIES_HEADCOUNT);
95                  menuItem.addItem(GOVERNMENT_BODIES_INCOME, VaadinIcons.MONEY, PageCommandGovernmentBodyRankingConstants.COMMAND_GOVERNMENT_BODIES_INCOME);
96                  menuItem.addItem(GOVERNMENT_BODIES_EXPENDITURE, VaadinIcons.MONEY_WITHDRAW,
97                  		PageCommandGovernmentBodyRankingConstants.COMMAND_GOVERNMENT_BODIES_EXPENDITURE);
98  
99                  menuItem.addItem(HEADCOUNT, VaadinIcons.USERS, PageCommandGovernmentBodyRankingConstants.COMMAND_GOVERNMENT_BODIES_HEADCOUNT);
100                 menuItem.addItem(INCOME, VaadinIcons.MONEY_DEPOSIT, PageCommandGovernmentBodyRankingConstants.COMMAND_GOVERNMENT_BODIES_INCOME);
101                 menuItem.addItem(EXPENDITURE, VaadinIcons.MONEY_WITHDRAW, PageCommandGovernmentBodyRankingConstants.COMMAND_GOVERNMENT_BODIES_EXPENDITURE);
102                 menuItem.addItem(RANKING_PAGE_VISIT_TEXT, VaadinIcons.CHART,
103                 		GOVERNMENT_RANKING_COMMAND_PAGEVISIT_HISTORY);
104         }
105 
106         /**
107          * Creates the overview page with quick-access links to government body ranking
108          * topics.
109          * Each button uses an icon and a brief description to guide the user:
110          * <ul>
111          * <li>Government Bodies: Building icon, structural roles</li>
112          * <li>Headcount: Users icon, staffing influence</li>
113          * <li>Income: Money deposit icon, revenue streams</li>
114          * <li>Expenditure: Money withdraw icon, spending focus</li>
115          * <li>Page Visit History: History icon, past engagement</li>
116          * </ul>
117          *
118          * @param panelContent the {@link VerticalLayout} container that holds the
119          *                     overview
120          *                     content. New UI components will be added here.
121          */
122         @Override
123         public void createOverviewPage(final VerticalLayout panelContent) {
124                 final ResponsiveRow grid = RowUtil.createGridLayout(panelContent);
125 
126                 createButtonLink(grid, GOVERNMENT_BODIES, VaadinIcons.BUILDING_O, PageCommandGovernmentBodyRankingConstants.COMMAND_GOVERNMENT_BODY_RANKING_DATAGRID,
127                                 GOVERNMENT_BODIES_DESCRIPTION);
128                 createButtonLink(grid, GOVERNMENT_BODIES_HEADCOUNT, VaadinIcons.USER_CHECK,
129                 		PageCommandGovernmentBodyRankingConstants.COMMAND_GOVERNMENT_BODIES_HEADCOUNT, GOVERNMENT_BODIES_HEADCOUNT_DESCRIPTION);
130                 createButtonLink(grid, GOVERNMENT_BODIES_INCOME, VaadinIcons.MONEY, PageCommandGovernmentBodyRankingConstants.COMMAND_GOVERNMENT_BODIES_INCOME,
131                                 GOVERNMENT_BODIES_INCOME_DESCRIPTION);
132                 createButtonLink(grid, GOVERNMENT_BODIES_EXPENDITURE, VaadinIcons.MONEY_WITHDRAW,
133                 		PageCommandGovernmentBodyRankingConstants.COMMAND_GOVERNMENT_BODIES_EXPENDITURE, GOVERNMENT_BODIES_EXPENDITURE_DESCRIPTION);
134 
135                 createButtonLink(grid, HEADCOUNT, VaadinIcons.USERS, PageCommandGovernmentBodyRankingConstants.COMMAND_GOVERNMENT_BODIES_HEADCOUNT,
136                                 HEADCOUNT_DESCRIPTION);
137                 createButtonLink(grid, INCOME, VaadinIcons.MONEY_DEPOSIT, PageCommandGovernmentBodyRankingConstants.COMMAND_GOVERNMENT_BODIES_INCOME,
138                                 INCOME_DESCRIPTION);
139                 createButtonLink(grid, EXPENDITURE, VaadinIcons.MONEY_WITHDRAW, PageCommandGovernmentBodyRankingConstants.COMMAND_GOVERNMENT_BODIES_EXPENDITURE,
140                                 EXPENDITURE_DESCRIPTION);
141                 createButtonLink(grid, RANKING_PAGE_VISIT_TEXT, VaadinIcons.CHART,
142                 		GOVERNMENT_RANKING_COMMAND_PAGEVISIT_HISTORY, RANKING_PAGE_VISIT_DESC);
143         }
144 }
