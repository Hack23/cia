120 	/**
121 	 * Create content for the dashboard overview page. Displays KPIs, statistics,
122 	 * and financial info for monarch, government, parliament, and related entities.
123 	 *
124 	 * @param parameters the navigation parameters
125 	 * @param menuBar    the main menu bar
126 	 * @param panel      the main panel holding the content
127 	 * @return a Layout containing the dashboard overview
128 	 */
129 	@Secured({ "ROLE_ANONYMOUS", "ROLE_USER", "ROLE_ADMIN" })
130 	@Override
131 	public Layout createContent(final String parameters, final MenuBar menuBar, final Panel panel) {
132 		final VerticalLayout panelContent = createPanelContent();
133 		final String pageId = getPageId(parameters);
134 
135 		getMenuItemFactory().createMainPageMenuBar(menuBar);
136 		CardInfoRowUtil.createPageHeader(panel, panelContent,
137 				CommonViewConstants.DASHBOARD_TITLE,
138 				CommonViewConstants.DASHBOARD_HEADER,
139 				CommonViewConstants.DASHBOARD_DESCRIPTION);
140 
141 		final ResponsiveRow row = RowUtil.createGridLayout(panelContent);
142 
143 		createDashboardMonarch(row);
144 		createDashboardGovernment(row);
145 		createDashboardParliament(row);
146 		createDashboardPartRiskByType(row);
147 		createDashboardPartRiskBySeverity(row);
148 
149 		getPageActionEventHelper().createPageEvent(ViewAction.VISIT_DASHBOARD_VIEW, ApplicationEventGroup.USER,
150 				CommonsViews.DASHBOARD_VIEW_NAME, parameters, pageId);
151 
152 		return panelContent;
153 	}

472 	/**
473 	 * Create a card showing headcount for a given orgId.
474 	 *
475 	 * @param panelContent the panel content
476 	 * @param orgId the org id
477 	 */
478 	private void createHeadCountCard(final Layout panelContent, final String orgId) {
479 		final Map<String, List<GovernmentBodyAnnualSummary>> yearlyData = esvApi.getData().get(2024).stream()
480 				.collect(Collectors.groupingBy(GovernmentBodyAnnualSummary::getOrgNumber));
481 
482 		final List<GovernmentBodyAnnualSummary> summaries = yearlyData.get(getPageId(orgId));
483 
484 		if (summaries != null && !summaries.isEmpty()) {
485 			final GovernmentBodyAnnualSummary bodySummary = summaries.get(0);
486 			if (bodySummary != null) {
487 				final Map<Integer,GovernmentBodyAnnualSummary> dataPerBody = esvApi.getDataPerGovernmentBody(bodySummary.getName());
488 				final Set<Integer> yearKeys = dataPerBody.keySet();
489 				final GovernmentBodyAnnualSummary latestSummary = dataPerBody.get(Collections.max(yearKeys));
490 
491 				panelContent.addComponent(
492 						new CounterStatisticsCard(
493 								VaadinIcons.WARNING,
494 								new CounterStatisticModel("Headcount", latestSummary.getHeadCount())
495 										.withShow(StatisticShow.Sum).withIconHidden().withShowOnlyStatistic(true),
496 								"Headcount"));
497 			}
498 		}
499 	}
