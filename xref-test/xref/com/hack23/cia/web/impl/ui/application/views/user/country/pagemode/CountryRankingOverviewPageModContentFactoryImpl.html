33  /**
34   * The Class CountryRankingOverviewPageModContentFactoryImpl.
35   */
36  @Component
37  public final class CountryRankingOverviewPageModContentFactoryImpl extends AbstractCountryPageModContentFactoryImpl {
38  
39  	/**
40  	 * Instantiates a new country ranking overview page mod content factory
41  	 * impl.
42  	 */
43  	public CountryRankingOverviewPageModContentFactoryImpl() {
44  		super();
45  	}
46  
47  	@Secured({ "ROLE_ANONYMOUS", "ROLE_USER", "ROLE_ADMIN" })
48  	@Override
49  	public Layout createContent(final String parameters, final MenuBar menuBar, final Panel panel) {
50  		final VerticalLayout panelContent = createPanelContent();
51  		getCountryMenuItemFactory().createCountryTopicMenu(menuBar);
52  		CardInfoRowUtil.createPageHeader(panel, panelContent,
53              CountryViewConstants.RANKING_OVERVIEW_TITLE,
54              CountryViewConstants.RANKING_OVERVIEW_SUBTITLE,
55              CountryViewConstants.RANKING_OVERVIEW_DESC);
56  		final String pageId = getPageId(parameters);
57  		getCountryMenuItemFactory().createOverviewPage(panelContent);
58  		getPageActionEventHelper().createPageEvent(ViewAction.VISIT_COUNTRY_VIEW, ApplicationEventGroup.USER, NAME, parameters, pageId);
59  
60  		return panelContent;
61  
62  	}
63  
64  	@Override
65  	public boolean matches(final String page, final String parameters) {
66  		return PageCommandCountryRankingConstants.COMMAND_COUNTRY_RANKING_OVERVIEW.matches(page, parameters);
67  	}

69  }
