18   * The Class GovernmentBodyMenuItemFactoryImpl.
19   */
20  @Service
21  public final class GovernmentBodyMenuItemFactoryImpl extends AbstractMenuItemFactoryImpl
22          implements GovernmentBodyMenuItemFactory {
23  
24      /** The application menu item factory. */
25      @Autowired
26      private ApplicationMenuItemFactory applicationMenuItemFactory;
27  
28      /** The government body ranking menu item factory. */
29      @Autowired
30      private GovernmentBodyRankingMenuItemFactory governmentBodyRankingMenuItemFactory;
31  
32      /**
33       * Instantiates a new government body menu item factory impl.
34       */
35      public GovernmentBodyMenuItemFactoryImpl() {
36          super();
37      }
38  
39      @Override
40      public void createGovernmentBodyMenuBar(final MenuBar menuBar, final String pageId, final String title) {
41          initApplicationMenuBar(menuBar);
42  
43          applicationMenuItemFactory.addRankingMenu(menuBar);
44  
45          governmentBodyRankingMenuItemFactory.createGovernmentBodyRankingTopics(
46                  menuBar.addItem(GOVERNMENT_BODY_RANKING, VaadinIcons.BUILDING_O, null));
47  
48          final MenuItem governmentBodyItem = menuBar.addItem(title, VaadinIcons.BUILDING_O, null);
49  
50          governmentBodyItem.addItem(GOVERNMENT_BODY_OVERVIEW_TEXT, VaadinIcons.FILE_TEXT,
51                  PageCommandGovernmentBodyConstants.COMMAND_GOVERNMENT_BODY_OVERVIEW.createItemPageCommand(pageId));
52  
53          governmentBodyItem.addItem(HEADCOUNT_CHART, VaadinIcons.USER,
54                  PageCommandGovernmentBodyConstants.COMMAND_GOVERNMENT_BODY_HEADCOUNT.createItemPageCommand(pageId));
55  
56          governmentBodyItem.addItem(INCOME, VaadinIcons.MONEY,
57                  PageCommandGovernmentBodyConstants.COMMAND_GOVERNMENT_BODY_INCOME.createItemPageCommand(pageId));
58  
59          governmentBodyItem.addItem(EXPENDITURE, VaadinIcons.CREDIT_CARD,
60                  PageCommandGovernmentBodyConstants.COMMAND_GOVERNMENT_BODY_EXPENDITURE.createItemPageCommand(pageId));
61      }
62  
63      @Override
64      public void createOverviewPage(final VerticalLayout panelContent, final String pageId) {
65          final ResponsiveRow grid = RowUtil.createGridLayout(panelContent);
66  
67          createButtonLink(grid, HEADCOUNT_CHART, VaadinIcons.USER,
68                  PageCommandGovernmentBodyConstants.COMMAND_GOVERNMENT_BODY_HEADCOUNT.createItemPageCommand(pageId),
69                  HEADCOUNT_DESCRIPTION);
70  
71          createButtonLink(grid, INCOME, VaadinIcons.MONEY,
72                  PageCommandGovernmentBodyConstants.COMMAND_GOVERNMENT_BODY_INCOME.createItemPageCommand(pageId),
73                  INCOME_DESCRIPTION);
74  
75          createButtonLink(grid, EXPENDITURE, VaadinIcons.CREDIT_CARD,
76                  PageCommandGovernmentBodyConstants.COMMAND_GOVERNMENT_BODY_EXPENDITURE.createItemPageCommand(pageId),
77                  EXPENDITURE_DESCRIPTION);
78      }
79  }
