43  /**
44   * The Class AdminApplicationEventsPageModContentFactoryImpl.
45   */
46  @Component
47  public final class AdminApplicationEventsPageModContentFactoryImpl extends AbstractAdminSystemPageModContentFactoryImpl {
48  
49  	/** The Constant APPLICATION_ACTION_EVENT. */
50  	private static final String APPLICATION_ACTION_EVENT = "ApplicationActionEvent";
51  
52  	/** The Constant COLUMN_ORDER. */
53  	private static final String[] COLUMN_ORDER = { "hjid", "createdDate", "userId", "actionName", "errorMessage",
54  			"applicationMessage", "page", "pageMode", "elementId", "modelObjectVersion" };
55  
56  	/** The Constant HIDE_COLUMNS. */
57  	private static final String[] HIDE_COLUMNS = { "hjid", "modelObjectId", "modelObjectVersion", "sessionId",
58  			"eventGroup", "applicationOperation" };
59  
60  	/** The Constant LISTENER. */
61  	private static final PageItemPropertyClickListener LISTENER = new PageItemPropertyClickListener(
62  			AdminViews.ADMIN_APPLICATIONS_EVENTS_VIEW_NAME, "hjid");
63  
64  	/** The Constant NAME. */
65  	public static final String NAME = AdminViews.ADMIN_APPLICATIONS_EVENTS_VIEW_NAME;
66  
67  	/**
68  	 * Instantiates a new admin application events page mod content factory impl.
69  	 */
70  	public AdminApplicationEventsPageModContentFactoryImpl() {
71  		super(NAME);
72  	}
73  
74  	@Secured({ "ROLE_ADMIN" })
75  	@Override
76  	public Layout createContent(final String parameters, final MenuBar menuBar, final Panel panel) {
77  		final VerticalLayout content = createPanelContent();
78  
79  		final String pageId = getPageId(parameters);
80  		final int pageNr = getPageNr(parameters);
81  
82  		getMenuItemFactory().createMainPageMenuBar(menuBar);
83  
84  		CardInfoRowUtil.createPageHeader(panel, content, AdminViewConstants.ADMIN_APPLICATION_EVENTS, AdminViewConstants.EVENT_DETAILS,
85  				AdminViewConstants.EVENT_REVIEW);
86  
87  		final DataContainer<ApplicationActionEvent, Long> dataContainer = getApplicationManager()
88  				.getDataContainer(ApplicationActionEvent.class);
89  
90  		final List<ApplicationActionEvent> pageOrderBy = dataContainer.getPageOrderBy(pageNr, DEFAULT_RESULTS_PER_PAGE,
91  				ApplicationActionEvent_.createdDate);
92  
93  		getPagingUtil().createPagingControls(content, NAME, pageId, dataContainer.getSize(), pageNr,
94  				DEFAULT_RESULTS_PER_PAGE);
95  
96  		getGridFactory().createBasicBeanItemGrid(content, ApplicationActionEvent.class, pageOrderBy,
97  				APPLICATION_ACTION_EVENT, COLUMN_ORDER, HIDE_COLUMNS, LISTENER, null, null);
98  
99  		if (pageId != null && !pageId.isEmpty()) {
100 			final ApplicationActionEvent applicationActionEvent = dataContainer.load(Long.valueOf(pageId));
101 			if (applicationActionEvent != null) {
102 				// Instead of form fields, use a card layout similar to the party overview.
103 				final Panel cardPanel = new Panel();
104 				cardPanel.addStyleName("politician-overview-card"); // Reuse existing style
105 				cardPanel.setWidth("100%");
106 				cardPanel.setHeightUndefined();
107 				Responsive.makeResponsive(cardPanel);
108 
109 				final VerticalLayout cardContent = new VerticalLayout();
110 				cardContent.setMargin(true);
111 				cardContent.setSpacing(true);
112 				cardContent.setWidth("100%");
113 				cardPanel.setContent(cardContent);
114 
115 				content.addComponent(cardPanel);
116 
117 				CardInfoRowUtil.createCardHeader(cardContent, AdminViewConstants.EVENT_DETAILS);
118 
119 				// Attributes layout
120 				final VerticalLayout attributesLayout = new VerticalLayout();
121 				attributesLayout.setSpacing(true);
122 				attributesLayout.setWidth("100%");
123 				cardContent.addComponent(attributesLayout);
124 
125 				CardInfoRowUtil.addInfoRowIfNotNull(attributesLayout, AdminViewConstants.CREATED_DATE,
126 						String.valueOf(applicationActionEvent.getCreatedDate()), VaadinIcons.CALENDAR);
127 				CardInfoRowUtil.addInfoRowIfNotNull(attributesLayout, AdminViewConstants.EVENT_GROUP,
128 						String.valueOf(applicationActionEvent.getEventGroup()), VaadinIcons.INFO);
129 				CardInfoRowUtil.addInfoRowIfNotNull(attributesLayout, AdminViewConstants.APPLICATION_OPERATION,
130 						String.valueOf(applicationActionEvent.getApplicationOperation()), VaadinIcons.TOOLS);
131 				CardInfoRowUtil.addInfoRowIfNotNull(attributesLayout, AdminViewConstants.PAGE,
132 						String.valueOf(applicationActionEvent.getPage()), VaadinIcons.FILE_TEXT);
133 				CardInfoRowUtil.addInfoRowIfNotNull(attributesLayout, AdminViewConstants.PAGE_MODE,
134 						String.valueOf(applicationActionEvent.getPageMode()), VaadinIcons.LIST);
135 				CardInfoRowUtil.addInfoRowIfNotNull(attributesLayout, AdminViewConstants.ELEMENT_ID,
136 						String.valueOf(applicationActionEvent.getElementId()), VaadinIcons.POINTER);
137 				CardInfoRowUtil.addInfoRowIfNotNull(attributesLayout, AdminViewConstants.ACTION_NAME,
138 						String.valueOf(applicationActionEvent.getActionName()), VaadinIcons.PLAY_CIRCLE);
139 				CardInfoRowUtil.addInfoRowIfNotNull(attributesLayout, AdminViewConstants.USER_ID,
140 						String.valueOf(applicationActionEvent.getUserId()), VaadinIcons.USER);
141 				CardInfoRowUtil.addInfoRowIfNotNull(attributesLayout, AdminViewConstants.SESSION_ID,
142 						String.valueOf(applicationActionEvent.getSessionId()), VaadinIcons.KEY);
143 				CardInfoRowUtil.addInfoRowIfNotNull(attributesLayout, AdminViewConstants.ERROR_MESSAGE,
144 						String.valueOf(applicationActionEvent.getErrorMessage()), VaadinIcons.WARNING);
145 				CardInfoRowUtil.addInfoRowIfNotNull(attributesLayout, AdminViewConstants.APPLICATION_MESSAGE,
146 						String.valueOf(applicationActionEvent.getApplicationMessage()), VaadinIcons.INFO_CIRCLE);
147 			}
148 		}
149 
150 		getPageActionEventHelper().createPageEvent(ViewAction.VISIT_ADMIN_APPLICATION_EVENTS_VIEW,
151 				ApplicationEventGroup.ADMIN, NAME, null, pageId);
152 
153 		return content;
154 	}
155 
156 	@Override
157 	public boolean matches(final String page, final String parameters) {
158 		return PageCommandAdminConstants.COMMAND_APPLICATION_EVENTS.matches(page, parameters);
159 	}
