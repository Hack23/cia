20  public final class ChartUtils {

22      /** The Constant CHART_BOTTOM_MARGIN_SIZE. */
23      private static final int CHART_BOTTOM_MARGIN_SIZE = 2;
24  
25      /** The Constant CHART_LEFT_MARGIN. */
26      private static final int CHART_LEFT_MARGIN = 2;
27  
28      /**  The Constant CHART_RIGHT_MARGIN = 2;. */
29      private static final int CHART_RIGHT_MARGIN = 2;
30  
31      /** The Constant CHART_TOP_MARGIN_SIZE. */
32      private static final int CHART_TOP_MARGIN_SIZE = 2;
33  
34      /** The Constant CHART_WIDTH_REDUCTION. */
35      private static final int CHART_WIDTH_REDUCTION = 50;
36  
37      /** The Constant HEIGHT_PERCENTAGE_FULL_PAGE. */
38      private static final double HEIGHT_PERCENTAGE_FULL_PAGE = 0.8;
39  
40      /** The Constant HEIGHT_PERCETAGE_HALF_PAGE. */
41      private static final double HEIGHT_PERCETAGE_HALF_PAGE = 0.5;
42  
43      /** The Constant MINIMUM_CHART_HEIGHT_FULL_PAGE. */
44      private static final int MINIMUM_CHART_HEIGHT_FULL_PAGE = 400;
45  
46      /** The Constant MINIMUM_CHART_WIDTH. */
47      private static final int MINIMUM_CHART_WIDTH = 600;
48  
49      /** The Constant NINIMUM_CHART_HEIGHT_HALF_PAGE. */
50      private static final int NINIMUM_CHART_HEIGHT_HALF_PAGE = 200;
51  
52      /**
53       * Instantiates a new chart utils.
54       */
55      private ChartUtils() {
56          // Utility class
57      }
58  
59      /**
60       * Gets the chart window height.
61       *
62       * @param isFullPage the is full page
63       * @return the chart window height
64       */
65      public static int getChartWindowHeight(final boolean isFullPage) {
66          if (isFullPage) {
67              return Math.max((int) (Page.getCurrent().getBrowserWindowHeight() * HEIGHT_PERCENTAGE_FULL_PAGE), MINIMUM_CHART_HEIGHT_FULL_PAGE);
68          } else {
69              return Math.max((int) (Page.getCurrent().getBrowserWindowHeight() * HEIGHT_PERCETAGE_HALF_PAGE), NINIMUM_CHART_HEIGHT_HALF_PAGE);
70          }
71      }
72  
73      /**
74       * Gets the chart window width.
75       *
76       * @return the chart window width
77       */
78      public static int getChartWindowWidth() {
79          return Math.max(Page.getCurrent().getBrowserWindowWidth() - CHART_WIDTH_REDUCTION, MINIMUM_CHART_WIDTH);
80      }
81  
82      /**
83       * Adds the chart.
84       *
85       * @param layout the layout
86       * @param caption the caption
87       * @param chart the chart
88       * @param isFullPage the is full page
89       */
90      public static void addChart(final AbstractOrderedLayout layout, final String caption, final DCharts chart, final boolean isFullPage) {
91          final HorizontalLayout horizontalLayout = new HorizontalLayout();
92  
93          final int browserWindowWidth = getChartWindowWidth();
94          final int browserWindowHeight = getChartWindowHeight(isFullPage);
95  
96          horizontalLayout.setWidth(browserWindowWidth, Unit.PIXELS);
97          horizontalLayout.setHeight(browserWindowHeight, Unit.PIXELS);
98          horizontalLayout.setMargin(true);
99          horizontalLayout.setSpacing(false);
100         horizontalLayout.addStyleName("v-layout-content-overview-panel-level1");
101 
102         final Panel formPanel = new Panel();
103         formPanel.setSizeFull();
104         formPanel.setContent(horizontalLayout);
105         formPanel.setCaption(caption);
106 
107         layout.addComponent(formPanel);
108         layout.setExpandRatio(formPanel, ContentRatio.LARGE);
109 
110         chart.setWidth(100, Unit.PERCENTAGE);
111         chart.setHeight(100, Unit.PERCENTAGE);
112         chart.setMarginRight(CHART_RIGHT_MARGIN);
113         chart.setMarginLeft(CHART_LEFT_MARGIN);
114         chart.setMarginBottom(CHART_BOTTOM_MARGIN_SIZE);
115         chart.setMarginTop(CHART_TOP_MARGIN_SIZE);
116 
117         horizontalLayout.addComponent(chart);
118         chart.setCaption(caption);
119     }
120 
121     /**
122      * Gets the party name.
123      *
124      * @param applicationManager the application manager
125      * @param partySummary the party summary
126      * @return the party name
127      */
128     public static String getPartyName(final ApplicationManager applicationManager, final String partySummary) {
129         final DataContainer<ViewRiksdagenParty, String> partyDataContainer = applicationManager.getDataContainer(ViewRiksdagenParty.class);
130         final ViewRiksdagenParty party = partyDataContainer.load(partySummary);
131         return Optional.ofNullable(party).map(ViewRiksdagenParty::getPartyName).orElse(null);
132     }
133 
134     /**
135      * Gets the role color.
136      *
137      * @param roleCode the role code
138      * @param role1 the role 1
139      * @param role2 the role 2
140      * @return the role color
141      */
142     public static String getRoleColor(final String roleCode, final String role1, final String role2) {
143         if (roleCode.equalsIgnoreCase(role1)) {
144             return "red";
145         } else if (roleCode.equalsIgnoreCase(role2)) {
146             return "blue";
147         } else {
148             return "green";
149         }
150     }
