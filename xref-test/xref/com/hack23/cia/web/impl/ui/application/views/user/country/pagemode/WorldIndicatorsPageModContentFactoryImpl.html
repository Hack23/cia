53  @Component
54  public final class WorldIndicatorsPageModContentFactoryImpl extends AbstractCountryPageModContentFactoryImpl {
55  
56  	/** The chart data manager. */
57  	@Autowired
58  	private WorldIndicatorChartDataManager chartDataManager;
59  
60  	/**
61  	 * Instantiates a new world indicators page mod content factory impl.
62  	 */
63  	public WorldIndicatorsPageModContentFactoryImpl() {
64  		super();
65  	}
66  
67  	@Secured({ "ROLE_ANONYMOUS", "ROLE_USER", "ROLE_ADMIN" })
68  	@Override
69  	public Layout createContent(final String parameters, final MenuBar menuBar, final Panel panel) {
70  		final VerticalLayout panelContent = createPanelContent();
71  		getCountryMenuItemFactory().createCountryTopicMenu(menuBar);
72  		CardInfoRowUtil.createPageHeader(panel, panelContent,
73  			CountryViewConstants.INDICATOR_TITLE,
74  			CountryViewConstants.INDICATOR_SUBTITLE,
75  			CountryViewConstants.INDICATOR_DESC);
76  		final String pageId = getPageId(parameters);
77  
78  		final String indicator = parameters.substring(PageMode.INDICATORS.toString().length() + "/".length());
79  
80  		createDataIndicatorSummaryChartPanel(panelContent, indicator, panel);
81  
82  		getPageActionEventHelper().createPageEvent(ViewAction.VISIT_COUNTRY_VIEW, ApplicationEventGroup.USER, NAME,
83  				parameters, pageId);
84  
85  		return panelContent;
86  	}
87  
88  	/**
89  	 * Creates the data indicator summary chart panel.
90  	 *
91  	 * @param verticalLayout the vertical layout
92  	 * @param indicator      the indicator
93  	 * @param panel          the panel
94  	 */
95  	private void createDataIndicatorSummaryChartPanel(final VerticalLayout verticalLayout, final String indicator,
96  			final Panel panel) {
97  		final DataContainer<ViewWorldbankIndicatorDataCountrySummary, WorldbankIndicatorDataCountrySummaryEmbeddedId> indicatorDataSummaryContainer = getApplicationManager()
98  				.getDataContainer(ViewWorldbankIndicatorDataCountrySummary.class);
99  
100 		final Optional<ViewWorldbankIndicatorDataCountrySummary> indicatorSummary = indicatorDataSummaryContainer
101 				.getAll().parallelStream()
102 				.filter(t -> t != null && t.getEmbeddedId().getIndicatorId().equals(indicator)).findFirst();
103 
104 		ViewWorldbankIndicatorDataCountrySummary indicatorSummaryValue = null;
105 		if (indicatorSummary.isPresent()) {
106 			indicatorSummaryValue = indicatorSummary.get();
107 
108 			// Instead of formFactory, we create a card for the indicator summary
109 			final Label sectionHeader = new Label(CountryViewConstants.INDICATOR_SUMMARY_HEADER);
110 			sectionHeader.addStyleName(CountryViewConstants.SECTION_HEADER_STYLE);
111 			verticalLayout.addComponent(sectionHeader);
112 
113 			final VerticalLayout card = new VerticalLayout();
114 			card.setMargin(true);
115 			card.setSpacing(true);
116 			card.addStyleName(CountryViewConstants.INDICATOR_SUMMARY_CARD_STYLE);
117 			Responsive.makeResponsive(card);
118 
119 			// For each field in AS_LIST, we create a row
120 			if (indicatorSummaryValue.getIndicatorName() != null) {
121 				card.addComponent(CardInfoRowUtil.createInfoRow(
122 					CountryViewConstants.INDICATOR_NAME_LABEL,
123 					indicatorSummaryValue.getIndicatorName(),
124 					VaadinIcons.INFO_CIRCLE,
125 					CountryViewConstants.INDICATOR_NAME_TOOLTIP));
126 			}
127 			if (indicatorSummaryValue.getSourceValue() != null) {
128 				card.addComponent(CardInfoRowUtil.createInfoRow("Source Value:", indicatorSummaryValue.getSourceValue(),
129 						VaadinIcons.GLOBE, "Source of this indicator data"));
130 			}
131 			if (indicatorSummaryValue.getSourceOrganization() != null) {
132 				card.addComponent(CardInfoRowUtil.createInfoRow("Source Organization:", indicatorSummaryValue.getSourceOrganization(),
133 						VaadinIcons.INSTITUTION, "Organization providing this data"));
134 			}
135 
136 			verticalLayout.addComponent(card);
137 		}
138 
139 		final DataContainer<WorldBankData, Serializable> dataContainer = getApplicationManager()
140 				.getDataContainer(WorldBankData.class);
141 
142 		final List<WorldBankData> dataList = dataContainer.findListByEmbeddedProperty(WorldBankData.class,
143 				WorldBankData_.indicator, Indicator.class, Indicator_.id, indicator);
144 
145 		chartDataManager.createIndicatorChart(verticalLayout, dataList, indicatorSummaryValue);
146 	}
147

151 		return PageCommandCountryRankingConstants.COMMAND_COUNTRY_RANKING_INDICATORS.matches(page, parameters);
152 	}
