71  	/**
72  	 * Creates the content.
73  	 *
74  	 * @param parameters the parameters
75  	 * @param menuBar the menu bar
76  	 * @param panel the panel
77  	 * @return the layout
78  	 */
79  	@Secured({ "ROLE_ANONYMOUS", "ROLE_USER", "ROLE_ADMIN" })
80  	@Override
81  	public Layout createContent(final String parameters, final MenuBar menuBar, final Panel panel) {
82  		panel.setSizeFull();
83  
84  		final VerticalLayout panelContent = createPanelContent();
85  		panelContent.setSizeFull();
86  		panel.setContent(panelContent);
87  
88  		getMinistryRankingMenuItemFactory().createMinistryRankingMenuBar(menuBar);
89  
90  		final String pageId = getPageId(parameters);
91  
92  		CardInfoRowUtil.createPageHeader(panel, panelContent,
93              MinistryRankingViewConstants.TITLE_MINISTRY_RANKINGS,
94              MinistryRankingViewConstants.LEADER_SCOREBOARD_TITLE,
95              MinistryRankingViewConstants.LEADER_SCOREBOARD_DESC);
96  
97  		final ResponsiveRow row = RowUtil.createGridLayout(panelContent);
98  		row.setSizeFull();
99  
100 		final List<ViewRiksdagenGovermentRoleMember> activeGovMembers = leaderCardUtil.loadActiveGovernmentRoleMembers();
101 		final Map<String, List<ViewRiksdagenPolitician>> activePoliticianMap = leaderCardUtil.loadActivePoliticiansByPersonId();
102 
103 		final Map<String, Boolean> partyLeaderMap = PartyLeaderUtil.computePartyLeaders( getApplicationManager() ,activePoliticianMap.keySet());
104 
105 		// Sort roles
106 		activeGovMembers.sort((a, b) -> {
107 			final boolean aLeader = partyLeaderMap.getOrDefault(a.getPersonId(), false);
108 			final boolean bLeader = partyLeaderMap.getOrDefault(b.getPersonId(), false);
109 			final int aPriority = getRolePriority(a.getRoleCode(), aLeader);
110 			final int bPriority = getRolePriority(b.getRoleCode(), bLeader);
111 			return Integer.compare(aPriority, bPriority);
112 		});
113 
114 		final Map<Integer, List<GovernmentBodyAnnualSummary>> dataMap = esvApi.getData();
115 		final List<GovernmentBodyAnnualSummary> currentYearGovernmentBodies = dataMap.get(CURRENT_YEAR);
116 		final Map<String, List<GovernmentBodyAnnualSummary>> governmentBodyByMinistry = currentYearGovernmentBodies
117 				.stream().collect(Collectors.groupingBy(GovernmentBodyAnnualSummary::getMinistry));
118 
119 		final Map<String, List<GovernmentBodyAnnualOutcomeSummary>> reportByMinistry = esvApi
120 				.getGovernmentBodyReportByMinistry();
121 
122 		for (final ViewRiksdagenGovermentRoleMember govMember : activeGovMembers) {
123 			final ViewRiksdagenPolitician politician = activePoliticianMap.get(govMember.getPersonId()).get(0);
124 			final ViewRiksdagenPoliticianBallotSummary ballotSummary = getApplicationManager()
125 					.getDataContainer(ViewRiksdagenPoliticianBallotSummary.class).load(govMember.getPersonId());
126 		    final ViewRiksdagenPoliticianExperienceSummary experienceSummary = getApplicationManager().getDataContainer(ViewRiksdagenPoliticianExperienceSummary.class).load(govMember.getPersonId());
127 
128 
129 			final Panel cardPanel = leaderCardUtil.createBaseballStyleCard(govMember, politician, ballotSummary, governmentBodyByMinistry,
130 					reportByMinistry,experienceSummary);
131 
132 			// Responsive column rules
133 			row.addColumn().withDisplayRules(DISPLAY_SIZE_XS_DEVICE, DISPLAYS_SIZE_XM_DEVICE, DISPLAY_SIZE_MD_DEVICE,
134 					DISPLAY_SIZE_LG_DEVICE).withComponent(cardPanel);
135 		}
136 
137 		getPageActionEventHelper().createPageEvent(ViewAction.VISIT_MINISTRY_RANKING_VIEW, ApplicationEventGroup.USER,
138 				NAME, parameters, pageId);
139 
140 		return panelContent;
141 	}

143 	/**
144 	 * Gets the role priority.
145 	 *
146 	 * @param role the role
147 	 * @param isPartyLeader the is party leader
148 	 * @return the role priority
149 	 */
150 	private int getRolePriority(final String role, final boolean isPartyLeader) {
151 		final String roleNormalized = role.toLowerCase(Locale.ROOT).trim();
152 		if ("statsminister".equals(roleNormalized)) {
153 			return 1;
154 		} else if (isPartyLeader) {
155 			return 2;
156 		} else if (roleNormalized.endsWith("minister")) {
157 			return 3;
158 		} else if ("statsråd".equalsIgnoreCase(roleNormalized)) {
159 			return 4;
160 		} else {
161 			return 5;
162 		}
163 	}
