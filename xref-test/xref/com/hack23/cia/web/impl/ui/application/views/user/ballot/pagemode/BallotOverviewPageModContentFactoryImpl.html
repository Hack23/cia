93  	/**
94  	 * Creates the content.
95  	 *
96  	 * @param parameters the parameters
97  	 * @param menuBar the menu bar
98  	 * @param panel the panel
99  	 * @return the layout
100 	 */
101 	@Secured({ "ROLE_ANONYMOUS", "ROLE_USER", "ROLE_ADMIN" })
102 	@Override
103 	public Layout createContent(final String parameters, final MenuBar menuBar, final Panel panel) {
104 		final VerticalLayout panelContent = createPanelContent();
105 		panel.setContent(panelContent);
106 
107 		final String pageId = getPageId(parameters);
108 		final List<ViewRiksdagenVoteDataBallotSummary> ballots = getItem(parameters);
109 
110 		if (!ballots.isEmpty()) {
111 			getBallotMenuItemFactory().createBallotMenuBar(menuBar, pageId);
112 
113 			final DataContainer<ViewRiksdagenVoteDataBallotPartySummary, RiksdagenVoteDataBallotPartyEmbeddedId> dataPartyContainer = getApplicationManager()
114 					.getDataContainer(ViewRiksdagenVoteDataBallotPartySummary.class);
115 
116 			final DataContainer<ViewRiksdagenCommitteeBallotDecisionSummary, ViewRiksdagenCommitteeBallotDecisionEmbeddedId> dataDecisionContainer = getApplicationManager()
117 					.getDataContainer(ViewRiksdagenCommitteeBallotDecisionSummary.class);
118 
119 			final List<ViewRiksdagenVoteDataBallotPartySummary> partyBallotList = dataPartyContainer
120 					.findListByEmbeddedProperty(ViewRiksdagenVoteDataBallotPartySummary.class,
121 							ViewRiksdagenVoteDataBallotPartySummary_.embeddedId,
122 							RiksdagenVoteDataBallotPartyEmbeddedId.class,
123 							RiksdagenVoteDataBallotPartyEmbeddedId_.ballotId, pageId);
124 
125 			final List<ViewRiksdagenCommitteeBallotDecisionSummary> decisionSummaries = dataDecisionContainer
126 					.getAllBy(ViewRiksdagenCommitteeBallotDecisionSummary_.ballotId, pageId);
127 
128 			final boolean useDecisionSummaries = !decisionSummaries.isEmpty();
129 
130 			String mainTitle;
131 			String subTitle;
132 			if (useDecisionSummaries) {
133 				mainTitle = BallotViewConstants.OVERVIEW_MAIN_TITLE_PREFIX + decisionSummaries.get(0).getTitle() + " - "
134 						+ decisionSummaries.get(0).getSubTitle();
135 			} else {
136 				final ViewRiksdagenVoteDataBallotSummary firstBallot = ballots.get(FIRST_OBJECT);
137 				mainTitle = BallotViewConstants.OVERVIEW_MAIN_TITLE_PREFIX + firstBallot.getEmbeddedId().getConcern();
138 			}
139 			subTitle = BallotViewConstants.OVERVIEW_CARD_BALLOT_INFO;
140 
141 			CardInfoRowUtil.createPageHeader(panel, panelContent, mainTitle, subTitle,
142 					BallotViewConstants.OVERVIEW_PAGE_DESCRIPTION);
143 
144 			// Card panel
145 			final Panel cardPanel = new Panel();
146 			cardPanel.addStyleName("politician-overview-card");
147 			cardPanel.setWidth("100%");
148 			cardPanel.setHeightUndefined();
149 			Responsive.makeResponsive(cardPanel);
150 
151 			final VerticalLayout cardContent = new VerticalLayout();
152 			cardContent.setMargin(true);
153 			cardContent.setSpacing(true);
154 			cardContent.setWidth("100%");
155 			cardPanel.setContent(cardContent);
156 
157 			panelContent.addComponent(cardPanel);
158 			panelContent.setExpandRatio(cardPanel, ContentRatio.GRID);
159 
160 			CardInfoRowUtil.createCardHeader(cardContent, BallotViewConstants.OVERVIEW_CARD_BALLOT_INFO);
161 
162 			// Two-column layout
163 			final HorizontalLayout attributesLayout = new HorizontalLayout();
164 			attributesLayout.setSpacing(true);
165 			attributesLayout.setWidth("100%");
166 			cardContent.addComponent(attributesLayout);
167 
168 			// Left column: Ballot Profile (Textual and identifying attributes)
169 			final VerticalLayout profileDetailsLayout = CardInfoRowUtil.createSectionLayout(BallotViewConstants.OVERVIEW_CARD_BALLOT_PROFILE);
170 
171 			// Right column: Voting Statistics (Numeric and outcome attributes)
172 			final VerticalLayout serviceStatsLayout = CardInfoRowUtil.createSectionLayout(BallotViewConstants.OVERVIEW_CARD_VOTING_STATS);
173 
174 			attributesLayout.addComponents(profileDetailsLayout, serviceStatsLayout);
175 
176 			// Populate fields depending on data
177 			if (useDecisionSummaries) {
178 				final ViewRiksdagenCommitteeBallotDecisionSummary ds = decisionSummaries.get(FIRST_OBJECT);
179 				// Show essential profile fields
180 				profileDetailsLayout.addComponent(CardInfoRowUtil.createInfoRow(BallotViewConstants.FIELD_VOTE_DATE, String.valueOf(ds.getVoteDate()), VaadinIcons.CALENDAR, "Date of the vote"));
181 				profileDetailsLayout.addComponent(CardInfoRowUtil.createInfoRow(BallotViewConstants.FIELD_TITLE, ds.getTitle(), VaadinIcons.FILE_TEXT_O, "Title of the ballot"));
182 				profileDetailsLayout.addComponent(CardInfoRowUtil.createInfoRow(BallotViewConstants.FIELD_SUBTITLE, ds.getSubTitle(), VaadinIcons.FILE_TEXT, "Subtitle of the ballot"));
183 				profileDetailsLayout.addComponent(CardInfoRowUtil.createInfoRow(BallotViewConstants.FIELD_DECISION_TYPE, ds.getDecisionType(), VaadinIcons.QUESTION_CIRCLE, "Type of decision"));
184 
185 				// Stats fields
186 				serviceStatsLayout.addComponent(CardInfoRowUtil.createInfoRow(BallotViewConstants.FIELD_CONCERN, ds.getEmbeddedId().getConcern(), VaadinIcons.CLIPBOARD, "Concern or topic of the ballot"));
187 				serviceStatsLayout.addComponent(CardInfoRowUtil.createInfoRow(BallotViewConstants.FIELD_BALLOT_TYPE, ds.getBallotType(), VaadinIcons.BULLETS, "Type of ballot"));
188 				serviceStatsLayout.addComponent(CardInfoRowUtil.createInfoRow(BallotViewConstants.FIELD_WINNER, ds.getWinner(), VaadinIcons.TROPHY, "Winner or outcome"));
189 				serviceStatsLayout.addComponent(CardInfoRowUtil.createInfoRow(BallotViewConstants.FIELD_APPROVED, String.valueOf(ds.isApproved()), VaadinIcons.CHECK, "Whether the proposal was approved"));
190 
191 				// Optional: If still too crowded, remove againstProposal fields
192 				// serviceStatsLayout.addComponent(createInfoRow("Against Proposal Parties:", ds.getAgainstProposalParties(), VaadinIcons.WARNING, "Parties against the proposal"));
193 				// serviceStatsLayout.addComponent(createInfoRow("Against Proposal Number:", String.valueOf(ds.getAgainstProposalNumber()), VaadinIcons.WARNING, "Number of parties against the proposal"));
194 			} else {
195 				// Non-decision scenario
196 				final ViewRiksdagenVoteDataBallotSummary bs = ballots.get(FIRST_OBJECT);
197 				// Profile fields
198 				profileDetailsLayout.addComponent(CardInfoRowUtil.createInfoRow(BallotViewConstants.FIELD_BALLOT_ID, bs.getEmbeddedId().getBallotId(), VaadinIcons.CLIPBOARD_USER, "Ballot identification"));
199 				profileDetailsLayout.addComponent(CardInfoRowUtil.createInfoRow(BallotViewConstants.FIELD_RM, bs.getRm(), VaadinIcons.CALENDAR, "Session (RM) identifier"));
200 				profileDetailsLayout.addComponent(CardInfoRowUtil.createInfoRow(BallotViewConstants.FIELD_VOTE_DATE, String.valueOf(bs.getVoteDate()), VaadinIcons.CALENDAR, "Date of the vote"));
201 				profileDetailsLayout.addComponent(CardInfoRowUtil.createInfoRow(BallotViewConstants.FIELD_ISSUE, bs.getEmbeddedId().getIssue(), VaadinIcons.CLIPBOARD_TEXT, "The specific issue voted upon"));
202 				profileDetailsLayout.addComponent(CardInfoRowUtil.createInfoRow(BallotViewConstants.FIELD_CONCERN, bs.getEmbeddedId().getConcern(), VaadinIcons.CLIPBOARD, "Concern or topic of the ballot"));
203 				profileDetailsLayout.addComponent(CardInfoRowUtil.createInfoRow(BallotViewConstants.FIELD_BALLOT_TYPE, bs.getBallotType(), VaadinIcons.BULLETS, "Type of ballot"));
204 				profileDetailsLayout.addComponent(CardInfoRowUtil.createInfoRow(BallotViewConstants.FIELD_LABEL, bs.getLabel(), VaadinIcons.FILE_TEXT, "Short label or description"));
205 
206 				// Stats fields
207 				serviceStatsLayout.addComponent(CardInfoRowUtil.createInfoRow(BallotViewConstants.FIELD_TOTAL_VOTES, String.valueOf(bs.getTotalVotes()), VaadinIcons.GROUP, "Total number of votes cast"));
208 				serviceStatsLayout.addComponent(CardInfoRowUtil.createInfoRow(BallotViewConstants.FIELD_YES_VOTES, String.valueOf(bs.getYesVotes()), VaadinIcons.THUMBS_UP, "Number of 'yes' votes"));
209 				serviceStatsLayout.addComponent(CardInfoRowUtil.createInfoRow(BallotViewConstants.FIELD_NO_VOTES, String.valueOf(bs.getNoVotes()), VaadinIcons.THUMBS_DOWN, "Number of 'no' votes"));
210 				serviceStatsLayout.addComponent(CardInfoRowUtil.createInfoRow(BallotViewConstants.FIELD_ABSTAIN_VOTES, String.valueOf(bs.getAbstainVotes()), VaadinIcons.SPLIT, "Number of abstentions"));
211 				serviceStatsLayout.addComponent(CardInfoRowUtil.createInfoRow(BallotViewConstants.FIELD_ABSENT_VOTES, String.valueOf(bs.getAbsentVotes()), VaadinIcons.EXIT_O, "Number of absent voters"));
212 				serviceStatsLayout.addComponent(CardInfoRowUtil.createInfoRow(BallotViewConstants.FIELD_APPROVED, String.valueOf(bs.isApproved()), VaadinIcons.CHECK, "Whether the proposal was approved"));
213 			}
214 
215 			// Spacer before Party Ballot Summary
216 			final Label tableDivider = new Label("<hr/>", ContentMode.HTML);
217 			tableDivider.addStyleName("card-divider");
218 			tableDivider.setWidth("100%");
219 			panelContent.addComponent(tableDivider);
220 
221 			// Party Ballot Summary table
222 			getGridFactory().createBasicBeanItemNestedPropertiesGrid(panelContent,
223 					ViewRiksdagenVoteDataBallotPartySummary.class, partyBallotList,
224 					BallotViewConstants.GRID_PARTY_BALLOT_SUMMARY,
225 					NESTED_PROPERTIES, COLUMN_ORDER, HIDE_COLUMNS, LISTENER,
226 					BallotViewConstants.GRID_EMBEDDED_ID_PARTY, null);
227 
228 			// Overview layout after table
229 			final VerticalLayout overviewLayout = new VerticalLayout();
230 			overviewLayout.setSizeFull();
231 			panelContent.addComponent(overviewLayout);
232 			panelContent.setExpandRatio(overviewLayout, ContentRatio.LARGE_FORM);
233 			getBallotMenuItemFactory().createOverviewPage(overviewLayout, pageId);
234 
235 			getPageActionEventHelper().createPageEvent(ViewAction.VISIT_BALLOT_VIEW, ApplicationEventGroup.USER, NAME,
236 					parameters, pageId);
237 		}
238 
239 		return panelContent;
240 	}
