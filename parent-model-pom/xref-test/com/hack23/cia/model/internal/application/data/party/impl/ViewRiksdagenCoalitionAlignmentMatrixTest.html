130 	/**
131 	 * Asserts that a field has the expected column name mapping.
132 	 * 
133 	 * @param fieldName the field name
134 	 * @param expectedColumnName the expected column name
135 	 * @throws Exception if reflection fails
136 	 */
137 	private void assertColumnMapping(final String fieldName, final String expectedColumnName) throws Exception {
138 		final Field field = ViewRiksdagenCoalitionAlignmentMatrix.class.getDeclaredField(fieldName);
139 		final Column columnAnnotation = field.getAnnotation(Column.class);
140 		
141 		assertNotNull("Field " + fieldName + " should have @Column annotation", columnAnnotation);
142 		assertEquals("Field " + fieldName + " should map to column " + expectedColumnName,
143 			expectedColumnName, columnAnnotation.name());
144 	}
