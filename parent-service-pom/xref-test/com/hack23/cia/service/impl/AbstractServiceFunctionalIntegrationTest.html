98  	/**
99  	 * Configure mail server.
100 	 *
101 	 * @param createTestApplicationSession
102 	 *            the create test application session
103 	 * @return the simple smtp server
104 	 * @throws IOException
105 	 *             Signals that an I/O exception has occurred.
106 	 */
107 	protected final SimpleSmtpServer configureMailServer(CreateApplicationSessionRequest createTestApplicationSession) throws IOException {
108 		final SimpleSmtpServer dumbster = SimpleSmtpServer.start(SimpleSmtpServer.AUTO_SMTP_PORT);
109 
110 		setAuthenticatedAdminuser();
111 
112 		createTestApplicationSession = createTestApplicationSession();
113 
114 		final ApplicationConfiguration sendEmail = applicationConfigurationService.checkValueOrLoadDefault(
115 				"Email configuration send emails", "Send email", ConfigurationGroup.EXTERNAL_SERVICES,
116 				EmailServiceImpl.class.getSimpleName(), "Send email", "Responsible for sending email",
117 				"application.email.send.email", "false");
118 
119 		final ApplicationConfiguration smtpPort = applicationConfigurationService.checkValueOrLoadDefault(
120 				"Email configuration smtp port", "Smtp port", ConfigurationGroup.EXTERNAL_SERVICES,
121 				EmailServiceImpl.class.getSimpleName(), "Smtp port", "Responsible for sending email",
122 				"application.email.smtp.port", "587");
123 
124 
125 		final ApplicationConfiguration smtpHost = applicationConfigurationService.checkValueOrLoadDefault(
126 				"Email configuration smtp host", "Smtp host", ConfigurationGroup.EXTERNAL_SERVICES,
127 				EmailServiceImpl.class.getSimpleName(), "Smtp host","Responsible for sending email", "application.email.smtp.host", "localhost");
128 
129 		updateApplicationConfiguration(createTestApplicationSession, sendEmail, "true");
130 		updateApplicationConfiguration(createTestApplicationSession, smtpPort,Integer.toString(dumbster.getPort()));
131 		updateApplicationConfiguration(createTestApplicationSession, smtpHost,"localhost");
132 
133 
134 
135 		return dumbster;
136 	}

148 	/**
149 	 * Creates the test application session.
150 	 *
151 	 * @return the creates the application session request
152 	 */
153 	protected final CreateApplicationSessionRequest createTestApplicationSession() {
154 		final Random r = new Random();
155 		final CreateApplicationSessionRequest serviceRequest = new CreateApplicationSessionRequest();
156 		serviceRequest.setIpInformation(r.nextInt(256) + "." + r.nextInt(256) + "." + r.nextInt(256) + "." + r.nextInt(256));
157 		serviceRequest.setLocale("en_US.UTF-8");
158 		serviceRequest.setOperatingSystem("LINUX");
159 		serviceRequest.setSessionId(UUID.randomUUID().toString());
160 		serviceRequest.setSessionType(ApplicationSessionType.ANONYMOUS);
161 		serviceRequest.setUserAgentInformation(
162 				"Mozilla/5.0 (X11; Ubuntu; Linux x86_64; rv:42.0) Gecko/20100101 Firefox/42.0");
163 
164 		final CreateApplicationSessionResponse response = (CreateApplicationSessionResponse) applicationManager
165 				.service(serviceRequest);
166 		assertNotNull(EXPECT_A_RESULT, response);
167 		assertEquals(EXPECT_SUCCESS,ServiceResult.SUCCESS, response.getResult());
168 		return serviceRequest;
169 	}

175 	 * @param createTestApplicationSession
176 	 *            the create test application session
177 	 * @param dumbster
178 	 *            the dumbster
179 	 */
180 	protected final void restoreMailConfiguration(final CreateApplicationSessionRequest createTestApplicationSession,final SimpleSmtpServer dumbster) {
181 		final ApplicationConfiguration sendEmail = applicationConfigurationService.checkValueOrLoadDefault(
182 				"Email configuration send emails", "Send email", ConfigurationGroup.EXTERNAL_SERVICES,
183 				EmailServiceImpl.class.getSimpleName(), "Send email", "Responsible for sending email",
184 				"application.email.send.email", "false");
185 
186 		final ApplicationConfiguration smtpPort = applicationConfigurationService.checkValueOrLoadDefault(
187 				"Email configuration smtp port", "Smtp port", ConfigurationGroup.EXTERNAL_SERVICES,
188 				EmailServiceImpl.class.getSimpleName(), "Smtp port", "Responsible for sending email",
189 				"application.email.smtp.port", "587");
190 
191 		updateApplicationConfiguration(createTestApplicationSession, sendEmail, "false");
192 		updateApplicationConfiguration(createTestApplicationSession, smtpPort,"587");
193 		dumbster.stop();
194 
195 	}

197 	/**
198 	 * Sets the authenticated adminuser.
199 	 */
200 	protected final void setAuthenticatedAdminuser() {
201 		final Collection<SimpleGrantedAuthority> authorities = new ArrayList<>();
202 		authorities.add(new SimpleGrantedAuthority(ROLE_ADMIN));
203 		authorities.add(new SimpleGrantedAuthority(ROLE_ANONYMOUS));
204 
205 		SecurityContextHolder.getContext()
206 				.setAuthentication(new AnonymousAuthenticationToken(KEY, PRINCIPAL, authorities));
207 	}
208 
209 	/**
210 	 * Sets the authenticated anonymous user.
211 	 */
212 	protected final void setAuthenticatedAnonymousUser() {
213 		final Collection<SimpleGrantedAuthority> authorities = new ArrayList<>();
214 		authorities.add(new SimpleGrantedAuthority(ROLE_ANONYMOUS));
215 		SecurityContextHolder.getContext()
216 				.setAuthentication(new AnonymousAuthenticationToken(KEY, PRINCIPAL, authorities));
217 	}

219 	/**
220 	 * Update application configuration.
221 	 *
222 	 * @param createTestApplicationSession
223 	 *            the create test application session
224 	 * @param applicationConfiguration
225 	 *            the application configuration
226 	 * @param propertyValue
227 	 *            the property value
228 	 */
229 	private void updateApplicationConfiguration(final CreateApplicationSessionRequest createTestApplicationSession,
230 			final ApplicationConfiguration applicationConfiguration, final String propertyValue) {
231 
232 		final UpdateApplicationConfigurationRequest serviceRequest = new UpdateApplicationConfigurationRequest();
233 		serviceRequest.setApplicationConfigurationId(applicationConfiguration.getHjid());
234 		serviceRequest.setSessionId(createTestApplicationSession.getSessionId());
235 
236 		serviceRequest.setComponentDescription(applicationConfiguration.getComponentDescription());
237 		serviceRequest.setConfigDescription(applicationConfiguration.getConfigDescription());
238 		serviceRequest.setConfigTitle(applicationConfiguration.getConfigTitle());
239 		serviceRequest.setComponentTitle(applicationConfiguration.getComponentTitle());
240 		serviceRequest.setPropertyValue(propertyValue);
241 
242 		final UpdateApplicationConfigurationResponse response = (UpdateApplicationConfigurationResponse) applicationManager
243 				.service(serviceRequest);
244 		assertEquals(EXPECT_SUCCESS, ServiceResult.SUCCESS, response.getResult());
245 	}
