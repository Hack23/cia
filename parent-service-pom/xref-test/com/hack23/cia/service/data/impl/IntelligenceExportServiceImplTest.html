86  				.sorted((a, b) -> b.compareTo(a))
87  				.forEach(path -> {

213 	@Test
214 	void testWriteJsonToFileRejectsPathTraversal() {
215 		// Arrange
216 		final String jsonContent = "{\"test\": \"data\"}";
217 
218 		// Act & Assert - Test various path traversal attempts
219 		assertThrows(IllegalArgumentException.class, () -> 
220 			service.writeJsonToFile(jsonContent, "../etc/passwd", tempDir.toString()));
221 		
222 		assertThrows(IllegalArgumentException.class, () -> 
223 			service.writeJsonToFile(jsonContent, "..\\windows\\system32\\config", tempDir.toString()));
224 		
225 		assertThrows(IllegalArgumentException.class, () -> 
226 			service.writeJsonToFile(jsonContent, "path/to/file.json", tempDir.toString()));
227 		
228 		assertThrows(IllegalArgumentException.class, () -> 
229 			service.writeJsonToFile(jsonContent, "path\\to\\file.json", tempDir.toString()));
230 	}
