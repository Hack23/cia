80  	@Override
81  	public String exportRiskAssessments() throws IOException {
82  		final List<RuleViolation> violations = ruleViolationDAO.getAll();
83  
84  		final RiskAssessmentExportDTO dto = new RiskAssessmentExportDTO();
85  
86  		final ExportMetadata metadata = new ExportMetadata();
87  		metadata.setSchema("intelligence-schema");
88  		metadata.setRecordCount(violations.size());
89  		metadata.setDataDate(new Date());
90  		dto.setMetadata(metadata);
91  
92  		final List<RiskViolation> riskViolations = new ArrayList<>();
93  		for (final RuleViolation violation : violations) {
94  			final RiskViolation riskViolation = new RiskViolation();
95  			riskViolation.setId(violation.getId());
96  			riskViolation.setDetectedDate(violation.getDetectedDate());
97  			riskViolation.setReferenceId(violation.getReferenceId());
98  			riskViolation.setName(violation.getName());
99  			riskViolation.setResourceType(violation.getResourceType() != null ? violation.getResourceType().toString() : null);
100 			riskViolation.setRuleName(violation.getRuleName());
101 			riskViolation.setRuleDescription(violation.getRuleDescription());
102 			riskViolation.setRuleGroup(violation.getRuleGroup());
103 			riskViolation.setStatus(violation.getStatus() != null ? violation.getStatus().toString() : null);
104 			riskViolation.setPositive(violation.getPositive());
105 			riskViolations.add(riskViolation);
106 		}
107 		dto.setViolations(riskViolations);
108 
109 		return objectMapper.writeValueAsString(dto);
110 	}

112 	@Override
113 	public String exportCoalitionAlignment() throws IOException {
114 		final List<ViewRiksdagenCoalitionAlignmentMatrix> alignments = coalitionAlignmentDAO.getAll();
115 
116 		final CoalitionAlignmentExportDTO dto = new CoalitionAlignmentExportDTO();
117 
118 		final ExportMetadata metadata = new ExportMetadata();
119 		metadata.setSchema("intelligence-schema");
120 		metadata.setRecordCount(alignments.size());
121 		metadata.setDataDate(new Date());
122 		dto.setMetadata(metadata);
123 
124 		final List<PartyAlignment> partyAlignments = new ArrayList<>();
125 		for (final ViewRiksdagenCoalitionAlignmentMatrix alignment : alignments) {
126 			final PartyAlignment partyAlignment = new PartyAlignment();
127 			if (alignment.getEmbeddedId() != null) {
128 				partyAlignment.setParty1(alignment.getEmbeddedId().getParty1());
129 				partyAlignment.setParty2(alignment.getEmbeddedId().getParty2());
130 			}
131 			partyAlignment.setAlignmentRate(alignment.getAlignmentRate());
132 			partyAlignment.setSharedVotes(alignment.getSharedVotes());
133 			partyAlignment.setAlignedVotes(alignment.getAlignedVotes());
134 			partyAlignments.add(partyAlignment);
135 		}
136 		dto.setAlignments(partyAlignments);
137 
138 		return objectMapper.writeValueAsString(dto);
139 	}

181 	@Override
182 	public void writeJsonToFile(final String jsonContent, final String fileName, final String outputDirectory)
183 			throws IOException {
184 		// Validate fileName doesn't contain path separators (subdirectories not supported)
185 		if (fileName.contains("/") || fileName.contains("\\")) {
186 			throw new IllegalArgumentException("Invalid file name: subdirectories not supported, must be a flat file name");
187 		}
188 
189 		// Validate against path traversal by ensuring resolved path stays within target directory
190 		final Path targetDir = Paths.get(outputDirectory).toAbsolutePath().normalize();
191 		final Path resolvedPath = targetDir.resolve(fileName).normalize();
192 		if (!resolvedPath.startsWith(targetDir)) {
193 			throw new IllegalArgumentException("Invalid file name: path traversal detected");
194 		}
195 
196 		final File directory = new File(outputDirectory);
197 		if (!directory.exists() && !directory.mkdirs()) {
198 			throw new IOException("Failed to create directory: " + outputDirectory);
199 		}
200 
201 		Files.write(resolvedPath, jsonContent.getBytes(StandardCharsets.UTF_8));
202 	}
