19  package com.hack23.cia.service.data.impl;
20  
21  import java.util.List;
22  
23  import javax.sql.DataSource;
24  
25  import org.springframework.beans.factory.annotation.Autowired;
26  import org.springframework.jdbc.core.JdbcTemplate;
27  import org.springframework.stereotype.Service;
28  
29  import com.hack23.cia.service.data.api.ViewDataManager;
30  
31  /**
32   * The Class ViewDataManagerImpl.
33   * 
34   * Note: REFRESH MATERIALIZED VIEW (without CONCURRENTLY) in PostgreSQL takes an
35   * ACCESS EXCLUSIVE lock on the materialized view for the entire duration of the
36   * refresh, blocking both reads and writes. If minimal read disruption is required,
37   * use REFRESH MATERIALIZED VIEW CONCURRENTLY (which requires a unique index on the
38   * materialized view). See PostgreSQL documentation for details on concurrent refresh.
39   */
40  @Service
41  final class ViewDataManagerImpl implements ViewDataManager {
42  
43  	/** The data source. */
44  	@Autowired
45  	private DataSource dataSource;
46  
47  	/**
48  	 * Instantiates a new view data manager impl.
49  	 */
50  	public ViewDataManagerImpl() {
51  		super();
52  	}
53  
54  	@Override
55  	public void refreshViews() {
56  		final JdbcTemplate jdbcTemplate = new JdbcTemplate(dataSource);
57  		
58  		// Set statement timeout to 60 minutes (3600 seconds) for long-running refreshes
59  		// This provides a safety limit when running outside a Spring transaction
60  		jdbcTemplate.setQueryTimeout(3600);
61  
62  		// Handle FP changed to L for folkpartiet name changed to liberalerna.
63  		jdbcTemplate.execute("update vote_data set gender='MAN' where gender='M'");
64  		jdbcTemplate.execute("update vote_data set gender='KVINNA' where gender='K'");
65  		jdbcTemplate.execute("update vote_data set gender='MAN' where gender='man'");
66  		jdbcTemplate.execute("update vote_data set gender='KVINNA' where gender='kvinna'");
67  		jdbcTemplate.execute("update vote_data set party='L' where party='FP'");
68  		jdbcTemplate.execute("update person_data set party='L' where party='FP'");
69  		jdbcTemplate.execute("update document_element set org='L' where org='FP' or org='fp'");
70  		jdbcTemplate.execute("update document_data set org='L' where org='FP' or org='fp'");
71  		jdbcTemplate.execute("update committee_document_data set org='L' where org='FP' or org='fp'");
72  		jdbcTemplate.execute("update document_person_reference_da_0 set party_short_code='L' where party_short_code='FP' or party_short_code='fp'");
73  		
74  		// Dynamically refresh materialized views in dependency order
75  		// Query uses same logic as refresh-all-views.sql to calculate dependency levels
76  		// Returns schema-qualified, properly quoted refresh statements
77  		final String dependencyOrderQuery = """
78  			WITH RECURSIVE 
79  			view_deps AS (
80  			    SELECT DISTINCT
81  			        dependent_view.relname AS view_name,
82  			        source_table.relname AS depends_on,
83  			        CASE WHEN source_table.relkind = 'm' THEN TRUE ELSE FALSE END AS depends_on_matview
84  			    FROM pg_depend
85  			    JOIN pg_rewrite ON pg_depend.objid = pg_rewrite.oid
86  			    JOIN pg_class AS dependent_view ON pg_rewrite.ev_class = dependent_view.oid
87  			    JOIN pg_class AS source_table ON pg_depend.refobjid = source_table.oid
88  			    JOIN pg_namespace dependent_ns ON dependent_ns.oid = dependent_view.relnamespace
89  			    JOIN pg_namespace source_ns ON source_ns.oid = source_table.relnamespace
90  			    WHERE dependent_view.relkind = 'm'
91  			      AND source_table.relkind IN ('v', 'm', 'r')
92  			      AND pg_depend.deptype = 'n'
93  			      AND dependent_ns.nspname = 'public'
94  			      AND source_ns.nspname = 'public'
95  			),
96  			mv_dependencies AS (
97  			    SELECT view_name, depends_on FROM view_deps WHERE depends_on_matview = TRUE
98  			),
99  			all_mvs AS (
100 			    SELECT schemaname, matviewname FROM pg_matviews WHERE schemaname = 'public'
101 			),
102 			dependency_depth AS (
103 			    SELECT am.schemaname, am.matviewname AS view_name, 0 AS depth, ARRAY[am.matviewname] AS path
104 			    FROM all_mvs am
105 			    WHERE NOT EXISTS (SELECT 1 FROM mv_dependencies md WHERE md.view_name = am.matviewname)
106 			    UNION ALL
107 			    SELECT am.schemaname, md.view_name, dd.depth + 1, dd.path || md.view_name
108 			    FROM mv_dependencies md
109 			    JOIN dependency_depth dd ON md.depends_on = dd.view_name
110 			    JOIN all_mvs am ON am.matviewname = md.view_name
111 			    WHERE NOT (md.view_name = ANY(dd.path))
112 			),
113 			max_depths AS (
114 			    SELECT schemaname, view_name, MAX(depth) as max_depth 
115 			    FROM dependency_depth 
116 			    GROUP BY schemaname, view_name
117 			)
118 			SELECT format('REFRESH MATERIALIZED VIEW %I.%I', schemaname, view_name) AS refresh_stmt
119 			FROM max_depths 
120 			ORDER BY max_depth, schemaname, view_name
121 			""";
122 		
123 		// Get ordered list of properly-quoted refresh statements
124 		final List<String> refreshStatements = jdbcTemplate.queryForList(dependencyOrderQuery, String.class);
125 		
126 		// Refresh each view in dependency order
127 		for (final String refreshStatement : refreshStatements) {
128 			jdbcTemplate.execute(refreshStatement);
129 		}
130 	}
131 
132 }
